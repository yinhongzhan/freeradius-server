00060 /*
00061  *      If we're limiting outstanding packets, then mark the response
00062  *      as being sent.
00063  */

00065 {
00066         int rtt;
00067         struct timeval now;

00069 
00070         rad_assert(request->listener == listener);

00072 
00073         /*
00074          *      This request timed out.  Remember that, and tell the
00075          *      caller it's OK to read more "detail" file stuff.
00076          */
00077         if (request->reply->code == 0) {
00078                 data->delay_time = data->retry_interval * USEC;
00079                 data->signal = 1;
00080                 data->state = STATE_NO_REPLY;
00081 
00082                 RDEBUG("Detail - No response configured for request %d.  Will retry in %d seconds",
00083                        request->number, data->retry_interval);
00084 
00085                 radius_signal_self(RADIUS_SIGNAL_SELF_DETAIL);
00086                 return 0;
00087         }
00088 
00089         /*
00090          *      We call gettimeofday a lot.  But it should be OK,
00091          *      because there's nothing else to do.
00092          */
00093         gettimeofday(&now, NULL);
00094 
00095         /*
00096          *      If we haven't sent a packet in the last second, reset
00097          *      the RTT.
00098          */
00099         now.tv_sec -= 1;
00100         if (timercmp(&data->last_packet, &now, <)) {
00101                 data->has_rtt = FALSE;
00102         }
00103         now.tv_sec += 1;
00104 
00105         /*
00106          *      Only one detail packet may be outstanding at a time,
00107          *      so it's safe to update some entries in the detail
00108          *      structure.
00109          *
00110          *      We keep smoothed round trip time (SRTT), but not round
00111          *      trip timeout (RTO).  We use SRTT to calculate a rough
00112          *      load factor.
00113          */
00114         rtt = now.tv_sec - request->packet->timestamp.tv_sec;
00115         rtt *= USEC;
00116         rtt += now.tv_usec;
00117         rtt -= request->packet->timestamp.tv_usec;
00118 
00119         /*
00120          *      If we're proxying, the RTT is our processing time,
00121          *      plus the network delay there and back, plus the time
00122          *      on the other end to process the packet.  Ideally, we
00123          *      should remove the network delays from the RTT, but we
00124          *      don't know what they are.
00125          *
00126          *      So, to be safe, we over-estimate the total cost of
00127          *      processing the packet.
00128          */
00129         if (!data->has_rtt) {
00130                 data->has_rtt = TRUE;
00131                 data->srtt = rtt;
00132                 data->rttvar = rtt / 2;
00133 
00134         } else {
00135                 data->rttvar -= data->rttvar >> 2;
00136                 data->rttvar += (data->srtt - rtt);
00137                 data->srtt -= data->srtt >> 3;
00138                 data->srtt += rtt >> 3;
00139         }
00140 
00141         /*
00142          *      Calculate the time we wait before sending the next
00143          *      packet.
00144          *
00145          *      rtt / (rtt + delay) = load_factor / 100
00146          */
00147         data->delay_time = (data->srtt * (100 - data->load_factor)) / (data->load_factor);
00148 
00149         /*
00150          *      Cap delay at 4 packets/s.  If the end system can't
00151          *      handle this, then it's very broken.
00152          */
00153         if (data->delay_time > (USEC / 4)) data->delay_time= USEC / 4;
00154         
00155         RDEBUG3("Received response for request %d.  Will read the next packet in %d seconds",
00156                 request->number, data->delay_time / USEC);
00157         
00158         data->last_packet = now;
00159         data->signal = 1;
00160         data->state = STATE_REPLIED;
00161         data->counter++;
00162         radius_signal_self(RADIUS_SIGNAL_SELF_DETAIL);
00163 
00164         return 0;
00165 }
