00168 static size_t linelog_escape_func(UNUSED REQUEST *request,
00169                 char *out, size_t outlen, const char *in,
00170                 UNUSED void *arg)
00171 {
00172         int len = 0;
00173 
00174         if (outlen == 0) return 0;
00175         if (outlen == 1) {
00176                 *out = '\0';
00177                 return 0;
00178         }
00179 
00180         while (in[0]) {
00181                 if (in[0] >= ' ') {
00182                         if (in[0] == '\\') {
00183                                 if (outlen <= 2) break;
00184                                 outlen--;
00185                                 *out++ = '\\';
00186                                 len++;
00187                         }
00188 
00189                         outlen--;
00190                         if (outlen == 1) break;
00191                         *out++ = *in++;
00192                         len++;
00193                         continue;
00194                 }
00195 
00196                 switch (in[0]) {
00197                 case '\n':
00198                         if (outlen <= 2) break;
00199                         *out++ = '\\';
00200                         *out++ = 'n';
00201                         in++;

00203                         break;
00204
