00163                 radlog(L_ERR, "Path too long in socket filename.");

00180          *      Check the path.
00181          */

00183                 if (errno != ENOENT) {
00184                         radlog(L_ERR, "Failed to stat %s: %s",
00185                                path, strerror(errno));
00186                         close(sockfd);
00187                         return -1;
00188                 }
00189 
00190                 /*
00191                  *      FIXME: Check the enclosing directory?
00192                  */
00193         } else {                /* it exists */
00194                 if (!S_ISREG(buf.st_mode)
00195 #ifdef S_ISSOCK

00201                         return -1;

00209                         close(sockfd);
00210                         return -1;

00214                         radlog(L_ERR, "Failed to delete %s: %s",

00236                 return -1;

00240                 radlog(L_ERR, "Failed listening to %s: %s",

01479                         rcode = i;
