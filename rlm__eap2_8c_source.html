00219 /*
00220  *      Find a a previous EAP-Request sent by us, which matches
00221  *      the current EAP-Response.
00222  *
00223  *      Then, release the handle from the list, and return it to
00224  *      the caller.
00225  *
00226  *      Also since we fill the eap_ds with the present EAP-Response we

00237          *      We key the sessions off of the 'state' attribute, so it
00238          *      must exist.
00239          */

00241         if (!state ||
00242             (state->length != EAP_STATE_LEN)) {
00243                 return NULL;
00244         }
00245 
00246         myHandler.src_ipaddr = request->packet->src_ipaddr;
00247         memcpy(myHandler.state, state->vp_strvalue, sizeof(myHandler.state));
00248 
00249         /*
00250          *      Playing with a data structure shared among threads
00251          *      means that we need a lock, to avoid conflict.
00252          */
00253         pthread_mutex_lock(&(inst->session_mutex));
00254 
00255         /*
00256          *      Check the first few handlers in the list, and delete
00257          *      them if they're too old.  We don't need to check them
00258          *      all, as incoming requests will quickly cause older
00259          *      handlers to be deleted.
00260          *
00261          */
00262         for (i = 0; i < 2; i++) {
00263                 handler = inst->session_head;
00264                 if (handler &&
00265                     ((request->timestamp - handler->timestamp) > inst->timer_limit)) {
00266                         node = rbtree_find(inst->session_tree, handler);
00267                         rad_assert(node != NULL);
00268                         rbtree_delete(inst->session_tree, node);
00269 
00270                         /*
00271                          *      handler == inst->session_head
00272                          */

00274                         if (handler->next) {
00275                                 handler->next->prev = NULL;
00276                         } else {
00277                                 inst->session_head = NULL;
00278                         }
00279                         eap_handler_free(handler);
00280                 }
00281         }
00282 
00283         handler = NULL;

00285         if (node) {
00286                 handler = rbtree_node2data(inst->session_tree, node);
00287 
00288                 /*
00289                  *      Delete old handler from the tree.
00290                  */
00291                 rbtree_delete(inst->session_tree, node);
00292                 
00293                 /*
00294                  *      And unsplice it from the linked list.
00295                  */
00296                 if (handler->prev) {
00297                         handler->prev->next = handler->next;
00298                 } else {
00299                         inst->session_head = handler->next;
00300                 }
00301                 if (handler->next) {
00302                         handler->next->prev = handler->prev;
00303                 } else {
00304                         inst->session_tail = handler->prev;
00305                 }
00306                 handler->prev = handler->next = NULL;
00307         }
00308 
00309         pthread_mutex_unlock(&(inst->session_mutex));
00310 
00311         /*
00312          *      Not found.
00313          */
00314         if (!node) {
00315                 RDEBUG2("Request not found in the list");
00316                 return NULL;
00317         }
00318 
00319         /*
00320          *      Found, but state verification failed.
00321          */
00322         if (!handler) {
00323                 radlog(L_ERR, "rlm_eap2: State verification failed.");
00324                 return NULL;
00325         }
00326 
00327         RDEBUG2("Request found, released from the list");
00328 
00329         return handler;
00330 }

00760 /*
00761  * Handles multiple EAP-Message attrs
00762  * ie concatenates all to get the complete EAP packet.
00763  *
00764  * NOTE: Sometimes Framed-MTU might contain the length of EAP-Message,
00765  *      refer fragmentation in rfc2869.
00766  */
00767 static int eap_vp2data(VALUE_PAIR *vps, void **data, int *data_len)
00768 {
00769         VALUE_PAIR *first, *vp;
00770         unsigned char *ptr;
00771         uint16_t len;
00772         int total_len;
00773 
00774         /*
00775          *      Get only EAP-Message attribute list
00776          */
00777         first = pairfind(vps, PW_EAP_MESSAGE, 0);
00778         if (first == NULL) {
00779                 radlog(L_ERR, "rlm_eap2: EAP-Message not found");
00780                 return -1;
00781         }
00782 
00783         /*
00784          *      Sanity check the length before doing anything.
00785          */
00786         if (first->length < 4) {
00787                 radlog(L_ERR, "rlm_eap2: EAP packet is too short.");
00788                 return -1;
00789         }
00790 
00791         /*
00792          *      Get the Actual length from the EAP packet
00793          *      First EAP-Message contains the EAP packet header
00794          */
00795         memcpy(&len, first->vp_strvalue + 2, sizeof(len));
00796         len = ntohs(len);
00797 
00798         /*
00799          *      Take out even more weird things.
00800          */
00801         if (len < 4) {
00802                 radlog(L_ERR, "rlm_eap2: EAP packet has invalid length.");
00803                 return -1;
00804         }
00805 
00806         /*
00807          *      Sanity check the length, BEFORE malloc'ing memory.
00808          */
00809         total_len = 0;
00810         for (vp = first; vp; vp = pairfind(vp->next, PW_EAP_MESSAGE, 0)) {
00811                 total_len += vp->length;
00812 
00813                 if (total_len > len) {
00814                         radlog(L_ERR, "rlm_eap2: Malformed EAP packet.  Length in packet header does not match actual length");
00815                         return -1;
00816                 }
00817         }
00818 
00819         /*
00820          *      If the length is SMALLER, die, too.
00821          */
00822         if (total_len < len) {
00823                 radlog(L_ERR, "rlm_eap2: Malformed EAP packet.  Length in packet header does not match actual length");
00824                 return -1;
00825         }
00826 
00827         /*
00828          *      Now that we know the lengths are OK, allocate memory.
00829          */
00830         *data = malloc(len);
00831         if (!*data) {
00832                 radlog(L_ERR, "rlm_eap2: out of memory");
00833                 return -1;
00834         }
00835         *data_len = len;
00836 
00837         /*
00838          *      Copy the data from EAP-Message's over to our EAP packet.
00839          */
00840         ptr = *data;
00841 
00842         /* RADIUS ensures order of attrs, so just concatenate all */
00843         for (vp = first; vp; vp = pairfind(vp->next, PW_EAP_MESSAGE, 0)) {
00844                 memcpy(ptr, vp->vp_strvalue, vp->length);
00845                 ptr += vp->length;
00846         }
00847 
00848         return 0;
00849 }
