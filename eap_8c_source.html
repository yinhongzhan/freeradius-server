00073  * Get all the supported EAP-types from config file.

00075 int eaptype_load(EAP_TYPES **type, int eap_type, CONF_SECTION *cs)
00076 {
00077         char            buffer[64];
00078         char            namebuf[64];
00079         const char      *eaptype_name;
00080         EAP_TYPES       *node;
00081 
00082         eaptype_name = eaptype_type2name(eap_type, namebuf, sizeof(namebuf));
00083         snprintf(buffer, sizeof(buffer), "rlm_eap_%s", eaptype_name);
00084 
00085         /* Make room for the EAP-Type */
00086         node = (EAP_TYPES *)malloc(sizeof(EAP_TYPES));
00087         if (node == NULL) {
00088                 radlog(L_ERR, "rlm_eap: out of memory");
00089                 return -1;
00090         }
00091         memset(node, 0, sizeof(*node));
00092 
00093         /* fill in the structure */
00094         node->cs = cs;
00095 
00096         /*
00097          *      In general, this is a terrible idea.  It works here
00098          *      solely because the eap_type2name function returns a
00099          *      'static const char *' pointer sometimes, and we can
00100          *      ONLY link to module which are named in that static
00101          *      array.
00102          */
00103         node->typename = eaptype_name;
00104         node->type_data = NULL;
00105 
00106 #if !defined(WITH_LIBLTDL) && defined(HAVE_DLFCN_H) && defined(RTLD_SELF)

00108         if (node->type) goto open_self;

00112         node->handle = lt_dlopenext(buffer);
00113         if (node->handle == NULL) {

00117                 return -1;
00118         }
00119 
00120         node->type = (EAP_TYPE *)lt_dlsym(node->handle, buffer);
00121         if (!node->type) {
00122                 radlog(L_ERR, "rlm_eap: Failed linking to %s structure in %s: %s",
00123                                 buffer, eaptype_name, lt_dlerror());
00124                 lt_dlclose(node->handle);       /* ignore any errors */
00125                 free(node);
00126                 return -1;
00127         }
00128 
00129 #if !defined(WITH_LIBLTDL) && defined(HAVE_DLFCN_H) && defined(RTLD_SELF)
00130 open_self:
00131 #endif
00132         cf_log_module(cs, "Linked to sub-module %s", buffer);
00133 
00134         cf_log_module(cs, "Instantiating eap-%s", eaptype_name);
00135 
00136         if ((node->type->attach) &&
00137             ((node->type->attach)(node->cs, &(node->type_data)) < 0)) {
00138 
00139                 radlog(L_ERR, "rlm_eap: Failed to initialize type %s",
00140                        eaptype_name);

00150 /*
00151  * Call the appropriate handle with the right eap_type.
00152  */
00153 static int eaptype_call(EAP_TYPES *atype, EAP_HANDLER *handler)
00154 {
00155         int rcode = 1;
00156         REQUEST *request = handler->request;
00157         const char *module = request->module;
00158 
00159         RDEBUG2("processing type %s", atype->typename);
00160         request->module = atype->typename;
00161 
00162         rad_assert(atype != NULL);
00163 
00164         switch (handler->stage) {
00165         case INITIATE:
00166                 if (!atype->type->initiate(atype->type_data, handler))
00167                         rcode = 0;
00168                 break;
00169 
00170         case AUTHORIZE:
00171                 /*
00172                  *   The called function updates the EAP reply packet.
00173                  */
00174                 if (!atype->type->authorize ||
00175                     !atype->type->authorize(atype->type_data, handler))
00176                         rcode = 0;
00177                 break;
00178 
00179         case AUTHENTICATE:
00180                 /*
00181                  *   The called function updates the EAP reply packet.
00182                  */
00183                 if (!atype->type->authenticate ||
00184                     !atype->type->authenticate(atype->type_data, handler))
00185                         rcode = 0;
00186                 break;
00187 
00188         default:
00189                 /* Should never enter here */
00190                 RDEBUG("Internal sanity check failed on eap_type");
00191                 rcode = 0;
00192                 break;
00193         }
00194 
00195         request->module = module;
00196         return rcode;
00197 }

00199 /*
00200  * Based on TYPE, call the appropriate EAP-type handler
00201  * Default to the configured EAP-Type
00202  * for all Unsupported EAP-Types
00203  */
00204 int eaptype_select(rlm_eap_t *inst, EAP_HANDLER *handler)
00205 {
00206         size_t          i;
00207         unsigned int    default_eap_type = inst->default_eap_type;
00208         eaptype_t       *eaptype;
00209         VALUE_PAIR      *vp;
00210         char            namebuf[64];
00211         const char      *eaptype_name;
00212         REQUEST         *request = handler->request;
00213 
00214         eaptype = &handler->eap_ds->response->type;
00215 
00216         /*
00217          *      Don't trust anyone.
00218          */
00219         if ((eaptype->type == 0) ||
00220             (eaptype->type > PW_EAP_MAX_TYPES)) {
00221                 RDEBUG2("Asked to select bad type");
00222                 return EAP_INVALID;
00223         }
00224 
00225         /*
00226          *      Multiple levels of nesting are invalid.
00227          */
00228         if (handler->request->parent && handler->request->parent->parent) {
00229                 RDEBUG2("Multiple levels of TLS nesting is invalid.");
00230                 return EAP_INVALID;
00231         }
00232 
00233         /*
00234          *      Figure out what to do.
00235          */
00236         switch(eaptype->type) {
00237         case PW_EAP_IDENTITY:
00238                 RDEBUG2("EAP Identity");
00239 
00240                 /*
00241                  *      Allow per-user configuration of EAP types.
00242                  */
00243                 vp = pairfind(handler->request->config_items,
00244                               PW_EAP_TYPE, 0);
00245                 if (vp) default_eap_type = vp->vp_integer;
00246 
00247         do_initiate:
00248                 /*
00249                  *      Ensure it's valid.
00250                  */
00251                 if ((default_eap_type < PW_EAP_MD5) ||
00252                     (default_eap_type > PW_EAP_MAX_TYPES) ||
00253                     (inst->types[default_eap_type] == NULL)) {
00254                         RDEBUG2("No such EAP type %s",
00255                                eaptype_type2name(default_eap_type,
00256                                                  namebuf, sizeof(namebuf)));
00257                         return EAP_INVALID;
00258                 }
00259 
00260                 handler->stage = INITIATE;
00261                 handler->eap_type = default_eap_type;
00262 
00263                 if ((default_eap_type == PW_EAP_TNC) &&
00264                     !handler->request->parent) {
00265                         RDEBUG2("ERROR: EAP-TNC must be run inside of a TLS method.");
00266                         return EAP_INVALID;
00267                 }
00268 
00269                 if (eaptype_call(inst->types[default_eap_type],
00270                                  handler) == 0) {
00271                         RDEBUG2("Default EAP type %s failed in initiate",
00272                                eaptype_type2name(default_eap_type,
00273                                                  namebuf, sizeof(namebuf)));
00274                         return EAP_INVALID;
00275                 }
00276                 break;
00277 
00278         case PW_EAP_NAK:
00279                 /*
00280                  *      The NAK data is the preferred EAP type(s) of
00281                  *      the client.
00282                  *
00283                  *      RFC 3748 says to list one or more proposed
00284                  *      alternative types, one per octet, or to use
00285                  *      0 for no alternative.
00286                  */
00287                 RDEBUG2("EAP NAK");
00288 
00289                 /*
00290                  *      Delete old data, if necessary.
00291                  */
00292                 if (handler->opaque && handler->free_opaque) {
00293                         handler->free_opaque(handler->opaque);
00294                         handler->free_opaque = NULL;
00295                         handler->opaque = NULL;
00296                 }
00297 
00298                 if (eaptype->data == NULL) {
00299                         RDEBUG2("Empty NAK packet, cannot decide what EAP type the client wants.");
00300                         return EAP_INVALID;
00301                 }
00302 
00303                 /*
00304                  *      Pick one type out of the one they asked for,
00305                  *      as they may have asked for many.
00306                  */
00307                 default_eap_type = 0;
00308                 vp = pairfind(handler->request->config_items,
00309                               PW_EAP_TYPE, 0);
00310                 for (i = 0; i < eaptype->length; i++) {
00311                         /*
00312                          *      It is invalid to request identity,
00313                          *      notification & nak in nak.
00314                          *
00315                          *      Type 0 is valid, and means there are no
00316                          *      common choices.
00317                          */
00318                         if (eaptype->data[i] < PW_EAP_MD5) {
00319                                 RDEBUG2("NAK asked for bad type %d",
00320                                        eaptype->data[i]);
00321                                 return EAP_INVALID;
00322                         }
00323 
00324                         if ((eaptype->data[i] > PW_EAP_MAX_TYPES) ||
00325                             !inst->types[eaptype->data[i]]) {
00326                                 DICT_VALUE *dv;
00327 
00328                                 dv = dict_valbyattr(PW_EAP_TYPE, 0, eaptype->data[i]);
00329                                 if (dv) {
00330                                         RDEBUG2("NAK asked for unsupported type %s",
00331                                                 dv->name);
00332                                 } else {
00333                                         RDEBUG2("NAK asked for unsupported type %d",
00334                                                 eaptype->data[i]);
00335                                 }
00336                                 continue;
00337                         }
00338 
00339                         eaptype_name = eaptype_type2name(eaptype->data[i],
00340                                                          namebuf,
00341                                                          sizeof(namebuf));
00342                         
00343                         /*
00344                          *      Prevent a firestorm if the client is confused.
00345                          */
00346                         if (handler->eap_type == eaptype->data[i]) {
00347                                 RDEBUG2("ERROR! Our request for %s was NAK'd with a request for %s.  Skipping the requested type.",
00348                                        eaptype_name, eaptype_name);
00349                                 continue;
00350                         }
00351 
00352                         /*
00353                          *      Enforce per-user configuration of EAP
00354                          *      types.
00355                          */
00356                         if (vp && (vp->vp_integer != eaptype->data[i])) {
00357                                 char    mynamebuf[64];
00358                                 RDEBUG2("Client wants %s, while we require %s.  Skipping the requested type.",
00359                                        eaptype_name,
00360                                        eaptype_type2name(vp->vp_integer,
00361                                                          mynamebuf,
00362                                                          sizeof(mynamebuf)));
00363                                 continue;
00364                         }
00365 
00366                         default_eap_type = eaptype->data[i];
00367                         break;
00368                 }
00369 
00370                 /*
00371                  *      We probably want to return 'fail' here...
00372                  */
00373                 if (!default_eap_type) {
00374                         RDEBUG2("No common EAP types found.");
00375                         return EAP_INVALID;
00376                 }
00377                 eaptype_name = eaptype_type2name(default_eap_type,
00378                                                  namebuf, sizeof(namebuf));
00379                 RDEBUG2("EAP-NAK asked for EAP-Type/%s",
00380                        eaptype_name);
00381 
00382                 goto do_initiate;
00383                 break;
00384 
00385                 /*
00386                  *      Key off of the configured sub-modules.
00387                  */
00388                 default:
00389                         eaptype_name = eaptype_type2name(eaptype->type,
00390                                                          namebuf,
00391                                                          sizeof(namebuf));
00392                         RDEBUG2("EAP/%s", eaptype_name);
00393 
00394                         /*
00395                          *      We haven't configured it, it doesn't exit.
00396                          */
00397                         if (!inst->types[eaptype->type]) {
00398                                 RDEBUG2("EAP type %d is unsupported",
00399                                        eaptype->type);
00400                                 return EAP_INVALID;
00401                         }
00402 
00403                         rad_assert(handler->stage == AUTHENTICATE);
00404                         handler->eap_type = eaptype->type;
00405                         if (eaptype_call(inst->types[eaptype->type],
00406                                          handler) == 0) {
00407                                 RDEBUG2("Handler failed in EAP/%s",
00408                                        eaptype_name);
00409                                 return EAP_INVALID;
00410                         }
00411                 break;
00412         }
00413 
00414         return EAP_OK;
00415 }

00418 /*
00419  *      compose EAP reply packet in EAP-Message attr of RADIUS.  If
00420  *      EAP exceeds 253, frame it in multiple EAP-Message attrs.
00421  *
00422  *      Set the RADIUS reply codes based on EAP request codes.  Append
00423  *      any additonal VPs to RADIUS reply
00424  */
00425 int eap_compose(EAP_HANDLER *handler)
00426 {
00427         VALUE_PAIR *vp;
00428         eap_packet_t *eap_packet;
00429         REQUEST *request = handler->request;
00430         EAP_DS *eap_ds = handler->eap_ds;
00431         EAP_PACKET *reply = eap_ds->request;
00432         int rcode;
00433 
00434         /*
00435          *      The Id for the EAP packet to the NAS wasn't set.
00436          *      Do so now.
00437          *
00438          *      LEAP requires the Id to be incremented on EAP-Success
00439          *      in Stage 4, so that we can carry on the conversation
00440          *      where the client asks us to authenticate ourselves
00441          *      in stage 5.
00442          */
00443         if (!eap_ds->set_request_id) {
00444                 /*
00445                  *      Id serves to suppport request/response
00446                  *      retransmission in the EAP layer and as such
00447                  *      must be different for 'adjacent' packets
00448                  *      except in case of success/failure-replies.
00449                  *
00450                  *      RFC2716 (EAP-TLS) requires this to be
00451                  *      incremented, RFC2284 only makes the above-
00452                  *      mentioned restriction.
00453                  */
00454                 reply->id = handler->eap_ds->response->id;
00455 
00456                 switch (reply->code) {
00457                         /*
00458                          *      The Id is a simple "ack" for success
00459                          *      and failure.
00460                          *
00461                          *      RFC 3748 section 4.2 says
00462                          *
00463                          *      ... The Identifier field MUST match
00464                          *      the Identifier field of the Response
00465                          *      packet that it is sent in response
00466                          *      to.
00467                          */
00468                 case PW_EAP_SUCCESS:
00469                 case PW_EAP_FAILURE:
00470                         break;
00471 
00472                         /*
00473                          *      We've sent a response to their
00474                          *      request, the Id is incremented.
00475                          */
00476                 default:
00477                         ++reply->id;
00478                 }
00479         } else {
00480                 RDEBUG2("Underlying EAP-Type set EAP ID to %d",
00481                        reply->id);
00482         }
00483 
00484         /*
00485          *      For Request & Response packets, set the EAP sub-type,
00486          *      if the EAP sub-module didn't already set it.
00487          *
00488          *      This allows the TLS module to be "morphic", and means
00489          *      that the TTLS and PEAP modules can call it to do most
00490          *      of their dirty work.
00491          */
00492         if (((eap_ds->request->code == PW_EAP_REQUEST) ||
00493              (eap_ds->request->code == PW_EAP_RESPONSE)) &&
00494             (eap_ds->request->type.type == 0)) {
00495                 rad_assert(handler->eap_type >= PW_EAP_MD5);
00496                 rad_assert(handler->eap_type <= PW_EAP_MAX_TYPES);
00497 
00498                 eap_ds->request->type.type = handler->eap_type;
00499         }
00500 
00501         /*
00502          *      FIXME: We malloc memory for the eap packet, and then
00503          *      immediately copy that data into VALUE_PAIRs.  This
00504          *      could be done more efficiently...
00505          */
00506         if (eap_wireformat(reply) == EAP_INVALID) {
00507                 return RLM_MODULE_INVALID;
00508         }
00509         eap_packet = (eap_packet_t *)reply->packet;
00510 
00511         vp = eap_packet2vp(eap_packet);
00512         if (!vp) return RLM_MODULE_INVALID;
00513         pairadd(&(request->reply->vps), vp);
00514 
00515         /*
00516          *      EAP-Message is always associated with
00517          *      Message-Authenticator but not vice-versa.
00518          *
00519          *      Don't add a Message-Authenticator if it's already
00520          *      there.
00521          */
00522         vp = pairfind(request->reply->vps, PW_MESSAGE_AUTHENTICATOR, 0);
00523         if (!vp) {
00524                 vp = paircreate(PW_MESSAGE_AUTHENTICATOR, 0, PW_TYPE_OCTETS);
00525                 memset(vp->vp_octets, 0, AUTH_VECTOR_LEN);
00526                 vp->length = AUTH_VECTOR_LEN;
00527                 pairadd(&(request->reply->vps), vp);
00528         }
00529 
00530         /* Set request reply code, but only if it's not already set. */
00531         rcode = RLM_MODULE_OK;
00532         if (!request->reply->code) switch(reply->code) {
00533         case PW_EAP_RESPONSE:
00534                 request->reply->code = PW_AUTHENTICATION_ACK;
00535                 rcode = RLM_MODULE_HANDLED; /* leap weirdness */
00536                 break;
00537         case PW_EAP_SUCCESS:
00538                 request->reply->code = PW_AUTHENTICATION_ACK;
00539                 rcode = RLM_MODULE_OK;
00540                 break;
00541         case PW_EAP_FAILURE:
00542                 request->reply->code = PW_AUTHENTICATION_REJECT;
00543                 rcode = RLM_MODULE_REJECT;
00544                 break;
00545         case PW_EAP_REQUEST:
00546                 request->reply->code = PW_ACCESS_CHALLENGE;
00547                 rcode = RLM_MODULE_HANDLED;
00548                 break;
00549         default:
00550                 /*
00551                  *      When we're pulling MS-CHAPv2 out of EAP-MS-CHAPv2,
00552                  *      we do so WITHOUT setting a reply code, as the
00553                  *      request is being proxied.
00554                  */
00555                 if (request->options & RAD_REQUEST_OPTION_PROXY_EAP) {
00556                         return RLM_MODULE_HANDLED;
00557                 }
00558 
00559                 /* Should never enter here */
00560                 radlog(L_ERR, "rlm_eap: reply code %d is unknown, Rejecting the request.", reply->code);
00561                 request->reply->code = PW_AUTHENTICATION_REJECT;
00562                 reply->code = PW_EAP_FAILURE;
00563                 rcode = RLM_MODULE_REJECT;
00564                 break;
00565         }
00566 
00567         return rcode;
00568 }

00570 /*
00571  * Radius criteria, EAP-Message is invalid without Message-Authenticator
00572  * For EAP_START, send Access-Challenge with EAP Identity request.
00573  */
00574 int eap_start(rlm_eap_t *inst, REQUEST *request)
00575 {
00576         VALUE_PAIR *vp, *proxy;
00577         VALUE_PAIR *eap_msg;
00578 
00579         eap_msg = pairfind(request->packet->vps, PW_EAP_MESSAGE, 0);
00580         if (eap_msg == NULL) {
00581                 RDEBUG2("No EAP-Message, not doing EAP");
00582                 return EAP_NOOP;
00583         }
00584 
00585         /*
00586          *      Look for EAP-Type = None (FreeRADIUS specific attribute)
00587          *      this allows you to NOT do EAP for some users.
00588          */
00589         vp = pairfind(request->packet->vps, PW_EAP_TYPE, 0);
00590         if (vp && vp->vp_integer == 0) {
00591                 RDEBUG2("Found EAP-Message, but EAP-Type = None, so we're not doing EAP.");
00592                 return EAP_NOOP;
00593         }
00594 
00595         /*
00596          *      http://www.freeradius.org/rfc/rfc2869.html#EAP-Message
00597          *
00598          *      Checks for Message-Authenticator are handled by rad_recv().
00599          */
00600 
00601         /*
00602          *      Check for a Proxy-To-Realm.  Don't get excited over LOCAL
00603          *      realms (sigh).
00604          */
00605         proxy = pairfind(request->config_items, PW_PROXY_TO_REALM, 0);
00606         if (proxy) {
00607                 REALM *realm;
00608 
00609                 /*
00610                  *      If it's a LOCAL realm, then we're not proxying
00611                  *      to it.
00612                  */
00613                 realm = realm_find(proxy->vp_strvalue);
00614                 if (!realm || (realm && (realm->auth_pool == NULL))) {
00615                         proxy = NULL;
00616                 }
00617         }
00618 
00619         /*
00620          *      Check the length before de-referencing the contents.
00621          *
00622          *      Lengths of zero are required by the RFC for EAP-Start,
00623          *      but we've never seen them in practice.
00624          *
00625          *      Lengths of two are what we see in practice as
00626          *      EAP-Starts.
00627          */
00628         if ((eap_msg->length == 0) || (eap_msg->length == 2)) {
00629                 EAP_DS *eap_ds;
00630                 EAP_HANDLER handler;
00631 
00632                 /*
00633                  *      It's a valid EAP-Start, but the request
00634                  *      was marked as being proxied.  So we don't
00635                  *      do EAP, as the home server will do it.
00636                  */
00637                 if (proxy) {
00638                 do_proxy:
00639                         RDEBUG2("Request is supposed to be proxied to Realm %s.  Not doing EAP.", proxy->vp_strvalue);
00640                         return EAP_NOOP;
00641                 }
00642 
00643                 RDEBUG2("Got EAP_START message");
00644                 if ((eap_ds = eap_ds_alloc()) == NULL) {
00645                         RDEBUG2("EAP Start failed in allocation");
00646                         return EAP_FAIL;
00647                 }
00648 
00649                 /*
00650                  *      It's an EAP-Start packet.  Tell them to stop wasting
00651                  *      our time, and give us an EAP-Identity packet.
00652                  *
00653                  *      Hmm... we should probably check the contents of the
00654                  *      EAP-Start packet for something...
00655                  */
00656                 eap_ds->request->code = PW_EAP_REQUEST;
00657                 eap_ds->request->type.type = PW_EAP_IDENTITY;
00658 
00659                 /*
00660                  *      We don't have a handler, but eap_compose needs one,
00661                  *      (for various reasons), so we fake it out here.
00662                  */
00663                 memset(&handler, 0, sizeof(handler));
00664                 handler.request = request;
00665                 handler.eap_ds = eap_ds;
00666 
00667                 eap_compose(&handler);
00668 
00669                 eap_ds_free(&eap_ds);
00670                 return EAP_FOUND;
00671         } /* end of handling EAP-Start */
00672 
00673         /*
00674          *      The EAP packet header is 4 bytes, plus one byte of
00675          *      EAP sub-type.  Short packets are discarded, unless
00676          *      we're proxying.
00677          */
00678         if (eap_msg->length < (EAP_HEADER_LEN + 1)) {
00679                 if (proxy) goto do_proxy;
00680 
00681                 RDEBUG2("Ignoring EAP-Message which is too short to be meaningful.");
00682                 return EAP_FAIL;
00683         }
00684 
00685         /*
00686          *      Create an EAP-Type containing the EAP-type
00687          *      from the packet.
00688          */
00689         vp = paircreate(PW_EAP_TYPE, 0, PW_TYPE_INTEGER);
00690         if (vp) {
00691                 vp->vp_integer = eap_msg->vp_octets[4];
00692                 pairadd(&(request->packet->vps), vp);
00693         }
00694 
00695         /*
00696          *      If the request was marked to be proxied, do it now.
00697          *      This is done after checking for a valid length
00698          *      (which may not be good), and after adding the EAP-Type
00699          *      attribute.  This lets other modules selectively cancel
00700          *      proxying based on EAP-Type.
00701          */
00702         if (proxy) goto do_proxy;
00703 
00704         /*
00705          *      From now on, we're supposed to be handling the
00706          *      EAP packet.  We better understand it...
00707          */
00708 
00709         /*
00710          *      We're allowed only a few codes.  Request, Response,
00711          *      Success, or Failure.
00712          */
00713         if ((eap_msg->vp_octets[0] == 0) ||
00714             (eap_msg->vp_octets[0] > PW_EAP_MAX_CODES)) {
00715                 RDEBUG2("Unknown EAP packet");
00716         } else {
00717                 RDEBUG2("EAP packet type %s id %d length %d",
00718                        eap_codes[eap_msg->vp_octets[0]],
00719                        eap_msg->vp_octets[1],
00720                        eap_msg->length);
00721         }
00722 
00723         /*
00724          *      We handle request and responses.  The only other defined
00725          *      codes are success and fail.  The client SHOULD NOT be
00726          *      sending success/fail packets to us, as it doesn't make
00727          *      sense.
00728          */
00729         if ((eap_msg->vp_octets[0] != PW_EAP_REQUEST) &&
00730             (eap_msg->vp_octets[0] != PW_EAP_RESPONSE)) {
00731                 RDEBUG2("Ignoring EAP packet which we don't know how to handle.");
00732                 return EAP_FAIL;
00733         }
00734 
00735         /*
00736          *      We've been told to ignore unknown EAP types, AND it's
00737          *      an unknown type.  Return "NOOP", which will cause the
00738          *      eap_authorize() to return NOOP.
00739          *
00740          *      EAP-Identity, Notification, and NAK are all handled
00741          *      internally, so they never have handlers.
00742          */
00743         if ((eap_msg->vp_octets[4] >= PW_EAP_MD5) &&
00744             inst->ignore_unknown_eap_types &&
00745             ((eap_msg->vp_octets[4] == 0) ||
00746              (eap_msg->vp_octets[4] > PW_EAP_MAX_TYPES) ||
00747              (inst->types[eap_msg->vp_octets[4]] == NULL))) {
00748                 RDEBUG2(" Ignoring Unknown EAP type");
00749                 return EAP_NOOP;
00750         }
00751 
00752         /*
00753          *      They're NAKing the EAP type we wanted to use, and
00754          *      asking for one which we don't support.
00755          *
00756          *      NAK is code + id + length1 + length + NAK
00757          *             + requested EAP type(s).
00758          *
00759          *      We know at this point that we can't handle the
00760          *      request.  We could either return an EAP-Fail here, but
00761          *      it's not too critical.
00762          *
00763          *      By returning "noop", we can ensure that authorize()
00764          *      returns NOOP, and another module may choose to proxy
00765          *      the request.
00766          */
00767         if ((eap_msg->vp_octets[4] == PW_EAP_NAK) &&
00768             (eap_msg->length >= (EAP_HEADER_LEN + 2)) &&
00769             inst->ignore_unknown_eap_types &&
00770             ((eap_msg->vp_octets[5] == 0) ||
00771              (eap_msg->vp_octets[5] > PW_EAP_MAX_TYPES) ||
00772              (inst->types[eap_msg->vp_octets[5]] == NULL))) {
00773                 RDEBUG2("Ignoring NAK with request for unknown EAP type");
00774                 return EAP_NOOP;
00775         }
00776 
00777         if ((eap_msg->vp_octets[4] == PW_EAP_TTLS) ||
00778             (eap_msg->vp_octets[4] == PW_EAP_PEAP)) {
00779                 RDEBUG2("Continuing tunnel setup.");
00780                 return EAP_OK;
00781         }
00782         /*
00783          * We return ok in response to EAP identity and MSCHAP success/fail
00784          * This means we can write:
00785          *
00786          * eap {
00787          *   ok = return
00788          * }
00789          * ldap
00790          * sql
00791          *
00792          * ...in the inner-tunnel, to avoid expensive and unnecessary SQL/LDAP lookups
00793          */
00794         if (eap_msg->vp_octets[4] == PW_EAP_IDENTITY) {
00795                 RDEBUG2("EAP-Identity reply, returning 'ok' so we can short-circuit the rest of authorize");
00796                 return EAP_OK;
00797         }
00798         if ((eap_msg->vp_octets[4] == PW_EAP_MSCHAPV2) && (eap_msg->length >= 6)) {
00799                 switch (eap_msg->vp_octets[5]) {
00800                         case 3:
00801                                 RDEBUG2("EAP-MSCHAPV2 success, returning short-circuit ok");
00802                                 return EAP_OK;
00803                         case 4:
00804                                 RDEBUG2("EAP-MSCHAPV2 failure, returning short-circuit ok");
00805                                 return EAP_OK;
00806                 }
00807         }
00808 
00809         /*
00810          *      Later EAP messages are longer than the 'start'
00811          *      message, so if everything is OK, this function returns
00812          *      'no start found', so that the rest of the EAP code can
00813          *      use the State attribute to match this EAP-Message to
00814          *      an ongoing conversation.
00815          */
00816         RDEBUG2("No EAP Start, assuming it's an on-going EAP conversation");
00817 
00818         return EAP_NOTFOUND;
00819 }

00850  */
00851 static int eap_validation(REQUEST *request, eap_packet_t *eap_packet)
00852 {
00853         uint16_t len;
00854 
00855         memcpy(&len, eap_packet->length, sizeof(uint16_t));
00856         len = ntohs(len);
00857 
00858         /*
00859          *      High level EAP packet checks
00860          */
00861         if ((len <= EAP_HEADER_LEN) ||
00862             ((eap_packet->code != PW_EAP_RESPONSE) &&
00863              (eap_packet->code != PW_EAP_REQUEST)) ||
00864             (eap_packet->data[0] <= 0) ||
00865             (eap_packet->data[0] > PW_EAP_MAX_TYPES)) {
00866 
00867                 radlog_request(L_AUTH, 0, request, 
00868                                "Badly formatted EAP Message: Ignoring the packet");
00869                 return EAP_INVALID;
00870         }
00871 
00872         /* we don't expect notification, but we send it */
00873         if (eap_packet->data[0] == PW_EAP_NOTIFICATION) {
00874                 radlog_request(L_AUTH, 0, request, "Got NOTIFICATION, "
00875                                "Ignoring the packet");
00876                 return EAP_INVALID;
00877         }
00878 
00879         return EAP_VALID;
00880 }

00915 /*
00916  *      Create our Request-Response data structure with the eap packet
00917  */
00918 static EAP_DS *eap_buildds(eap_packet_t **eap_packet_p)
00919 {
00920         EAP_DS          *eap_ds = NULL;
00921         eap_packet_t    *eap_packet = *eap_packet_p;
00922         int             typelen;
00923         uint16_t        len;
00924 
00925         if ((eap_ds = eap_ds_alloc()) == NULL) {
00926                 return NULL;
00927         }
00928 
00929         eap_ds->response->packet = (unsigned char *)eap_packet;
00930         eap_ds->response->code = eap_packet->code;
00931         eap_ds->response->id = eap_packet->id;
00932         eap_ds->response->type.type = eap_packet->data[0];
00933 
00934         memcpy(&len, eap_packet->length, sizeof(uint16_t));
00935         len = ntohs(len);
00936         eap_ds->response->length = len;
00937 
00938         /*
00939          *      We've eaten the eap packet into the eap_ds.
00940          */
00941         *eap_packet_p = NULL;
00942 
00943         /*
00944          *      First 5 bytes in eap, are code + id + length(2) + type.
00945          *
00946          *      The rest is type-specific data.  We skip type while
00947          *      getting typedata from data.
00948          */
00949         typelen = len - 5/*code + id + length + type */;
00950         if (typelen > 0) {
00951                 /*
00952                  *      Since the packet contains the complete
00953                  *      eap_packet, typedata will be a ptr in packet
00954                  *      to its typedata
00955                  */
00956                 eap_ds->response->type.data = eap_ds->response->packet + 5/*code+id+length+type*/;
00957                 eap_ds->response->type.length = typelen;
00958         } else {
00959                 eap_ds->response->type.length = 0;
00960                 eap_ds->response->type.data = NULL;
00961         }
00962 
00963         return eap_ds;
00964 }

01103                         *      Paranoia.  If the NAS *did* set the
