00034 static int chap_authorize(void *instance, REQUEST *request)
00035 {
00036 
00037         /* quiet the compiler */
00038         instance = instance;
00039         request = request;
00040 
00041         if (!pairfind(request->packet->vps, PW_CHAP_PASSWORD, 0)) {
00042                 return RLM_MODULE_NOOP;
00043         }
00044 
00045         if (pairfind(request->config_items, PW_AUTHTYPE, 0) != NULL) {
00046                 RDEBUG2("WARNING: Auth-Type already set.  Not setting to CHAP");
00047                 return RLM_MODULE_NOOP;
00048         }
00049 
00050         RDEBUG("Setting 'Auth-Type := CHAP'");
00051         pairadd(&request->config_items,
00052                 pairmake("Auth-Type", "CHAP", T_OP_EQ));
00053         return RLM_MODULE_OK;
00054 }
