00141 /*************************************************************************
00142  *
00143  *      Function: sql_userparse
00144  *
00145  *      Purpose: Read entries from the database and fill VALUE_PAIR structures
00146  *
00147  *************************************************************************/
00148 int sql_userparse(VALUE_PAIR ** first_pair, SQL_ROW row)
00149 {
00150         VALUE_PAIR *pair;
00151         const char *ptr, *value;
00152         char buf[MAX_STRING_LEN];
00153         char do_xlat = 0;
00154         FR_TOKEN token, operator = T_EOL;
00155 
00156         /*
00157          *      Verify the 'Attribute' field
00158          */
00159         if (row[2] == NULL || row[2][0] == '\0') {
00160                 radlog(L_ERR, "rlm_sql: The 'Attribute' field is empty or NULL, skipping the entire row.");
00161                 return -1;
00162         }
00163 
00164         /*
00165          *      Verify the 'op' field
00166          */
00167         if (row[4] != NULL && row[4][0] != '\0') {
00168                 ptr = row[4];
00169                 operator = gettoken(&ptr, buf, sizeof(buf));
00170                 if ((operator < T_OP_ADD) ||
00171                     (operator > T_OP_CMP_EQ)) {
00172                         radlog(L_ERR, "rlm_sql: Invalid operator \"%s\" for attribute %s", row[4], row[2]);
00173                         return -1;
00174                 }
00175 
00176         } else {
00177                 /*
00178                  *  Complain about empty or invalid 'op' field
00179                  */
00180                 operator = T_OP_CMP_EQ;
00181                 radlog(L_ERR, "rlm_sql: The 'op' field for attribute '%s = %s' is NULL, or non-existent.", row[2], row[3]);
00182                 radlog(L_ERR, "rlm_sql: You MUST FIX THIS if you want the configuration to behave as you expect.");
00183         }
00184 
00185         /*
00186          *      The 'Value' field may be empty or NULL
00187          */
00188         value = row[3];
00189         /*
00190          *      If we have a new-style quoted string, where the
00191          *      *entire* string is quoted, do xlat's.
00192          */
00193         if (row[3] != NULL &&
00194            ((row[3][0] == '\'') || (row[3][0] == '`') || (row[3][0] == '"')) &&
00195            (row[3][0] == row[3][strlen(row[3])-1])) {
00196 
00197                 token = gettoken(&value, buf, sizeof(buf));
00198                 switch (token) {
00199                         /*
00200                          *      Take the unquoted string.
00201                          */
00202                 case T_SINGLE_QUOTED_STRING:
00203                 case T_DOUBLE_QUOTED_STRING:
00204                         value = buf;
00205                         break;
00206 
00207                         /*
00208                          *      Mark the pair to be allocated later.
00209                          */
00210                 case T_BACK_QUOTED_STRING:
00211                         value = NULL;
00212                         do_xlat = 1;
00213                         break;
00214 
00215                         /*
00216                          *      Keep the original string.
00217                          */
00218                 default:
00219                         value = row[3];
00220                         break;
00221                 }
00222         }
00223 
00224         /*
00225          *      Create the pair
00226          */
00227         if (do_xlat) {
00228                 pair = pairmake_xlat(row[2], value, operator);
00229         } else {
00230                 pair = pairmake(row[2], value, operator);
00231         }
00232         if (pair == NULL) {
00233                 radlog(L_ERR, "rlm_sql: Failed to create the pair: %s", fr_strerror());
00234                 return -1;
00235         }
00236 
00237         /*
00238          *      Add the pair into the packet
00239          */
00240         pairadd(first_pair, pair);
00241         return 0;
00242 }

00287          *      If there's no query, return an error.

00303                 /*
00304                  * Run through all available sockets until we exhaust all existing
00305                  * sockets in the pool and fail to establish a *new* connection.
00306                  */
00307                 if (ret == SQL_DOWN) {
00308                         sql_down:
00309                         *sqlsocket = fr_connection_reconnect(inst->pool, *sqlsocket);
00310                         if (!*sqlsocket) return SQL_DOWN;
00311                         
00312                         continue;
00313                 }
00314                 
00315                 if (ret < 0) {
00316                         radlog(L_ERR,
00317                                    "rlm_sql (%s): Database query error: '%s'",
00318                                    inst->config->xlat_name,
00319                                    (inst->module->sql_error)(*sqlsocket, inst->config));
00320                 }
00321                 
00322                 return ret;
