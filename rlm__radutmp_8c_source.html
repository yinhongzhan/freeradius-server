00118 static int radutmp_zap(UNUSED rlm_radutmp_t *inst,
00119                        const char *filename,
00120                        uint32_t nasaddr,
00121                        time_t t)
00122 {
00123         struct radutmp  u;
00124         int             fd;
00125 
00126         if (t == 0) time(&t);
00127 
00128         fd = open(filename, O_RDWR);
00129         if (fd < 0) {
00130                 radlog(L_ERR, "rlm_radutmp: Error accessing file %s: %s",
00131                        filename, strerror(errno));
00132                 return RLM_MODULE_FAIL;
00133         }
00134 
00135         /*
00136          *      Lock the utmp file, prefer lockf() over flock().
00137          */
00138         rad_lockfd(fd, LOCK_LEN);
00139 
00140         /*
00141          *      Find the entry for this NAS / portno combination.
00142          */
00143         while (read(fd, &u, sizeof(u)) == sizeof(u)) {
00144           if ((nasaddr != 0 && nasaddr != u.nas_address) ||
00145               u.type != P_LOGIN)
00146             continue;
00147           /*
00148            *    Match. Zap it.
00149            */
00150           if (lseek(fd, -(off_t)sizeof(u), SEEK_CUR) < 0) {
00151             radlog(L_ERR, "rlm_radutmp: radutmp_zap: negative lseek!");
00152             lseek(fd, (off_t)0, SEEK_SET);
00153           }
00154           u.type = P_IDLE;
00155           u.time = t;
00156           write(fd, &u, sizeof(u));
00157         }
00158         close(fd);      /* and implicitely release the locks */
00159 
00160         return 0;
00161 }
