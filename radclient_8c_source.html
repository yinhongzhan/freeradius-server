00136         prev = radclient->prev;
00137         next = radclient->next;
00138 
00139         if (prev) {
00140                 assert(radclient_head != radclient);
00141                 prev->next = next;
00142         } else if (radclient_head) {
00143                 assert(radclient_head == radclient);
00144                 radclient_head = next;
00145         }
00146 
00147         if (next) {
00148                 assert(radclient_tail != radclient);
00149                 next->prev = prev;

00151                 assert(radclient_tail == radclient);
00152                 radclient_tail = prev;

00398                         fprintf(stderr, "radclient: Request was \"auto\", but request %d in file %s did not contain Packet-Type\n",
00399                                 radclient->packet_number, radclient->filename);
00400                         return -1;
00401                 }
00402 
00403                 radclient->request->code = packet_code;
00404         }
00405         radclient->request->sockfd = -1;

00559                                                 radclient->request, NULL);

00566                                                                 &server_ipaddr,

00573                                 exit(1);

00579                                 exit(1);

00591                 assert(radclient->request->data == NULL);

00649                 radclient->timestamp = time(NULL);
00650                 radclient->tries = 1;
00651                 radclient->resend++;
00652 
00653                 /*
00654                  *      Duplicate found.  Serious error!
00655                  */
00656                 if (!fr_packet_list_insert(pl, &radclient->request)) {
00657                         assert(0 == 1);
00658                 }
00659 
00660 #ifdef WITH_TCP
00661                 /*
00662                  *      WTF?
00663                  */
00664                 if (client_port == 0) {
00665                         client_ipaddr = radclient->request->src_ipaddr;
00666                         client_port = radclient->request->src_port;
00667                 }
00668 #endif
00669 
00670         } else {                /* radclient->request->id >= 0 */
00671                 time_t now = time(NULL);
00672 
00673                 /*
00674                  *      FIXME: Accounting packets are never retried!
00675                  *      The Acct-Delay-Time attribute is updated to
00676                  *      reflect the delay, and the packet is re-sent
00677                  *      from scratch!
00678                  */
00679 
00680                 /*
00681                  *      Not time for a retry, do so.
00682                  */
00683                 if ((now - radclient->timestamp) < timeout) {
00684                         /*
00685                          *      When we walk over the tree sending
00686                          *      packets, we update the minimum time
00687                          *      required to sleep.
00688                          */
00689                         if ((sleep_time == -1) ||
00690                             (sleep_time > (now - radclient->timestamp))) {
00691                                 sleep_time = now - radclient->timestamp;
00692                         }
00693                         return 0;
00694                 }
00695 
00696                 /*
00697                  *      We're not trying later, maybe the packet is done.
00698                  */
00699                 if (radclient->tries == retries) {
00700                         assert(radclient->request->id >= 0);
00701 
00702                         /*
00703                          *      Delete the request from the tree of
00704                          *      outstanding requests.
00705                          */
00706                         fr_packet_list_yank(pl, radclient->request);

00711                         /*
00712                          *      Normally we mark it "done" when we've received
00713                          *      the response, but this is a special case.
00714                          */
00715                         if (radclient->resend == resend_count) {
00716                                 radclient->done = 1;
00717                         }
00718                         totallost++;
00719                         return -1;
00720                 }
00721 
00722                 /*
00723                  *      We are trying later.
00724                  */
00725                 radclient->timestamp = now;
00726                 radclient->tries++;
00727         }
00728 
00729 
00730         /*
00731          *      Send the packet.
00732          */
00733         if (rad_send(radclient->request, NULL, secret) < 0) {
00734                 fprintf(stderr, "radclient: Failed to send packet for ID %d: %s\n",
00735                         radclient->request->id, fr_strerror());
00736         }
00737 
00738         if (fr_debug_flag > 2) print_hex(radclient->request);
00739 
00740         return 0;
00741 }

00743 /*
00744  *      Receive one packet, maybe.
00745  */
00746 static int recv_one_packet(int wait_time)
00747 {
00748         fd_set          set;
00749         struct timeval  tv;
00750         radclient_t     *radclient;
00751         RADIUS_PACKET   *reply, **request_p;
00752         volatile int max_fd;
00753 
00754         /* And wait for reply, timing out as necessary */
00755         FD_ZERO(&set);
00756 
00757         max_fd = fr_packet_list_fd_set(pl, &set);
00758         if (max_fd < 0) exit(1); /* no sockets to listen on! */
00759 
00760         if (wait_time <= 0) {
00761                 tv.tv_sec = 0;
00762         } else {
00763                 tv.tv_sec = wait_time;
00764         }
00765         tv.tv_usec = 0;
00766 
00767         /*
00768          *      No packet was received.
00769          */
00770         if (select(max_fd, &set, NULL, NULL, &tv) <= 0) {
00771                 return 0;
00772         }
00773 
00774         /*
00775          *      Look for the packet.
00776          */
00777 
00778         reply = fr_packet_list_recv(pl, &set);
00779         if (!reply) {
00780                 fprintf(stderr, "radclient: received bad packet: %s\n",
00781                         fr_strerror());
00782 #ifdef WITH_TCP
00783                 /*
00784                  *      If the packet is bad, we close the socket.
00785                  *      I'm not sure how to do that now, so we just
00786                  *      die...
00787                  */
00788                 if (proto) exit(1);
00789 #endif
00790                 return -1;      /* bad packet */
00791         }
00792 
00793         /*
00794          *      udpfromto issues.  We may have bound to "*",
00795          *      and we want to find the replies that are sent to
00796          *      (say) 127.0.0.1.
00797          */
00798         reply->dst_ipaddr = client_ipaddr;
00799         reply->dst_port = client_port;
00800 #ifdef WITH_TCP
00801         reply->src_ipaddr = server_ipaddr;
00802         reply->src_port = server_port;
00803 #endif
00804 
00805         if (fr_debug_flag > 2) print_hex(reply);
00806 
00807         request_p = fr_packet_list_find_byreply(pl, reply);
00808         if (!request_p) {
00809                 fprintf(stderr, "radclient: received response to request we did not send. (id=%d socket %d)\n", reply->id, reply->sockfd);
00810                 rad_free(&reply);
00811                 return -1;      /* got reply to packet we didn't send */
00812         }
00813         radclient = fr_packet2myptr(radclient_t, request, request_p);
00814 
00815         /*
00816          *      Fails the signature validation: not a real reply.
00817          *      FIXME: Silently drop it and listen for another packet.
00818          */
00819         if (rad_verify(reply, radclient->request, secret) < 0) {
00820                 fr_perror("rad_verify");
00821                 totallost++;
00822                 goto packet_done; /* shared secret is incorrect */
00823         }
00824 
00825         fr_packet_list_yank(pl, radclient->request);
00826         if (print_filename) printf("%s:%d %d\n",
00827                                    radclient->filename,
00828                                    radclient->packet_number,
00829                                    reply->code);
00830         deallocate_id(radclient);
00831         radclient->reply = reply;
00832         reply = NULL;
00833 
00834         /*
00835          *      If this fails, we're out of memory.
00836          */
00837         if (rad_decode(radclient->reply, radclient->request, secret) != 0) {
00838                 fr_perror("rad_decode");
00839                 totallost++;
00840                 goto packet_done;
00841         }
00842 
00843         /* libradius debug already prints out the value pairs for us */
00844         if (!fr_debug_flag && do_output) {
00845                 printf("Received response ID %d, code %d, length = %zd\n",
00846                        radclient->reply->id, radclient->reply->code,
00847                        radclient->reply->data_len);
00848                 vp_printlist(stdout, radclient->reply->vps);
00849         }
00850 
00851         if ((radclient->reply->code == PW_AUTHENTICATION_ACK) ||
00852             (radclient->reply->code == PW_ACCOUNTING_RESPONSE) ||
00853             (radclient->reply->code == PW_COA_ACK) ||
00854             (radclient->reply->code == PW_DISCONNECT_ACK)) {
00855                 success = 1;            /* have a good response */
00856                 totalapp++;
00857         } else {
00858                 totaldeny++;
00859         }
00860         
00861         if (radclient->resend == resend_count) {
00862                 radclient->done = 1;
00863         }
00864 
00865  packet_done:
00866         rad_free(&radclient->reply);
00867         rad_free(&reply);       /* may be NULL */
00868 
00869         return 0;
00870 }
