00157 static int
00158 send_pwd_request (pwd_session_t *sess, EAP_DS *eap_ds)
00159 {
00160     int len;
00161     uint16_t totlen;
00162     pwd_hdr *hdr;
00163 
00164     len = (sess->out_buf_len - sess->out_buf_pos) + sizeof(pwd_hdr);
00165     rad_assert(len > 0);
00166     eap_ds->request->code = PW_EAP_REQUEST;
00167     eap_ds->request->type.type = PW_EAP_PWD;
00168     eap_ds->request->type.length = (len > sess->mtu) ? sess->mtu : len;
00169     eap_ds->request->type.data = malloc(eap_ds->request->type.length);
00170     memset(eap_ds->request->type.data, 0, eap_ds->request->type.length);
00171 
00172     hdr = (pwd_hdr *)eap_ds->request->type.data;
00173     switch (sess->state) {
00174         case PWD_STATE_ID_REQ:
00175             EAP_PWD_SET_EXCHANGE(hdr, EAP_PWD_EXCH_ID);
00176             break;
00177         case PWD_STATE_COMMIT:
00178             EAP_PWD_SET_EXCHANGE(hdr, EAP_PWD_EXCH_COMMIT);
00179             break;
00180         case PWD_STATE_CONFIRM:
00181             EAP_PWD_SET_EXCHANGE(hdr, EAP_PWD_EXCH_CONFIRM);
00182             break;
00183         default:
00184             DEBUG2("pwd state is messed up! Cannot send request");
00185             return 0;
00186     }
00187     /*
00188      * are we fragmenting?
00189      */
00190     if ((int)((sess->out_buf_len - sess->out_buf_pos) + sizeof(pwd_hdr)) > sess->mtu) {
00191         EAP_PWD_SET_MORE_BIT(hdr);
00192         if (sess->out_buf_pos == 0) {
00193             /*
00194              * the first fragment, add the total length
00195              */
00196             EAP_PWD_SET_LENGTH_BIT(hdr);
00197             totlen = ntohs(sess->out_buf_len);
00198             memcpy(hdr->data, (char *)&totlen, sizeof(totlen));
00199             memcpy(hdr->data + sizeof(uint16_t),
00200                    sess->out_buf,
00201                    sess->mtu - sizeof(pwd_hdr) - sizeof(uint16_t));
00202             sess->out_buf_pos += (sess->mtu - sizeof(pwd_hdr) - sizeof(uint16_t));
00203         } else {
00204             /*
00205              * an intermediate fragment
00206              */
00207             memcpy(hdr->data, sess->out_buf + sess->out_buf_pos, (sess->mtu - sizeof(pwd_hdr)));
00208             sess->out_buf_pos += (sess->mtu - sizeof(pwd_hdr));
00209         }
00210     } else {
00211         /*
00212          * either it's not a fragment or it's the last fragment. 
00213          * The out buffer isn't needed anymore though so get rid of it.
00214          */
00215         memcpy(hdr->data, sess->out_buf + sess->out_buf_pos, 
00216                (sess->out_buf_len - sess->out_buf_pos));
00217         free(sess->out_buf);
00218         sess->out_buf = NULL;
00219         sess->out_buf_pos = sess->out_buf_len = 0;
00220     }
00221     return 1;
00222 }

00344     response = handler->eap_ds->response;
00345     hdr = (pwd_hdr *)response->type.data;
00346 
00347     buf = hdr->data;
00348     len = response->type.length - sizeof(pwd_hdr);
00349 
00350     /*
00351      * see if we're fragmenting, if so continue until we're done
00352      */
00353     if (pwd_session->out_buf_pos) {
00354         if (len) {
00355             RDEBUG2("pwd got something more than an ACK for a fragment");
00356         }
00357         return send_pwd_request(pwd_session, eap_ds);
00358     }
00359 
00360     /*
00361      * the first fragment will have a total length, make a
00362      * buffer to hold all the fragments
00363      */
00364     if (EAP_PWD_GET_LENGTH_BIT(hdr)) {
00365         if (pwd_session->in_buf) {
00366             RDEBUG2("pwd already alloced buffer for fragments");
00367             return 0;

00370         if ((pwd_session->in_buf = malloc(pwd_session->in_buf_len)) == NULL) {

00372                     pwd_session->in_buf_len);

00378         len -= sizeof(uint16_t);
00379     }
00380

00383      * buffer those fragments!
00384      */
00385     if (EAP_PWD_GET_MORE_BIT(hdr)) {
00386         rad_assert(pwd_session->in_buf != NULL);
00387         if ((pwd_session->in_buf_pos + len) > pwd_session->in_buf_len) {
00388             RDEBUG2("pwd will not overflow a fragment buffer. Nope, not prudent.");
00389             return 0;
