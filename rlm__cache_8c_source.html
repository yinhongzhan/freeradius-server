00140 /*
00141  *      Find a cached entry.
00142  */
00143 static rlm_cache_entry_t *cache_find(rlm_cache_t *inst, REQUEST *request,
00144                                      const char *key)
00145 {
00146         int ttl;
00147         rlm_cache_entry_t *c, my_c;
00148         VALUE_PAIR *vp;
00149 
00150         /*
00151          *      Look at the expiry heap.
00152          */
00153         c = fr_heap_peek(inst->heap);
00154         if (!c) {
00155                 rad_assert(rbtree_num_elements(inst->cache) == 0);
00156                 return NULL;
00157         }
00158 
00159         /*
00160          *      If it's time to expire an old entry, do so now.
00161          */
00162         if (c->expires < request->timestamp) {
00163                 fr_heap_extract(inst->heap, c);
00164                 rbtree_deletebydata(inst->cache, c);
00165         }
00166 
00167         /*
00168          *      Is there an entry for this key?
00169          */
00170         my_c.key = key;
00171         c = rbtree_finddata(inst->cache, &my_c);
00172         if (!c) return NULL;
00173 
00174         /*
00175          *      Yes, but it expired, OR the "forget all" epoch has
00176          *      passed.  Delete it, and pretend it doesn't exist.
00177          */
00178         if ((c->expires < request->timestamp) ||
00179             (c->created < inst->epoch)) {
00180         delete:
00181                 DEBUG("rlm_cache: Entry has expired, removing");
00182 
00183                 fr_heap_extract(inst->heap, c);
00184                 rbtree_deletebydata(inst->cache, c);
00185                 
00186                 return NULL;
00187         }
00188 
00189         DEBUG("rlm_cache: Found entry for \"%s\"", key);
00190 
00191         /*
00192          *      Update the expiry time based on the TTL.
00193          *      A TTL of 0 means "delete from the cache".
00194          */
00195         vp = pairfind(request->config_items, PW_CACHE_TTL, 0);
00196         if (vp) {
00197                 if (vp->vp_integer == 0) goto delete;
00198                 
00199                 ttl = vp->vp_integer;
00200                 c->expires = request->timestamp + ttl;
00201                 DEBUG("rlm_cache: Adding %d to the TTL", ttl);
00202         }
00203         c->hits++;
00204 
00205         return c;
00206 }

00564 /*
00565  *      Do caching checks.  Since we can update ANY VP list, we do
00566  *      exactly the same thing for all sections (autz / auth / etc.)
00567  *
00568  *      If you want to cache something different in different sections,
00569  *      configure another cache module.
00570  */
00571 static int cache_it(void *instance, REQUEST *request)
00572 {
00573         rlm_cache_entry_t *c;
00574         rlm_cache_t *inst = instance;
00575         VALUE_PAIR *vp;
00576         char buffer[1024];
00577 
00578         radius_xlat(buffer, sizeof(buffer), inst->key, request, NULL, NULL);
00579 
00580         c = cache_find(inst, request, buffer);
00581         
00582         /*
00583          *      If yes, only return whether we found a valid cache entry
00584          */
00585         vp = pairfind(request->config_items, PW_CACHE_STATUS_ONLY, 0);
00586         if (vp && vp->vp_integer) {
00587                 return c ?
00588                         RLM_MODULE_OK:
00589                         RLM_MODULE_NOTFOUND;
00590         }
00591         
00592         if (c) {
00593                 cache_merge(inst, request, c);
00594                 return RLM_MODULE_OK;
00595         }
00596 
00597         c = cache_add(inst, request, buffer);
00598         if (!c) return RLM_MODULE_NOOP;
00599 
00600         cache_merge(inst, request, c);
00601 
00602         return RLM_MODULE_UPDATED;
00603 }
