00662 /*
00663  * this code runs the EAP-SIM client state machine.
00664  * the *request* is from the server.
00665  * the *reponse* is to the server.
00666  *
00667  */
00668 static int respond_eap_sim(RADIUS_PACKET *req,
00669                            RADIUS_PACKET *resp)
00670 {
00671         enum eapsim_clientstates state, newstate;
00672         enum eapsim_subtype subtype;
00673         VALUE_PAIR *vp, *statevp, *radstate, *eapid;
00674         char statenamebuf[32], subtypenamebuf[32];
00675 
00676         if ((radstate = paircopy2(req->vps, PW_STATE, 0)) == NULL)
00677         {
00678                 return 0;
00679         }
00680 
00681         if ((eapid = paircopy2(req->vps, ATTRIBUTE_EAP_ID, 0)) == NULL)
00682         {
00683                 return 0;
00684         }
00685 
00686         /* first, dig up the state from the request packet, setting
00687          * outselves to be in EAP-SIM-Start state if there is none.
00688          */
00689 
00690         if((statevp = pairfind(resp->vps, ATTRIBUTE_EAP_SIM_STATE, 0)) == NULL)
00691         {
00692                 /* must be initial request */
00693                 statevp = paircreate(ATTRIBUTE_EAP_SIM_STATE, 0, PW_TYPE_INTEGER);
00694                 statevp->vp_integer = eapsim_client_init;
00695                 pairreplace(&(resp->vps), statevp);
00696         }
00697         state = statevp->vp_integer;
00698 
00699         /*
00700          * map the attributes, and authenticate them.
00701          */
00702         unmap_eapsim_types(req);
00703 
00704         if((vp = pairfind(req->vps, ATTRIBUTE_EAP_SIM_SUBTYPE, 0)) == NULL)
00705         {
00706                 return 0;
00707         }
00708         subtype = vp->vp_integer;
00709 
00710         /*
00711          * look for the appropriate state, and process incoming message
00712          */
00713         switch(state) {
00714         case eapsim_client_init:
00715                 switch(subtype) {
00716                 case eapsim_start:
00717                         newstate = process_eap_start(req, resp);
00718                         break;
00719 
00720                 case eapsim_challenge:
00721                 case eapsim_notification:
00722                 case eapsim_reauth:
00723                 default:
00724                         fprintf(stderr, "radeapclient: sim in state %s message %s is illegal. Reply dropped.\n",
00725                                 sim_state2name(state, statenamebuf, sizeof(statenamebuf)),
00726                                 sim_subtype2name(subtype, subtypenamebuf, sizeof(subtypenamebuf)));
00727                         /* invalid state, drop message */
00728                         return 0;
00729                 }
00730                 break;
00731 
00732         case eapsim_client_start:
00733                 switch(subtype) {
00734                 case eapsim_start:
00735                         /* NOT SURE ABOUT THIS ONE, retransmit, I guess */
00736                         newstate = process_eap_start(req, resp);
00737                         break;
00738 
00739                 case eapsim_challenge:
00740                         newstate = process_eap_challenge(req, resp);
00741                         break;
00742 
00743                 default:
00744                         fprintf(stderr, "radeapclient: sim in state %s message %s is illegal. Reply dropped.\n",
00745                                 sim_state2name(state, statenamebuf, sizeof(statenamebuf)),
00746                                 sim_subtype2name(subtype, subtypenamebuf, sizeof(subtypenamebuf)));
00747                         /* invalid state, drop message */
00748                         return 0;
00749                 }
00750                 break;
00751 
00752 
00753         default:
00754                 fprintf(stderr, "radeapclient: sim in illegal state %s\n",
00755                         sim_state2name(state, statenamebuf, sizeof(statenamebuf)));
00756                 return 0;
00757         }
00758 
00759         /* copy the eap state object in */
00760         pairreplace(&(resp->vps), eapid);
00761 
00762         /* update stete info, and send new packet */
00763         map_eapsim_types(resp);
00764 
00765         /* copy the radius state object in */
00766         pairreplace(&(resp->vps), radstate);
00767 
00768         statevp->vp_integer = newstate;
00769         return 1;
00770 }

00772 static int respond_eap_md5(RADIUS_PACKET *req,
00773                            RADIUS_PACKET *rep)
00774 {
00775         VALUE_PAIR *vp, *id, *state;
00776         size_t valuesize, namesize;
00777         uint8_t identifier;
00778         uint8_t *value;
00779         uint8_t *name;
00780         FR_MD5_CTX      context;
00781         uint8_t    response[16];
00782 
00783         cleanresp(rep);
00784 
00785         if ((state = paircopy2(req->vps, PW_STATE, 0)) == NULL)
00786         {
00787                 fprintf(stderr, "radeapclient: no state attribute found\n");
00788                 return 0;
00789         }
00790 
00791         if ((id = paircopy2(req->vps, ATTRIBUTE_EAP_ID, 0)) == NULL)
00792         {
00793                 fprintf(stderr, "radeapclient: no EAP-ID attribute found\n");
00794                 return 0;
00795         }
00796         identifier = id->vp_integer;
00797 
00798         if ((vp = pairfind(req->vps, ATTRIBUTE_EAP_BASE+PW_EAP_MD5, 0)) == NULL)
00799         {
00800                 fprintf(stderr, "radeapclient: no EAP-MD5 attribute found\n");
00801                 return 0;
00802         }
00803 
00804         /* got the details of the MD5 challenge */
00805         valuesize = vp->vp_octets[0];
00806         value = &vp->vp_octets[1];
00807         name  = &vp->vp_octets[valuesize+1];
00808         namesize = vp->length - (valuesize + 1);
00809 
00810         /* sanitize items */
00811         if(valuesize > vp->length)
00812         {
00813                 fprintf(stderr, "radeapclient: md5 valuesize if too big (%u > %u)\n",
00814                         (unsigned int) valuesize, (unsigned int) vp->length);
00815                 return 0;
00816         }
00817 
00818         /* now do the CHAP operation ourself, rather than build the
00819          * buffer. We could also call rad_chap_encode, but it wants
00820          * a CHAP-Challenge, which we don't want to bother with.
00821          */
00822         fr_MD5Init(&context);
00823         fr_MD5Update(&context, &identifier, 1);
00824         fr_MD5Update(&context, (uint8_t *) password, strlen(password));
00825         fr_MD5Update(&context, value, valuesize);
00826         fr_MD5Final(response, &context);
00827 
00828         vp = paircreate(ATTRIBUTE_EAP_BASE+PW_EAP_MD5, 0, PW_TYPE_OCTETS);
00829         vp->vp_octets[0]=16;
00830         memcpy(&vp->vp_strvalue[1], response, 16);
00831         vp->length = 17;
00832 
00833         pairreplace(&(rep->vps), vp);
00834 
00835         pairreplace(&(rep->vps), id);
00836 
00837         /* copy the state object in */
00838         pairreplace(&(rep->vps), state);
00839 
00840         return 1;
00841 }

00845 static int sendrecv_eap(RADIUS_PACKET *rep)
00846 {
00847         RADIUS_PACKET *req = NULL;
00848         VALUE_PAIR *vp, *vpnext;
00849         int tried_eap_md5 = 0;
00850 
00851         /*
00852          *      Keep a copy of the the User-Password attribute.
00853          */
00854         if ((vp = pairfind(rep->vps, PW_CLEARTEXT_PASSWORD, 0)) != NULL) {
00855                 strlcpy(password, (char *)vp->vp_strvalue, sizeof(vp->vp_strvalue));
00856 
00857         } else  if ((vp = pairfind(rep->vps, PW_USER_PASSWORD, 0)) != NULL) {
00858                 strlcpy(password, (char *)vp->vp_strvalue, sizeof(vp->vp_strvalue));
00859                 /*
00860                  *      Otherwise keep a copy of the CHAP-Password attribute.
00861                  */
00862         } else if ((vp = pairfind(rep->vps, PW_CHAP_PASSWORD, 0)) != NULL) {
00863                 strlcpy(password, (char *)vp->vp_strvalue, sizeof(vp->vp_strvalue));
00864         } else {
00865                 *password = '\0';
00866         }
00867 
00868  again:
00869         rep->id++;
00870 
00871         /*
00872          * if there are EAP types, encode them into an EAP-Message
00873          *
00874          */
00875         map_eap_types(rep);
00876 
00877         /*
00878          *  Fix up Digest-Attributes issues
00879          */
00880         for (vp = rep->vps; vp != NULL; vp = vp->next) {
00881                 switch (vp->attribute) {
00882                 default:
00883                         break;
00884 
00885                 case PW_DIGEST_REALM:
00886                 case PW_DIGEST_NONCE:
00887                 case PW_DIGEST_METHOD:
00888                 case PW_DIGEST_URI:
00889                 case PW_DIGEST_QOP:
00890                 case PW_DIGEST_ALGORITHM:
00891                 case PW_DIGEST_BODY_DIGEST:
00892                 case PW_DIGEST_CNONCE:
00893                 case PW_DIGEST_NONCE_COUNT:
00894                 case PW_DIGEST_USER_NAME:
00895                         /* overlapping! */
00896                         memmove(&vp->vp_strvalue[2], &vp->vp_octets[0], vp->length);
00897                         vp->vp_octets[0] = vp->attribute - PW_DIGEST_REALM + 1;
00898                         vp->length += 2;
00899                         vp->vp_octets[1] = vp->length;
00900                         vp->attribute = PW_DIGEST_ATTRIBUTES;
00901                         break;
00902                 }
00903         }
00904 
00905         /*
00906          *      If we've already sent a packet, free up the old
00907          *      one, and ensure that the next packet has a unique
00908          *      ID and authentication vector.
00909          */
00910         if (rep->data) {
00911                 free(rep->data);
00912                 rep->data = NULL;
00913         }
00914 
00915         fr_md5_calc(rep->vector, rep->vector,
00916                         sizeof(rep->vector));
00917 
00918         if (*password != '\0') {
00919                 if ((vp = pairfind(rep->vps, PW_CLEARTEXT_PASSWORD, 0)) != NULL) {
00920                         strlcpy((char *)vp->vp_strvalue, password, sizeof(vp->vp_strvalue));
00921                         vp->length = strlen(password);
00922 
00923                 } else if ((vp = pairfind(rep->vps, PW_USER_PASSWORD, 0)) != NULL) {
00924                         strlcpy((char *)vp->vp_strvalue, password, sizeof(vp->vp_strvalue));
00925                         vp->length = strlen(password);
00926 
00927                 } else if ((vp = pairfind(rep->vps, PW_CHAP_PASSWORD, 0)) != NULL) {
00928                         strlcpy((char *)vp->vp_strvalue, password, sizeof(vp->vp_strvalue));
00929                         vp->length = strlen(password);
00930 
00931                         rad_chap_encode(rep, vp->vp_octets, rep->id, vp);
00932                         vp->length = 17;
00933                 }
00934         } /* there WAS a password */
00935 
00936         /* send the response, wait for the next request */
00937         send_packet(rep, &req);
00938 
00939         /* okay got back the packet, go and decode the EAP-Message. */
00940         unmap_eap_types(req);
00941 
00942         debug_packet(req, R_RECV);
00943 
00944         /* now look for the code type. */
00945         for (vp = req->vps; vp != NULL; vp = vpnext) {
00946                 vpnext = vp->next;
00947 
00948                 switch (vp->attribute) {
00949                 default:
00950                         break;
00951 
00952                 case ATTRIBUTE_EAP_BASE+PW_EAP_MD5:
00953                         if(respond_eap_md5(req, rep) && tried_eap_md5 < 3)
00954                         {
00955                                 tried_eap_md5++;
00956                                 goto again;
00957                         }
00958                         break;
00959 
00960                 case ATTRIBUTE_EAP_BASE+PW_EAP_SIM:
00961                         if(respond_eap_sim(req, rep))
00962                         {
00963                                 goto again;
00964                         }
00965                         break;
00966                 }
00967         }
00968 
00969         return 1;
00970 }

01245  * given a radius request with some attributes in the EAP range, build
01246  * them all into a single EAP-Message body.
01247  *
01248  * Note that this function will build multiple EAP-Message bodies
01249  * if there are multiple eligible EAP-types. This is incorrect, as the
01250  * recipient will in fact concatenate them.
01251  *
01252  * XXX - we could break the loop once we process one type. Maybe this
01253  *       just deserves an assert?

01255  */
01256 static void map_eap_types(RADIUS_PACKET *req)
01257 {
01258         VALUE_PAIR *vp, *vpnext;
01259         int id, eapcode;
01260         EAP_PACKET ep;
01261         int eap_type;
01262 
01263         vp = pairfind(req->vps, ATTRIBUTE_EAP_ID, 0);
01264         if(vp == NULL) {
01265                 id = ((int)getpid() & 0xff);
01266         } else {
01267                 id = vp->vp_integer;
01268         }
01269 
01270         vp = pairfind(req->vps, ATTRIBUTE_EAP_CODE, 0);
01271         if(vp == NULL) {
01272                 eapcode = PW_EAP_REQUEST;
01273         } else {
01274                 eapcode = vp->vp_integer;
01275         }
01276 
01277 
01278         for(vp = req->vps; vp != NULL; vp = vpnext) {
01279                 /* save it in case it changes! */
01280                 vpnext = vp->next;
01281 
01282                 if(vp->attribute >= ATTRIBUTE_EAP_BASE &&
01283                    vp->attribute < ATTRIBUTE_EAP_BASE+256) {
01284                         break;
01285                 }
01286         }
01287 
01288         if(vp == NULL) {
01289                 return;
01290         }
01291 
01292         eap_type = vp->attribute - ATTRIBUTE_EAP_BASE;
01293 
01294         switch(eap_type) {
01295         case PW_EAP_IDENTITY:
01296         case PW_EAP_NOTIFICATION:
01297         case PW_EAP_NAK:
01298         case PW_EAP_MD5:
01299         case PW_EAP_OTP:
01300         case PW_EAP_GTC:
01301         case PW_EAP_TLS:
01302         case PW_EAP_LEAP:
01303         case PW_EAP_TTLS:
01304         case PW_EAP_PEAP:
01305         default:
01306                 /*
01307                  * no known special handling, it is just encoded as an
01308                  * EAP-message with the given type.
01309                  */
01310 
01311                 /* nuke any existing EAP-Messages */
01312                 pairdelete(&req->vps, PW_EAP_MESSAGE, 0);
01313 
01314                 memset(&ep, 0, sizeof(ep));
01315                 ep.code = eapcode;
01316                 ep.id   = id;
01317                 ep.type.type = eap_type;
01318                 ep.type.length = vp->length;
01319                 ep.type.data = malloc(vp->length);
01320                 memcpy(ep.type.data,vp->vp_octets, vp->length);

01325 /*
01326  * given a radius request with an EAP-Message body, decode it specific
01327  * attributes.
01328  */
01329 static void unmap_eap_types(RADIUS_PACKET *rep)
01330 {
01331         VALUE_PAIR *eap1;
01332         eap_packet_t *e;
01333         int len;
01334         int type;
01335 
01336         /* find eap message */
01337         e = eap_vp2packet(rep->vps);
01338 
01339         /* nothing to do! */
01340         if(e == NULL) return;
01341 
01342         /* create EAP-ID and EAP-CODE attributes to start */
01343         eap1 = paircreate(ATTRIBUTE_EAP_ID, 0, PW_TYPE_INTEGER);
01344         eap1->vp_integer = e->id;
01345         pairadd(&(rep->vps), eap1);
01346 
01347         eap1 = paircreate(ATTRIBUTE_EAP_CODE, 0, PW_TYPE_INTEGER);
01348         eap1->vp_integer = e->code;
01349         pairadd(&(rep->vps), eap1);
01350 
01351         switch(e->code)
01352         {
01353         default:
01354         case PW_EAP_SUCCESS:
01355         case PW_EAP_FAILURE:
01356                 /* no data */
01357                 break;
01358 
01359         case PW_EAP_REQUEST:
01360         case PW_EAP_RESPONSE:
01361                 /* there is a type field, which we use to create
01362                  * a new attribute */
01363 
01364                 /* the length was decode already into the attribute
01365                  * length, and was checked already. Network byte
01366                  * order, just pull it out using math.
01367                  */
01368                 len = e->length[0]*256 + e->length[1];
01369 
01370                 /* verify the length is big enough to hold type */
01371                 if(len < 5)
01372                 {
01373                         free(e);
01374                         return;
01375                 }
01376 
01377                 type = e->data[0];
01378 
01379                 type += ATTRIBUTE_EAP_BASE;
01380                 len -= 5;
01381 
01382                 if(len > MAX_STRING_LEN) {
01383                         len = MAX_STRING_LEN;
01384                 }
01385 
01386                 eap1 = paircreate(type, 0, PW_TYPE_OCTETS);
01387                 memcpy(eap1->vp_strvalue, &e->data[1], len);
01388                 eap1->length = len;
01389                 pairadd(&(rep->vps), eap1);
01390                 break;
01391         }
01392 
01393         free(e);
01394         return;
01395 }
