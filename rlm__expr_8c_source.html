00086 static int get_number(REQUEST *request, const char **string, int64_t *answer)
00087 {
00088         int             i, found;
00089         int64_t         result;
00090         int64_t         x;
00091         const char      *p;
00092         expr_token_t    this;
00093 
00094         /*
00095          *  Loop over the input.
00096          */
00097         result = 0;
00098         this = TOKEN_NONE;
00099 
00100         for (p = *string; *p != '\0'; /* nothing */) {
00101                 if ((*p == ' ') ||
00102                     (*p == '\t')) {
00103                         p++;
00104                         continue;
00105                 }
00106 
00107                 /*
00108                  *  Discover which token it is.
00109                  */
00110                 found = FALSE;
00111                 for (i = 0; map[i].token != TOKEN_LAST; i++) {
00112                         if (*p == map[i].op) {
00113                                 if (this != TOKEN_NONE) {
00114                                         RDEBUG2("Invalid operator at \"%s\"", p);
00115                                         return -1;
00116                                 }
00117                                 this = map[i].token;
00118                                 p++;
00119                                 found = TRUE;
00120                                 break;
00121                         }
00122                 }
00123 
00124                 /*
00125                  *  Found the algebraic operator.  Get the next number.
00126                  */
00127                 if (found) {
00128                         continue;
00129                 }
00130 
00131                 /*
00132                  *  End of a group.  Stop.
00133                  */
00134                 if (*p == ')') {
00135                         if (this != TOKEN_NONE) {
00136                                 RDEBUG2("Trailing operator before end sub-expression at \"%s\"", p);
00137                                 return -1;
00138                         }
00139                         p++;
00140                         break;
00141                 }
00142 
00143                 /*
00144                  *  Start of a group.  Call ourselves recursively.
00145                  */
00146                 if (*p == '(') {
00147                         p++;
00148 
00149                         found = get_number(request, &p, &x);
00150                         if (found < 0) {
00151                                 return -1;
00152                         }
00153                 } else {
00154                         /*
00155                          *  No algrebraic operator found, the next thing
00156                          *  MUST be a number.
00157                          *
00158                          *  If it isn't, then we die.
00159                          */
00160                         if ((*p < '0') || (*p > '9')) {
00161                                 RDEBUG2("Not a number at \"%s\"", p);
00162                                 return -1;
00163                         }
00164 
00165                         /*
00166                          *  This is doing it the hard way, but it also allows
00167                          *  us to increment 'p'.
00168                          */
00169                         x = 0;
00170                         while ((*p >= '0') && (*p <= '9')) {
00171                                 x *= 10;
00172                                 x += (*p - '0');
00173                                 p++;
00174                         }
00175                 }
00176 
00177                 switch (this) {
00178                 default:
00179                 case TOKEN_NONE:
00180                         result = x;
00181                         break;
00182 
00183                 case TOKEN_ADD:
00184                         result += x;
00185                         break;
00186 
00187                 case TOKEN_SUBTRACT:
00188                         result -= x;
00189                         break;
00190 
00191                 case TOKEN_DIVIDE:
00192                         if (x == 0) {
00193                                 result = 0; /* we don't have NaN for integers */
00194                                 break;
00195                         }
00196                         result /= x;
00197                         break;
00198 
00199                 case TOKEN_REMAINDER:
00200                         if (x == 0) {
00201                                 result = 0; /* we don't have NaN for integers */
00202                                 break;
00203                         }
00204                         result %= x;
00205                         break;
00206 
00207                 case TOKEN_MULTIPLY:
00208                         result *= x;
00209                         break;
00210 
00211                 case TOKEN_AND:
00212                         result &= x;
00213                         break;
00214 
00215                 case TOKEN_OR:
00216                         result |= x;
00217                         break;
00218                 }
00219 
00220                 /*
00221                  *  We've used this token.
00222                  */
00223                 this = TOKEN_NONE;
00224         }
00225 
00226         /*
00227          *  And return the answer to the caller.
00228          */
00229         *string = p;
00230         *answer = result;
00231         return 0;
00232 }

00360

00362                          *  Alpha numeric
