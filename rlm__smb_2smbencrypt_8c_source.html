00087 /*
00088  * Creates the MD4 Hash of the users password in NT UNICODE.
00089  */
00090 
00091 void E_md4hash(uchar *passwd, uchar *p16)
00092 {
00093         int len;
00094         int16 wpwd[129];
00095 
00096         /* Password cannot be longer than 128 characters */
00097         len = strlen((char *)passwd);
00098         if(len > 128)
00099                 len = 128;
00100         /* Password must be converted to NT unicode */
00101         _my_mbstowcs(wpwd, passwd, len);
00102         wpwd[len] = 0; /* Ensure string is null terminated */
00103         /* Calculate length in bytes */
00104         len = _my_wcslen(wpwd) * sizeof(int16);
00105 
00106         mdfour(p16, (unsigned char *)wpwd, len);
00107 }
