00024  *   CRAM mail authentication (APOP, CRAM-MD5)
00025  *   by 3APA3A
00026  *
00027  *   rlm_cram module is a part of Mail authorization/authentication
00028  *   support.
00029  *
00030  *   Attributes used (Vendor Code/PEN: 11406, you may change it to your own)
00031  *      101 (Sandy-Mail-Authtype), selects CRAM protocol, possible values:
00032  *              2: CRAM-MD5
00033  *              3: APOP
00034  *              8: CRAM-MD4
00035  *              9: CRAM-SHA1
00036  *      102 (Sandy-Mail-Challenge), contains server's challenge (usually
00037  *      text banner)
00038  *      103 (Sandy-Mail-Response), contains client's response, 16 octets
00039  *      for APOP/CRAM-MD5/CRAM-MD4, 20 octets for CRAM-SHA1

00130 static int cram_authenticate(UNUSED void * instance, REQUEST *request)
00131 {
00132         VALUE_PAIR *authtype, *challenge, *response, *password;
00133         char buffer[64];
00134 
00135         password = pairfind(request->config_items, PW_CLEARTEXT_PASSWORD, 0);

00137                 radlog(L_AUTH, "rlm_cram: Cleartext-Password is required for authentication.");

00140         authtype = pairfind(request->packet->vps, SM_AUTHTYPE, VENDORPEC_SM);
00141         if(!authtype) {
00142                 radlog(L_AUTH, "rlm_cram: Required attribute Sandy-Mail-Authtype missed");
00143                 return RLM_MODULE_INVALID;

00146         if(!challenge) {
00147                 radlog(L_AUTH, "rlm_cram: Required attribute Sandy-Mail-Challenge missed");
00148                 return RLM_MODULE_INVALID;
00149         }
00150         response = pairfind(request->packet->vps, SM_RESPONSE, VENDORPEC_SM);
00151         if(!response) {
00152                 radlog(L_AUTH, "rlm_cram: Required attribute Sandy-Mail-Response missed");
00153                 return RLM_MODULE_INVALID;
00154         }
00155         switch(authtype->vp_integer){
00156                 case 2:                         /*      CRAM-MD5        */
00157                         if(challenge->length < 5 || response->length != 16) {
00158                                 radlog(L_AUTH, "rlm_cram: invalid MD5 challenge/response length");

00160                         }
00161                         calc_md5_digest(buffer, challenge->vp_strvalue, challenge->length, password->vp_strvalue);
00162                         if(!memcmp(buffer, response->vp_strvalue, 16)) return RLM_MODULE_OK;
00163                         break;
