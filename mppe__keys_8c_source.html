00052  * TLS PRF from RFC 2246
00053  */
00054 static void P_hash(const EVP_MD *evp_md,
00055                    const unsigned char *secret, unsigned int secret_len,
00056                    const unsigned char *seed,   unsigned int seed_len,
00057                    unsigned char *out, unsigned int out_len)
00058 {
00059         HMAC_CTX ctx_a, ctx_out;
00060         unsigned char a[HMAC_MAX_MD_CBLOCK];
00061         unsigned int size;
00062 
00063         HMAC_CTX_init(&ctx_a);
00064         HMAC_CTX_init(&ctx_out);
00065         HMAC_Init_ex(&ctx_a, secret, secret_len, evp_md, NULL);
00066         HMAC_Init_ex(&ctx_out, secret, secret_len, evp_md, NULL);
00067 
00068         size = HMAC_size(&ctx_out);
00069 
00070         /* Calculate A(1) */
00071         HMAC_Update(&ctx_a, seed, seed_len);
00072         HMAC_Final(&ctx_a, a, NULL);
00073 
00074         while (1) {
00075                 /* Calculate next part of output */
00076                 HMAC_Update(&ctx_out, a, size);
00077                 HMAC_Update(&ctx_out, seed, seed_len);
00078 
00079                 /* Check if last part */
00080                 if (out_len < size) {
00081                         HMAC_Final(&ctx_out, a, NULL);
00082                         memcpy(out, a, out_len);
00083                         break;
00084                 }
00085 
00086                 /* Place digest in output buffer */
00087                 HMAC_Final(&ctx_out, out, NULL);
00088                 HMAC_Init_ex(&ctx_out, NULL, 0, NULL, NULL);
00089                 out += size;
00090                 out_len -= size;
00091 
00092                 /* Calculate next A(i) */
00093                 HMAC_Init_ex(&ctx_a, NULL, 0, NULL, NULL);
00094                 HMAC_Update(&ctx_a, a, size);
00095                 HMAC_Final(&ctx_a, a, NULL);
00096         }
00097 
00098         HMAC_CTX_cleanup(&ctx_a);
00099         HMAC_CTX_cleanup(&ctx_out);
00100         memset(a, 0, sizeof(a));
00101 }

00103 static void PRF(const unsigned char *secret, unsigned int secret_len,
00104                 const unsigned char *seed,   unsigned int seed_len,
00105                 unsigned char *out, unsigned char *buf, unsigned int out_len)
00106 {
00107         unsigned int i;
00108         unsigned int len = (secret_len + 1) / 2;
00109         const unsigned char *s1 = secret;
00110         const unsigned char *s2 = secret + (secret_len - len);
00111 
00112         P_hash(EVP_md5(),  s1, len, seed, seed_len, out, out_len);
00113         P_hash(EVP_sha1(), s2, len, seed, seed_len, buf, out_len);
00114 
00115         for (i=0; i < out_len; i++) {
00116                 out[i] ^= buf[i];
00117         }
00118 }
