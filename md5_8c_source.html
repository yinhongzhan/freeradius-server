00116 fr_MD5Update(FR_MD5_CTX *ctx, const unsigned char *input, size_t len)
00117 {
00118         size_t have, need;
00119 
00120         /* Check how many bytes we already have and how many more we need. */
00121         have = (size_t)((ctx->count[0] >> 3) & (MD5_BLOCK_LENGTH - 1));
00122         need = MD5_BLOCK_LENGTH - have;
00123 
00124         /* Update bitcount */
00125 /*      ctx->count += (uint64_t)len << 3;*/
00126         if ((ctx->count[0] += ((uint32_t)len << 3)) < (uint32_t)len) {
00127         /* Overflowed ctx->count[0] */
00128                 ctx->count[1]++;
00129         }
00130         ctx->count[1] += ((uint32_t)len >> 29);
00131 
00132         if (len >= need) {
00133                 if (have != 0) {
00134                         memcpy(ctx->buffer + have, input, need);
00135                         fr_MD5Transform(ctx->state, ctx->buffer);
00136                         input += need;
00137                         len -= need;
00138                         have = 0;
00139                 }
00140 
00141                 /* Process data in MD5_BLOCK_LENGTH-byte chunks. */
00142                 while (len >= MD5_BLOCK_LENGTH) {
00143                         fr_MD5Transform(ctx->state, input);
00144                         input += MD5_BLOCK_LENGTH;
00145                         len -= MD5_BLOCK_LENGTH;
00146                 }
00147         }
00148 
00149         /* Handle any remaining bytes of data. */
00150         if (len != 0)
00151                 memcpy(ctx->buffer + have, input, len);
00152 }

00154 /*
00155  * Final wrapup - pad to 64-byte boundary with the bit pattern
00156  * 1 0* (64-bit count of bits processed, MSB-first)
00157  */
00158 void
00159 fr_MD5Final(uint8_t digest[MD5_DIGEST_LENGTH], FR_MD5_CTX *ctx)
00160 {
00161         uint8_t count[8];
00162         size_t padlen;
00163         int i;
00164 
00165         /* Convert count to 8 bytes in little endian order. */
00166         PUT_64BIT_LE(count, ctx->count);
00167 
00168         /* Pad out to 56 mod 64. */
00169         padlen = MD5_BLOCK_LENGTH -
00170             ((ctx->count[0] >> 3) & (MD5_BLOCK_LENGTH - 1));
00171         if (padlen < 1 + 8)
00172                 padlen += MD5_BLOCK_LENGTH;
00173         fr_MD5Update(ctx, PADDING, padlen - 8); /* padlen - 8 <= 64 */
00174         fr_MD5Update(ctx, count, 8);
00175 
00176         if (digest != NULL) {
00177                 for (i = 0; i < 4; i++)
00178                         PUT_32BIT_LE(digest + i * 4, ctx->state[i]);
00179         }
00180         memset(ctx, 0, sizeof(*ctx));   /* in case it's sensitive */
00181 }
