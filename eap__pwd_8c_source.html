00069 /* a counter-based KDF based on NIST SP800-108 */
00070 static void 
00071 eap_pwd_kdf(uint8_t *key, int keylen, const char *label, int labellen,
00072             uint8_t *result, int resultbitlen)
00073 {
00074     HMAC_CTX hctx;
00075     uint8_t digest[SHA256_DIGEST_LENGTH];
00076     uint16_t i, ctr, L;
00077     int resultbytelen, len = 0;
00078     unsigned int mdlen = SHA256_DIGEST_LENGTH;
00079     uint8_t mask = 0xff;
00080 
00081     resultbytelen = (resultbitlen + 7)/8;
00082     ctr = 0;
00083     L = htons(resultbitlen);
00084     while (len < resultbytelen) {
00085         ctr++; i = htons(ctr);
00086         HMAC_Init(&hctx, key, keylen, EVP_sha256());
00087         if (ctr > 1) {
00088             HMAC_Update(&hctx, digest, mdlen);
00089         }
00090         HMAC_Update(&hctx, (uint8_t *) &i, sizeof(uint16_t));
00091         HMAC_Update(&hctx, (const uint8_t *)label, labellen);
00092         HMAC_Update(&hctx, (uint8_t *) &L, sizeof(uint16_t));
00093         HMAC_Final(&hctx, digest, &mdlen);
00094         if ((len + (int) mdlen) > resultbytelen) {
00095             memcpy(result + len, digest, resultbytelen - len);
00096         } else {
00097             memcpy(result + len, digest, mdlen);
00098         }
00099         len += mdlen;
00100         HMAC_CTX_cleanup(&hctx);
00101     }
00102 
00103     /* since we're expanding to a bit length, mask off the excess */
00104     if (resultbitlen % 8) {
00105         mask <<= (8 - (resultbitlen % 8));
00106         result[resultbytelen - 1] &= mask;
00107     }
00108 }

00168         DEBUG("unable to get order for curve");

00172         DEBUG("unable to get cofactor for curve");
00173         goto fail;

00178         DEBUG("unable to malloc space for prf buffer");
00179         goto fail;
00180     }
00181     memset(prfbuf, 0, primebytelen);
00182     ctr = 0;
00183     while (1) {
00184         if (ctr > 10) {
00185             DEBUG("unable to find random point on curve for group %d, something's fishy", grp_num);
00186             goto fail;
00187         }
00188         ctr++;
00189 
00190         /*
00191          * compute counter-mode password value and stretch to prime
00192          *    pwd-seed = H(token | peer-id | server-id | password |
00193          *                 counter)
00194          */
00195         H_Init(&ctx);
00196         H_Update(&ctx, (uint8_t *)token, sizeof(*token));
00197         H_Update(&ctx, (uint8_t *)id_peer, id_peer_len);
00198         H_Update(&ctx, (uint8_t *)id_server, id_server_len);
00199         H_Update(&ctx, (uint8_t *)password, password_len);
00200         H_Update(&ctx, (uint8_t *)&ctr, sizeof(ctr));
00201         H_Final(&ctx, pwe_digest);

00205                     "EAP-pwd Hunting And Pecking",
00206                     strlen("EAP-pwd Hunting And Pecking"),
00207                     prfbuf, primebitlen);

00210         /*

00215          * we have to shift right the amount we masked off.

00218             BN_rshift(x_candidate, x_candidate, (8 - (primebitlen % 8)));

00221             continue;
00222         }

00224          * need to unambiguously identify the solution, if there is

00229         } else {
00230             is_odd = 0;
00231         }
00232         /*
00233          * solve the quadratic equation, if it's not solvable then we
00234          * don't have a point
00235          */
00236         if (!EC_POINT_set_compressed_coordinates_GFp(sess->group,
00237                                                      sess->pwe,
00238                                                      x_candidate,
00239                                                      is_odd, NULL)) {
00240             continue;
00241         }
00242         /*
00243          * If there's a solution to the equation then the point must be
00244          * on the curve so why check again explicitly? OpenSSL code
00245          * says this is required by X9.62. We're not X9.62 but it can't
00246          * hurt just to be sure.
00247          */
00248         if (!EC_POINT_is_on_curve(sess->group, sess->pwe, NULL)) {
00249             DEBUG("EAP-pwd: point is not on curve");
00250             continue;
00251         }
00252 
00253         if (BN_cmp(cofactor, BN_value_one())) {
00254             /* make sure the point is not in a small sub-group */
00255             if (!EC_POINT_mul(sess->group, sess->pwe, NULL, sess->pwe,
00256                               cofactor, NULL)) {
00257                 DEBUG("EAP-pwd: cannot multiply generator by order");
00258                 continue;
00259             }
00260             if (EC_POINT_is_at_infinity(sess->group, sess->pwe)) {
00261                 DEBUG("EAP-pwd: point is at infinity");
00262                 continue;
00263             }
00264         }
00265         /* if we got here then we have a new generator. */
00266         break;
00267     }
00268     sess->group_num = grp_num;
00269     if (0) {
00270 fail:
00271         EC_GROUP_free(sess->group);
00272         EC_POINT_free(sess->pwe);
00273         BN_free(sess->order);
00274         BN_free(sess->prime);
00275         free(prfbuf);
00276         prfbuf = NULL;
00277         free(sess);
00278         sess = NULL;
00279         ret = -1;
00280     }
00281     /* cleanliness and order.... */
00282     BN_free(cofactor);
00283     BN_free(x_candidate);
00284     BN_free(rnd);
00285     free(prfbuf);
00286 
00287     return ret;
00288 }

00328 int
00329 process_peer_commit (pwd_session_t *sess, uint8_t *commit, BN_CTX *bnctx)
00330 {
00331     uint8_t *ptr;
00332     BIGNUM *x = NULL, *y = NULL, *cofactor = NULL;
00333     EC_POINT *K = NULL, *point = NULL;
00334     int res = 1;
00335 
00336     if (((sess->peer_scalar = BN_new()) == NULL) ||
00337         ((sess->k = BN_new()) == NULL) ||
00338         ((cofactor = BN_new()) == NULL) ||
00339         ((x = BN_new()) == NULL) ||
00340         ((y = BN_new()) == NULL) ||
00341         ((point = EC_POINT_new(sess->group)) == NULL) ||
00342         ((K = EC_POINT_new(sess->group)) == NULL) ||
00343         ((sess->peer_element = EC_POINT_new(sess->group)) == NULL)) {
00344         DEBUG2("pwd: failed to allocate room to process peer's commit");
00345         goto fin;
00346     }
00347 
00348     if (!EC_GROUP_get_cofactor(sess->group, cofactor, NULL)) {
00349         DEBUG2("pwd: unable to get group co-factor");
00350         goto fin;
00351     }
00352 
00353     /* element, x then y, followed by scalar */
00354     ptr = (uint8_t *)commit;
00355     BN_bin2bn(ptr, BN_num_bytes(sess->prime), x);
00356     ptr += BN_num_bytes(sess->prime);
00357     BN_bin2bn(ptr, BN_num_bytes(sess->prime), y);
00358     ptr += BN_num_bytes(sess->prime);
00359     BN_bin2bn(ptr, BN_num_bytes(sess->order), sess->peer_scalar);
00360     if (!EC_POINT_set_affine_coordinates_GFp(sess->group,
00361                                              sess->peer_element, x, y,
00362                                              bnctx)) {
00363         DEBUG2("pwd: unable to get coordinates of peer's element");
00364         goto fin;
00365     }
00366 
00367     /* check to ensure peer's element is not in a small sub-group */
00368     if (BN_cmp(cofactor, BN_value_one())) {
00369         if (!EC_POINT_mul(sess->group, point, NULL,
00370                           sess->peer_element, cofactor, NULL)) {
00371             DEBUG2("pwd: unable to multiply element by co-factor");
00372             goto fin;
00373         }
00374         if (EC_POINT_is_at_infinity(sess->group, point)) {
00375             DEBUG2("pwd: peer's element is in small sub-group");
00376             goto fin;
00377         }
00378     }
00379 
00380     /* compute the shared key, k */
00381     if ((!EC_POINT_mul(sess->group, K, NULL, sess->pwe,
00382                        sess->peer_scalar, bnctx)) ||
00383         (!EC_POINT_add(sess->group, K, K, sess->peer_element,
00384                        bnctx)) ||
00385         (!EC_POINT_mul(sess->group, K, NULL, K, sess->private_value,
00386                        bnctx))) {
00387         DEBUG2("pwd: unable to compute shared key, k");
00388         goto fin;
00389     }
00390 
00391     /* ensure that the shared key isn't in a small sub-group */
00392     if (BN_cmp(cofactor, BN_value_one())) {
00393         if (!EC_POINT_mul(sess->group, K, NULL, K, cofactor,
00394                           NULL)) {
00395             DEBUG2("pwd: unable to multiply k by co-factor");
00396             goto fin;
00397         }
00398     }
00399 
00400     /*
00401      * This check is strictly speaking just for the case above where
00402      * co-factor > 1 but it was suggested that even though this is probably
00403      * never going to happen it is a simple and safe check "just to be
00404      * sure" so let's be safe.
00405      */
00406     if (EC_POINT_is_at_infinity(sess->group, K)) {
00407         DEBUG2("pwd: k is point-at-infinity!");
00408         goto fin;
00409     }
00410     if (!EC_POINT_get_affine_coordinates_GFp(sess->group, K, sess->k,
00411                                              NULL, bnctx)) {
00412         DEBUG2("pwd: unable to get shared secret from K");
00413         goto fin;
00414     }
00415     res = 0;
00416 
00417   fin:
00418     EC_POINT_free(K);
00419     EC_POINT_free(point);
00420     BN_free(cofactor);
00421     BN_free(x);
00422     BN_free(y);
00423 
00424     return res;
00425 }
