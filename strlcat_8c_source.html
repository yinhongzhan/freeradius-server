00035 /*
00036  * Appends src to string dst of size siz (unlike strncat, siz is the
00037  * full size of dst, not space left).  At most siz-1 characters
00038  * will be copied.  Always NUL terminates (unless siz <= strlen(dst)).
00039  * Returns strlen(src) + MIN(siz, strlen(initial dst)).
00040  * If retval >= siz, truncation occurred.
00041  */
00042 size_t
00043 strlcat(char *dst, const char *src, size_t siz)
00044 {
00045         char *d = dst;
00046         const char *s = src;
00047         size_t n = siz;
00048         size_t dlen;
00049 
00050         /* Find the end of dst and adjust bytes left but don't go past end */
00051         while (n-- != 0 && *d != '\0')
00052                 d++;
00053         dlen = d - dst;
00054         n = siz - dlen;
00055 
00056         if (n == 0)
00057                 return(dlen + strlen(s));
00058         while (*s != '\0') {
00059                 if (n != 1) {
00060                         *d++ = *s;
00061                         n--;
00062                 }
00063                 s++;
00064         }
00065         *d = '\0';
00066 
00067         return(dlen + (s - src));       /* count does not include NUL */
00068 }
