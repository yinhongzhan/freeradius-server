00401 /*
00402  *      Writes a packet, assuming it's already been encoded.
00403  *
00404  *      It returns the number of bytes written, which MAY be less than
00405  *      the packet size (data_len).  It is the caller's responsibility
00406  *      to check the return code, and to schedule writes again.
00407  */
00408 ssize_t fr_tcp_write_packet(RADIUS_PACKET *packet)
00409 {
00410         ssize_t rcode;
00411 
00412         if (!packet || !packet->data) return 0;
00413 
00414         if (packet->partial >= packet->data_len) return packet->data_len;
00415 
00416         rcode = write(packet->sockfd, packet->data + packet->partial,
00417                       packet->data_len - packet->partial);
00418         if (rcode < 0) return packet->partial; /* ignore most errors */
00419 
00420         packet->partial += rcode;
00421 
00422         return packet->partial;
00423 }
