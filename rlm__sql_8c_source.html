00405 {
00406         SQL_INST *inst = arg;
00407         size_t len = 0;
00408 
00409         while (in[0]) {
00410                 /*
00411                  *      Non-printable characters get replaced with their
00412                  *      mime-encoded equivalents.
00413                  */
00414                 if ((in[0] < 32) ||
00415                     strchr(inst->config->allowed_chars, *in) == NULL) {
00416                         /*
00417                          *      Only 3 or less bytes available.
00418                          */
00419                         if (outlen <= 3) {
00420                                 break;
00421                         }
00422 
00423                         snprintf(out, outlen, "=%02X", (unsigned char) in[0]);
00424                         in++;
00425                         out += 3;
00426                         outlen -= 3;
00427                         len += 3;
00428                         continue;
00429                 }
00430 
00431                 /*
00432                  *      Only one byte left.
00433                  */
00434                 if (outlen <= 1) {
00435                         break;
00436                 }
00437 
00438                 /*
00439                  *      Allowed character.
00440                  */
00441                 *out = *in;
00442                 out++;
00443                 in++;
00444                 outlen--;
00445                 len++;
00446         }
00447         *out = '\0';
00448         return len;

01049          *  After this point, ALL 'return's MUST release the SQL socket!
01050          */

01069                 return RLM_MODULE_FAIL;

01073                  */

01075                         found = 1;

01160                                 }
01161                         }
01162                 }
01163
