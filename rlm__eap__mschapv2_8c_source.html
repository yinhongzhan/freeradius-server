00241         int             i;

00244 
00245         type_data = type_data;  /* -Wunused */
00246 
00247         challenge = pairmake("MS-CHAP-Challenge", "0x00", T_OP_EQ);
00248         if (!challenge) {
00249                 radlog(L_ERR, "rlm_eap_mschapv2: out of memory");
00250                 return 0;
00251         }
00252 
00253         /*
00254          *      Get a random challenge.
00255          */
00256         challenge->length = MSCHAPV2_CHALLENGE_LEN;
00257         for (i = 0; i < MSCHAPV2_CHALLENGE_LEN; i++) {
00258                 challenge->vp_strvalue[i] = fr_rand();
00259         }
00260         DEBUG2("rlm_eap_mschapv2: Issuing Challenge");
00261 
00262         /*
00263          *      Keep track of the challenge.
00264          */
00265         data = malloc(sizeof(mschapv2_opaque_t));
00266         rad_assert(data != NULL);
00267 
00268         /*
00269          *      We're at the stage where we're challenging the user.
00270          */
00271         data->code = PW_EAP_MSCHAPV2_CHALLENGE;
00272         memcpy(data->challenge, challenge->vp_strvalue, MSCHAPV2_CHALLENGE_LEN);
00273         data->mppe_keys = NULL;
00274         data->reply = NULL;
00275

00278 
00279         /*
00280          *      Compose the EAP-MSCHAPV2 packet out of the data structure,

00301         handler->stage = AUTHENTICATE;
