00035 /*
00036  *      The signal() function in Solaris 2.5.1 sets SA_NODEFER in
00037  *      sa_flags, which causes grief if signal() is called in the
00038  *      handler before the cause of the signal has been cleared.
00039  *      (Infinite recursion).
00040  *
00041  *      The same problem appears on HPUX, so we avoid it, if we can.
00042  *
00043  *      Using sigaction() to reset the signal handler fixes the problem,
00044  *      so where available, we prefer that solution.
00045  */
00046 
00047 void (*reset_signal(int signo, void (*func)(int)))(int)
00048 {
00049 #ifdef HAVE_SIGACTION
00050         struct sigaction act, oact;
00051 
00052         memset(&act, 0, sizeof(act));
00053         act.sa_handler = func;
00054         sigemptyset(&act.sa_mask);
00055         act.sa_flags = 0;
00056 #ifdef  SA_INTERRUPT            /* SunOS */
00057         act.sa_flags |= SA_INTERRUPT;
00058 #endif
00059         if (sigaction(signo, &act, &oact) < 0)
00060                 return SIG_ERR;
00061         return oact.sa_handler;
00062 #else
00063 
00064         /*
00065          *      re-set by calling the 'signal' function, which
00066          *      may cause infinite recursion and core dumps due to
00067          *      stack growth.
00068          *
00069          *      However, the system is too dumb to implement sigaction(),
00070          *      so we don't have a choice.
00071          */
00072         signal(signo, func);
00073 
00074         return NULL;
00075 #endif
00076 }

00092  *
00093  *      The unique ptr is meant to be a malloc'd module configuration,
00094  *      and the unique integer allows the caller to have multiple
00095  *      opaque data associated with a REQUEST.

00098                      void *unique_ptr, int unique_int,
00099                      void *opaque, void (*free_opaque)(void *))
00100 {
00101         request_data_t *this, **last, *next;
00102 
00103         /*
00104          *      Some simple sanity checks.
00105          */

00107 
00108         this = next = NULL;
00109         for (last = &(request->data); *last != NULL; last = &((*last)->next)) {
00110                 if (((*last)->unique_ptr == unique_ptr) &&
00111                     ((*last)->unique_int == unique_int)) {

00113 
00114                         next = this->next;
00115 
00116                         if (this->opaque && /* free it, if necessary */
00117                             this->free_opaque)
00118                                 this->free_opaque(this->opaque);

00120                 }
00121         }
00122 
00123         if (!this) this = rad_malloc(sizeof(*this));
00124         memset(this, 0, sizeof(*this));

00126         this->next = next;

00128         this->unique_int = unique_int;

00286 /*
00287  *      Create possibly many directories.
00288  *
00289  *      Note that the input directory name is NOT a constant!
00290  *      This is so that IF an error is returned, the 'directory' ptr
00291  *      points to the name of the file which caused the error.
00292  */
00293 int rad_mkdir(char *directory, int mode)
00294 {
00295         int rcode;
00296         char *p;
00297         struct stat st;
00298 
00299         /*
00300          *      If the directory exists, don't do anything.
00301          */
00302         if (stat(directory, &st) == 0) {
00303                 return 0;
00304         }
00305 
00306         /*
00307          *      Look for the LAST directory name.  Try to create that,
00308          *      failing on any error.
00309          */
00310         p = strrchr(directory, FR_DIR_SEP);
00311         if (p != NULL) {
00312                 *p = '\0';
00313                 rcode = rad_mkdir(directory, mode);
00314 
00315                 /*
00316                  *      On error, we leave the directory name as the
00317                  *      one which caused the error.
00318                  */
00319                 if (rcode < 0) {
00320                         return rcode;
00321                 }
00322 
00323                 /*
00324                  *      Reset the directory delimiter, and go ask
00325                  *      the system to make the directory.
00326                  */
00327                 *p = FR_DIR_SEP;
00328         } else {
00329                 return 0;
00330         }
00331 
00332         /*
00333          *      Having done everything successfully, we do the
00334          *      system call to actually go create the directory.
00335          */
00336         return mkdir(directory, mode);
00337 }

00617 int rad_pps(int *past, int *present, time_t *then, struct timeval *now)
00618 {
00619         int pps;
00620 
00621         if (*then != now->tv_sec) {
00622                 *then = now->tv_sec;
00623                 *past = *present;
00624                 *present = 0;
00625         }
00626 
00627         /*
00628          *      Bootstrap PPS by looking at a percentage of
00629          *      the previous PPS.  This lets us take a moving
00630          *      count, without doing a moving average.  If
00631          *      we're a fraction "f" (0..1) into the current
00632          *      second, we can get a good guess for PPS by
00633          *      doing:
00634          *
00635          *      PPS = pps_now + pps_old * (1 - f)
00636          *
00637          *      It's an instantaneous measurement, rather than
00638          *      a moving average.  This will hopefully let it
00639          *      respond better to sudden spikes.
00640          *
00641          *      Doing the calculations by thousands allows us
00642          *      to not overflow 2^32, AND to not underflow
00643          *      when we divide by USEC.
00644          */
00645         pps = USEC - now->tv_usec; /* useconds left in previous second */
00646         pps /= 1000;               /* scale to milliseconds */
00647         pps *= *past;              /* multiply by past count to get fraction */
00648         pps /= 1000;               /* scale to usec again */
00649         pps += *present;           /* add in current count */
00650 
00651         return pps;
00652 }
