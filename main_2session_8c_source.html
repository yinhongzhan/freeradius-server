00160          *      Fork.
00161          */
00162         if ((pid = rad_fork()) < 0) { /* do wait for the fork'd result */
00163                 radlog(L_ERR, "Accounting: Failed in fork(): Cannot run checkrad\n");
00164                 return 2;
00165         }
00166 
00167         if (pid > 0) {
00168                 child_pid = rad_waitpid(pid, &status);
00169 
00170                 /*
00171                  *      It's taking too long.  Stop waiting for it.
00172                  *
00173                  *      Don't bother to kill it, as we don't care what

00177                         radlog(L_ERR, "Check-TS: timeout waiting for checkrad");
