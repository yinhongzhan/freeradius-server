00299 static size_t config_escape_func(UNUSED REQUEST *request, char *out, size_t outlen, const char *in, UNUSED void *arg)
00300 {
00301         size_t len = 0;
00302         static const char *disallowed = "%{}\\'\"`";
00303 
00304         while (in[0]) {
00305                 /*
00306                  *      Non-printable characters get replaced with their
00307                  *      mime-encoded equivalents.
00308                  */
00309                 if ((in[0] < 32)) {
00310                         if (outlen <= 3) break;
00311 
00312                         snprintf(out, outlen, "=%02X", (unsigned char) in[0]);
00313                         in++;
00314                         out += 3;
00315                         outlen -= 3;
00316                         len += 3;
00317                         continue;
00318 
00319                 } else if (strchr(disallowed, *in) != NULL) {
00320                         if (outlen <= 2) break;
00321 
00322                         out[0] = '\\';
00323                         out[1] = *in;
00324                         in++;
00325                         out += 2;
00326                         outlen -= 2;
00327                         len += 2;
00328                         continue;
00329                 }
00330 
00331                 /*
00332                  *      Only one byte left.
00333                  */
00334                 if (outlen <= 1) {
00335                         break;
00336                 }
00337 
00338                 /*
00339                  *      Allowed character.
00340                  */
00341                 *out = *in;
00342                 out++;
00343                 in++;
00344                 outlen--;
00345                 len++;
00346         }
00347         *out = '\0';
00348         return len;
00349 }

00354 static size_t xlat_config(void *instance, REQUEST *request,
00355                           const char *fmt, char *out,
00356                           size_t outlen)
00357 {
00358         const char *value;
00359         CONF_PAIR *cp;
00360         CONF_ITEM *ci;
00361         char buffer[1024];
00362 
00363         request = request;      /* -Wunused */
00364         instance = instance;    /* -Wunused */
00365 
00366         /*
00367          *      Expand it safely.
00368          */
00369         if (!radius_xlat(buffer, sizeof(buffer), fmt, request, config_escape_func, NULL)) {
00370                 return 0;
00371         }
00372 
00373         ci = cf_reference_item(request->root->config,
00374                                request->root->config, buffer);
00375         if (!ci || !cf_item_is_pair(ci)) {
00376                 *out = '\0';
00377                 return 0;
00378         }
00379 
00380         cp = cf_itemtopair(ci);
00381 
00382         /*
00383          *  Ensure that we only copy what's necessary.
00384          *
00385          *  If 'outlen' is too small, then the output is chopped to fit.
00386          */
00387         value = cf_pair_value(cp);
00388         if (value) {
00389                 if (outlen > strlen(value)) {
00390                         outlen = strlen(value) + 1;
00391                 }
00392         }
00393 
00394         strlcpy(out, value, outlen);
00395 
00396         return strlen(out);
00397 }

01007 {
01008         cached_config_t *cc;
01009         CONF_SECTION *cs;
01010         char buffer[1024];
01011 
01012         radlog(L_INFO, "HUP - Re-reading configuration files");
01013 
01014         /* Read the configuration file */
01015         snprintf(buffer, sizeof(buffer), "%.200s/%.50s.conf",
01016                  radius_dir, mainconfig.name);
01017         if ((cs = cf_file_read(buffer)) == NULL) {
01018                 radlog(L_ERR, "Failed to re-read or parse %s", buffer);
01019                 return;
01020         }
01021 
01022         cc = rad_malloc(sizeof(*cc));
01023         memset(cc, 0, sizeof(*cc));
01024 
01025         /*
01026          *      Save the current configuration.  Note that we do NOT
01027          *      free older ones.  We should probably do so at some
01028          *      point.  Doing so will require us to mark which modules
01029          *      are still in use, and which aren't.  Modules that
01030          *      can't be HUPed always use the original configuration.
01031          *      Modules that can be HUPed use one of the newer
01032          *      configurations.
01033          */
01034         cc->created = time(NULL);
01035         cc->cs = cs;
01036         cc->next = cs_cache;
01037         cs_cache = cc;
01038 
01039         /*
01040          *      Re-open the log file.  If we can't, then keep logging
01041          *      to the old log file.
01042          *
01043          *      The "open log file" code is here rather than in log.c,
01044          *      because it makes that function MUCH simpler.
01045          */
01046         if (mainconfig.radlog_dest == RADLOG_FILES) {
01047                 int fd, old_fd;
01048                 
01049                 fd = open(mainconfig.log_file,
01050                           O_WRONLY | O_APPEND | O_CREAT, 0640);
01051                 if (fd >= 0) {
01052                         /*
01053                          *      Atomic swap. We'd like to keep the old
01054                          *      FD around so that callers don't
01055                          *      suddenly find the FD closed, and the
01056                          *      writes go nowhere.  But that's hard to
01057                          *      do.  So... we have the case where a
01058                          *      log message *might* be lost on HUP.
01059                          */
01060                         old_fd = mainconfig.radlog_fd;
01061                         mainconfig.radlog_fd = fd;
01062                         close(old_fd);
01063                 }
01064         }
01065 
01066         radlog(L_INFO, "HUP - loading modules");
01067 
01068         /*
01069          *      Prefer the new module configuration.
01070          */
01071         module_hup(cf_section_sub_find(cs, "modules"));
01072 
01073         /*
01074          *      Load new servers BEFORE freeing old ones.
01075          */

01078         virtual_servers_free(cc->created - mainconfig.max_request_time * 4);
