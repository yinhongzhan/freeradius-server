00189 /*
00190  * Verify an otp by asking otpd.
00191  * Returns an OTP_* code, or -1 on system failure.
00192  * Fills in reply.
00193  */
00194 static int
00195 otp_verify(const otp_option_t *opt,
00196            const otp_request_t *request, otp_reply_t *reply)
00197 {
00198   otp_fd_t *fdp;
00199   int rc;
00200   int tryagain = 2;
00201 
00202 retry:
00203   if (!tryagain--)
00204     return -1;
00205   fdp = otp_getfd(opt);
00206   if (!fdp || fdp->fd == -1)
00207     return -1;
00208 
00209   if ((rc = otp_write(fdp, (const char *) request, sizeof(*request))) != sizeof(*request)) {
00210     if (rc == 0)
00211       goto retry;       /* otpd disconnect */   /*TODO: pause */
00212     else
00213       return -1;
00214   }
00215 
00216   if ((rc = otp_read(fdp, (char *) reply, sizeof(*reply))) != sizeof(*reply)) {
00217     if (rc == 0)
00218       goto retry;       /* otpd disconnect */   /*TODO: pause */
00219     else
00220       return -1;
00221   }
00222 
00223   /* validate the reply */
00224   if (reply->version != 1) {
00225     (void) radlog(L_AUTH, "rlm_otp: otpd reply for [%s] invalid "
00226                           "(version %d != 1)",
00227                   request->username, reply->version);
00228     otp_putfd(fdp, 1);
00229     return -1;
00230   }
00231 
00232   if (reply->passcode[OTP_MAX_PASSCODE_LEN] != '\0') {
00233     (void) radlog(L_AUTH, "rlm_otp: otpd reply for [%s] invalid (passcode)",
00234                   request->username);
00235     otp_putfd(fdp, 1);
00236     return -1;
00237   }
00238 
00239   otp_putfd(fdp, 0);
00240   return reply->rc;
00241 }

00243 /*
00244  * Full read with logging, and close on failure.
00245  * Returns nread on success, 0 on EOF, -1 on other failures.
00246  */
00247 static int
00248 otp_read(otp_fd_t *fdp, char *buf, size_t len)
00249 {
00250   ssize_t n;
00251   size_t nread = 0;     /* bytes read into buf */
00252 
00253   while (nread < len) {
00254     if ((n = read(fdp->fd, &buf[nread], len - nread)) == -1) {
00255       if (errno == EINTR) {
00256         continue;
00257       } else {
00258         (void) radlog(L_ERR, "rlm_otp: %s: read from otpd: %s",
00259                       __func__, strerror(errno));
00260         otp_putfd(fdp, 1);
00261         return -1;
00262       }
00263     }
00264     if (!n) {
00265       (void) radlog(L_ERR, "rlm_otp: %s: otpd disconnect", __func__);
00266       otp_putfd(fdp, 1);
00267       return 0;
00268     }
00269     nread += n;
00270   } /* while (more to read) */
00271 
00272   return nread;
00273 }

00275 /*
00276  * Full write with logging, and close on failure.
00277  * Returns 0 on success, errno on failure.
00278  */
00279 static int
00280 otp_write(otp_fd_t *fdp, const char *buf, size_t len)
00281 {
00282   size_t nleft = len;
00283   ssize_t nwrote;
00284 
00285   while (nleft) {
00286     if ((nwrote = write(fdp->fd, &buf[len - nleft], nleft)) == -1) {
00287       if (errno == EINTR) {
00288         continue;

00291                       __func__, strerror(errno));
00292         otp_putfd(fdp, 1);
00293         return errno;
00294       }
00295     }
00296     nleft -= nwrote;
00297   }

00299   return 0;
00300 }

00302 /* connect to otpd and return fd */
00303 static int
00304 otp_connect(const char *path)
00305 {
00306   int fd;
00307   struct sockaddr_un sa;
00308   size_t sp_len;                /* sun_path length (strlen) */
00309 
00310   /* setup for unix domain socket */
00311   sp_len = strlen(path);
00312   if (sp_len > sizeof(sa.sun_path) - 1) {
00313     (void) radlog(L_ERR, "rlm_otp: %s: rendezvous point name too long",
00314                   __func__);
00315     return -1;
00316   }
00317   sa.sun_family = AF_UNIX;
00318   (void) strcpy(sa.sun_path, path);
00319 
00320   /* connect to otpd */
00321   if ((fd = socket(PF_UNIX, SOCK_STREAM, 0)) == -1) {
00322     (void) radlog(L_ERR, "rlm_otp: %s: socket: %s", __func__, strerror(errno));
00323     return -1;
00324   }
00325   if (connect(fd, (struct sockaddr *) &sa,
00326               sizeof(sa.sun_family) + sp_len) == -1) {
00327     (void) radlog(L_ERR, "rlm_otp: %s: connect(%s): %s",
00328                   __func__, path, strerror(errno));
00329     (void) close(fd);
00330     return -1;
00331   }
00332   return fd;
00333 }

00335 /*
00336  * Retrieve an fd (from pool) to use for otpd connection.
00337  * It'd be simpler to use TLS but FR can have lots of threads
00338  * and we don't want to waste fd's that way.
00339  * We can't have a global fd because we'd then be pipelining
00340  * requests to otpd and we have no way to demultiplex
00341  * the responses.
00342  */
00343 static otp_fd_t *
00344 otp_getfd(const otp_option_t *opt)
00345 {
00346   int rc;
00347   otp_fd_t *fdp;
00348 
00349   /* walk the connection pool looking for an available fd */
00350   for (fdp = otp_fd_head; fdp; fdp = fdp->next) {
00351     rc = otp_pthread_mutex_trylock(&fdp->mutex);
00352     if (!rc)
00353       if (!strcmp(fdp->path, opt->otpd_rp))     /* could just use == */
00354         break;
00355   }
00356 
00357   if (!fdp) {
00358     /* no fd was available, add a new one */
00359     fdp = rad_malloc(sizeof(*fdp));
00360     otp_pthread_mutex_init(&fdp->mutex, NULL);
00361     otp_pthread_mutex_lock(&fdp->mutex);
00362     /* insert new fd at head */
00363     otp_pthread_mutex_lock(&otp_fd_head_mutex);
00364     fdp->next = otp_fd_head;
00365     otp_fd_head = fdp;
00366     otp_pthread_mutex_unlock(&otp_fd_head_mutex);
00367     /* initialize */
00368     fdp->path = opt->otpd_rp;
00369     fdp->fd = -1;
00370   }
00371 
00372   /* establish connection */
00373   if (fdp->fd == -1)
00374     fdp->fd = otp_connect(fdp->path);
00375 
00376   return fdp;
00377 }
