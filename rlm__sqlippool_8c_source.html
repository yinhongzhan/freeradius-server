00200 /*
00201  *      Replace %<whatever> in a string.
00202  *
00203  *      %P      pool_name
00204  *      %I      param
00205  *      %J      lease_duration
00206  *
00207  */
00208 static int sqlippool_expand(char * out, int outlen, const char * fmt,
00209                             rlm_sqlippool_t *data, char * param, int param_len)
00210 {
00211         char *q;
00212         const char *p;
00213         char tmp[40]; /* For temporary storing of integers */
00214 
00215         q = out;
00216         for (p = fmt; *p ; p++) {
00217                 int freespace;
00218                 int c;
00219 
00220                 /* Calculate freespace in output */
00221                 freespace = outlen - (q - out);
00222                 if (freespace <= 1)
00223                         break;
00224 
00225                 c = *p;
00226                 if (c != '%' && c != '$' && c != '\\') {
00227                         *q++ = *p;
00228                         continue;
00229                 }
00230 
00231                 if (*++p == '\0')
00232                         break;
00233 
00234                 if (c == '\\') {
00235                         switch(*p) {
00236                         case '\\':
00237                                 *q++ = '\\';
00238                                 break;
00239                         case 't':
00240                                 *q++ = '\t';
00241                                 break;
00242                         case 'n':
00243                                 *q++ = '\n';
00244                                 break;
00245                         default:
00246                                 *q++ = c;
00247                                 *q++ = *p;
00248                                 break;
00249                         }
00250                 }
00251                 else if (c == '%') {
00252                         switch(*p) {
00253                         case '%':
00254                                 *q++ = *p;
00255                                 break;
00256                         case 'P': /* pool name */
00257                                 strlcpy(q, data->pool_name, freespace);
00258                                 q += strlen(q);
00259                                 break;
00260                         case 'I': /* IP address */
00261                                 if (param && param_len > 0) {
00262                                         if (param_len > freespace) {
00263                                                 strlcpy(q, param, freespace);
00264                                                 q += strlen(q);
00265                                         }
00266                                         else {
00267                                                 memcpy(q, param, param_len);
00268                                                 q += param_len;
00269                                         }
00270                                 }
00271                                 break;
00272                         case 'J': /* lease duration */
00273                                 sprintf(tmp, "%d", data->lease_duration);
00274                                 strlcpy(q, tmp, freespace);
00275                                 q += strlen(q);
00276                                 break;
00277                         default:
00278                                 *q++ = '%';
00279                                 *q++ = *p;
00280                                 break;
00281                         }
00282                 }
00283         }
00284         *q = '\0';
00285 
00286 #if 0
00287         DEBUG2("sqlippool_expand: '%s'", out);
00288 #endif
00289 
00290         return strlen(out);
00291 }

00311                         radlog(L_ERR, "sqlippool_command: xlat failed on: '%s'", query);
00312                         return 0;

00315                 strcpy(query, expansion);

00317

00322                 radlog(L_ERR, "sqlippool_command: database query error in: '%s'", query);

00822 /*
00823  *      Check for an Accounting-Stop
00824  *      If we find one and we have allocated an IP to this nas/port
00825  *      combination, then deallocate it.
00826  */
00827 static int sqlippool_accounting(void * instance, REQUEST * request)
00828 {
00829         int rcode;
00830         VALUE_PAIR * vp;
00831         int acct_status_type;
00832         rlm_sqlippool_t * data = (rlm_sqlippool_t *) instance;
00833         SQLSOCK * sqlsocket;
00834         char sqlusername[MAX_STRING_LEN];
00835 
00836         vp = pairfind(request->packet->vps, PW_ACCT_STATUS_TYPE, 0);
00837         if (!vp) {
00838                 RDEBUG("Could not find account status type in packet.");
00839                 return RLM_MODULE_NOOP;
00840         }
00841         acct_status_type = vp->vp_integer;
00842 
00843         switch (acct_status_type) {
00844         case PW_STATUS_START:
00845         case PW_STATUS_ALIVE:
00846         case PW_STATUS_STOP:
00847         case PW_STATUS_ACCOUNTING_ON:
00848         case PW_STATUS_ACCOUNTING_OFF:
00849                 break;          /* continue through to the next section */
00850 
00851         default:
00852                 /* We don't care about any other accounting packet */
00853                 return RLM_MODULE_NOOP;
00854         }
00855 
00856         sqlsocket = data->sql_inst->sql_get_socket(data->sql_inst);
00857         if (sqlsocket == NULL) {
00858                 RDEBUG("cannot allocate sql connection");
00859                 return RLM_MODULE_NOOP;
00860         }
00861 
00862         if (data->sql_inst->sql_set_user(data->sql_inst, request, sqlusername, NULL) < 0) {
00863                 return RLM_MODULE_FAIL;
00864         }
00865 
00866         switch (acct_status_type) {
00867         case PW_STATUS_START:
00868                 rcode = sqlippool_accounting_start(sqlsocket, data, request);
00869                 break;
00870 
00871         case PW_STATUS_ALIVE:
00872                 rcode = sqlippool_accounting_alive(sqlsocket, data, request);
00873                 break;
00874 
00875         case PW_STATUS_STOP:
00876                 rcode = sqlippool_accounting_stop(sqlsocket, data, request);
00877                 break;
00878 
00879         case PW_STATUS_ACCOUNTING_ON:
00880                 rcode = sqlippool_accounting_on(sqlsocket, data, request);
00881                 break;
00882 
00883         case PW_STATUS_ACCOUNTING_OFF:
00884                 rcode = sqlippool_accounting_off(sqlsocket, data, request);
00885                 break;
00886 
00887         default:
00888                 /* We don't care about any other accounting packet */
00889                 return RLM_MODULE_NOOP;
00890         }
00891 
00892         data->sql_inst->sql_release_socket(data->sql_inst, sqlsocket);
00893
