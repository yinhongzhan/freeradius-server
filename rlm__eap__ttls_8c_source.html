00274 {

00280         REQUEST *request = handler->request;
00281

00283 
00284         tls_session->length_flag = inst->include_length;
00285 
00286         /*
00287          *      Process TLS layer until done.
00288          */
00289         status = eaptls_process(handler);
00290         RDEBUG2("eaptls_process returned %d\n", status);
00291         switch (status) {
00292                 /*
00293                  *      EAP-TLS handshake was successful, tell the
00294                  *      client to keep talking.
00295                  *
00296                  *      If this was EAP-TLS, we would just return
00297                  *      an EAP-TLS-Success packet here.
00298                  */
00299         case FR_TLS_SUCCESS:
00300                 if (SSL_session_reused(tls_session->ssl)) {
00301                         RDEBUG("Skipping Phase2 due to session resumption");
00302                         goto do_keys;
00303                 }
00304 
00305                 if (t && t->authenticated) {
00306                         RDEBUG2("Using saved attributes from the original Access-Accept");
00307                         debug_pair_list(t->accept_vps);
00308                         pairadd(&handler->request->reply->vps,
00309                                 t->accept_vps);
00310                         t->accept_vps = NULL;
00311                 do_keys:
00312                         /*
00313                          *      Success: Automatically return MPPE keys.
00314                          */
00315                         return eaptls_success(handler, 0);
00316                 } else {
00317                         eaptls_request(handler->eap_ds, tls_session);
00318                 }
00319                 return 1;
00320 
00321                 /*
00322                  *      The TLS code is still working on the TLS
00323                  *      exchange, and it's a valid TLS request.
00324                  *      do nothing.
00325                  */
00326         case FR_TLS_HANDLED:
00327                 return 1;
00328 
00329                 /*
00330                  *      Handshake is done, proceed with decoding tunneled
00331                  *      data.
00332                  */
00333         case FR_TLS_OK:
00334                 break;
00335 
00336                 /*
00337                  *      Anything else: fail.
00338                  */
00339         default:
00340                 return 0;
00341         }
00342 
00343         /*
00344          *      Session is established, proceed with decoding
00345          *      tunneled data.
00346          */
00347         RDEBUG2("Session established.  Proceeding to decode tunneled attributes.");
00348 
00349         /*
00350          *      We may need TTLS data associated with the session, so
00351          *      allocate it here, if it wasn't already alloacted.
00352          */
00353         if (!tls_session->opaque) {
00354                 tls_session->opaque = ttls_alloc(inst);
00355                 tls_session->free_opaque = ttls_free;
00356         }
00357 
00358         /*
00359          *      Process the TTLS portion of the request.
00360          */
00361         rcode = eapttls_process(handler, tls_session);
00362         switch (rcode) {
00363         case PW_AUTHENTICATION_REJECT:
00364                 eaptls_fail(handler, 0);
00365                 return 0;
00366 
00367                 /*
00368                  *      Access-Challenge, continue tunneled conversation.
00369                  */
00370         case PW_ACCESS_CHALLENGE:
00371                 eaptls_request(handler->eap_ds, tls_session);
00372                 return 1;
00373 
00374                 /*
00375                  *      Success: Automatically return MPPE keys.
00376                  */
00377         case PW_AUTHENTICATION_ACK:
00378                 return eaptls_success(handler, 0);
00379 
00380                 /*
00381                  *      No response packet, MUST be proxying it.
00382                  *      The main EAP module will take care of discovering
00383                  *      that the request now has a "proxy" packet, and
00384                  *      will proxy it, rather than returning an EAP packet.
00385                  */
00386         case PW_STATUS_CLIENT:
00387 #ifdef WITH_PROXY
00388                 rad_assert(handler->request->proxy != NULL);
00389 #endif
00390                 return 1;
00391                 break;
00392 
00393         default:
00394                 break;
00395         }
00396 
00397         /*
00398          *      Something we don't understand: Reject it.
00399          */
00400         eaptls_fail(handler, 0);
00401         return 0;
00402 }
