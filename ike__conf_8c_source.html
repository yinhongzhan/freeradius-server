00075 {
00076         int rcode;
00077         PAIR_LIST *users = NULL;
00078 
00079         rcode = pairlist_read(filename, &users, 1);
00080         if (rcode < 0) {
00081                 return -1;
00082         }
00083 
00084         /*
00085          *      Walk through the 'users' file list, if we're debugging,
00086          *      or if we're in compat_mode.
00087          */
00088         if ((debug_flag) ||
00089             (strcmp(compat_mode_str, "cistron") == 0)) {
00090                 PAIR_LIST *entry;
00091                 VALUE_PAIR *vp;
00092                 int compat_mode = FALSE;
00093 
00094                 if (strcmp(compat_mode_str, "cistron") == 0) {
00095                         compat_mode = TRUE;
00096                 }
00097 
00098                 entry = users;
00099                 while (entry) {
00100                         if (compat_mode) {
00101                                 DEBUG("[%s]:%d Cistron compatibility checks for entry %s ...",
00102                                                 filename, entry->lineno,
00103                                                 entry->name);
00104                         }
00105 
00106                         /*
00107                          *      Look for improper use of '=' in the
00108                          *      check items.  They should be using
00109                          *      '==' for on-the-wire RADIUS attributes,
00110                          *      and probably ':=' for server
00111                          *      configuration items.
00112                          */
00113                         for (vp = entry->check; vp != NULL; vp = vp->next) {
00114                                 /*
00115                                  *      Ignore attributes which are set
00116                                  *      properly.
00117                                  */
00118                                 if (vp->operator != T_OP_EQ) {
00119                                         continue;
00120                                 }
00121 
00122                                 /*
00123                                  *      If it's a vendor attribute,
00124                                  *      or it's a wire protocol,
00125                                  *      ensure it has '=='.
00126                                  */
00127                                 if ((vp->vendor!= 0) ||
00128                                                 (vp->attribute < 0x100)) {
00129                                         if (!compat_mode) {
00130                                                 DEBUG("[%s]:%d WARNING! Changing '%s =' to '%s =='\n\tfor comparing RADIUS attribute in check item list for user %s",
00131                                                                 filename, entry->lineno,
00132                                                                 vp->name, vp->name,
00133                                                                 entry->name);
00134                                         } else {
00135                                                 DEBUG("\tChanging '%s =' to '%s =='",
00136                                                                 vp->name, vp->name);
00137                                         }
00138                                         vp->operator = T_OP_CMP_EQ;
00139                                         continue;
00140                                 }
00141 
00142                                 /*
00143                                  *      Cistron Compatibility mode.
00144                                  *
00145                                  *      Re-write selected attributes
00146                                  *      to be '+=', instead of '='.
00147                                  *
00148                                  *      All others get set to '=='
00149                                  */
00150                                 if (compat_mode) {
00151                                         /*
00152                                          *      Non-wire attributes become +=
00153                                          *
00154                                          *      On the write attributes
00155                                          *      become ==
00156                                          */
00157                                         if ((vp->attribute >= 0x100) &&
00158                                                         (vp->attribute <= 0xffff) &&
00159                                                         (vp->attribute != PW_HINT) &&
00160                                                         (vp->attribute != PW_HUNTGROUP_NAME)) {
00161                                                 DEBUG("\tChanging '%s =' to '%s +='",
00162                                                                 vp->name, vp->name);
00163                                                 vp->operator = T_OP_ADD;
00164                                         } else {
00165                                                 DEBUG("\tChanging '%s =' to '%s =='",
00166                                                                 vp->name, vp->name);
00167                                                 vp->operator = T_OP_CMP_EQ;
00168                                         }
00169                                 }
00170 
00171                         } /* end of loop over check items */
00172 
00173 
00174                         /*
00175                          *      Look for server configuration items
00176                          *      in the reply list.
00177                          *
00178                          *      It's a common enough mistake, that it's
00179                          *      worth doing.
00180                          */
00181                         for (vp = entry->reply; vp != NULL; vp = vp->next) {
00182                                 /*
00183                                  *      If it's NOT a vendor attribute,
00184                                  *      and it's NOT a wire protocol
00185                                  *      and we ignore Fall-Through,
00186                                  *      then bitch about it, giving a
00187                                  *      good warning message.
00188                                  */
00189                                  if ((vp->vendor == 0) &&
00190                                         (vp->attribute > 0xff) &&
00191                                         (vp->attribute > 1000)) {
00192                                         log_debug("[%s]:%d WARNING! Check item \"%s\"\n"
00193                                                         "\tfound in reply item list for user \"%s\".\n"
00194                                                         "\tThis attribute MUST go on the first line"
00195                                                         " with the other check items",
00196                                                         filename, entry->lineno, vp->name,
00197                                                         entry->name);
00198                                 }
00199                         }
00200 
00201                         entry = entry->next;
00202                 }
00203         }
00204 
00205         *pair_list = users;
00206         return 0;

00257     radlog(L_DBG,IKEv2_LOG_PREFIX "Begin load transforms");
00258     CONF_PAIR *cp;
00259     int option_exists=0;
00260     int i=0;
00261     while(config_transforms[i].name)  {
00262         uint8_t id;
00263         uint16_t keylen;
00264         for(
00265                 cp=cf_pair_find(cf,config_transforms[i].name);
00266                 cp;
00267                 cp=cf_pair_find_next(cf,cp,config_transforms[i].name)

00270                 radlog(L_ERR,IKEv2_LOG_PREFIX "Unsupported %s transform: %s ",
00271                         config_transforms[i].name,cf_pair_value(cp));
00272                 return -1;
00273             }
00274             if(!AddTransform(prot,config_transforms[i].type,id,keylen)) {
00275                 radlog(L_ERR,IKEv2_LOG_PREFIX "Problem with transform %s:%s",
00276                         config_transforms[i].name,cf_pair_value(cp));
00277                 return -1;
00278             }
00279             option_exists|=config_transforms[i].exist_flag;
00280         }
00281         i++;
00282     }
00283     if((option_exists & OPT_NEEDED) != OPT_NEEDED ) {
00284         radlog(L_ERR,IKEv2_LOG_PREFIX "Not all mandatory transforms are set properly");
00285         radlog(L_DBG,IKEv2_LOG_PREFIX "Option flags: 0x%02X",option_exists);
00286         return -1;
