00065 {
00066         if (!client) return;
00067 
00068 #ifdef WITH_DYNAMIC_CLIENTS
00069         if (client->dynamic == 2) {
00070                 time_t now;
00071 
00072                 if (!deleted_clients) {
00073                         deleted_clients = fr_fifo_create(1024,
00074                                                          (void *) client_free);
00075                         if (!deleted_clients) return; /* MEMLEAK */

00078                 /*
00079                  *      Mark it as in the fifo, and remember when we
00080                  *      pushed it.
00081                  */

00084                 fr_fifo_push(deleted_clients, client);

00086                 /*
00087                  *      Peek at the head of the fifo.  If it might

00090                  */

00092                 if ((client->created + 120) >= now) return;
00093 
00094                 client = fr_fifo_pop(deleted_clients);
00095                 rad_assert(client != NULL);
00096         }
00097 #endif
00098 
00099         free(client->longname);
00100         free(client->secret);
00101         free(client->shortname);
00102         free(client->nastype);
00103         free(client->login);
00104         free(client->password);
00105         free(client->server);
00106 
00107 #ifdef WITH_DYNAMIC_CLIENTS

00197 /*
00198  *      Sanity check a client.
00199  */
00200 static int client_sane(RADCLIENT *client)
00201 {
00202         switch (client->ipaddr.af) {
00203         case AF_INET:
00204                 if (client->prefix > 32) {
00205                         return 0;
00206                 }
00207 
00208                 /*
00209                  *      Zero out the subnet bits.
00210                  */
00211                 if (client->prefix == 0) {
00212                         memset(&client->ipaddr.ipaddr.ip4addr, 0,
00213                                sizeof(client->ipaddr.ipaddr.ip4addr));
00214 
00215                 } else if (client->prefix < 32) {
00216                         uint32_t mask = ~0;
00217 
00218                         mask <<= (32 - client->prefix);
00219                         client->ipaddr.ipaddr.ip4addr.s_addr &= htonl(mask);
00220                 }
00221                 break;
00222 
00223         case AF_INET6:
00224                 if (client->prefix > 128) return 0;
00225 
00226                 if (client->prefix == 0) {
00227                         memset(&client->ipaddr.ipaddr.ip6addr, 0,
00228                                sizeof(client->ipaddr.ipaddr.ip6addr));
00229 
00230                 } else if (client->prefix < 128) {
00231                         uint32_t mask, *addr;
00232 
00233                         addr = (uint32_t *) &client->ipaddr.ipaddr.ip6addr;
00234 
00235                         if ((client->prefix & 0x1f) == 0) {
00236                                 mask = 0;
00237                         } else {
00238                                 mask = ~ ((uint32_t) 0);
00239                                 mask <<= (32 - (client->prefix & 0x1f));
00240                                 mask = htonl(mask);
00241                         }
00242 
00243                         switch (client->prefix >> 5) {
00244                         case 0:
00245                                 addr[0] &= mask;
00246                                 mask = 0;
00247                                 /* FALL-THROUGH */
00248                         case 1:
00249                                 addr[1] &= mask;
00250                                 mask = 0;
00251                                 /* FALL-THROUGH */
00252                         case 2:
00253                                 addr[2] &= mask;
00254                                 mask = 0;
00255                                 /* FALL-THROUGH */
00256                         case 3:
00257                                 addr[3] &= mask;
00258                           break;
00259                         }
00260                 }
00261                 break;
00262 
00263         default:
00264                 return 0;
00265         }
00266 
00267         return 1;
00268 }
