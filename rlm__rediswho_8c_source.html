00170 static int rediswho_accounting_all(REDISSOCK **dissocket_p,

00172                                    const char *insert,
00173                                    const char *trim,
00174                                    const char *expire)

00176         REDISSOCK *dissocket;
00177 
00178         if (!insert || !trim || !expire) return 0;
00179 
00180         if (rediswho_command(insert, dissocket_p, inst, request) < 0) {
00181                 return -1;

00183 
00184         /* Only trim if necessary */
00185         dissocket = *dissocket_p;
00186         if (dissocket->reply->type == REDIS_REPLY_INTEGER) {
00187                 if (dissocket->reply->integer > inst->trim_count) {
00188                         if (rediswho_command(trim, dissocket_p,
00189                                              inst, request) < 0) {
00190                                 return -1;

00192                 }
00193         }
00194 
00195         rediswho_command(expire, dissocket_p, inst, request);
00196 
00197         return RLM_MODULE_OK;
