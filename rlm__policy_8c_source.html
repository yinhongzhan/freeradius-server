00071 /*
00072  *      Do any per-module initialization that is separate to each
00073  *      configured instance of the module.  e.g. set up connections
00074  *      to external databases, read configuration files, set up
00075  *      dictionary entries, etc.
00076  *
00077  *      If configuration information is given in the config section
00078  *      that must be referenced in later calls, store a handle to it
00079  *      in *instance otherwise put a null pointer there.
00080  */
00081 static int policy_instantiate(CONF_SECTION *conf, void **instance)
00082 {
00083         rlm_policy_t *inst;
00084 
00085         /*
00086          *      Set up a storage area for instance data
00087          */
00088         inst = rad_malloc(sizeof(*inst));
00089         if (!inst) {
00090                 return -1;
00091         }
00092         memset(inst, 0, sizeof(*inst));
00093 
00094         /*
00095          *      If the configuration parameters can't be parsed, then
00096          *      fail.
00097          */
00098         if (cf_section_parse(conf, inst, module_config) < 0) {
00099                 policy_detach(inst);
00100                 return -1;
00101         }
00102 
00103         inst->policies = rbtree_create(policyname_cmp,
00104                                        (void (*)(void *)) rlm_policy_free_item,
00105                                        0);
00106         if (!inst->policies) {
00107                 policy_detach(inst);
00108                 return -1;
00109         }
00110 
00111         /*
00112          *      Parse the policy from the file.
00113          */
00114         if (!rlm_policy_parse(inst->policies, inst->filename)) {
00115                 policy_detach(inst);
00116                 return -1;
00117         }
00118 
00119         *instance = inst;
00120 
00121         return 0;
00122 }
