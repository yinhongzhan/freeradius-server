00139 /*
00140  *      Decode one base64 chunk
00141  */
00142 static int decode_it(const char *src, uint8_t *dst)
00143 {
00144         int i;
00145         unsigned int x = 0;
00146 
00147         for(i = 0; i < 4; i++) {
00148                 if (src[i] >= 'A' && src[i] <= 'Z')
00149                         x = (x << 6) + (unsigned int)(src[i] - 'A' + 0);
00150                 else if (src[i] >= 'a' && src[i] <= 'z')
00151                          x = (x << 6) + (unsigned int)(src[i] - 'a' + 26);
00152                 else if(src[i] >= '0' && src[i] <= '9')
00153                          x = (x << 6) + (unsigned int)(src[i] - '0' + 52);
00154                 else if(src[i] == '+')
00155                         x = (x << 6) + 62;
00156                 else if (src[i] == '/')
00157                         x = (x << 6) + 63;
00158                 else if (src[i] == '=')
00159                         x = (x << 6);
00160                 else return 0;
00161         }
00162 
00163         dst[2] = (unsigned char)(x & 255); x >>= 8;
00164         dst[1] = (unsigned char)(x & 255); x >>= 8;
00165         dst[0] = (unsigned char)(x & 255);
00166 
00167         return 1;
00168 }

00171 /*
00172  *      Base64 decoding.
00173  */
00174 static int base64_decode (const char *src, uint8_t *dst)
00175 {
00176         int length, equals;
00177         int i, num;
00178         uint8_t last[3];
00179 
00180         length = equals = 0;
00181         while (src[length] && src[length] != '=') length++;
00182 
00183         while (src[length + equals] == '=') equals++;
00184 
00185         num = (length + equals) / 4;
00186 
00187         for (i = 0; i < num - 1; i++) {
00188                 if (!decode_it(src, dst)) return 0;
00189                 src += 4;
00190                 dst += 3;
00191         }
00192 
00193         decode_it(src, last);
00194         for (i = 0; i < (3 - equals); i++) {
00195                 dst[i] = last[i];
00196         }
00197 
00198         return (num * 3) - equals;
00199 }

00203  *      Hex or base64 or bin auto-discovery.
00204  */
00205 static void normify(REQUEST *request, VALUE_PAIR *vp, size_t min_length)
00206 {
00207         size_t decoded;
00208         uint8_t buffer[64];
00209 
00210         if (min_length >= sizeof(buffer)) return; /* paranoia */
00211 
00212         /*
00213          *      Hex encoding.
00214          */
00215         if (vp->length >= (2 * min_length)) {
00216                 decoded = fr_hex2bin(vp->vp_strvalue, buffer, vp->length >> 1);
00217                 if (decoded == (vp->length >> 1)) {
00218                         RDEBUG2("Normalizing %s from hex encoding", vp->name);
00219                         memcpy(vp->vp_octets, buffer, decoded);
00220                         vp->length = decoded;
00221                         return;
00222                 }
00223         }
00224 
00225         /*
00226          *      Base 64 encoding.  It's at least 4/3 the original size,
00227          *      and we want to avoid division...
00228          */
00229         if ((vp->length * 3) >= ((min_length * 4))) {
00230                 decoded = base64_decode(vp->vp_strvalue, buffer);
00231                 if (decoded >= min_length) {
00232                         RDEBUG2("Normalizing %s from base64 encoding", vp->name);
00233                         memcpy(vp->vp_octets, buffer, decoded);
00234                         vp->length = decoded;
00235                         return;
00236                 }
00237         }
00238 
00239         /*
00240          *      Else unknown encoding, or already binary.  Leave it.
00241          */
00242 }
