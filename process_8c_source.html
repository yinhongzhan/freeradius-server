01027 STATE_MACHINE_DECL(request_finish)
01028 {
01029         VALUE_PAIR *vp;
01030 
01031         TRACE_STATE_MACHINE;
01032 
01033         action = action;        /* -Wunused */
01034 
01035         if (request->master_state == REQUEST_STOP_PROCESSING) return;
01036 
01037         /*
01038          *      Don't send replies if there are none to send.
01039          */
01040         if (!request->in_request_hash) return;
01041 
01042         /*
01043          *      Catch Auth-Type := Reject BEFORE proxying the packet.
01044          */
01045         if ((request->packet->code == PW_AUTHENTICATION_REQUEST) &&
01046             (request->reply->code == 0)) {
01047                 if (((vp = pairfind(request->config_items, PW_AUTH_TYPE, 0)) != NULL) &&
01048                     (vp->vp_integer == PW_AUTHTYPE_REJECT)) {
01049                         request->reply->code = PW_AUTHENTICATION_REJECT;
01050 
01051                 } else {
01052                         /*
01053                          *      Check if the lack of response is
01054                          *      intentional.
01055                          */
01056                         vp = pairfind(request->config_items,
01057                                       PW_RESPONSE_PACKET_TYPE, 0);
01058                         if (!vp) {
01059                                 RDEBUG2("There was no response configured: rejecting request");
01060                                 request->reply->code = PW_AUTHENTICATION_REJECT;
01061 
01062                         } else if (vp->vp_integer == 256) {
01063                                 RDEBUG2("Not responding to request");
01064 
01065                         } else {
01066                                 request->reply->code = vp->vp_integer;
01067                         }
01068                 }
01069         }
01070 
01071         /*
01072          *      Copy Proxy-State from the request to the reply.
01073          */
01074         vp = paircopy2(request->packet->vps, PW_PROXY_STATE, 0);
01075         if (vp) pairadd(&request->reply->vps, vp);
01076 
01077         /*
01078          *      Run rejected packets through
01079          *
01080          *      Post-Auth-Type = Reject
01081          */
01082         if (request->reply->code == PW_AUTHENTICATION_REJECT) {
01083                 pairdelete(&request->config_items, PW_POST_AUTH_TYPE, 0);
01084                 vp = radius_pairmake(request, &request->config_items,
01085                                      "Post-Auth-Type", "Reject",
01086                                      T_OP_SET);
01087                 if (vp) rad_postauth(request);
01088         }
01089 
01090         if (request->reply->code == PW_AUTHENTICATION_ACK) {
01091                 rad_postauth(request);
01092         }
01093 
01094         /*
01095          *      Send the reply here.
01096          */
01097         if ((request->reply->code != PW_AUTHENTICATION_REJECT) ||
01098             (request->root->reject_delay == 0)) {
01099                 DEBUG_PACKET(request, request->reply, 1);
01100                 request->listener->send(request->listener,
01101                                         request);
01102         }
01103 
01104         gettimeofday(&request->reply->timestamp, NULL);
01105 
01106         /*
01107          *      Clean up.  These are no longer needed.
01108          */
01109         pairfree(&request->config_items);
01110 
01111         pairfree(&request->packet->vps);
01112         request->username = NULL;
01113         request->password = NULL;
01114         
01115         if (request->reply->code != PW_AUTHENTICATION_REJECT) {
01116                 pairfree(&request->reply->vps);
01117         }
01118 
01119 #ifdef WITH_PROXY
01120         if (request->proxy) {
01121                 pairfree(&request->proxy->vps);
01122 
01123                 if (request->proxy_reply) {
01124                         pairfree(&request->proxy_reply->vps);
01125                 }
01126         }
01127 #endif
01128         
01129         RDEBUG2("Finished request %u.", request->number);
01130 }

01494                         listener->print(listener, buffer, sizeof(buffer));
01495                         DEBUG("Reached idle timeout on socket %s", buffer);
01496                         goto do_close;
01497                 }
01498 
01499                 /*
01500                  *      Enforce the minimum of idle timeout or lifetime.
01501                  */
01502                 if (timercmp(&idle, &end, <)) {
01503                         end = idle;
01504                 }
01505         }
01506 
01507         /*
01508          *      Wake up at t + 0.5s.  The code above checks if the timers
01509          *      are <= t.  This addition gives us a bit of leeway.
01510          */
01511         end.tv_usec = USEC / 2;
01512 
01513         if (!fr_event_insert(el, tcp_socket_timer, listener, &end, &sock->ev)) {
01514                 rad_panic("Failed to insert event");
01515         }
01516 }

01520  *      Add +/- 2s of jitter, as suggested in RFC 3539
01521  *      and in RFC 5080.
01522  */
01523 static void add_jitter(struct timeval *when)
01524 {
01525         uint32_t jitter;
01526 
01527         when->tv_sec -= 2;
01528 
01529         jitter = fr_rand();
01530         jitter ^= (jitter >> 10);
01531         jitter &= ((1 << 22) - 1); /* 22 bits of 1 */
01532 
01533         /*
01534          *      Add in ~ (4 * USEC) of jitter.
01535          */
01536         tv_add(when, jitter);

01589 static void remove_from_proxy_hash_nl(REQUEST *request)
01590 {
01591         fr_packet_list_yank(proxy_list, request->proxy);
01592         fr_packet_list_id_free(proxy_list, request->proxy);
01593 
01594         /*
01595          *      On the FIRST reply, decrement the count of outstanding
01596          *      requests.  Note that this is NOT the count of sent
01597          *      packets, but whether or not the home server has
01598          *      responded at all.
01599          */
01600         if (request->home_server &&
01601             request->home_server->currently_outstanding) {
01602                 request->home_server->currently_outstanding--;
01603         }
01604 
01605 #ifdef WITH_TCP
01606         request->proxy_listener->count--;
01607 #endif
01608         request->proxy_listener = NULL;
01609 
01610         /*
01611          *      Got from YES in hash, to NO, not in hash while we hold
01612          *      the mutex.  This guarantees that when another thread
01613          *      grabs the mutex, the "not in hash" flag is correct.
01614          */
01615         request->in_proxy_hash = FALSE;
01616 }

01618 static void remove_from_proxy_hash(REQUEST *request)
01619 {
01620         /*
01621          *      Check this without grabbing the mutex because it's a
01622          *      lot faster that way.
01623          */
01624         if (!request->in_proxy_hash) return;
01625 
01626         /*
01627          *      The "not in hash" flag is definitive.  However, if the
01628          *      flag says that it IS in the hash, there might still be
01629          *      a race condition where it isn't.
01630          */
01631         PTHREAD_MUTEX_LOCK(&proxy_mutex);
01632 
01633         if (!request->in_proxy_hash) {
01634                 PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01635                 return;
01636         }
01637 
01638         remove_from_proxy_hash_nl(request);
01639 
01640         PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01641 }

01643 static int insert_into_proxy_hash(REQUEST *request)
01644 {
01645         char buf[128];
01646         int rcode, tries;
01647         void *proxy_listener;
01648 
01649         rad_assert(request->proxy != NULL);
01650         rad_assert(proxy_list != NULL);
01651 
01652         tries = 1;
01653 retry:
01654         PTHREAD_MUTEX_LOCK(&proxy_mutex);
01655         rcode = fr_packet_list_id_alloc(proxy_list,
01656                                         request->home_server->proto,
01657                                         request->proxy, &proxy_listener);
01658         request->num_proxied_requests = 1;
01659         request->num_proxied_responses = 0;
01660         PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01661         
01662         if (!rcode) {
01663 #ifdef HAVE_PTHREAD_H
01664                 if (proxy_no_new_sockets) return 0;
01665 #endif
01666 
01667                 /*
01668                  *      Also locks the proxy mutex, so we have to call
01669                  *      it with the mutex unlocked.  Some systems
01670                  *      don't support recursive mutexes.
01671                  */
01672                 if (!proxy_new_listener(request->home_server, 0)) {
01673                         radlog(L_ERR, "Failed to create a new socket for proxying requests.");
01674                         return 0;
01675                 }
01676                 request->proxy->src_port = 0; /* Use any new socket */
01677 
01678                 tries++;
01679                 if (tries > 2) {
01680                         RDEBUG2("ERROR: Failed allocating Id for new socket when proxying requests.");
01681                         return 0;
01682                 }
01683                 
01684                 goto retry;
01685         }
01686 
01687         request->proxy_listener = proxy_listener;
01688 
01689         PTHREAD_MUTEX_LOCK(&proxy_mutex);
01690         if (!fr_packet_list_insert(proxy_list, &request->proxy)) {
01691                 fr_packet_list_id_free(proxy_list, request->proxy);
01692                 PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01693                 radlog_request(L_PROXY, 0, request, "Failed to insert entry into proxy list.");
01694                 return 0;
01695         }
01696 
01697         request->in_proxy_hash = TRUE;
01698 
01699         /*
01700          *      Keep track of maximum outstanding requests to a
01701          *      particular home server.  'max_outstanding' is
01702          *      enforced in home_server_ldb(), in realms.c.
01703          */
01704         if (request->home_server) {
01705                 request->home_server->currently_outstanding++;
01706         }
01707 
01708 #ifdef WITH_TCP
01709         request->proxy_listener->count++;
01710 #endif
01711 
01712         PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01713 
01714         RDEBUG3(" proxy: allocating destination %s port %d - Id %d",
01715                inet_ntop(request->proxy->dst_ipaddr.af,
01716                          &request->proxy->dst_ipaddr.ipaddr, buf, sizeof(buf)),
01717                request->proxy->dst_port,
01718                request->proxy->id);
01719 
01720         return 1;
01721 }

01723 static int process_proxy_reply(REQUEST *request)
01724 {
01725         int rcode;
01726         int post_proxy_type = 0;
01727         VALUE_PAIR *vp;
01728         
01729         /*
01730          *      Delete any reply we had accumulated until now.
01731          */
01732         pairfree(&request->reply->vps);
01733         
01734         /*
01735          *      Run the packet through the post-proxy stage,
01736          *      BEFORE playing games with the attributes.
01737          */
01738         vp = pairfind(request->config_items, PW_POST_PROXY_TYPE, 0);
01739         if (vp) {
01740                 RDEBUG2("  Found Post-Proxy-Type %s", vp->vp_strvalue);
01741                 post_proxy_type = vp->vp_integer;
01742         }
01743         
01744         if (request->home_pool && request->home_pool->virtual_server) {
01745                 const char *old_server = request->server;
01746                 
01747                 request->server = request->home_pool->virtual_server;
01748                 RDEBUG2(" server %s {", request->server);
01749                 rcode = module_post_proxy(post_proxy_type, request);
01750                 RDEBUG2(" }");
01751                 request->server = old_server;
01752         } else {
01753                 rcode = module_post_proxy(post_proxy_type, request);
01754         }
01755 
01756 #ifdef WITH_COA
01757         if (request->packet->code == request->proxy->code)
01758           /*
01759            *    Don't run the next bit if we originated a CoA
01760            *    packet, after receiving an Access-Request or
01761            *    Accounting-Request.
01762            */
01763 #endif
01764 
01765         /*
01766          *      There may NOT be a proxy reply, as we may be
01767          *      running Post-Proxy-Type = Fail.
01768          */
01769         if (request->proxy_reply) {
01770                 /*
01771                  *      Delete the Proxy-State Attributes from
01772                  *      the reply.  These include Proxy-State
01773                  *      attributes from us and remote server.
01774                  */
01775                 pairdelete(&request->proxy_reply->vps, PW_PROXY_STATE, 0);
01776                 
01777                 /*
01778                  *      Add the attributes left in the proxy
01779                  *      reply to the reply list.
01780                  */
01781                 pairadd(&request->reply->vps, request->proxy_reply->vps);
01782                 request->proxy_reply->vps = NULL;
01783                 
01784                 /*
01785                  *      Free proxy request pairs.
01786                  */
01787                 pairfree(&request->proxy->vps);
01788         }
01789         
01790         switch (rcode) {
01791         default:  /* Don't do anything */
01792                 break;
01793         case RLM_MODULE_FAIL:
01794                 return 0;
01795                 
01796         case RLM_MODULE_HANDLED:
01797                 return 0;
01798         }
01799 
01800         return 1;
01801 }

01803 int request_proxy_reply(RADIUS_PACKET *packet)
01804 {
01805         RADIUS_PACKET **proxy_p;
01806         REQUEST *request;
01807         struct timeval now;
01808         char buffer[128];
01809 
01810         PTHREAD_MUTEX_LOCK(&proxy_mutex);
01811         proxy_p = fr_packet_list_find_byreply(proxy_list, packet);
01812 
01813         if (!proxy_p) {
01814                 PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01815                 radlog(L_PROXY, "No outstanding request was found for reply from host %s port %d - ID %d",
01816                        inet_ntop(packet->src_ipaddr.af,
01817                                  &packet->src_ipaddr.ipaddr,
01818                                  buffer, sizeof(buffer)),
01819                        packet->src_port, packet->id);
01820                 return 0;
01821         }
01822 
01823         request = fr_packet2myptr(REQUEST, proxy, proxy_p);
01824         request->num_proxied_responses++; /* needs to be protected by lock */
01825 
01826         PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
01827 
01828         /*
01829          *      No reply, BUT the current packet fails verification:
01830          *      ignore it.  This does the MD5 calculations in the
01831          *      server core, but I guess we can fix that later.
01832          */
01833         if (!request->proxy_reply &&
01834             (rad_verify(packet, request->proxy,
01835                         request->home_server->secret) != 0)) {
01836                 DEBUG("Ignoring spoofed proxy reply.  Signature is invalid");
01837                 return 0;
01838         }
01839 
01840         /*
01841          *      The home server sent us a packet which doesn't match
01842          *      something we have: ignore it.  This is done only to
01843          *      catch the case of broken systems.
01844          */
01845         if (request->proxy_reply &&
01846             (memcmp(request->proxy_reply->vector,
01847                     packet->vector,
01848                     sizeof(request->proxy_reply->vector)) != 0)) {
01849                 RDEBUG2("Ignoring conflicting proxy reply");
01850                 return 0;
01851         }
01852 
01853         gettimeofday(&now, NULL);
01854 
01855         /*
01856          *      Status-Server packets don't count as real packets.
01857          */
01858         if (request->proxy->code != PW_STATUS_SERVER) {
01859                 listen_socket_t *sock = request->proxy_listener->data;
01860 
01861                 request->home_server->last_packet = now.tv_sec;
01862                 sock->last_packet = now.tv_sec;
01863         }
01864 
01865         /*
01866          *      If we have previously seen a reply, ignore the
01867          *      duplicate.
01868          */
01869         if (request->proxy_reply) {
01870                 RDEBUG2("Discarding duplicate reply from host %s port %d  - ID: %d",
01871                         inet_ntop(packet->src_ipaddr.af,
01872                                   &packet->src_ipaddr.ipaddr,
01873                                   buffer, sizeof(buffer)),
01874                         packet->src_port, packet->id);
01875                 return 0;
01876         }
01877 
01878         /*
01879          *      Call the state machine to do something useful with the
01880          *      request.
01881          */
01882         request->proxy_reply = packet;
01883         packet->timestamp = now;
01884         request->priority = RAD_LISTEN_PROXY;
01885 
01886 #ifdef WITH_STATS
01887         request->home_server->stats.last_packet = packet->timestamp.tv_sec;
01888         request->proxy_listener->stats.last_packet = packet->timestamp.tv_sec;
01889 
01890         if (request->proxy->code == PW_AUTHENTICATION_REQUEST) {
01891                 proxy_auth_stats.last_packet = packet->timestamp.tv_sec;
01892 #ifdef WITH_ACCOUNTING
01893         } else if (request->proxy->code == PW_ACCOUNTING_REQUEST) {
01894                 proxy_acct_stats.last_packet = packet->timestamp.tv_sec;
01895 #endif
01896         }
01897 #endif  /* WITH_STATS */
01898 
01899 #ifdef WITH_COA
01900         /*
01901          *      When we originate CoA requests, we patch them in here
01902          *      so that they don't affect the rest of the state
01903          *      machine.
01904          */
01905         if (request->parent) {
01906                 rad_assert(request->parent->coa == request);
01907                 rad_assert((request->proxy->code == PW_COA_REQUEST) ||
01908                            (request->proxy->code == PW_DISCONNECT_REQUEST));
01909                 rad_assert(request->process != NULL);
01910                 request_coa_separate(request);
01911         }
01912 #endif
01913 
01914         request->process(request, FR_ACTION_PROXY_REPLY);
01915         
01916         return 1;
01917 }

02171          *      doesn't need it.
02172          */
02173         vp = radius_paircreate(request, &request->proxy->vps,
02174                                PW_PROXY_STATE, 0, PW_TYPE_OCTETS);
02175         snprintf(vp->vp_strvalue, sizeof(vp->vp_strvalue), "%d",
02176                  request->packet->id);
02177         vp->length = strlen(vp->vp_strvalue);
02178 
02179         /*
02180          *      Should be done BEFORE inserting into proxy hash, as
02181          *      pre-proxy may use this information, or change it.
02182          */
02183         request->proxy->code = request->packet->code;
02184 
02185         /*
02186          *      Call the pre-proxy routines.
02187          */
02188         vp = pairfind(request->config_items, PW_PRE_PROXY_TYPE, 0);
02189         if (vp) {
02190                 RDEBUG2("  Found Pre-Proxy-Type %s", vp->vp_strvalue);
02191                 pre_proxy_type = vp->vp_integer;
02192         }
02193 
02194         rad_assert(request->home_pool != NULL);
02195 
02196         if (request->home_pool->virtual_server) {
02197                 const char *old_server = request->server;
02198                 
02199                 request->server = request->home_pool->virtual_server;
02200                 RDEBUG2(" server %s {", request->server);
02201                 rcode = module_pre_proxy(pre_proxy_type, request);
02202                 RDEBUG2(" }");
02203                         request->server = old_server;
02204         } else {
02205                 rcode = module_pre_proxy(pre_proxy_type, request);
02206         }
02207         switch (rcode) {
02208         case RLM_MODULE_FAIL:
02209         case RLM_MODULE_INVALID:
02210         case RLM_MODULE_NOTFOUND:
02211         case RLM_MODULE_USERLOCK:
02212         default:
02213                 /* FIXME: debug print failed stuff */
02214                 return -1;
02215 
02216         case RLM_MODULE_REJECT:
02217         case RLM_MODULE_HANDLED:
02218                 return 0;
02219 
02220         /*
02221          *      Only proxy the packet if the pre-proxy code succeeded.
02222          */
02223         case RLM_MODULE_NOOP:
02224         case RLM_MODULE_OK:
02225         case RLM_MODULE_UPDATED:
02226                 break;
02227         }
02228 
02229         return 1;

02233 {
02234         char buffer[128];
02235 
02236         rad_assert(request->parent == NULL);
02237         rad_assert(request->home_server != NULL);
02238 
02239         if (request->master_state == REQUEST_STOP_PROCESSING) return 0;
02240 
02241 #ifdef WITH_COA

02243                 RDEBUG("WARNING: Cannot proxy and originate CoA packets at the same time.  Cancelling CoA request");
02244                 request_done(request->coa, FR_ACTION_DONE);

02246 #endif
02247 
02248         /*
02249          *      The request may be sent to a virtual server.  If we're
02250          *      in a child thread, just process it here. If we're the

02252          *      processing.

02256                       request->home_server->server);

02260 #ifdef HAVE_PTHREAD_H

02271          *      We're actually sending a proxied packet.  Do that now.
02272          */

02300 /*
02301  *      Proxy the packet as if it was new.
02302  */
02303 static int request_proxy_anew(REQUEST *request)
02304 {
02305         /*
02306          *      Keep a copy of the old Id so that the
02307          *      re-transmitted request doesn't re-use the old
02308          *      Id.  Note that in certain cases (socket crash)
02309          *      there is no Id as they have been purged from
02310          *      proxy_list, but there should still be a leftover
02311          *      packet hung off this request.
02312          */
02313         RADIUS_PACKET old = *request->proxy;
02314         int old_hash = request->in_proxy_hash;
02315         home_server *home;
02316         home_server *old_home = request->home_server;
02317 #ifdef WITH_TCP
02318         rad_listen_t *listener = request->proxy_listener;
02319 #endif
02320 
02321         rad_assert(old_home != NULL);
02322         
02323         /*
02324          *      Find a live home server for the request.
02325          */
02326         home = home_server_ldb(NULL, request->home_pool, request);
02327         if (!home) {
02328                 RDEBUG2("ERROR: Failed to find live home server for request");
02329         post_proxy_fail:
02330                 remove_from_proxy_hash(request);
02331 
02332                 if (!setup_post_proxy_fail(request)) {
02333                         return 0;
02334                 }
02335                 
02336                 request_queue_or_run(request, proxy_running);
02337                 return 0;
02338         }
02339 
02340         /*
02341          *      Don't free the old Id (if any) on error.
02342          */
02343         if (!insert_into_proxy_hash(request)) {
02344                 radlog_request(L_PROXY, 0, request, "Failed to insert retransmission into the proxy list.");
02345                 goto post_proxy_fail;
02346         }
02347 
02348         /*
02349          *      Now that we have a new Id, free the old one (if any)
02350          *      and update the various statistics.
02351          */
02352         PTHREAD_MUTEX_LOCK(&proxy_mutex);
02353         if (old_hash) {
02354                 fr_packet_list_yank(proxy_list, &old);
02355                 fr_packet_list_id_free(proxy_list, &old);
02356                 old_home->currently_outstanding--;
02357 #ifdef WITH_TCP
02358                 if (listener) listener->count--;
02359 #endif
02360         }
02361         PTHREAD_MUTEX_UNLOCK(&proxy_mutex);
02362 
02363         /*
02364          *      Free the old packet, to force re-encoding
02365          */
02366         free(request->proxy->data);
02367         request->proxy->data = NULL;
02368         request->proxy->data_len = 0;
02369 
02370 #ifdef WITH_ACCOUNTING
02371         /*
02372          *      Update the Acct-Delay-Time attribute.
02373          */
02374         if (request->packet->code == PW_ACCOUNTING_REQUEST) {
02375                 VALUE_PAIR *vp;
02376 
02377                 vp = pairfind(request->proxy->vps, PW_ACCT_DELAY_TIME, 0);
02378                 if (!vp) vp = radius_paircreate(request,
02379                                                 &request->proxy->vps,
02380                                                 PW_ACCT_DELAY_TIME, 0,
02381                                                 PW_TYPE_INTEGER);
02382                 if (vp) {
02383                         struct timeval now;
02384                         
02385                         gettimeofday(&now, NULL);
02386                         vp->vp_integer += now.tv_sec - request->proxy_retransmit.tv_sec;
02387                 }
02388         }
02389 #endif
02390 
02391         if (request_proxy(request, 1) != 1) goto post_proxy_fail;
02392 
02393         return 1;
02394 }

02396 STATE_MACHINE_DECL(request_ping)
02397 {
02398         home_server *home = request->home_server;
02399         char buffer[128];
02400 
02401         TRACE_STATE_MACHINE;
02402         ASSERT_MASTER;
02403 
02404         switch (action) {
02405         case FR_ACTION_TIMER:
02406                 radlog(L_ERR, "No response to status check %d for home server %s port %d",
02407                        request->number,
02408                        inet_ntop(request->proxy->dst_ipaddr.af,
02409                                  &request->proxy->dst_ipaddr.ipaddr,
02410                                  buffer, sizeof(buffer)),
02411                        request->proxy->dst_port);
02412                 break;
02413 
02414         case FR_ACTION_PROXY_REPLY:
02415                 rad_assert(request->in_proxy_hash);
02416 
02417                 request->home_server->num_received_pings++;
02418                 radlog_request(L_PROXY, 0, request, "Received response to status check %d (%d in current sequence)",
02419                        request->number, home->num_received_pings);
02420 
02421                 /*
02422                  *      Remove the request from any hashes
02423                  */
02424                 fr_event_delete(el, &request->ev);
02425                 remove_from_proxy_hash(request);
02426 
02427                 /*
02428                  *      The control socket may have marked the home server as
02429                  *      alive.  OR, it may have suddenly started responding to
02430                  *      requests again.  If so, don't re-do the "make alive"
02431                  *      work.
02432                  */
02433                 if (home->state == HOME_STATE_ALIVE) break;
02434                 
02435                 /*
02436                  *      We haven't received enough ping responses to mark it
02437                  *      "alive".  Wait a bit.
02438                  */
02439                 if (home->num_received_pings < home->num_pings_to_alive) {
02440                         break;
02441                 }
02442 
02443                 /*
02444                  *      Mark it alive and delete any outstanding
02445                  *      pings.
02446                  */
02447                 home->state = HOME_STATE_ALIVE;
02448                 exec_trigger(request, request->home_server->cs, "home_server.alive", FALSE);
02449                 home->currently_outstanding = 0;
02450                 home->num_sent_pings = 0;
02451                 home->num_received_pings = 0;
02452                 gettimeofday(&home->revive_time, NULL);
02453                 
02454                 fr_event_delete(el, &home->ev);
02455 
02456                 radlog_request(L_PROXY, 0, request, "Marking home server %s port %d alive",
02457                        inet_ntop(request->proxy->dst_ipaddr.af,
02458                                  &request->proxy->dst_ipaddr.ipaddr,
02459                                  buffer, sizeof(buffer)),
02460                        request->proxy->dst_port);
02461                 break;
02462 
02463         default:
02464                 RDEBUG3("%s: Ignoring action %s", __FUNCTION__, action_codes[action]);
02465                 break;
02466         }
02467 
02468         rad_assert(!request->in_request_hash);
02469         rad_assert(request->ev == NULL);
02470         request_done(request, FR_ACTION_DONE);
02471 }

02480         REQUEST *request;
02481         VALUE_PAIR *vp;

02483

02485             (home->ping_check == HOME_PING_CHECK_NONE) ||
02486 #ifdef WITH_TCP

02488 #endif
02489             (home->ev != NULL)) {
02490                 return;
02491         }

02494 
02495         if (home->state == HOME_STATE_ZOMBIE) {

02522                 request->proxy->code = PW_AUTHENTICATION_REQUEST;
02523

02525                                 "User-Name", home->ping_user_name, T_OP_SET);

02528                 radius_pairmake(request, &request->proxy->vps,
02529                                 "Service-Type", "Authenticate-Only", T_OP_SET);
02530                 radius_pairmake(request, &request->proxy->vps,
02531                                 "Message-Authenticator", "0x00", T_OP_SET);

02534 #ifdef WITH_ACCOUNTING

02552                              "NAS-Identifier", "", T_OP_SET);

02554                 snprintf(vp->vp_strvalue, sizeof(vp->vp_strvalue),

02557                 vp->length = strlen(vp->vp_strvalue);

02567 #endif

02588          *      Set up the timer callback.
02589          */

02597         home->num_sent_pings++;
02598 
02599         rad_assert(request->proxy_listener != NULL);
02600         request->proxy_listener->send(request->proxy_listener,
02601                                       request);
02602 
02603         /*
02604          *      Add +/- 2s of jitter, as suggested in RFC 3539
02605          *      and in the Issues and Fixes draft.

02607         home->when = now;
02608         home->when.tv_sec += home->ping_interval;
02609 
02610         add_jitter(&home->when);
02611 
02612         DEBUG("PING: Next status packet in %u seconds", home->ping_interval);

02614 }

02737 {
02738         struct timeval now, when;
02739         home_server *home = request->home_server;
02740         char buffer[128];
02741 
02742         TRACE_STATE_MACHINE;
02743 
02744         rad_assert(request->packet->code != PW_STATUS_SERVER);
02745         rad_assert(request->home_server != NULL);
02746

02748 
02749         rad_assert(request->child_state != REQUEST_DONE);

02752                 request_done(request, FR_ACTION_DONE);
02753                 return;
02754         }

02757         case FR_ACTION_DUP:

02765                 }

02769                         DEBUG2("Suppressing duplicate proxied request to home server %s port %d proto TCP - ID: %d",
02770                                inet_ntop(request->proxy->dst_ipaddr.af,
02771                                          &request->proxy->dst_ipaddr.ipaddr,
02772                                          buffer, sizeof(buffer)),
02773                                request->proxy->dst_port,
02774                                request->proxy->id);
02775                         return;
02776                 }
02777 #endif
02778 
02779 #ifdef WITH_ACCOUNTING
02780                 /*
02781                  *      If we update the Acct-Delay-Time, we need to
02782                  *      get a new ID.
02783                  */

02785                     pairfind(request->proxy->vps, PW_ACCT_DELAY_TIME, 0)) {
02786                         request_proxy_anew(request);
02787                         return;
02788                 }
02789 #endif
02790

02792                         inet_ntop(request->proxy->dst_ipaddr.af,
02793                                   &request->proxy->dst_ipaddr.ipaddr,
02794                                   buffer, sizeof(buffer)),
02795                         request->proxy->dst_port,
02796                         request->proxy->id);
02797                 request->num_proxied_requests++;
02798 
02799                 rad_assert(request->proxy_listener != NULL);;
02800                 DEBUG_PACKET(request, request->proxy, 1);
02801                 FR_STATS_TYPE_INC(home->stats.total_requests);
02802                 request->proxy_listener->send(request->proxy_listener,
02803                                               request);
02804                 break;
02805 
02806         case FR_ACTION_TIMER:
02807                 /*
02808                  *      If we haven't received a packet for a while,
02809                  *      mark it as zombie.  If the connection is TCP,
02810                  *      then another "watchdog timer" function takes
02811                  *      care of pings, etc.
02812                  */
02813                 if ((home->state == HOME_STATE_ALIVE) &&
02814 #ifdef WITH_TCP
02815                     (home->proto != IPPROTO_TCP) &&
02816 #endif
02817                     ((home->last_packet + ((home->zombie_period + 3) / 4)) < now.tv_sec)) {
02818                         mark_home_server_zombie(home);
02819                 }
02820 
02821                 when = request->proxy->timestamp;
02822                 when.tv_sec += home->response_window;
02823 
02824                 /*
02825                  *      Not at the response window.  Set the timer for
02826                  *      that.
02827                  */
02828                 if (timercmp(&when, &now, >)) {
02829                         STATE_MACHINE_TIMER(FR_ACTION_TIMER);
02830                         return;
02831                 }
02832 
02833                 FR_STATS_TYPE_INC(home->stats.total_timeouts);
02834                 if (home->type == HOME_TYPE_AUTH) {
02835                         if (request->proxy_listener) FR_STATS_TYPE_INC(request->proxy_listener->stats.total_timeouts);
02836                         FR_STATS_TYPE_INC(proxy_auth_stats.total_timeouts);
02837                 }

02839                 else if (home->type == HOME_TYPE_ACCT) {
02840                         if (request->proxy_listener) FR_STATS_TYPE_INC(request->proxy_listener->stats.total_timeouts);
02841                         FR_STATS_TYPE_INC(proxy_acct_stats.total_timeouts);
02842                 }
02843 #endif
02844 
02845                 /*
02846                  *      FIXME: debug log no response to proxied request
02847                  */
02848 
02849                 /*
02850                  *      No response, but we're supposed to do nothing
02851                  *      when there's no response.  The request is finished.
02852                  */
02853                 if (!home->no_response_fail) {
02854 #ifdef HAVE_PTHREAD_H
02855                         request->child_pid = NO_SUCH_CHILD_PID;
02856 #endif
02857                         gettimeofday(&request->reply->timestamp, NULL);
02858 #ifdef DEBUG_STATE_MACHINE
02859                         if (debug_flag) printf("(%u) ********\tSTATE %s C%u -> C%u\t********\n", request->number, __FUNCTION__, request->child_state, REQUEST_DONE);
02860 #endif
02861 #ifdef HAVE_PTHREAD_H
02862                         rad_assert(request->child_pid == NO_SUCH_CHILD_PID);
02863 #endif
02864                         request->child_state = REQUEST_DONE;
02865                         request_process_timer(request);
02866                         return;
02867                 }

02869                 /*
02870                  *      Do "fail on no response".
02871                  */
02872                 radlog_request(L_ERR, 0, request, "Rejecting request (proxy Id %d) due to lack of any response from home server %s port %d",
02873                                request->proxy->id,
02874                                inet_ntop(request->proxy->dst_ipaddr.af,
02875                                          &request->proxy->dst_ipaddr.ipaddr,
02876                                          buffer, sizeof(buffer)),
02877                                request->proxy->dst_port);
02878 
02879                 if (!setup_post_proxy_fail(request)) {
02880                         return;
02881                 }
02882                 /* FALL-THROUGH */
02883 
02884                 /*
02885                  *      Duplicate proxy replies have been quenched by
02886                  *      now.  This state is only called ONCE, when we
02887                  *      receive a new reply from the home server.
02888                  */
02889         case FR_ACTION_PROXY_REPLY:
02890                 request_queue_or_run(request, proxy_running);
02891                 break;
02892 
02893         case FR_ACTION_CONFLICTING:
02894                 request_done(request, action);
02895                 return;
02896 
02897         default:
02898                 RDEBUG3("%s: Ignoring action %s", __FUNCTION__, action_codes[action]);
02899                 break;

03146 static void request_coa_timer(REQUEST *request)
03147 {
03148         int delay, frac;
03149         struct timeval now, when, mrd;
03150 
03151         rad_assert(request->parent == NULL);
03152 
03153         if (request->proxy_reply) return request_process_timer(request);
03154 
03155         gettimeofday(&now, NULL);
03156 
03157         if (request->delay == 0) {
03158                 /*
03159                  *      Implement re-transmit algorithm as per RFC 5080
03160                  *      Section 2.2.1.
03161                  *
03162                  *      We want IRT + RAND*IRT
03163                  *      or 0.9 IRT + rand(0,.2) IRT
03164                  *
03165                  *      2^20 ~ USEC, and we want 2.
03166                  *      rand(0,0.2) USEC ~ (rand(0,2^21) / 10)
03167                  */
03168                 delay = (fr_rand() & ((1 << 22) - 1)) / 10;
03169                 request->delay = delay * request->home_server->coa_irt;
03170                 delay = request->home_server->coa_irt * USEC;
03171                 delay -= delay / 10;
03172                 delay += request->delay;
03173                 request->delay = delay;
03174                 
03175                 when = request->proxy->timestamp;
03176                 tv_add(&when, delay);
03177 
03178                 if (timercmp(&when, &now, >)) {
03179                         STATE_MACHINE_TIMER(FR_ACTION_TIMER);
03180                         return;
03181                 }
03182         }
03183 
03184         /*
03185          *      Retransmit CoA request.
03186          */
03187 
03188         /*
03189          *      Cap count at MRC, if it is non-zero.
03190          */
03191         if (request->home_server->coa_mrc &&
03192             (request->num_coa_requests >= request->home_server->coa_mrc)) {
03193                 if (!setup_post_proxy_fail(request)) {
03194                         return;
03195                 }
03196                 
03197                 request_queue_or_run(request, proxy_running);
03198                 return;
03199         }
03200 
03201         /*
03202          *      RFC 5080 Section 2.2.1
03203          *
03204          *      RT = 2*RTprev + RAND*RTprev
03205          *         = 1.9 * RTprev + rand(0,.2) * RTprev
03206          *         = 1.9 * RTprev + rand(0,1) * (RTprev / 5)
03207          */
03208         delay = fr_rand();
03209         delay ^= (delay >> 16);
03210         delay &= 0xffff;
03211         frac = request->delay / 5;
03212         delay = ((frac >> 16) * delay) + (((frac & 0xffff) * delay) >> 16);
03213 
03214         delay += (2 * request->delay) - (request->delay / 10);
03215 
03216         /*
03217          *      Cap delay at MRT, if MRT is non-zero.
03218          */
03219         if (request->home_server->coa_mrt &&
03220             (delay > (request->home_server->coa_mrt * USEC))) {
03221                 int mrt_usec = request->home_server->coa_mrt * USEC;
03222 
03223                 /*
03224                  *      delay = MRT + RAND * MRT
03225                  *            = 0.9 MRT + rand(0,.2)  * MRT
03226                  */
03227                 delay = fr_rand();
03228                 delay ^= (delay >> 15);
03229                 delay &= 0x1ffff;
03230                 delay = ((mrt_usec >> 16) * delay) + (((mrt_usec & 0xffff) * delay) >> 16);
03231                 delay += mrt_usec - (mrt_usec / 10);
03232         }
03233 
03234         request->delay = delay;
03235         when = now;
03236         tv_add(&when, request->delay);
03237         mrd = request->proxy->timestamp;
03238         mrd.tv_sec += request->home_server->coa_mrd;
03239 
03240         /*
03241          *      Cap duration at MRD.
03242          */
03243         if (timercmp(&mrd, &when, <)) {
03244                 when = mrd;
03245         }
03246         STATE_MACHINE_TIMER(FR_ACTION_TIMER);
03247 
03248         request->num_coa_requests++; /* is NOT reset by code 3 lines above! */
03249 
03250         FR_STATS_TYPE_INC(request->home_server->stats.total_requests);
03251         request->proxy_listener->send(request->proxy_listener,
03252                                       request);
03253 }
