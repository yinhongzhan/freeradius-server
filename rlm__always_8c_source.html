00058 static int str2rcode(const char *s)
00059 {
00060         if(!strcasecmp(s, "reject"))
00061                 return RLM_MODULE_REJECT;
00062         else if(!strcasecmp(s, "fail"))
00063                 return RLM_MODULE_FAIL;
00064         else if(!strcasecmp(s, "ok"))
00065                 return RLM_MODULE_OK;
00066         else if(!strcasecmp(s, "handled"))
00067                 return RLM_MODULE_HANDLED;
00068         else if(!strcasecmp(s, "invalid"))
00069                 return RLM_MODULE_INVALID;
00070         else if(!strcasecmp(s, "userlock"))
00071                 return RLM_MODULE_USERLOCK;
00072         else if(!strcasecmp(s, "notfound"))
00073                 return RLM_MODULE_NOTFOUND;
00074         else if(!strcasecmp(s, "noop"))
00075                 return RLM_MODULE_NOOP;
00076         else if(!strcasecmp(s, "updated"))
00077                 return RLM_MODULE_UPDATED;
00078         else {
00079                 radlog(L_ERR|L_CONS,
00080                         "rlm_always: Unknown module rcode '%s'.\n", s);
00081                 return -1;
00082         }
00083 }

00085 static int always_instantiate(CONF_SECTION *conf, void **instance)
00086 {
00087         rlm_always_t *data;
00088 
00089         /*
00090          *      Set up a storage area for instance data
00091          */
00092         data = rad_malloc(sizeof(*data));
00093         if (!data) {
00094                 return -1;
00095         }
00096         memset(data, 0, sizeof(*data));
00097 
00098         /*
00099          *      If the configuration parameters can't be parsed, then
00100          *      fail.
00101          */
00102         if (cf_section_parse(conf, data, module_config) < 0) {
00103                 free(data);
00104                 return -1;
00105         }
00106 
00107         /*
00108          *      Convert the rcode string to an int, and get rid of it
00109          */
00110         data->rcode = str2rcode(data->rcode_str);
00111         if (data->rcode == -1) {
00112                 free(data);
00113                 return -1;
00114         }
00115 
00116         *instance = data;
00117 
00118         return 0;
00119 }
