00152                  * vendor opt 220/0xdc - SoH payload, or null byte to probe, or string

00165                                         if (vlen <= 1) {
00166                                                 RDEBUG("SoH adding NAP marker to DHCP reply");
00167                                                 /* client probe; send "NAP" in the reply */

00169                                                 vp->vp_octets[0] = 220;

00200 {
00201         VALUE_PAIR *vp;

00204         /* try to find the MS-SoH payload */
