00035  */
00036 void
00037 otp_get_random(char *rnd_data, size_t len)
00038 {
00039   size_t bytes_read = 0;
00040 
00041   while (bytes_read < len) {
00042     int n;
00043     unsigned int bytes_left = len - bytes_read;
00044     uint32_t r = fr_rand();
00045 
00046     n = sizeof(r) < bytes_left ? sizeof(r) : bytes_left;
00047     (void) memcpy(rnd_data + bytes_read, &r, n);
00048     bytes_read += n;
00049   }
00050 }

00071 /* ascii to hex; returns HEXlen on success or -1 on error */
00072 int
00073 otp_a2x(const char *s, unsigned char x[])
00074 {
00075   unsigned i;
00076   size_t l = strlen(s);
00077 
00078   /*
00079    * We could just use sscanf, but we do this a lot, and have very
00080    * specific needs, and it's easy to implement, so let's go for it!
00081    */
00082   for (i = 0; i < l / 2; ++i) {
00083     unsigned int n[2];
00084     int j;
00085 
00086     /* extract 2 nibbles */
00087     n[0] = *s++;
00088     n[1] = *s++;
00089 
00090     /* verify range */
00091     for (j = 0; j < 2; ++j) {
00092       if ((n[j] >= '0' && n[j] <= '9') ||
00093           (n[j] >= 'A' && n[j] <= 'F') ||
00094           (n[j] >= 'a' && n[j] <= 'f'))
00095         continue;
00096       return -1;
00097     }
00098 
00099     /* convert ASCII hex digits to numeric values */
00100     n[0] -= '0';
00101     n[1] -= '0';
00102     if (n[0] > 9) {
00103       if (n[0] > 'F' - '0')
00104         n[0] -= 'a' - '9' - 1;
00105       else
00106         n[0] -= 'A' - '9' - 1;
00107     }
00108     if (n[1] > 9) {
00109       if (n[1] > 'F' - '0')
00110         n[1] -= 'a' - '9' - 1;
00111       else
00112         n[1] -= 'A' - '9' - 1;
00113     }
00114 
00115     /* store as octets */
00116     x[i]  = n[0] << 4;
00117     x[i] += n[1];
00118   } /* for (each octet) */
00119 
00120   return l/2;
00121 }

00131 /*
00132  * hex to ascii
00133  * Fills in s, which must point to at least len*2+1 bytes of space.
00134  */
00135 void
00136 otp_x2a(const unsigned char *x, size_t len, char *s)
00137 {
00138   unsigned i;
00139 
00140   for (i = 0; i < len; ++i) {
00141     unsigned n[2];
00142 
00143     n[0] = (x[i] >> 4) & 0x0f;
00144     n[1] = x[i] & 0x0f;
00145     s[2 * i + 0] = otp_hex_conversion[n[0]];
00146     s[2 * i + 1] = otp_hex_conversion[n[1]];
00147   }
00148   s[2 * len] = '\0';
