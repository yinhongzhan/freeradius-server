00031 /*
00032  *      Find the client definition.
00033  */
00034 static int dynamic_client_authorize(UNUSED void *instance, REQUEST *request)
00035 {
00036         size_t length;
00037         const char *value;
00038         CONF_PAIR *cp;
00039         RADCLIENT *c;
00040         char buffer[2048];
00041 
00042         /*
00043          *      Ensure we're only being called from the main thread,
00044          *      with fake packets.
00045          */
00046         if ((request->packet->src_port != 0) || (request->packet->vps != NULL) ||
00047             (request->parent != NULL)) {
00048                 RDEBUG("Improper configuration");
00049                 return RLM_MODULE_NOOP;
00050         }
00051 
00052         if (!request->client || !request->client->cs) {
00053                 RDEBUG("Unknown client definition");
00054                 return RLM_MODULE_NOOP;
00055         }
00056 
00057         cp = cf_pair_find(request->client->cs, "directory");
00058         if (!cp) {
00059                 RDEBUG("No directory configuration in the client");
00060                 return RLM_MODULE_NOOP;
00061         }
00062 
00063         value = cf_pair_value(cp);
00064         if (!value) {
00065                 RDEBUG("No value given for the directory entry in the client.");
00066                 return RLM_MODULE_NOOP;
00067         }
00068 
00069         length = strlen(value);
00070         if (length > (sizeof(buffer) - 256)) {
00071                 RDEBUG("Directory name too long");
00072                 return RLM_MODULE_NOOP;
00073         }
00074 
00075         memcpy(buffer, value, length + 1);
00076         ip_ntoh(&request->packet->src_ipaddr,
00077                 buffer + length, sizeof(buffer) - length - 1);
00078 
00079         /*
00080          *      Read the buffer and generate the client.
00081          */
00082         c = client_read(buffer, (request->client->server != NULL), TRUE);
00083         if (!c) return RLM_MODULE_FAIL;
00084 
00085         /*
00086          *      Replace the client.  This is more than a bit of a
00087          *      hack.
00088          */
00089         request->client = c;
00090 
00091         return RLM_MODULE_OK;
00092 }
