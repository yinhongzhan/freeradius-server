00065 static int str2rcode(const char *s)
00066 {
00067         if(!strcasecmp(s, "reject"))
00068                 return RLM_MODULE_REJECT;
00069         else if(!strcasecmp(s, "fail"))
00070                 return RLM_MODULE_FAIL;
00071         else if(!strcasecmp(s, "ok"))
00072                 return RLM_MODULE_OK;
00073         else if(!strcasecmp(s, "handled"))
00074                 return RLM_MODULE_HANDLED;
00075         else if(!strcasecmp(s, "invalid"))
00076                 return RLM_MODULE_INVALID;
00077         else if(!strcasecmp(s, "userlock"))
00078                 return RLM_MODULE_USERLOCK;
00079         else if(!strcasecmp(s, "notfound"))
00080                 return RLM_MODULE_NOTFOUND;
00081         else if(!strcasecmp(s, "noop"))
00082                 return RLM_MODULE_NOOP;
00083         else if(!strcasecmp(s, "updated"))
00084                 return RLM_MODULE_UPDATED;
00085         else {
00086                 radlog(L_ERR|L_CONS,
00087                         "rlm_sometimes: Unknown module rcode '%s'.\n", s);
00088                 return -1;
00089         }
00090 }

00146 /*
00147  *      A lie!  It always returns!
00148  */
00149 static int sometimes_return(void *instance, RADIUS_PACKET *packet,
00150                             RADIUS_PACKET *reply)
00151 {
00152         uint32_t hash;
00153         int value;
00154         rlm_sometimes_t *inst = instance;
00155         VALUE_PAIR *vp;
00156 
00157         /*
00158          *      Set it to NOOP and the module will always do nothing
00159          */
00160         if (inst->rcode == RLM_MODULE_NOOP) return inst->rcode;
00161 
00162         /*
00163          *      Hash based on the given key.  Usually User-Name.
00164          */
00165         vp = pairfind(packet->vps, inst->da->attr, inst->da->vendor);
00166         if (!vp) return RLM_MODULE_NOOP;
00167 
00168         hash = fr_hash(&vp->data, vp->length);
00169         hash &= 0xff;           /* ensure it's 0..255 */
00170         value = hash;
00171 
00172         /*
00173          *      Ranges are INCLUSIVE.
00174          *      [start,end] returns "rcode"
00175          *      Everything else returns "noop"
00176          */
00177         if (value < inst->start) return RLM_MODULE_NOOP;
00178         if (value > inst->end) return RLM_MODULE_NOOP;
00179 
00180         /*
00181          *      If we're returning "handled", then set the packet
00182          *      code in the reply, so that the server responds.
00183          */
00184         if ((inst->rcode == RLM_MODULE_HANDLED) && reply) {
00185                 switch (packet->code) {
00186                 case PW_AUTHENTICATION_REQUEST:
00187                         reply->code = PW_AUTHENTICATION_ACK;
00188                         break;
00189 
00190                 case PW_ACCOUNTING_REQUEST:
00191                         reply->code = PW_ACCOUNTING_RESPONSE;
00192                         break;
00193 
00194                 case PW_COA_REQUEST:
00195                         reply->code = PW_COA_ACK;
00196                         break;
00197 
00198                 case PW_DISCONNECT_REQUEST:
00199                         reply->code = PW_DISCONNECT_ACK;
00200                         break;
00201 
00202                 default:
00203                         break;
00204                 }
00205         }
00206 
00207         return inst->rcode;
00208 }
