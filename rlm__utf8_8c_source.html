00034 {
00035         size_t i, len;
00036         VALUE_PAIR *vp, *next;
00037 
00038         /* quiet the compiler */
00039         instance = instance;
00040 
00041         for (vp = request->packet->vps; vp != NULL; vp = next) {
00042                 next = vp->next;
00043 
00044                 if (vp->type != PW_TYPE_STRING) continue;
00045 
00046                 for (i = 0; i < vp->length; i += len) {
00047                         len = fr_utf8_char(&vp->vp_octets[i]);
00048                         if (len == 0) return RLM_MODULE_FAIL;
00049                 }
00050         }
00051 
00052         return RLM_MODULE_NOOP;
