00239 static int find_next_reset(rlm_counter_t *data, time_t timeval)
00240 {
00241         int ret = 0;
00242         size_t len;
00243         unsigned int num = 1;
00244         char last = '\0';
00245         struct tm *tm, s_tm;
00246         char sCurrentTime[40], sNextTime[40];
00247 
00248         tm = localtime_r(&timeval, &s_tm);
00249         len = strftime(sCurrentTime, sizeof(sCurrentTime), "%Y-%m-%d %H:%M:%S", tm);
00250         if (len == 0) *sCurrentTime = '\0';
00251         tm->tm_sec = tm->tm_min = 0;
00252 
00253         if (data->reset == NULL)
00254                 return -1;
00255         if (isdigit((int) data->reset[0])){
00256                 len = strlen(data->reset);
00257                 if (len == 0)
00258                         return -1;
00259                 last = data->reset[len - 1];
00260                 if (!isalpha((int) last))
00261                         last = 'd';
00262                 num = atoi(data->reset);
00263                 DEBUG("rlm_counter: num=%d, last=%c",num,last);
00264         }
00265         if (strcmp(data->reset, "hourly") == 0 || last == 'h') {
00266                 /*
00267                  *  Round up to the next nearest hour.
00268                  */
00269                 tm->tm_hour += num;
00270                 data->reset_time = mktime(tm);
00271         } else if (strcmp(data->reset, "daily") == 0 || last == 'd') {
00272                 /*
00273                  *  Round up to the next nearest day.
00274                  */
00275                 tm->tm_hour = 0;
00276                 tm->tm_mday += num;
00277                 data->reset_time = mktime(tm);
00278         } else if (strcmp(data->reset, "weekly") == 0 || last == 'w') {
00279                 /*
00280                  *  Round up to the next nearest week.
00281                  */
00282                 tm->tm_hour = 0;
00283                 tm->tm_mday += (7 - tm->tm_wday) +(7*(num-1));
00284                 data->reset_time = mktime(tm);
00285         } else if (strcmp(data->reset, "monthly") == 0 || last == 'm') {
00286                 tm->tm_hour = 0;
00287                 tm->tm_mday = 1;
00288                 tm->tm_mon += num;
00289                 data->reset_time = mktime(tm);
00290         } else if (strcmp(data->reset, "never") == 0) {
00291                 data->reset_time = 0;
00292         } else {
00293                 radlog(L_ERR, "rlm_counter: Unknown reset timer \"%s\"",
00294                         data->reset);
00295                 return -1;
00296         }
00297 
00298         len = strftime(sNextTime, sizeof(sNextTime), "%Y-%m-%d %H:%M:%S", tm);
00299         if (len == 0) *sNextTime = '\0';
00300         DEBUG2("rlm_counter: Current Time: %li [%s], Next reset %li [%s]",
00301                 timeval, sCurrentTime, data->reset_time, sNextTime);
00302 
00303         return ret;
00304 }
