00152 /*
00153  *      Reverse a key.
00154  */
00155 static uint32_t reverse(uint32_t key)
00156 {
00157         return ((reversed_byte[key & 0xff] << 24) |
00158                 (reversed_byte[(key >> 8) & 0xff] << 16) |
00159                 (reversed_byte[(key >> 16) & 0xff] << 8) |
00160                 (reversed_byte[(key >> 24) & 0xff]));
00161 }

00163 /*
00164  *      Take the parent by discarding the highest bit that is set.
00165  */
00166 static uint32_t parent_of(uint32_t key)
00167 {
00168         if (key > 0x00ffffff)
00169                 return (key & 0x00ffffff) | (parent_byte[key >> 24] << 24);
00170 
00171         if (key > 0x0000ffff)
00172                 return (key & 0x0000ffff) | (parent_byte[key >> 16] << 16);
00173 
00174         if (key > 0x000000ff)
00175                 return (key & 0x000000ff) | (parent_byte[key >> 8] << 8);
00176 
00177         return parent_byte[key];

00302 /*
00303  *      If the current bucket is uninitialized, initialize it
00304  *      by recursively copying information from the parent.
00305  *
00306  *      We may have a situation where entry E is a parent to 2 other
00307  *      entries E' and E".  If we split E into E and E', then the
00308  *      nodes meant for E" end up in E or E', either of which is
00309  *      wrong.  To solve that problem, we walk down the whole chain,
00310  *      inserting the elements into the correct place.
00311  */
00312 static void fr_hash_table_fixup(fr_hash_table_t *ht, uint32_t entry)
00313 {
00314         uint32_t parent_entry;
00315         fr_hash_entry_t **last, *cur;
00316         uint32_t this;
00317 
00318         parent_entry = parent_of(entry);
00319 
00320         /* parent_entry == entry if and only if entry == 0 */
00321 
00322         if (!ht->buckets[parent_entry]) {
00323                 fr_hash_table_fixup(ht, parent_entry);
00324         }
00325 
00326         /*
00327          *      Keep walking down cur, trying to find entries that
00328          *      don't belong here any more.  There may be multiple
00329          *      ones, so we can't have a naive algorithm...
00330          */
00331         last = &ht->buckets[parent_entry];
00332         this = parent_entry;
00333 
00334         for (cur = *last; cur != &ht->null; cur = cur->next) {
00335                 uint32_t real_entry;
00336 
00337                 real_entry = cur->key & ht->mask;
00338                 if (real_entry != this) { /* ht->buckets[real_entry] == NULL */
00339                         *last = &ht->null;
00340                         ht->buckets[real_entry] = cur;
00341                         this = real_entry;
00342                 }
00343 
00344                 last = &(cur->next);
00345         }
00346 
00347         /*
00348          *      We may NOT have initialized this bucket, so do it now.
00349          */
00350         if (!ht->buckets[entry]) ht->buckets[entry] = &ht->null;
00351 }

00386 /*
00387  *      Insert data.
00388  */
00389 int fr_hash_table_insert(fr_hash_table_t *ht, void *data)
00390 {
00391         uint32_t key;
00392         uint32_t entry;
00393         uint32_t reversed;
00394         fr_hash_entry_t *node;
00395 
00396         if (!ht || !data) return 0;
00397 
00398         key = ht->hash(data);
00399         entry = key & ht->mask;
00400         reversed = reverse(key);
00401 
00402         if (!ht->buckets[entry]) fr_hash_table_fixup(ht, entry);
00403 
00404         /*
00405          *      If we try to do our own memory allocation here, the
00406          *      speedup is only ~15% or so, which isn't worth it.
00407          */
00408         node = malloc(sizeof(*node));
00409         if (!node) return 0;
00410         memset(node, 0, sizeof(*node));
00411 
00412         node->next = &ht->null;
00413         node->reversed = reversed;
00414         node->key = key;
00415         node->data = data;
00416 
00417         /* already in the table, can't insert it */
00418         if (!list_insert(ht, &ht->buckets[entry], node)) {
00419                 free(node);
00420                 return 0;
00421         }
00422 
00423         /*
00424          *      Check the load factor, and grow the table if
00425          *      necessary.
00426          */
00427         ht->num_elements++;
00428         if (ht->num_elements >= ht->next_grow) {
00429                 fr_hash_table_grow(ht);
00430         }
00431 
00432         return 1;
00433 }

00747 /*
00748  *      Return a "folded" hash, where the lower "bits" are the
00749  *      hash, and the upper bits are zero.
00750  *
00751  *      If you need a non-power-of-two hash, cope.
00752  */
00753 uint32_t fr_hash_fold(uint32_t hash, int bits)
00754 {
00755         int count;
00756         uint32_t result;
00757 
00758         if ((bits <= 0) || (bits >= 32)) return hash;
00759 
00760         result = hash;
00761 
00762         /*
00763          *      Never use the same bits twice in an xor.
00764          */
00765         for (count = 0; count < 32; count += bits) {
00766                 hash >>= bits;
00767                 result ^= hash;
00768         }
00769 
00770         return result & (((uint32_t) (1 << bits)) - 1);
00771 }
