00052 /*
00053  * These are used in case ldap_search returns LDAP_SERVER_DOWN
00054  * In that case we do conn->failed_conns++ and then check it:
00055  * If conn->failed_conns <= MAX_FAILED_CONNS_START then we try
00056  * to reconnect
00057  * conn->failed_conns is also checked on entrance in perform_search:
00058  * If conn->failed_conns > MAX_FAILED_CONNS_START then we don't
00059  * try to do anything and we just do conn->failed_conns++ and
00060  * return RLM_MODULE_FAIL
00061  * if conn->failed_conns >= MAX_FAILED_CONNS_END then we give it
00062  * another chance and we set it to MAX_FAILED_CONNS_RESTART and
00063  * try to reconnect.
00064  *
00065  *
00066  * We are assuming that the majority of the LDAP_SERVER_DOWN cases
00067  * will either be an ldap connection timeout or a temporary ldap
00068  * server problem.
00069  * As a result we make a few attempts to reconnect hoping that the problem
00070  * will soon go away. If it does not go away then we just return
00071  * RLM_MODULE_FAIL on entrance in perform_search until conn->failed_conns
00072  * gets to MAX_FAILED_CONNS_END. After that we give it one more chance by
00073  * going back to MAX_FAILED_CONNS_RESTART
00074  *
00075  */
00076 
00077 #define MAX_FAILED_CONNS_END            20
00078 #define MAX_FAILED_CONNS_RESTART        4
00079 #define MAX_FAILED_CONNS_START          5

00719 /*
00720  *      read_mappings(...) reads a ldap<->radius mappings file to
00721  *      inst->reply_item_map and inst->check_item_map
00722  */
00723 #define MAX_LINE_LEN 160
00724 #define GENERIC_ATTRIBUTE_ID "$GENERIC$"
00725 
00726 static int
00727 read_mappings(ldap_instance* inst)
00728 {
00729         FILE* mapfile;
00730         char *filename;
00731 
00732         /*
00733          *      All buffers are of MAX_LINE_LEN so we can use sscanf
00734          *      without being afraid of buffer overflows
00735          */
00736         char buf[MAX_LINE_LEN], itemType[MAX_LINE_LEN];
00737         char radiusAttribute[MAX_LINE_LEN], ldapAttribute[MAX_LINE_LEN];
00738         int linenumber;
00739         FR_TOKEN operator;
00740         char opstring[MAX_LINE_LEN];
00741 
00742         /* open the mappings file for reading */
00743 
00744         filename = inst->dictionary_mapping;
00745         DEBUG("rlm_ldap: reading ldap<->radius mappings from file %s", filename);
00746         mapfile = fopen(filename, "r");
00747 
00748         if (mapfile == NULL) {
00749                 radlog(L_ERR, "rlm_ldap: Opening file %s failed: %s",
00750                        filename, strerror(errno));
00751                 return -1; /* error */
00752         }
00753 
00754         /*
00755          *      read file line by line. Note that if line length
00756          *      exceeds MAX_LINE_LEN, line numbers will be mixed up
00757          */
00758         linenumber = 0;
00759 
00760         while (fgets(buf, sizeof buf, mapfile)!=NULL) {
00761                 char* ptr;
00762                 int token_count;
00763                 TLDAP_RADIUS* pair;
00764 
00765                 linenumber++;
00766 
00767                 /* strip comments */
00768                 ptr = strchr(buf, '#');
00769                 if (ptr) *ptr = 0;
00770 
00771                 /* empty line */
00772                 if (buf[0] == 0) continue;
00773 
00774                 /* extract tokens from the string */
00775                 token_count = sscanf(buf, "%s %s %s %s",
00776                                      itemType, radiusAttribute,
00777                                      ldapAttribute, opstring);
00778 
00779                 if (token_count <= 0) /* no tokens */
00780                         continue;
00781 
00782                 if ((token_count < 3) || (token_count > 4)) {
00783                         radlog(L_ERR, "rlm_ldap: Skipping %s line %i: %s",
00784                                filename, linenumber, buf);
00785                         radlog(L_ERR, "rlm_ldap: Expected 3 to 4 tokens "
00786                                "(Item type, RADIUS Attribute and LDAP Attribute) but found only %i", token_count);
00787                         continue;
00788                 }
00789 
00790                 if (token_count == 3) {
00791                         operator = T_OP_INVALID; /* use defaults */
00792                 } else {
00793                         ptr = opstring;
00794                         operator = gettoken((void*)&ptr, buf, sizeof(buf));
00795                         if ((operator < T_OP_ADD) || (operator > T_OP_CMP_EQ)) {
00796                                 radlog(L_ERR, "rlm_ldap: file %s: skipping line %i: unknown or invalid operator %s",
00797                                        filename, linenumber, opstring);
00798                                 continue;
00799                         }
00800                 }
00801 
00802                 /* create new TLDAP_RADIUS list node */
00803                 pair = rad_malloc(sizeof(*pair));
00804 
00805                 pair->attr = strdup(ldapAttribute);
00806                 pair->radius_attr = strdup(radiusAttribute);
00807                 pair->operator = operator;
00808 
00809                 if ( (pair->attr == NULL) || (pair->radius_attr == NULL) ) {
00810                         radlog(L_ERR, "rlm_ldap: Out of memory");
00811                         if (pair->attr) free(pair->attr);
00812                         if (pair->radius_attr) free(pair->radius_attr);
00813                         free(pair);
00814                         fclose(mapfile);
00815                         return -1;
00816                 }
00817 
00818                 /* push node to correct list */
00819                 if (strcasecmp(itemType, "checkItem") == 0) {
00820                         pair->next = inst->check_item_map;
00821                         inst->check_item_map = pair;
00822                 } else if (strcasecmp(itemType, "replyItem") == 0) {
00823                         pair->next = inst->reply_item_map;
00824                         inst->reply_item_map = pair;
00825                 } else {
00826                         radlog(L_ERR, "rlm_ldap: file %s: skipping line %i: unknown itemType %s",
00827                                filename, linenumber, itemType);
00828                         free(pair->attr);
00829                         free(pair->radius_attr);
00830                         free(pair);
00831                         continue;
00832                 }
00833 
00834                 DEBUG("rlm_ldap: LDAP %s mapped to RADIUS %s",
00835                       pair->attr, pair->radius_attr);
00836         }
00837 
00838         fclose(mapfile);
00839 
00840         return 0; /* success */
00841 }

00844                           char *search_basedn, int scope, char *filter,
00845                           char **attrs, LDAPMessage ** result)
00846 {
00847         int             res = RLM_MODULE_OK;
00848         int             ldap_errno = 0;
00849         ldap_instance  *inst = instance;

00851         struct timeval  tv;
00852 
00853         *result = NULL;
00854 
00855         if (!conn){
00856                 radlog(L_ERR, "  [%s] NULL connection handle passed",
00857                         inst->xlat_name);

00859         }

00861                 conn->failed_conns++;

00863                         conn->failed_conns = MAX_FAILED_CONNS_RESTART;

00872                                 inst->xlat_name);
00873                         ldap_unbind_s(conn->ld);

00876                                              inst->password, 0, &res, NULL)) == NULL) {

00880                                 conn->failed_conns++;

00882                 }
00883                 conn->bound = 1;
00884                 conn->failed_conns = 0;
00885         }
00886 
00887         tv.tv_sec = inst->timeout;

00890                search_basedn ? search_basedn : "(null)" , filter);

00892                                attrs, 0, &tv, result)) {

00918                 return RLM_MODULE_FAIL;
00919         case LDAP_FILTER_ERROR:
00920                 radlog(L_ERR, "  [%s] ldap_search() failed: Bad search filter: %s", inst->xlat_name,filter);
00921                 ldap_msgfree(*result);
00922                 return RLM_MODULE_FAIL;
00923         case LDAP_TIMELIMIT_EXCEEDED:
00924                 exec_trigger(NULL, inst->cs, "modules.ldap.timeout", TRUE);
00925 
00926         case LDAP_BUSY:
00927         case LDAP_UNAVAILABLE:
00928                 /* We don't need to reconnect in these cases so we don't set conn->bound */
00929                 ldap_get_option(conn->ld, LDAP_OPT_ERROR_NUMBER, &ldap_errno);
00930                 radlog(L_ERR, "  [%s] ldap_search() failed: %s", inst->xlat_name,
00931                        ldap_err2string(ldap_errno));
00932                 ldap_msgfree(*result);
00933                 return (RLM_MODULE_FAIL);
00934         default:
00935                 ldap_get_option(conn->ld, LDAP_OPT_ERROR_NUMBER, &ldap_errno);

00937                        ldap_err2string(ldap_errno));
00938                 conn->bound = 0;
00939                 ldap_msgfree(*result);
00940                 return (RLM_MODULE_FAIL);
00941         }
00942

00944         if (ldap_errno != 1) {

00946                         DEBUG("  [%s] object not found", inst->xlat_name);
00947                 } else {
00948                         DEBUG("  [%s] got ambiguous search result (%d results)", inst->xlat_name, ldap_errno);
00949                 }
00950                 res = RLM_MODULE_NOTFOUND;
00951                 ldap_msgfree(*result);
00952         }
00953         return res;
00954 }

00957 /*
00958  *      Translate the LDAP queries.
00959  */
00960 static size_t ldap_escape_func(UNUSED REQUEST *request, char *out, size_t outlen, const char *in, void *arg)
00961 {
00962         ldap_instance   *inst = arg;
00963         size_t len = 0;
00964 
00965         while (in[0]) {
00966                 /*
00967                  *      Encode unsafe characters.
00968                  */
00969                 if (((len == 0) &&
00970                     ((in[0] == ' ') || (in[0] == '#'))) ||
00971                     (strchr(",+\"\\<>;*=()", *in))) {
00972                         static const char hex[] = "0123456789abcdef";
00973 
00974                         /*
00975                          *      Only 3 or less bytes available.
00976                          */
00977                         if (outlen <= 3) {
00978                                 break;
00979                         }
00980 
00981                         *(out++) = '\\';
00982                         *(out++) = hex[((*in) >> 4) & 0x0f];
00983                         *(out++) = hex[(*in) & 0x0f];
00984                         outlen -= 3;
00985                         len += 3;
00986                         in++;
00987                         continue;
00988                 }
00989 
00990                 /*
00991                  *      Only one byte left.
00992                  */
00993                 if (outlen <= 1) {
00994                         break;
00995                 }
00996 
00997                 /*
00998                  *      Allowed character.
00999                  */
01000                 *(out++) = *(in++);
01001                 outlen--;
01002                 len++;
01003         }
01004         *out = '\0';
01005         return len;
01006 }
