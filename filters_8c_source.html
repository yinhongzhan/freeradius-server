00353 /*
00354  *      ascend_parse_ipx_net
00355  *
00356  *      srcipxnet nnnn srcipxnode mmmmm [srcipxsoc cmd value ]
00357  */
00358 static int ascend_parse_ipx_net(int argc, char **argv,
00359                                 ascend_ipx_net_t *net, uint8_t *comp)
00360 {
00361         int             token;
00362         const char      *p;
00363 
00364         if (argc < 3) return -1;
00365 
00366         /*
00367          *      Parse the net, which is a hex number.
00368          */
00369         net->net = htonl(strtol(argv[0], NULL, 16));
00370 
00371         /*
00372          *      Parse the node.
00373          */
00374         token = fr_str2int(filterKeywords, argv[1], -1);
00375         switch (token) {
00376         case FILTER_IPX_SRC_IPXNODE:
00377         case FILTER_IPX_DST_IPXNODE:
00378                 break;
00379 
00380         default:
00381                 return -1;
00382         }
00383 
00384         /*
00385          *      Can have a leading "0x" or "0X"
00386          */
00387         p = argv[2];
00388         if ((memcmp(p, "0X", 2) == 0) ||
00389             (memcmp(p, "0x", 2) == 0)) p += 2;
00390 
00391         /*
00392          *      Node must be 6 octets long.
00393          */
00394         token = fr_hex2bin(p, net->node, IPX_NODE_ADDR_LEN);
00395         if (token != IPX_NODE_ADDR_LEN) return -1;
00396 
00397         /*
00398          *      Nothing more, die.
00399          */
00400         if (argc == 3) return 3;
00401 
00402         /*
00403          *      Can't be too little or too much.
00404          */
00405         if (argc != 6) return -1;
00406 
00407         /*
00408          *      Parse the socket.
00409          */
00410         token = fr_str2int(filterKeywords, argv[3], -1);
00411         switch (token) {
00412         case FILTER_IPX_SRC_IPXSOCK:
00413         case FILTER_IPX_DST_IPXSOCK:
00414                 break;
00415 
00416         default:
00417                 return -1;
00418         }
00419 
00420         /*
00421          *      Parse the command "<", ">", "=" or "!="
00422          */
00423         token = fr_str2int(filterCompare, argv[4], -1);
00424         switch (token) {
00425         case RAD_COMPARE_LESS:
00426         case RAD_COMPARE_EQUAL:
00427         case RAD_COMPARE_GREATER:
00428         case RAD_COMPARE_NOT_EQUAL:
00429                 *comp = token;
00430                 break;
00431 
00432         default:
00433                 return -1;
00434         }
00435 
00436         /*
00437          *      Parse the value.
00438          */
00439         token = strtoul(argv[5], NULL, 16);
00440         if (token > 65535) return -1;
00441 
00442         net->socket = token;
00443         net->socket = htons(net->socket);
00444 
00445 
00446         /*
00447          *      Everything's OK, we parsed 6 entries.
00448          */
00449         return 6;
00450 }

00452 /*
00453  *      ascend_parse_ipx_filter
00454  *
00455  *      This routine parses an IPX filter string from a string.
00456  *      The format of the string is:
00457  *
00458  *      [ srcipxnet nnnn srcipxnode mmmmm [srcipxsoc cmd value ]]
00459  *      [ dstipxnet nnnn dstipxnode mmmmm [dstipxsoc cmd value ]]
00460  *
00461  * Fields in [...] are optional.
00462  *      where:
00463  *
00464  *  srcipxnet:      Keyword for source IPX address.
00465  *                  nnnn = IPX Node address.
00466  *
00467  *  srcipxnode:     Keyword for source IPX Node address.
00468  *                  mmmmm = IPX Node Address, could be FFFFFF.
00469  *                  A vlid ipx node number should accompany ipx net number.
00470  *
00471  *      srcipxsoc:      Keyword for source IPX socket address.
00472  *
00473  *      cmd:            One of ">" or "<" or "=" or "!=".
00474  *
00475  *      value:          Socket value to be compared against, in hex.
00476  *
00477  *      dstipxnet:      Keyword for destination IPX address.
00478  *                      nnnn = IPX Node address.
00479  *
00480  *      dstipxnode:     Keyword for destination IPX Node address.
00481  *              mmmmm = IPX Node Address, could be FFFFFF.
00482  *                     A valid ipx node number should accompany ipx net number.
00483  *
00484  *      dstipxsoc:      Keyword for destination IPX socket address.
00485  *
00486  *      cmd:            One of ">" or "<" or "=" or "!=".
00487  *
00488  *      value:          Socket value to be compared against, in hex.
00489  */
00490 static int ascend_parse_ipx(int argc, char **argv, ascend_ipx_filter_t *filter)
00491 {
00492         int rcode;
00493         int token;
00494         int flags = 0;
00495 
00496         /*
00497          *      We may have nothing, in which case we simply return.
00498          */
00499         if (argc == 0) return 0;
00500 
00501         /*
00502          *      Must have "net N node M"
00503          */
00504         if (argc < 4) return -1;
00505 
00506         while ((argc > 0) && (flags != 0x03)) {
00507                 token = fr_str2int(filterKeywords, argv[0], -1);
00508                 switch (token) {
00509                 case FILTER_IPX_SRC_IPXNET:
00510                         if (flags & 0x01) return -1;
00511                         rcode = ascend_parse_ipx_net(argc - 1, argv + 1,
00512                                                      &(filter->src),
00513                                                      &(filter->srcSocComp));
00514                         if (rcode < 0) return -1;
00515                         argc -= (rcode + 1);
00516                         argv += rcode + 1;
00517                         flags |= 0x01;
00518                         break;
00519 
00520                 case FILTER_IPX_DST_IPXNET:
00521                         if (flags & 0x02) return -1;
00522                         rcode = ascend_parse_ipx_net(argc - 1, argv + 1,
00523                                                      &(filter->dst),
00524                                                      &(filter->dstSocComp));
00525                         if (rcode < 0) return -1;
00526                         argc -= (rcode + 1);
00527                         argv += rcode + 1;
00528                         flags |= 0x02;
00529                         break;
00530 
00531                 default:
00532                         fr_strerror_printf("Unknown string \"%s\" in IPX data filter",
00533                                    argv[0]);
00534                         return -1;
00535                 }
00536         }
00537 
00538         /*
00539          *      Arguments left over: die.
00540          */
00541         if (argc != 0) return -1;
00542 
00543         /*
00544          *      Everything's OK.
00545          */
00546         return 0;
00547 }

00550 /*
00551  *      Parse an IP address and optionally a netmask, to a uint32_t.
00552  *
00553  *      ipaddr should already be initialized to zero.
00554  *      ipaddr is in network byte order.
00555  *
00556  *      Returns -1 on error, or the number of bits in the netmask, otherwise.
00557  */
00558 static int ascend_parse_ipaddr(uint32_t *ipaddr, char *str)
00559 {
00560         int             count = 0;
00561         int             ip[4];
00562         int             masklen;
00563         uint32_t        netmask = 0;
00564 
00565         /*
00566          *      Look for IP's.
00567          */
00568         count = 0;
00569         while (*str && (count < 4) && (netmask == 0)) {
00570         next:
00571                 ip[count] = 0;
00572 
00573                 while (*str) {
00574                         switch (*str) {
00575                         case '0': case '1': case '2': case '3':
00576                         case '4': case '5': case '6': case '7':
00577                         case '8': case '9':
00578                                 ip[count] *= 10;
00579                                 ip[count] += (*str) - '0';
00580                                 str++;
00581                                 break;
00582 
00583 
00584                         case '.': /* dot between IP numbers. */
00585                                 str++;
00586                                 if (ip[count] > 255) return -1;
00587 
00588                                 /*
00589                                  *      24, 16, 8, 0, done.
00590                                  */
00591                                 *ipaddr |= (ip[count] << (8 * (3 - count)));
00592                                 count++;
00593                                 goto next;
00594 
00595                         case '/': /* netmask  */
00596                                 str++;
00597                                 masklen = atoi(str);
00598                                 if ((masklen < 0) || (masklen > 32)) return -1;
00599                                 str += strspn(str, "0123456789");
00600                                 netmask = masklen;
00601                                 goto finalize;
00602                                 break;
00603 
00604                         default:
00605                                 fr_strerror_printf("Invalid character in IP address");
00606                                 return -1;
00607                         }
00608                 } /* loop over one character */
00609         } /* loop until the count hits 4 */
00610 
00611         if (count == 3) {
00612         finalize:
00613                 /*
00614                  *      Do the last one, too.
00615                  */
00616                 if (ip[count] > 255) return -1;
00617 
00618                 /*
00619                  *      24, 16, 8, 0, done.
00620                  */
00621                 *ipaddr |= (ip[count] << (8 * (3 - count)));
00622         }
00623 
00624         /*
00625          *      We've hit the end of the IP address, and there's something
00626          *      else left over: die.
00627          */
00628         if (*str) return -1;
00629 
00630         /*
00631          *      Set the default netmask.
00632          */
00633         if (!netmask) {
00634                 if (!*ipaddr) {
00635                         netmask = 0;
00636                 } else if ((*ipaddr & 0x80000000) == 0) {
00637                         netmask = 8;
00638                 } else if ((*ipaddr & 0xc0000000) == 0x80000000) {
00639                         netmask = 16;
00640                 } else if ((*ipaddr & 0xe0000000) == 0xc0000000) {
00641                         netmask = 24;
00642                 } else {
00643                         netmask = 32;
00644                 }
00645         }
00646 
00647         *ipaddr = htonl(*ipaddr);
00648         return netmask;
00649 }

00652  *      ascend_parse_port:  Parse a comparator and port.
00653  *
00654  *      Returns -1 on error, or the comparator.
00655  */
00656 static int ascend_parse_port(uint16_t *port, char *compare, char *str)
00657 {
00658         int rcode, token = -1;
00659 
00660         /*
00661          *      There MUST be a comparison string.

00663         rcode = fr_str2int(filterCompare, compare, -1);
00664         if (rcode < 0) return rcode;

00666         if (strspn(str, "0123456789") == strlen(str)) {

00673 
00674         *port = token;

00844  *

00846  *

00848  *                      to start comparing.
00849  *
00850  *      mask:           A hexadecimal mask of bits to compare.
00851  *
00852  *      value:          A value to compare with the masked data.
00853  *
00854  *      compNeq:        Defines type of comparison. ( "==" or "!=")
00855  *                      Default is "==".
00856  *

00869          */

00871 
00872         /*
00873          *      We need at least "offset mask value"
00874          */
00875         if (argc < 3) return -1;
00876 
00877         /*
00878          *      No more than optional comparison and "more"
00879          */
00880         if (argc > 5) return -1;
00881 
00882         /*
00883          *      Offset is a uint16_t number.
00884          */
00885         if (strspn(argv[0], "0123456789") != strlen(argv[0])) return -1;
00886 
00887         rcode = atoi(argv[0]);
00888         if (rcode > 65535) return -1;
00889 
00890         filter->offset = rcode;
00891         filter->offset = htons(filter->offset);
00892 
00893         rcode = fr_hex2bin(argv[1], filter->mask, sizeof(filter->mask));
00894         if (rcode != sizeof(filter->mask)) return -1;
00895 
00896         token = fr_hex2bin(argv[2], filter->value, sizeof(filter->value));
00897         if (token != sizeof(filter->value)) return -1;
00898 
00899         /*
00900          *      The mask and value MUST be the same length.
00901          */
00902         if (rcode != token) return -1;
00903 
00904         filter->len = rcode;
00905         filter->len = htons(filter->len);
00906 
00907         /*
00908          *      Nothing more.  Exit.

00927                         flags |= 0x01;
00928                         break;

00930                 case FILTER_MORE:
00931                         if (flags & 0x02) return -1;
00932                         filter->more = htons( 1 );
00933                         flags |= 0x02;
00934                         break;
00935 
00936                 default:
00937                         fr_strerror_printf("Invalid string \"%s\" in generic data filter",
00938                                    argv[0]);
00939                         return -1;
00940                 }
00941 
00942                 argc--;
00943                 argv++;
00944         }
00945 
00946         return 0;
