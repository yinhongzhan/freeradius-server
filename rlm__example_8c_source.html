00030 /*
00031  *      Define a structure for our module configuration.
00032  *
00033  *      These variables do not need to be in a structure, but it's

00044 /*
00045  *      A mapping of configuration file names to internal variables.
00046  *
00047  *      Note that the string is dynamically allocated, so it MUST
00048  *      be freed.  When the configuration file parse re-reads the string,
00049  *      it free's the old one, and strdup's the new one, placing the pointer
00050  *      to the strdup'd string into 'config.string'.  This gets around
00051  *      buffer over-flows.
00052  */
00053 static const CONF_PARSER module_config[] = {
00054   { "integer", PW_TYPE_INTEGER,    offsetof(rlm_example_t,value), NULL,   "1" },
00055   { "boolean", PW_TYPE_BOOLEAN,    offsetof(rlm_example_t,boolean), NULL, "no"},
00056   { "string",  PW_TYPE_STRING_PTR, offsetof(rlm_example_t,string), NULL,  NULL},
00057   { "ipaddr",  PW_TYPE_IPADDR,     offsetof(rlm_example_t,ipaddr), NULL,  "*" },
00058 
00059   { NULL, -1, 0, NULL, NULL }           /* end the list */

00063 /*
00064  *      Do any per-module initialization that is separate to each
00065  *      configured instance of the module.  e.g. set up connections
00066  *      to external databases, read configuration files, set up
00067  *      dictionary entries, etc.
00068  *
00069  *      If configuration information is given in the config section
00070  *      that must be referenced in later calls, store a handle to it
00071  *      in *instance otherwise put a null pointer there.
00072  */
00073 static int example_instantiate(CONF_SECTION *conf, void **instance)
00074 {
00075         rlm_example_t *data;
00076 
00077         /*
00078          *      Set up a storage area for instance data
00079          */
00080         data = rad_malloc(sizeof(*data));
00081         if (!data) {
00082                 return -1;
00083         }
00084         memset(data, 0, sizeof(*data));
00085 
00086         /*
00087          *      If the configuration parameters can't be parsed, then
00088          *      fail.
00089          */
00090         if (cf_section_parse(conf, data, module_config) < 0) {
00091                 free(data);
00092                 return -1;
00093         }
00094 
00095         *instance = data;
00096 
00097         return 0;
00098 }

00100 /*
00101  *      Find the named user in this modules database.  Create the set
00102  *      of attribute-value pairs to check and reply with for this user
00103  *      from the database. The authentication code only needs to check
00104  *      the password, the rest is done here.
00105  */
00106 static int example_authorize(void *instance, REQUEST *request)
00107 {
00108         VALUE_PAIR *state;
00109         VALUE_PAIR *reply;
00110 
00111         /* quiet the compiler */
00112         instance = instance;
00113         request = request;
00114 
00115         /*
00116          *  Look for the 'state' attribute.
00117          */
00118         state =  pairfind(request->packet->vps, PW_STATE, 0);
00119         if (state != NULL) {
00120                 RDEBUG("Found reply to access challenge");
00121                 return RLM_MODULE_OK;
00122         }
00123 
00124         /*
00125          *  Create the challenge, and add it to the reply.
00126          */
00127         reply = pairmake("Reply-Message", "This is a challenge", T_OP_EQ);
00128         pairadd(&request->reply->vps, reply);
00129         state = pairmake("State", "0", T_OP_EQ);
00130         pairadd(&request->reply->vps, state);
00131 
00132         /*
00133          *  Mark the packet as an Access-Challenge packet.
00134          *
00135          *  The server will take care of sending it to the user.
00136          */
00137         request->reply->code = PW_ACCESS_CHALLENGE;
00138         RDEBUG("Sending Access-Challenge.");
00139 
00140         return RLM_MODULE_HANDLED;
00141 }
