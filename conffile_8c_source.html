01260                         }
01261                         p++;

01268                                 return 0; /* no trailing slash */

01305                 case '/':

01396                  */

01415                         ptr = cbuf;
01416                         while (isspace((int) *ptr)) ptr++;

01420                                 len -= (ptr - cbuf);

01426                  *      conditions.
01427                  */

01430                         
01431                         ptr = buf;
01432                         while (*ptr && isspace((int) *ptr)) ptr++;
