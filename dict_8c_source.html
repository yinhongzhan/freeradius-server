00942 static int sscanf_i(const char *str, unsigned int *pvalue)
00943 {
00944         int rcode = 0;
00945         int base = 10;
00946         static const char *tab = "0123456789";
00947 
00948         if ((str[0] == '0') &&
00949             ((str[1] == 'x') || (str[1] == 'X'))) {
00950                 tab = "0123456789abcdef";
00951                 base = 16;
00952 
00953                 str += 2;
00954         }
00955 
00956         while (*str) {
00957                 const char *c;
00958 
00959                 if (*str == '.') break;
00960 
00961                 c = memchr(tab, tolower((int) *str), base);
00962                 if (!c) return 0;
00963 
00964                 rcode *= base;
00965                 rcode += (c - tab);
00966                 str++;
00967         }
00968 
00969         *pvalue = rcode;
00970         return 1;
00971 }

01584 /*
01585  *      Process the VENDOR command
01586  */

01588                           int argc)
01589 {

01591         int     continuation = 0;
01592         const   char *format = NULL;
01593 
01594         if ((argc < 2) || (argc > 3)) {
01595                 fr_strerror_printf( "dict_init: %s[%d] invalid VENDOR entry",
01596                             fn, line);
01597                 return -1;
01598         }
01599 
01600         /*
01601          *       Validate all entries
01602          */
01603         if (!isdigit((int) argv[1][0])) {
01604                 fr_strerror_printf("dict_init: %s[%d]: invalid value",
01605                         fn, line);
01606                 return -1;
01607         }
01608         value = atoi(argv[1]);
01609 
01610         /* Create a new VENDOR entry for the list */
01611         if (dict_addvendor(argv[0], value) < 0) {
01612                 char buffer[256];
01613 
01614                 strlcpy(buffer, fr_strerror(), sizeof(buffer));
01615 
01616                 fr_strerror_printf("dict_init: %s[%d]: %s",
01617                            fn, line, buffer);
01618                 return -1;
01619         }
01620 
01621         /*
01622          *      Look for a format statement
01623          */
01624         if (argc == 3) {
01625                 format = argv[2];
01626 
01627         } else if (value == VENDORPEC_USR) { /* catch dictionary screw-ups */
01628                 format = "format=4,0";
01629 
01630         } else if (value == VENDORPEC_LUCENT) {
01631                 format = "format=2,1";
01632 
01633         } else if (value == VENDORPEC_STARENT) {
01634                 format = "format=2,2";
01635 
01636         } /* else no fixups to do */
01637 
01638         if (format) {
01639                 int type, length;
01640                 const char *p;
01641                 DICT_VENDOR *dv;
01642 
01643                 if (strncasecmp(format, "format=", 7) != 0) {
01644                         fr_strerror_printf("dict_init: %s[%d]: Invalid format for VENDOR.  Expected \"format=\", got \"%s\"",
01645                                    fn, line, format);
01646                         return -1;
01647                 }
01648 
01649                 p = format + 7;
01650                 if ((strlen(p) < 3) ||
01651                     !isdigit((int) p[0]) ||
01652                     (p[1] != ',') ||
01653                     !isdigit((int) p[2]) ||
01654                     (p[3] && (p[3] != ','))) {
01655                         fr_strerror_printf("dict_init: %s[%d]: Invalid format for VENDOR.  Expected text like \"1,1\", got \"%s\"",
01656                                    fn, line, p);
01657                         return -1;
01658                 }
01659 
01660                 type = (int) (p[0] - '0');
01661                 length = (int) (p[2] - '0');
01662 
01663                 if (p[3] == ',') {
01664                         if (!p[4]) {
01665                                 fr_strerror_printf("dict_init: %s[%d]: Invalid format for VENDOR.  Expected text like \"1,1\", got \"%s\"",
01666                                    fn, line, p);
01667                                 return -1;
01668                         }
01669 
01670                         if ((p[4] != 'c') ||
01671                             (p[5] != '\0')) {
01672                                 fr_strerror_printf("dict_init: %s[%d]: Invalid format for VENDOR.  Expected text like \"1,1\", got \"%s\"",
01673                                            fn, line, p);
01674                                 return -1;
01675                         }
01676                         continuation = 1;
01677 
01678                         if ((value != VENDORPEC_WIMAX) ||
01679                             (type != 1) || (length != 1)) {
01680                                 fr_strerror_printf("dict_init: %s[%d]: Only WiMAX VSAs can have continuations",
01681                                            fn, line);
01682                                 return -1;
01683                         }
01684                 }
01685 
01686                 dv = dict_vendorbyvalue(value);
01687                 if (!dv) {
01688                         fr_strerror_printf("dict_init: %s[%d]: Failed adding format for VENDOR",
01689                                    fn, line);
01690                         return -1;
01691                 }
01692 
01693                 if ((type != 1) && (type != 2) && (type != 4)) {
01694                         fr_strerror_printf("dict_init: %s[%d]: invalid type value %d for VENDOR",
01695                                    fn, line, type);
01696                         return -1;
01697                 }
01698 
01699                 if ((length != 0) && (length != 1) && (length != 2)) {
01700                         fr_strerror_printf("dict_init: %s[%d]: invalid length value %d for VENDOR",
01701                                    fn, line, length);
01702                         return -1;
01703                 }
01704 
01705                 dv->type = type;
01706                 dv->length = length;
01707                 dv->flags = continuation;
01708         }
01709 
01710         return 0;
01711 }

01713 /*
01714  *      String split routine.  Splits an input string IN PLACE
01715  *      into pieces, based on spaces.
01716  */
01717 int str2argv(char *str, char **argv, int max_argc)
01718 {
01719         int argc = 0;
01720 
01721         while (*str) {
01722                 if (argc >= max_argc) break;
01723 
01724                 /*
01725                  *      Chop out comments early.
01726                  */
01727                 if (*str == '#') {
01728                         *str = '\0';
01729                         break;
01730                 }
01731 
01732                 while ((*str == ' ') ||
01733                        (*str == '\t') ||
01734                        (*str == '\r') ||
01735                        (*str == '\n')) *(str++) = '\0';
01736 
01737                 if (!*str) break;
01738 
01739                 argv[argc] = str;
01740                 argc++;
01741 
01742                 while (*str &&
01743                        (*str != ' ') &&
01744                        (*str != '\t') &&
01745                        (*str != '\r') &&
01746                        (*str != '\n')) str++;
01747         }
01748 
01749         return argc;
01750 }

01756  */

01758                         const char *src_file, int src_line)
01759 {
01760         FILE    *fp;
01761         char    dirtmp[256];
01762         char    buf[256];
01763         char    *p;
01764         int     line = 0;
01765         unsigned int    vendor;
01766         unsigned int    block_vendor;
01767         struct stat statbuf;
01768         char    *argv[MAX_ARGV];
01769         int     argc;
01770         DICT_ATTR *da, *block_tlv[MAX_TLV_NEST + 1];
01771         int     which_block_tlv = 0;
01772 
01773         block_tlv[0] = NULL;
01774         block_tlv[1] = NULL;
01775         block_tlv[2] = NULL;
01776         block_tlv[3] = NULL;
01777 
01778         if (strlen(fn) >= sizeof(dirtmp) / 2 ||
01779             strlen(dir) >= sizeof(dirtmp) / 2) {
01780                 fr_strerror_printf("dict_init: filename name too long");
01781                 return -1;
01782         }
01783 
01784         /*
01785          *      First see if fn is relative to dir. If so, create
01786          *      new filename. If not, remember the absolute dir.
01787          */
01788         if ((p = strrchr(fn, FR_DIR_SEP)) != NULL) {
01789                 strcpy(dirtmp, fn);
01790                 dirtmp[p - fn] = 0;
01791                 dir = dirtmp;
01792         } else if (dir && dir[0] && strcmp(dir, ".") != 0) {
01793                 snprintf(dirtmp, sizeof(dirtmp), "%s/%s", dir, fn);
01794                 fn = dirtmp;

01798                 if (!src_file) {

01803                                    src_file, src_line, fn, strerror(errno));

01812                            fn);

01817          *      Globally writable dictionaries means that users can control
