00100 static void add_vp_tuple(VALUE_PAIR **vpp, VALUE rb_value,
00101         const char *function_name) {
00102     int i, outertuplesize;
00103     VALUE_PAIR *vp;
00104 
00105     /* If the Ruby function gave us nil for the tuple, then just return. */
00106     if (NIL_P(rb_value)) {
00107         return;
00108     }
00109 
00110     if (TYPE(rb_value) != T_ARRAY) {
00111         radlog(L_ERR, "add_vp_tuple, %s: non-array passed", function_name);
00112         return;
00113     }
00114 
00115     /* Get the array size. */
00116     outertuplesize = RARRAY(rb_value)->len;
00117 
00118     for (i = 0; i < outertuplesize; i++) {
00119         VALUE pTupleElement = rb_ary_entry(rb_value, i);
00120 
00121         if ((pTupleElement != 0) &&
00122                 (TYPE(pTupleElement) == T_ARRAY)) {
00123 
00124             /* Check if it's a pair */
00125             int tuplesize;
00126 
00127             if ((tuplesize = RARRAY(pTupleElement)->len) != 2) {
00128                 radlog(L_ERR, "%s: tuple element %d is a tuple "
00129                         " of size %d. must be 2\n", function_name,
00130                         i, tuplesize);
00131             } else {
00132                 VALUE pString1, pString2;
00133 
00134                 pString1 = rb_ary_entry(pTupleElement, 0);
00135                 pString2 = rb_ary_entry(pTupleElement, 1);
00136 
00137                 if ((TYPE(pString1) == T_STRING) &&
00138                         (TYPE(pString2) == T_STRING)) {
00139 
00140 
00141                     const char *s1, *s2;
00142 
00143                     /* pairmake() will convert and find any
00144                      * errors in the pair.
00145                      */
00146 
00147                     s1 = STR2CSTR(pString1);
00148                     s2 = STR2CSTR(pString2);
00149 
00150                     if ((s1 != NULL) && (s2 != NULL)) {
00151                         radlog(L_DBG, "%s: %s = %s ",
00152                                 function_name, s1, s2);
00153 
00154                         /* xxx Might need to support other T_OP */
00155                         vp = pairmake(s1, s2, T_OP_EQ);
00156                         if (vp != NULL) {
00157                             pairadd(vpp, vp);
00158                             radlog(L_DBG, "%s: s1, s2 OK\n",
00159                                     function_name);
00160                         } else {
00161                             radlog(L_DBG, "%s: s1, s2 FAILED\n",
00162                                     function_name);
00163                         }
00164                     } else {
00165                         radlog(L_ERR, "%s: string conv failed\n",
00166                                 function_name);
00167                     }
00168 
00169                 } else {
00170                     radlog(L_ERR, "%s: tuple element %d must be "
00171                             "(string, string)", function_name, i);
00172                 }
00173             }
00174         } else {
00175             radlog(L_ERR, "%s: tuple element %d is not a tuple\n",

00187 static int ruby_function(REQUEST *request, int func, VALUE module, const char *function_name) {
00188 #define BUF_SIZE 1024
00189 
00190     char buf[BUF_SIZE]; /* same size as vp_print buffer */
00191 
00192     VALUE_PAIR *vp;
00193     VALUE rb_request, rb_result, rb_reply_items, rb_config_items;
00194 
00195     int n_tuple, return_value;
00196     radlog(L_DBG, "Calling ruby function %s which has id: %d\n", function_name, func);
00197 
00198     /* Return with "OK, continue" if the function is not defined. 
00199      * TODO: Should check with rb_respond_to each time, just because ruby can define function dynamicly?
00200      */
00201     if (func == 0) {
00202         return RLM_MODULE_OK;
00203     }
00204 
00205     /* Default return value is "OK, continue" */
00206     return_value = RLM_MODULE_OK;
00207 
00208     n_tuple = 0;
00209 
00210     if (request != NULL) {
00211         for (vp = request->packet->vps; vp; vp = vp->next) {
00212             n_tuple++;
00213         }
00214     }
00215     /*
00216         Creating ruby array, that contains arrays of [name,value]
00217         Maybe we should use hash instead? Can this names repeat?
00218      */
00219     rb_request = rb_ary_new2(n_tuple);
00220     if (request != NULL) {
00221         for (vp = request->packet->vps; vp; vp = vp->next) {
00222             VALUE tmp = rb_ary_new2(2);
00223 
00224             /* The name. logic from vp_prints, lib/print.c */
00225             if (vp->flags.has_tag) {
00226                 snprintf(buf, BUF_SIZE, "%s:%d", vp->name, vp->flags.tag);
00227             } else {
00228                 strcpy(buf, vp->name);
00229             }
00230             VALUE rbString1 = rb_str_new2(buf);
00231             /* The value. Use delimiter - don't know what that means */
00232             vp_prints_value(buf, sizeof (buf), vp, 1);
00233             VALUE rbString2 = rb_str_new2(buf);
00234 
00235             rb_ary_push(tmp, rbString1);
00236             rb_ary_push(tmp, rbString2);
00237             rb_ary_push(rb_request, tmp);
00238         }
00239     }
00240 
00241     /* Calling corresponding ruby function, passing request and catching result */
00242     rb_result = rb_funcall(module, func, 1, rb_request);
00243 
00244     /* Checking result, it can be array of type [result, [array of reply pairs],[array of config pairs]],
00245      * It can also be just a fixnum, which is a result itself.
00246      */
00247     if (TYPE(rb_result) == T_ARRAY) {
00248         if (!FIXNUM_P(rb_ary_entry(rb_result, 0))) {
00249             radlog(L_ERR, "First element of an array was not a FIXNUM(Which has to be a return_value)");
00250         } else
00251             return_value = FIX2INT(rb_ary_entry(rb_result, 0));
00252         rb_reply_items = rb_ary_entry(rb_result, 1);
00253         rb_config_items = rb_ary_entry(rb_result, 2);
00254         add_vp_tuple(&request->reply->vps, rb_reply_items, function_name);
00255         add_vp_tuple(&request->config_items, rb_config_items, function_name);
00256     } else if (FIXNUM_P(rb_result)) {
00257         return_value = FIX2INT(rb_result);
00258     }
00259     return return_value;
00260 }

00286  * Import a user module and load a function from it
00287  */
00288 static int load_ruby_function(const char *f_name, int *func, VALUE module) {
00289     if (f_name == NULL) {
00290         *func = 0;
00291     } else {
00292         *func = rb_intern(f_name);
00293         /* rb_intern returns a symbol of a function, not a function itself
00294             it can be aplied to any recipient,
00295             so we should check it for our module recipient
00296          */
00297         if (!rb_respond_to(module, *func))
00298             *func = 0;
00299     }
00300     radlog(L_DBG, "load_ruby_function %s, result: %d", f_name, *func);
00301     return 0;
00302 }
