01432                 dv = dict_vendorbyvalue(vendor & (FR_MAX_VENDOR - 1));
01433                 if (dv) {
01434                         dv_type = dv->type;
01435                         if (dv_type > 3) dv_type = 3; /* hack */
01436                 }
01437         }

01440          *      Parse the next number.  It could be a Vendor-Type
01441          *      of 1..2^24, or it could be a TLV.
01442          */

01444                 attr = strtol(p + 1, &q, 10);
01445                 if (attr == 0) {
01446                         fr_strerror_printf("Invalid attribute number");
01447                         return NULL;
01448                 }
01449 
01450                 if (*q) {
01451                         if (*q != '.') goto invalid;
01452                         if (dv_type != 1) goto invalid;
01453                 }

01456         }

01458         /*
01459          *      Enforce a maximum value on the attribute number.
01460          */

01462 
01463         if (*p == '.') {
01464                 if (!dict_str2oid(p + 1, &attr, vendor, 1)) {
01465                         return NULL;
01466                 }
01467         }

01469         /*
01470          *      Maybe we're reading an old detail/config file, where
01471          *      it didn't know about a particular attribute and dumped
01472          *      it as hex.  Now the dictionaries have been updated,
01473          *      and we know about it.  So... convert it to the
01474          *      appropriate type.
01475          *
01476          *      FIXME: call data2vp_any.
01477          */
01478         da = dict_attrbyvalue(attr, vendor);
01479         if (da) {
01480                 /*
01481                  *      FIXME: convert hex to the data type.
01482                  */
01483                 return NULL;
01484         }
01485         
01486         /*
01487          *      We've now parsed the attribute properly, Let's create
01488          *      it.  This next stop also looks the attribute up in the
01489          *      dictionary, and creates the appropriate type for it.
01490          */
01491         if ((vp = paircreate(attr, vendor, PW_TYPE_OCTETS)) == NULL) {
01492                 fr_strerror_printf("out of memory");
01493                 return NULL;
01494         }
01495 
01496         vp->operator = (operator == 0) ? T_OP_EQ : operator;
01497         if (!value) return vp;
01498 
01499         size = strlen(value + 2);
01500         data = vp->vp_octets;
01501 
01502         vp->length = size >> 1;
01503         if (vp->length > sizeof(vp->vp_octets)) {
01504                 vp->vp_tlv = malloc(vp->length);
01505                 if (!vp->vp_tlv) {
01506                         fr_strerror_printf("Out of memory");
01507                         free(vp);
01508                         return NULL;
01509                 }
01510                 data = vp->vp_tlv;
01511                 vp->type |= PW_FLAG_LONG;
01512         }
01513 
01514         if (fr_hex2bin(value + 2, data, size) != vp->length) {
01515                 fr_strerror_printf("Invalid hex string");
01516                 free(vp);
01517                 return NULL;
01518         }
01519 
01520         return vp;

01524 /*
01525  *      Create a VALUE_PAIR from an ASCII attribute and value.
01526  */
01527 VALUE_PAIR *pairmake(const char *attribute, const char *value, int operator)
01528 {
01529         DICT_ATTR       *da;
01530         VALUE_PAIR      *vp;
01531         char            *tc, *ts;
01532         signed char     tag;
01533         int             found_tag;
01534         char            buffer[256];
01535         const char      *attrname = attribute;
01536 
01537         /*
01538          *    Check for tags in 'Attribute:Tag' format.
01539          */
01540         found_tag = 0;
01541         tag = 0;
01542 
01543         ts = strrchr(attribute, ':');
01544         if (ts && !ts[1]) {
01545                 fr_strerror_printf("Invalid tag for attribute %s", attribute);
01546                 return NULL;
01547         }
01548 
01549         if (ts && ts[1]) {
01550                 strlcpy(buffer, attribute, sizeof(buffer));
01551                 attrname = buffer;
01552                 ts = strrchr(attrname, ':');
01553 
01554                  /* Colon found with something behind it */
01555                  if (ts[1] == '*' && ts[2] == 0) {
01556                          /* Wildcard tag for check items */
01557                          tag = TAG_ANY;
01558                          *ts = 0;
01559                  } else if ((ts[1] >= '0') && (ts[1] <= '9')) {
01560                          /* It's not a wild card tag */
01561                          tag = strtol(ts + 1, &tc, 0);
01562                          if (tc && !*tc && TAG_VALID_ZERO(tag))
01563                                  *ts = 0;
01564                          else tag = 0;
01565                  } else {
01566                          fr_strerror_printf("Invalid tag for attribute %s", attribute);
01567                          return NULL;
01568                  }
01569                  found_tag = 1;
01570         }
01571 
01572         /*
01573          *      It's not found in the dictionary, so we use
01574          *      another method to create the attribute.
01575          */
01576         if ((da = dict_attrbyname(attrname)) == NULL) {
01577                 return pairmake_any(attrname, value, operator);
01578         }
01579 
01580         if ((vp = pairalloc(da)) == NULL) {
01581                 return NULL;
01582         }
01583 
01584         vp->operator = (operator == 0) ? T_OP_EQ : operator;
01585 
01586         /*      Check for a tag in the 'Merit' format of:
01587          *      :Tag:Value.  Print an error if we already found
01588          *      a tag in the Attribute.
01589          */
01590 
01591         if (value && (*value == ':' && da->flags.has_tag)) {
01592                 /* If we already found a tag, this is invalid */
01593                 if(found_tag) {
01594                         fr_strerror_printf("Duplicate tag %s for attribute %s",
01595                                    value, vp->name);
01596                         DEBUG("Duplicate tag %s for attribute %s\n",
01597                                    value, vp->name);
01598                         pairbasicfree(vp);
01599                         return NULL;
01600                 }
01601                 /* Colon found and attribute allows a tag */
01602                 if (value[1] == '*' && value[2] == ':') {
01603                        /* Wildcard tag for check items */
01604                        tag = TAG_ANY;
01605                        value += 3;
01606                 } else {
01607                        /* Real tag */
01608                        tag = strtol(value + 1, &tc, 0);
01609                        if (tc && *tc==':' && TAG_VALID_ZERO(tag))
01610                             value = tc + 1;
01611                        else tag = 0;
01612                 }
01613                 found_tag = 1;
01614         }
01615 
01616         if (found_tag) {
01617           vp->flags.tag = tag;
01618         }
01619 
01620         switch (vp->operator) {
01621         default:
01622                 break;
01623 
01624                 /*
01625                  *      For =* and !* operators, the value is irrelevant
01626                  *      so we return now.
01627                  */
01628         case T_OP_CMP_TRUE:
01629         case T_OP_CMP_FALSE:
01630                 vp->vp_strvalue[0] = '\0';
01631                 vp->length = 0;
01632                 return vp;
01633                 break;
01634 
01635                 /*
01636                  *      Regular expression comparison of integer attributes
01637                  *      does a STRING comparison of the names of their
01638                  *      integer attributes.
01639                  */
01640         case T_OP_REG_EQ:       /* =~ */
01641         case T_OP_REG_NE:       /* !~ */
01642 #ifndef WITH_REGEX
01643                 fr_strerror_printf("Regular expressions are not supported");
01644                 return NULL;
01645 
01646 #else
01647                 if (!value) {
01648                         fr_strerror_printf("No regular expression found in %s",
01649                                            vp->name);
01650                         pairbasicfree(vp);
01651                         return NULL;
01652                 }
01653 
01654                 pairbasicfree(vp);
01655                 
01656                 if (1) {
01657                         int compare;
01658                         regex_t reg;
01659                         
01660                         compare = regcomp(&reg, value, REG_EXTENDED);
01661                         if (compare != 0) {
01662                                 regerror(compare, &reg, buffer, sizeof(buffer));
01663                                 fr_strerror_printf("Illegal regular expression in attribute: %s: %s",
01664                                            attribute, buffer);
01665                                 return NULL;
01666                         }
01667                 }
01668 
01669                 return pairmake_xlat(attribute, value, operator);
01670 #endif
01671         }
01672 
01673         /*
01674          *      FIXME: if (strcasecmp(attribute, vp->name) != 0)
01675          *      then the user MAY have typed in the attribute name
01676          *      as Vendor-%d-Attr-%d, and the value MAY be octets.
01677          *
01678          *      We probably want to fix pairparsevalue to accept
01679          *      octets as values for any attribute.
01680          */
01681         if (value && (pairparsevalue(vp, value) == NULL)) {
01682                 pairbasicfree(vp);
01683                 return NULL;
01684         }
01685 
01686         return vp;
01687 }

01730 /*
01731  *      Read a valuepair from a buffer, and advance pointer.
01732  *      Sets *eol to T_EOL if end of line was encountered.
01733  */
01734 VALUE_PAIR *pairread(const char **ptr, FR_TOKEN *eol)
01735 {
01736         char            buf[64];
01737         char            attr[64];
01738         char            value[1024], *q;
01739         const char      *p;
01740         FR_TOKEN        token, t, xlat;
01741         VALUE_PAIR      *vp;
01742         size_t          len;
01743 
01744         *eol = T_OP_INVALID;
01745 
01746         p = *ptr;
01747         while ((*p == ' ') || (*p == '\t')) p++;
01748 
01749         if (!*p) {
01750                 *eol = T_OP_INVALID;
01751                 fr_strerror_printf("No token read where we expected an attribute name");
01752                 return NULL;
01753         }
01754 
01755         if (*p == '#') {
01756                 *eol = T_HASH;
01757                 fr_strerror_printf("Read a comment instead of a token");
01758                 return NULL;
01759         }
01760 
01761         q = attr;
01762         for (len = 0; len < sizeof(attr); len++) {
01763                 if (valid_attr_name[(int)*p]) {
01764                         *q++ = *p++;
01765                         continue;
01766                 }
01767                 break;
01768         }
01769 
01770         if (len == sizeof(attr)) {
01771                 *eol = T_OP_INVALID;
01772                 fr_strerror_printf("Attribute name is too long");
01773                 return NULL;
01774         }
01775 
01776         /*
01777          *      We may have Foo-Bar:= stuff, so back up.
01778          */
01779         if ((len > 0) && (attr[len - 1] == ':')) {
01780                 p--;
01781                 len--;
01782         }
01783 
01784         attr[len] = '\0';
01785         *ptr = p;
01786 
01787         /* Now we should have an operator here. */
01788         token = gettoken(ptr, buf, sizeof(buf));
01789         if (token < T_EQSTART || token > T_EQEND) {
01790                 fr_strerror_printf("expecting operator");
01791                 return NULL;
01792         }
01793 
01794         /* Read value.  Note that empty string values are allowed */
01795         xlat = gettoken(ptr, value, sizeof(value));
01796         if (xlat == T_EOL) {
01797                 fr_strerror_printf("failed to get value");
01798                 return NULL;
01799         }
01800 
01801         /*
01802          *      Peek at the next token. Must be T_EOL, T_COMMA, or T_HASH
01803          */
01804         p = *ptr;
01805         t = gettoken(&p, buf, sizeof(buf));
01806         if (t != T_EOL && t != T_COMMA && t != T_HASH) {
01807                 fr_strerror_printf("Expected end of line or comma");
01808                 return NULL;
01809         }
01810 
01811         *eol = t;
01812         if (t == T_COMMA) {
01813                 *ptr = p;
01814         }
01815 
01816         vp = NULL;
01817         switch (xlat) {
01818                 /*
01819                  *      Make the full pair now.
01820                  */
01821         default:
01822                 vp = pairmake(attr, value, token);
01823                 break;
01824 
01825                 /*
01826                  *      Perhaps do xlat's
01827                  */
01828         case T_DOUBLE_QUOTED_STRING:
01829                 p = strchr(value, '%');
01830                 if (p && (p[1] == '{')) {
01831                         if (strlen(value) >= sizeof(vp->vp_strvalue)) {
01832                                 fr_strerror_printf("Value too long");
01833                                 return NULL;
01834                         }
01835                         vp = pairmake_xlat(attr, value, token);
01836                         if (!vp) {
01837                                 *eol = T_OP_INVALID;
01838                                 return NULL;
01839                         }
01840 
01841                 } else {
01842                         /*
01843                          *      Parse && escape it, as defined by the
01844                          *      data type.
01845                          */
01846                         vp = pairmake(attr, value, token);
01847                         if (!vp) {
01848                                 *eol = T_OP_INVALID;
01849                                 return NULL;
01850                         }
01851                 }
01852                 break;
01853 
01854         case T_SINGLE_QUOTED_STRING:
01855                 vp = pairmake(attr, NULL, token);
01856                 if (!vp) {
01857                         *eol = T_OP_INVALID;
01858                         return NULL;
01859                 }
01860 
01861                 /*
01862                  *      String and octet types get copied verbatim.
01863                  */
01864                 if ((vp->type == PW_TYPE_STRING) ||
01865                     (vp->type == PW_TYPE_OCTETS)) {
01866                         strlcpy(vp->vp_strvalue, value,
01867                                 sizeof(vp->vp_strvalue));
01868                         vp->length = strlen(vp->vp_strvalue);
01869 
01870                         /*
01871                          *      Everything else gets parsed: it's
01872                          *      DATA, not a string!
01873                          */
01874                 } else if (!pairparsevalue(vp, value)) {
01875                         pairfree(&vp);
01876                         *eol = T_OP_INVALID;
01877                         return NULL;
01878                 }
01879                 break;
01880 
01881                 /*
01882                  *      Mark the pair to be allocated later.
01883                  */
01884         case T_BACK_QUOTED_STRING:
01885                 if (strlen(value) >= sizeof(vp->vp_strvalue)) {
01886                         fr_strerror_printf("Value too long");
01887                         return NULL;
01888                 }
01889 
01890                 vp = pairmake_xlat(attr, value, token);
01891                 if (!vp) {
01892                         *eol = T_OP_INVALID;
01893                         return NULL;
01894                 }
01895                 break;
01896         }
01897 
01898         /*
01899          *      If we didn't make a pair, return an error.
01900          */
01901         if (!vp) {
01902                 *eol = T_OP_INVALID;
01903                 return NULL;
01904         }
01905 
01906         return vp;
01907 }

01910  *      Read one line of attribute/value pairs. This might contain
01911  *      multiple pairs seperated by comma's.
01912  */
01913 FR_TOKEN userparse(const char *buffer, VALUE_PAIR **first_pair)
01914 {
01915         VALUE_PAIR      *vp;
01916         const char      *p;
01917         FR_TOKEN        last_token = T_OP_INVALID;
01918         FR_TOKEN        previous_token;
01919 
01920         /*
01921          *      We allow an empty line.
01922          */
01923         if (buffer[0] == 0)
01924                 return T_EOL;
01925 
01926         p = buffer;
01927         do {
01928                 previous_token = last_token;
01929                 if ((vp = pairread(&p, &last_token)) == NULL) {
01930                         return last_token;
01931                 }

01937          */
01938         if (last_token == T_HASH) {
01939                 return previous_token;
01940         }
01941 
01942         /*
01943          *      And return the last token which we read.
01944          */
01945         return last_token;
01946 }

02007 /*
02008  *      Compare two pairs, using the operator from "one".
02009  *
02010  *      i.e. given two attributes, it does:
02011  *
02012  *      (two->data) (one->operator) (one->data)
02013  *
02014  *      e.g. "foo" != "bar"
02015  *
02016  *      Returns true (comparison is true), or false (comparison is not true);
02017  *
02018  *      FIXME: Ignores tags!
02019  */
02020 int paircmp(VALUE_PAIR *one, VALUE_PAIR *two)
02021 {
02022         int compare;
02023 
02024         switch (one->operator) {
02025         case T_OP_CMP_TRUE:
02026                 return (two != NULL);
02027 
02028         case T_OP_CMP_FALSE:
02029                 return (two == NULL);
02030 
02031                 /*
02032                  *      One is a regex, compile it, print two to a string,
02033                  *      and then do string comparisons.
02034                  */
02035         case T_OP_REG_EQ:
02036         case T_OP_REG_NE:
02037 #ifndef WITH_REGEX
02038                 return -1;
02039 #else
02040                 {
02041                         regex_t reg;
02042                         char buffer[MAX_STRING_LEN * 4 + 1];
02043 
02044                         compare = regcomp(&reg, one->vp_strvalue,
02045                                           REG_EXTENDED);
02046                         if (compare != 0) {
02047                                 regerror(compare, &reg, buffer, sizeof(buffer));
02048                                 fr_strerror_printf("Illegal regular expression in attribute: %s: %s",
02049                                            one->name, buffer);
02050                                 return -1;
02051                         }
02052 
02053                         vp_prints_value(buffer, sizeof(buffer), two, 0);
02054 
02055                         /*
02056                          *      Don't care about substring matches,
02057                          *      oh well...
02058                          */
02059                         compare = regexec(&reg, buffer, 0, NULL, 0);
02060 
02061                         regfree(&reg);
02062                         if (one->operator == T_OP_REG_EQ) return (compare == 0);
02063                         return (compare != 0);
02064                 }
02065 #endif
02066 
02067         default:                /* we're OK */
02068                 break;
02069         }
02070 
02071         /*
02072          *      After doing the previous check for special comparisons,
02073          *      do the per-type comparison here.
02074          */
02075         switch (one->type) {
02076         case PW_TYPE_ABINARY:
02077         case PW_TYPE_OCTETS:
02078         {
02079                 size_t length;
02080 
02081                 if (one->length < two->length) {
02082                         length = one->length;
02083                 } else {
02084                         length = two->length;
02085                 }
02086 
02087                 if (length) {
02088                         compare = memcmp(two->vp_octets, one->vp_octets,
02089                                          length);
02090                         if (compare != 0) break;
02091                 }
02092 
02093                 /*
02094                  *      Contents are the same.  The return code
02095                  *      is therefore the difference in lengths.
02096                  *
02097                  *      i.e. "0x00" is smaller than "0x0000"
02098                  */
02099                 compare = two->length - one->length;
02100         }
02101                 break;
02102 
02103         case PW_TYPE_STRING:
02104                 compare = strcmp(two->vp_strvalue, one->vp_strvalue);
02105                 break;
02106 
02107         case PW_TYPE_BYTE:
02108         case PW_TYPE_SHORT:
02109         case PW_TYPE_INTEGER:
02110         case PW_TYPE_DATE:
02111                 compare = two->vp_integer - one->vp_integer;
02112                 break;
02113 
02114         case PW_TYPE_INTEGER64:
02115                 /*
02116                  *      Don't want integer overflow!
02117                  */
02118                 if (two->vp_integer64 < one->vp_integer64) {
02119                         compare = -1;
02120                 } else if (two->vp_integer64 > one->vp_integer64) {
02121                         compare = +1;
02122                 } else {
02123                         compare = 0;
02124                 }
02125                 break;
02126         case PW_TYPE_IPADDR:
02127                 compare = ntohl(two->vp_ipaddr) - ntohl(one->vp_ipaddr);
02128                 break;
02129 
02130         case PW_TYPE_IPV6ADDR:
