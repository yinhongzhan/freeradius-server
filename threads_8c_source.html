00428         REQUEST *request;

00432 
00433 #ifdef WITH_STATS
00434 #ifdef WITH_ACCOUNTING
00435         if (thread_pool.auto_limit_acct) {
00436                 struct timeval now;
00437 
00438                 gettimeofday(&now, NULL);
00439                 
00440                 thread_pool.pps_out.pps  = rad_pps(&thread_pool.pps_out.pps_old,
00441                                                    &thread_pool.pps_out.pps_now,
00442                                                    &thread_pool.pps_out.time_old,
00443                                                    &now);
00444                 thread_pool.pps_out.pps_now++;
00445         }
00446 #endif
00447 #endif
00448 
00449         /*
00450          *      Clear old requests from all queues.
00451          *
00452          *      We only do one pass over the queue, in order to
00453          *      amortize the work across the child threads.  Since we
00454          *      do N checks for one request de-queued, the old
00455          *      requests will be quickly cleared.
00456          */
00457         for (i = 0; i < RAD_LISTEN_MAX; i++) {
00458                 request = fr_fifo_peek(thread_pool.fifo[i]);
00459                 if (!request) continue;
00460 
00461                 rad_assert(request->magic == REQUEST_MAGIC);
00462 
00463                 if (request->master_state != REQUEST_STOP_PROCESSING) {
00464                         continue;
00465                 }
00466 
00467                 /*
00468                  *      This entry was marked to be stopped.  Acknowledge it.
00469                  */
00470                 request = fr_fifo_pop(thread_pool.fifo[i]);
00471                 rad_assert(request != NULL);
00472                 request->child_state = REQUEST_DONE;
00473                 thread_pool.num_queued--;
00474         }
00475 
00476         start = 0;
00477  retry:
00478         /*
00479          *      Pop results from the top of the queue
00480          */
00481         for (i = start; i < RAD_LISTEN_MAX; i++) {
00482                 request = fr_fifo_pop(thread_pool.fifo[i]);
00483                 if (request) {
00484                         start = i;
00485                         break;
00486                 }
00487         }
00488 
00489         if (!request) {
00490                 pthread_mutex_unlock(&thread_pool.queue_mutex);
00491                 *prequest = NULL;
00492                 return 0;
00493         }
00494 
00495         rad_assert(thread_pool.num_queued > 0);
00496         thread_pool.num_queued--;
00497         *prequest = request;
00498 
00499         rad_assert(*prequest != NULL);
00500         rad_assert(request->magic == REQUEST_MAGIC);
00501 
00502         request->component = "<core>";
00503         request->module = "<thread>";
00504 
00505         /*
00506          *      If the request has sat in the queue for too long,
00507          *      kill it.
00508          *
00509          *      The main clean-up code can't delete the request from
00510          *      the queue, and therefore won't clean it up until we
00511          *      have acknowledged it as "done".
00512          */
00513         if (request->master_state == REQUEST_STOP_PROCESSING) {
00514                 request->module = "<done>";
00515                 request->child_state = REQUEST_DONE;
00516                 goto retry;
00517         }
00518 
00519         /*
00520          *      The thread is currently processing a request.
00521          */
00522         thread_pool.active_threads++;
00523 
00524         blocked = time(NULL);
00525         if ((blocked - request->timestamp) > 5) {
00526                 if (last_complained < blocked) {
00527                         blocked -= request->timestamp;
00528                 } else {
00529                         blocked = 0;
00530                 }
00531         } else {
00532                 blocked = 0;
00533         }
00534 
00535         pthread_mutex_unlock(&thread_pool.queue_mutex);
00536 
00537         if (blocked) {
00538                 radlog(L_ERR, "Request %u has been waiting in the processing queue for %d seconds.  Check that all databases are running properly!",
00539                        request->number, (int) blocked);
00540         }
00541 
00542         return 1;
00543 }

00999  *      If there are too many or too few threads waiting, then we
01000  *      either create some more, or delete some.
01001  */
01002 static void thread_pool_manage(time_t now)
01003 {
01004         int spare;
01005         int i, total;
01006         THREAD_HANDLE *handle, *next;
01007         int active_threads;
01008 
01009         /*
01010          *      We don't need a mutex lock here, as we're reading
01011          *      active_threads, and not modifying it.  We want a close
01012          *      approximation of the number of active threads, and this
01013          *      is good enough.
01014          */
01015         active_threads = thread_pool.active_threads;
01016         spare = thread_pool.total_threads - active_threads;
01017         if (debug_flag) {
01018                 static int old_total = -1;
01019                 static int old_active = -1;
01020 
01021                 if ((old_total != thread_pool.total_threads) ||
01022                                 (old_active != active_threads)) {
01023                         DEBUG2("Threads: total/active/spare threads = %d/%d/%d",
01024                                         thread_pool.total_threads, active_threads, spare);
01025                         old_total = thread_pool.total_threads;
01026                         old_active = active_threads;
01027                 }
01028         }
01029 
01030         /*
01031          *      If there are too few spare threads.  Go create some more.
01032          */
01033         if ((thread_pool.total_threads < thread_pool.max_threads) &&
01034             (spare < thread_pool.min_spare_threads)) {
01035                 total = thread_pool.min_spare_threads - spare;
01036 
01037                 if ((total + thread_pool.total_threads) > thread_pool.max_threads) {
01038                         total = thread_pool.max_threads - thread_pool.total_threads;
01039                 }
01040 
01041                 DEBUG2("Threads: Spawning %d spares", total);
01042 
01043                 /*
01044                  *      Create a number of spare threads.
01045                  */
01046                 for (i = 0; i < total; i++) {
01047                         handle = spawn_thread(now, 1);
01048                         if (handle == NULL) {
01049                                 return;
01050                         }
01051                 }
01052 
01053                 return;         /* there aren't too many spare threads */
01054         }
01055 
01056         /*
01057          *      Only delete spare threads if we haven't already done
01058          *      so this second.
01059          */
01060         if (now == last_cleaned) {
01061                 return;
01062         }
01063         last_cleaned = now;
01064 
01065         /*
01066          *      Loop over the thread pool, deleting exited threads.
01067          */
01068         for (handle = thread_pool.head; handle; handle = next) {
01069                 next = handle->next;
01070 
01071                 /*
01072                  *      Maybe we've asked the thread to exit, and it
01073                  *      has agreed.
01074                  */
01075                 if (handle->status == THREAD_EXITED) {
01076                         delete_thread(handle);
01077                 }
01078         }

01082          *      passed since we last created one.  This helps to minimize
01083          *      the amount of create/delete cycles.
01084          */
01085         if ((now - thread_pool.time_last_spawned) < thread_pool.cleanup_delay) {
01086                 return;
01087         }
01088 
01089         /*
01090          *      If there are too many spare threads, delete one.
01091          *
01092          *      Note that we only delete ONE at a time, instead of
01093          *      wiping out many.  This allows the excess servers to
01094          *      be slowly reaped, just in case the load spike comes again.
01095          */
01096         if (spare > thread_pool.max_spare_threads) {
01097 
01098                 spare -= thread_pool.max_spare_threads;
01099 
01100                 DEBUG2("Threads: deleting 1 spare out of %d spares", spare);
01101 
01102                 /*
01103                  *      Walk through the thread pool, deleting the
01104                  *      first idle thread we come across.
01105                  */
01106                 for (handle = thread_pool.head; (handle != NULL) && (spare > 0) ; handle = next) {
01107                         next = handle->next;
01108 
01109                         /*

01111                          *      request, but still live, then tell it
01112                          *      to exit.
01113                          *
01114                          *      It will eventually wake up, and realize
01115                          *      it's been told to commit suicide.
01116                          */
01117                         if ((handle->request == NULL) &&
01118                             (handle->status == THREAD_RUNNING)) {
01119                                 handle->status = THREAD_CANCELLED;
01120                                 /*
01121                                  *      Post an extra semaphore, as a
01122                                  *      signal to wake up, and exit.
01123                                  */
01124                                 sem_post(&thread_pool.semaphore);
01125                                 spare--;
01126                                 break;
01127                         }
01128                 }
01129         }
01130 
01131         /*
01132          *      If the thread has handled too many requests, then make it
01133          *      exit.
01134          */
01135         if (thread_pool.max_requests_per_thread > 0) {
01136                 for (handle = thread_pool.head; handle; handle = next) {
01137                         next = handle->next;
01138 
01139                         /*
01140                          *      Not handling a request, but otherwise
01141                          *      live, we can kill it.
01142                          */

01144                             (handle->status == THREAD_RUNNING) &&
01145                             (handle->request_count > thread_pool.max_requests_per_thread)) {
01146                                 handle->status = THREAD_CANCELLED;
01147                                 sem_post(&thread_pool.semaphore);
