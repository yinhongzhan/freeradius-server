00058 /*
00059  * gethostbyaddr() & gethostbyname() return hostent structure
00060  * To make these functions thread safe, we need to
00061  * copy the data and not pointers
00062  *
00063  * struct hostent {
00064  *    char    *h_name;        * official name of host *
00065  *    char    **h_aliases;    * alias list *
00066  *    int     h_addrtype;     * host address type *
00067  *    int     h_length;       * length of address *
00068  *    char    **h_addr_list;  * list of addresses *
00069  * }
00070  * This struct contains 3 pointers as members.
00071  * The data from these pointers is copied into a buffer.
00072  * The buffer is formatted as below to store the data
00073  *  ---------------------------------------------------------------
00074  * | h_name\0alias_array\0h_aliases\0..\0addr_array\0h_addr_list\0 |
00075  *  ---------------------------------------------------------------
00076  */
00077 #if defined(LOCAL_GETHOSTBYNAMER) || defined(LOCAL_GETHOSTBYADDRR)
00078 #define BUFFER_OVERFLOW 255
00079 static int copy_hostent(struct hostent *from, struct hostent *to,
00080                         char *buffer, int buflen, int *error)
00081 {
00082     int i, len;
00083     char *ptr = buffer;
00084 
00085     *error = 0;
00086     to->h_addrtype = from->h_addrtype;
00087     to->h_length = from->h_length;
00088     to->h_name = (char *)ptr;
00089 
00090     /* copy hostname to buffer */
00091     len=strlen(from->h_name)+1;
00092     strcpy(ptr, from->h_name);
00093     ptr += len;
00094 
00095     /* copy aliases to buffer */
00096     to->h_aliases = (char**)ptr;
00097     for(i = 0; from->h_aliases[i]; i++);
00098     ptr += (i+1) * sizeof(char *);
00099 
00100     for(i = 0; from->h_aliases[i]; i++) {
00101        len = strlen(from->h_aliases[i])+1;
00102        if ((ptr-buffer)+len < buflen) {
00103            to->h_aliases[i] = ptr;
00104                strcpy(ptr, from->h_aliases[i]);
00105            ptr += len;
00106        } else {
00107            *error = BUFFER_OVERFLOW;
00108            return *error;
00109        }
00110     }
00111     to->h_aliases[i] = NULL;
00112
