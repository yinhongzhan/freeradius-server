00042 void
00043 fr_hmac_md5(const uint8_t *text, int text_len,
00044               const uint8_t *key, int key_len,
00045               uint8_t *digest)
00046 {
00047         FR_MD5_CTX context;
00048         uint8_t k_ipad[65];    /* inner padding -
00049                                       * key XORd with ipad
00050                                       */
00051         uint8_t k_opad[65];    /* outer padding -
00052                                       * key XORd with opad
00053                                       */
00054         uint8_t tk[16];
00055         int i;
00056         /* if key is longer than 64 bytes reset it to key=MD5(key) */
00057         if (key_len > 64) {
00058 
00059                 FR_MD5_CTX      tctx;
00060 
00061                 fr_MD5Init(&tctx);
00062                 fr_MD5Update(&tctx, key, key_len);
00063                 fr_MD5Final(tk, &tctx);
00064 
00065                 key = tk;
00066                 key_len = 16;
00067         }
00068 
00069         /*
00070          * the HMAC_MD5 transform looks like:
00071          *
00072          * MD5(K XOR opad, MD5(K XOR ipad, text))
00073          *
00074          * where K is an n byte key
00075          * ipad is the byte 0x36 repeated 64 times
00076 
00077          * opad is the byte 0x5c repeated 64 times
00078          * and text is the data being protected
00079          */
00080 
00081         /* start out by storing key in pads */
00082         memset( k_ipad, 0, sizeof(k_ipad));
00083         memset( k_opad, 0, sizeof(k_opad));
00084         memcpy( k_ipad, key, key_len);
00085         memcpy( k_opad, key, key_len);
00086 
00087         /* XOR key with ipad and opad values */
00088         for (i = 0; i < 64; i++) {
00089                 k_ipad[i] ^= 0x36;
00090                 k_opad[i] ^= 0x5c;
00091         }
00092         /*
00093          * perform inner MD5
00094          */
00095         fr_MD5Init(&context);                   /* init context for 1st
00096                                               * pass */
00097         fr_MD5Update(&context, k_ipad, 64);      /* start with inner pad */
00098         fr_MD5Update(&context, text, text_len); /* then text of datagram */
00099         fr_MD5Final(digest, &context);          /* finish up 1st pass */
00100         /*
00101          * perform outer MD5
00102          */
00103         fr_MD5Init(&context);                   /* init context for 2nd
00104                                               * pass */
00105         fr_MD5Update(&context, k_opad, 64);     /* start with outer pad */
00106         fr_MD5Update(&context, digest, 16);     /* then results of 1st
00107                                               * hash */
00108         fr_MD5Final(digest, &context);          /* finish up 2nd pass */
00109 }
