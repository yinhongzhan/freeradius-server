00093 static int rlm_rest_perform (rlm_rest_t *instance, rlm_rest_section_t *section,
00094                              void *handle, REQUEST *request)
00095 {
00096         size_t uri_len;
00097         char uri[REST_URI_MAX_LEN];
00098         
00099         int ret;
00100         
00101         RDEBUG("Expanding URI components");
00102         /*
00103          *      Build xlat'd URI, this allows REST servers to be specified by
00104          *      request attributes.
00105          */
00106         uri_len = rest_uri_build(instance, section, request, uri, sizeof(uri));
00107         if (uri_len <= 0) return -1;
00108 
00109         RDEBUG("Sending HTTP %s to \"%s\"",
00110                 fr_int2str(http_method_table, section->method, NULL),
00111                 uri);
00112 
00113         /*
00114          *      Configure various CURL options, and initialise the read/write
00115          *      context data.
00116          */
00117         ret = rest_request_config(instance, section, request,
00118                 handle,
00119                 section->method,
00120                 section->body,
00121                 uri);
00122         if (ret <= 0) return -1;
00123 
00124         /*
00125          *      Send the CURL request, pre-parse headers, aggregate incoming
00126          *      HTTP body data into a single contiguous buffer.
00127          */
00128         ret = rest_request_perform(instance, section, handle);
00129         if (ret <= 0) return -1;
00130 
00131         return 1;
00132 }

00282 /*
00283  *      Find the named user in this modules database.  Create the set
00284  *      of attribute-value pairs to check and reply with for this user
00285  *      from the database. The authentication code only needs to check
00286  *      the password, the rest is done here.
00287  */
00288 static int rlm_rest_authorize(void *instance, REQUEST *request)
00289 {
00290         rlm_rest_t *my_instance = instance;
00291         rlm_rest_section_t *section = &my_instance->authorize;
00292 
00293         void *handle;
00294         int hcode;
00295         int rcode = RLM_MODULE_OK;
00296         int ret;
00297 
00298         handle = fr_connection_get(my_instance->conn_pool);
00299         if (!handle) return RLM_MODULE_FAIL;
00300 
00301         ret = rlm_rest_perform(instance, section, handle, request);
00302         if (ret < 0) { 
00303                 rcode = RLM_MODULE_FAIL;
00304                 goto end;
00305         }
00306 
00307         hcode = rest_get_handle_code(handle);
00308 
00309         switch (hcode) {
00310                 case 404:
00311                 case 410:
00312                         rcode = RLM_MODULE_NOTFOUND;
00313                         break;
00314                 case 403:
00315                         rcode = RLM_MODULE_USERLOCK;
00316                         break;
00317                 case 401:
00318                         /*
00319                          *      Attempt to parse content if there was any.
00320                          */
00321                         ret = rest_request_decode(my_instance, section,
00322                                                   request, handle);
00323                         if (ret < 0) {
00324                                 rcode = RLM_MODULE_FAIL;
00325                                 break;
00326                         }
00327 
00328                         rcode = RLM_MODULE_REJECT;
00329                         break;
00330                 case 204:
00331                         rcode = RLM_MODULE_OK;
00332                         break;
00333                 default:
00334                         /*
00335                          *      Attempt to parse content if there was any.
00336                          */
00337                         if ((hcode >= 200) && (hcode < 300)) {
00338                                 ret = rest_request_decode(my_instance, section,
00339                                                           request, handle);
00340                                 if (ret < 0)       rcode = RLM_MODULE_FAIL;
00341                                 else if (ret == 0) rcode = RLM_MODULE_OK;
00342                                 else               rcode = RLM_MODULE_UPDATED;
00343                                 break;
00344                         } else if (hcode < 500) {
00345                                 rcode = RLM_MODULE_INVALID;
00346                         } else {
00347                                 rcode = RLM_MODULE_FAIL;
00348                         }
00349         }
00350 
00351         end:
00352 
00353         rlm_rest_cleanup(instance, section, handle);
00354 
00355         fr_connection_release(my_instance->conn_pool, handle);
00356 
00357         return rcode;
00358 }
