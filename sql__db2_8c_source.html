00184  *
00185  *      Purpose: database specific fetch_row. Returns a SQL_ROW struct

00190 static int sql_fetch_row(SQLSOCK * sqlsocket, SQL_CONFIG *config)
00191 {
00192         int c, i;
00193         SQLINTEGER len, slen;
00194         SQL_ROW retval;
00195         rlm_sql_db2_sock *sock;
00196 
00197         sock = sqlsocket->conn;
00198 
00199         c = sql_num_fields(sqlsocket, config);
00200         retval = (SQL_ROW)rad_malloc(c*sizeof(char*)+1);
00201         /* advance cursor */
00202         if(SQLFetch(sock->stmt) == SQL_NO_DATA_FOUND) {
00203                 sqlsocket->row = NULL;
00204                 return 0;
00205         }
00206 
00207         for(i = 0; i < c; i++) {
00208                 /* get column length */
00209                 SQLColAttribute(sock->stmt,
00210                                 i+1, SQL_DESC_DISPLAY_SIZE,
00211                                 NULL, 0, NULL, &len);
00212                 retval[i] = (char*)rad_malloc(len+1);
00213                 /* get the actual column */
00214                 SQLGetData(sock->stmt,
00215                                 i+1, SQL_C_CHAR, retval[i], len+1, &slen);
00216                 if(slen == SQL_NULL_DATA)
00217                         retval[i][0] = '\0';
00218         }
00219 
00220         sqlsocket->row = retval;
