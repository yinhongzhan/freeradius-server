00070 /*
00071  * we do it in 8-bit chunks, because we have to keep the numbers
00072  * in network byte order (i.e. MSB)
00073  *
00074  * make it a structure so that we can do structure assignments.

00076 typedef struct onesixty {
00077         uint8_t p[20];

00080 static void onesixty_add_mod(onesixty *sum, onesixty *a, onesixty *b)
00081 {
00082         uint32_t s;
00083         int i, carry;
00084 
00085         carry = 0;
00086         for(i=19; i>=0; i--) {
00087 /*      for(i=0; i<20; i++) {  */
00088                 s = a->p[i] + b->p[i] + carry;
00089                 sum->p[i] = s & 0xff;
00090                 carry = s >> 8;
00091         }
00092 }

00094 /*
00095  * run the FIPS-186-2 PRF on the given Master Key (160 bits)
00096  * in order to derive 1280 bits (160 bytes) of keying data from
00097  * it.
00098  *
00099  * Given that in EAP-SIM, this is coming from a 64-bit Kc it seems
00100  * like an awful lot of "randomness" to pull out.. (MCR)
00101  *
00102  */
00103 void fips186_2prf(uint8_t mk[20], uint8_t finalkey[160])
00104 {
00105         fr_SHA1_CTX context;
00106         int j;
00107         onesixty xval, xkey, w_0, w_1, sum, one;
00108         uint8_t *f;
00109         uint8_t zeros[64];
00110 
00111         /*
00112          * let XKEY := MK,
00113          *
00114          * Step 3: For j = 0 to 3 do
00115          *   a. XVAL = XKEY
00116          *   b. w_0 = SHA1(XVAL)
00117          *   c. XKEY = (1 + XKEY + w_0) mod 2^160
00118          *   d. XVAL = XKEY
00119          *   e. w_1 = SHA1(XVAL)
00120          *   f. XKEY = (1 + XKEY + w_1) mod 2^160
00121          * 3.3 x_j = w_0|w_1
00122          *
00123          */
00124         memcpy(&xkey, mk, sizeof(xkey));
00125 
00126         /* make the value 1 */
00127         memset(&one,  0, sizeof(one));
00128         one.p[19]=1;
00129 
00130         f=finalkey;
00131 
00132         for(j=0; j<4; j++) {
00133                 /*   a. XVAL = XKEY  */
00134                 xval = xkey;
00135 
00136                 /*   b. w_0 = SHA1(XVAL)  */
00137                 fr_SHA1Init(&context);
00138 
00139                 memset(zeros, 0, sizeof(zeros));
00140                 memcpy(zeros, xval.p, 20);
00141 #ifndef WITH_OPENSSL_SHA1
00142                 fr_SHA1Transform(context.state, zeros);
00143 #else
00144                 fr_SHA1Transform(&context, zeros);
00145 #endif
00146                 fr_SHA1FinalNoLen(w_0.p, &context);
00147 
00148                 /*   c. XKEY = (1 + XKEY + w_0) mod 2^160 */
00149                 onesixty_add_mod(&sum,  &xkey, &w_0);
00150                 onesixty_add_mod(&xkey, &sum,  &one);
00151 
00152                 /*   d. XVAL = XKEY  */
00153                 xval = xkey;
00154 
00155                 /*   e. w_1 = SHA1(XVAL)  */
00156                 fr_SHA1Init(&context);
00157 
00158                 memset(zeros, 0, sizeof(zeros));
00159                 memcpy(zeros, xval.p, 20);
00160 #ifndef WITH_OPENSSL_SHA1
00161                 fr_SHA1Transform(context.state, zeros);
00162 #else
00163                 fr_SHA1Transform(&context, zeros);
00164 #endif
00165                 fr_SHA1FinalNoLen(w_1.p, &context);
00166 
00167                 /*   f. XKEY = (1 + XKEY + w_1) mod 2^160 */
00168                 onesixty_add_mod(&sum,  &xkey, &w_1);
00169                 onesixty_add_mod(&xkey, &sum,  &one);
00170 
00171                 /* now store it away */
00172                 memcpy(f, &w_0, 20);
00173                 f += 20;
00174 
00175                 memcpy(f, &w_1, 20);
00176                 f += 20;
00177         }
00178 }
