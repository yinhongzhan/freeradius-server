00250 static void python_vptuple(VALUE_PAIR **vpp, PyObject *pValue,
00251                            const char *funcname)
00252 {
00253         int             i;
00254         int             tuplesize;
00255         VALUE_PAIR      *vp;
00256 
00257         /*
00258          *      If the Python function gave us None for the tuple,
00259          *      then just return.
00260          */

00262                 return;
00263 
00264         if (!PyTuple_CheckExact(pValue)) {
00265                 radlog(L_ERR, "rlm_python:%s: non-tuple passed", funcname);
00266                 return;
00267         }
00268         /* Get the tuple tuplesize. */
00269         tuplesize = PyTuple_GET_SIZE(pValue);
00270         for (i = 0; i < tuplesize; i++) {
00271                 PyObject *pTupleElement = PyTuple_GET_ITEM(pValue, i);
00272                 PyObject *pStr1;
00273                 PyObject *pStr2;

00275                 const char *s1;

00277

00279                         radlog(L_ERR, "rlm_python:%s: tuple element %d is not a tuple", funcname, i);
00280                         continue;
00281                 }

00283                 if ((pairsize = PyTuple_GET_SIZE(pTupleElement)) != 2) {
00284                         radlog(L_ERR, "rlm_python:%s: tuple element %d is a tuple of size %d. Must be 2", funcname, i, pairsize);

00294                 s2 = PyString_AsString(pStr2);
00295                 /* xxx Might need to support other T_OP */

00297                 if (vp != NULL) {
00298                         pairadd(vpp, vp);
00299                         radlog(L_DBG, "rlm_python:%s: '%s' = '%s'", funcname, s1, s2);
00300                 } else {
00301                         radlog(L_DBG, "rlm_python:%s: Failed: '%s' = '%s'", funcname, s1, s2);

00343 static int python_function(REQUEST *request, PyObject *pFunc,
00344                            const char *funcname)
00345 {
00346         VALUE_PAIR      *vp;
00347         PyObject        *pRet = NULL;
00348         PyObject        *pArgs = NULL;
00349         int             tuplelen;
00350         int             ret;
00351         
00352         PyGILState_STATE gstate;
00353         
00354         /* Return with "OK, continue" if the function is not defined. */
00355         if (pFunc == NULL)
00356                 return RLM_MODULE_OK;
00357         
00358         /* Default return value is "OK, continue" */
00359         ret = RLM_MODULE_OK;
00360         
00361         /*
00362          *      We will pass a tuple containing (name, value) tuples
00363          *      We can safely use the Python function to build up a
00364          *      tuple, since the tuple is not used elsewhere.
00365          *
00366          *      Determine the size of our tuple by walking through the packet.
00367          *      If request is NULL, pass None.
00368          */
00369         tuplelen = 0;
00370         if (request != NULL) {
00371                 for (vp = request->packet->vps; vp; vp = vp->next)
00372                         tuplelen++;
00373         }
00374 
00375         gstate = PyGILState_Ensure();
00376         
00377         if (tuplelen == 0) {
00378                 Py_INCREF(Py_None);
00379                 pArgs = Py_None;
00380         } else {
00381                 int i = 0;
00382                 if ((pArgs = PyTuple_New(tuplelen)) == NULL)
00383                         goto failed;
00384 
00385                 for (vp = request->packet->vps;
00386                      vp != NULL;
00387                      vp = vp->next, i++) {
00388                         PyObject *pPair;
00389                         
00390                         /* The inside tuple has two only: */
00391                         if ((pPair = PyTuple_New(2)) == NULL)
00392                                 goto failed;
00393                         
00394                         if (python_populate_vptuple(pPair, vp) == 0) {
00395                                 /* Put the tuple inside the container */
00396                                 PyTuple_SET_ITEM(pArgs, i, pPair);
00397                         } else {
00398                                 Py_INCREF(Py_None);
00399                                 PyTuple_SET_ITEM(pArgs, i, Py_None);
00400                                 Py_DECREF(pPair);
00401                         }
00402                 }
00403         }
00404         
00405         /* Call Python function. */
00406         pRet = PyObject_CallFunctionObjArgs(pFunc, pArgs, NULL);
00407         
00408         if (pRet == NULL)
00409                 goto failed;
00410         
00411         if (request == NULL)
00412                 goto okay;
00413 
00414         /*
00415          *      The function returns either:
00416          *  1. (returnvalue, replyTuple, configTuple), where
00417          *   - returnvalue is one of the constants RLM_*
00418          *   - replyTuple and configTuple are tuples of string
00419          *      tuples of size 2
00420          *
00421          *  2. the function return value alone
00422          *
00423          *  3. None - default return value is set
00424          *
00425          * xxx This code is messy!
00426          */
00427         if (PyTuple_CheckExact(pRet)) {
00428                 PyObject *pTupleInt;
00429                 
00430                 if (PyTuple_GET_SIZE(pRet) != 3) {
00431                         radlog(L_ERR, "rlm_python:%s: tuple must be (return, replyTuple, configTuple)", funcname);
00432                         goto failed;
00433                 }
00434                 
00435                 pTupleInt = PyTuple_GET_ITEM(pRet, 0);
00436                 if (!PyInt_CheckExact(pTupleInt)) {
00437                         radlog(L_ERR, "rlm_python:%s: first tuple element not an integer", funcname);
00438                         goto failed;
00439                 }
00440                 /* Now have the return value */
00441                 ret = PyInt_AsLong(pTupleInt);
00442                 /* Reply item tuple */
00443                 python_vptuple(&request->reply->vps,
00444                                PyTuple_GET_ITEM(pRet, 1), funcname);
00445                 /* Config item tuple */
00446                 python_vptuple(&request->config_items,
00447                                PyTuple_GET_ITEM(pRet, 2), funcname);
00448 
00449         } else if (PyInt_CheckExact(pRet)) {
00450                 /* Just an integer */
00451                 ret = PyInt_AsLong(pRet);
00452 
00453         } else if (pRet == Py_None) {
00454                 /* returned 'None', return value defaults to "OK, continue." */
00455                 ret = RLM_MODULE_OK;
00456         } else {
00457                 /* Not tuple or None */
00458                 radlog(L_ERR, "rlm_python:%s: function did not return a tuple or None", funcname);
00459                 goto failed;
00460         }
00461 
00462  okay:
00463         Py_DECREF(pArgs);
00464         Py_DECREF(pRet);
00465         PyGILState_Release(gstate);
00466         return ret;
00467         
00468  failed:
00469         python_error();
00470         Py_XDECREF(pArgs);
00471         Py_XDECREF(pRet);
00472         PyGILState_Release(gstate);
00473         
00474         return -1;
00475 }
