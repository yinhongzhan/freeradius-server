00041 static int replicate_packet(void *instance, REQUEST *request)
00042 {
00043         int rcode = RLM_MODULE_NOOP;
00044         VALUE_PAIR *vp, *last;
00045         home_server *home;
00046         REALM *realm;
00047         home_pool_t *pool;
00048         RADIUS_PACKET *packet = NULL;
00049 
00050         instance = instance;    /* -Wunused */
00051         last = request->config_items;
00052 
00053         /*
00054          *      Send as many packets as necessary to different
00055          *      destinations.
00056          */
00057         while (1) {
00058                 vp = pairfind(last, PW_REPLICATE_TO_REALM, 0);
00059                 if (!vp) break;
00060 
00061                 last = vp->next;
00062 
00063                 realm = realm_find2(vp->vp_strvalue);
00064                 if (!realm) {
00065                         RDEBUG2("ERROR: Cannot Replicate to unknown realm %s", realm);
00066                         continue;
00067                 }
00068                 
00069                 /*
00070                  *      We shouldn't really do this on every loop.
00071                  */
00072                 switch (request->packet->code) {
00073                 default:
00074                         RDEBUG2("ERROR: Cannot replicate unknown packet code %d",
00075                                 request->packet->code);
00076                         cleanup(packet);
00077                         return RLM_MODULE_FAIL;
00078                 
00079                 case PW_AUTHENTICATION_REQUEST:
00080                         pool = realm->auth_pool;
00081                         break;
00082                         
00083 #ifdef WITH_ACCOUNTING
00084                         
00085                 case PW_ACCOUNTING_REQUEST:
00086                         pool = realm->acct_pool;
00087                         break;
00088 #endif
00089                         
00090 #ifdef WITH_COA
00091                 case PW_COA_REQUEST:
00092                 case PW_DISCONNECT_REQUEST:
00093                         pool = realm->acct_pool;
00094                         break;
00095 #endif
00096                 }
00097                 
00098                 if (!pool) {
00099                         RDEBUG2(" WARNING: Cancelling replication to Realm %s, as the realm is local.", realm->name);
00100                         continue;
00101                 }
00102                 
00103                 home = home_server_ldb(realm->name, pool, request);
00104                 if (!home) {
00105                         RDEBUG2("ERROR: Failed to find live home server for realm %s",
00106                                 realm->name);
00107                         continue;
00108                 }
00109                 
00110                 if (!packet) {
00111                         packet = rad_alloc(1);
00112                         if (!packet) return RLM_MODULE_FAIL;
00113                         packet->sockfd = -1;
00114                         packet->code = request->packet->code;
00115                         packet->id = fr_rand() & 0xff;
00116 
00117                         packet->sockfd = fr_socket(&home->src_ipaddr, 0);
00118                         if (packet->sockfd < 0) {
00119                                 RDEBUG("ERROR: Failed opening socket: %s", fr_strerror());
00120                                 cleanup(packet);
00121                                 return RLM_MODULE_FAIL;
00122                         }
00123 
00124                         packet->vps = paircopy(request->packet->vps);
00125                         if (!packet->vps) {
00126                                 RDEBUG("ERROR: Out of memory!");
00127                                 cleanup(packet);
00128                                 return RLM_MODULE_FAIL;
00129                         }
00130 
00131                         /*
00132                          *      For CHAP, create the CHAP-Challenge if
00133                          *      it doesn't exist.
00134                          */
00135                         if ((request->packet->code == PW_AUTHENTICATION_REQUEST) &&
00136                             (pairfind(request->packet->vps, PW_CHAP_PASSWORD, 0) != NULL) &&
00137                             (pairfind(request->packet->vps, PW_CHAP_CHALLENGE, 0) == NULL)) {
00138                                 vp = radius_paircreate(request, &packet->vps,
00139                                                        PW_CHAP_CHALLENGE, 0,
00140                                                        PW_TYPE_OCTETS);
00141                                 vp->length = AUTH_VECTOR_LEN;
00142                                 memcpy(vp->vp_strvalue, request->packet->vector,
00143                                        AUTH_VECTOR_LEN);
00144                         }
00145                 } else {
00146                         size_t i;
00147 
00148                         for (i = 0; i < sizeof(packet->vector); i++) {
00149                                 packet->vector[i] = fr_rand() & 0xff;
00150                         }
00151 
00152                         packet->id++;
00153                         free(packet->data);
00154                         packet->data = NULL;
00155                         packet->data_len = 0;
00156                 }
00157 
00158                 /*
00159                  *      (Re)-Write these.
00160                  */
00161                 packet->dst_ipaddr = home->ipaddr;
00162                 packet->dst_port = home->port;
00163                 memset(&packet->src_ipaddr, 0, sizeof(packet->src_ipaddr));
00164                 packet->src_port = 0;
00165                 
00166                 /*
00167                  *      Encode, sign and then send the packet.
00168                  */
00169                 RDEBUG("Replicating packet to Realm %s", realm->name);
00170                 if (rad_send(packet, NULL, home->secret) < 0) {
00171                         RDEBUG("ERROR: Failed replicating packet: %s",
00172                                fr_strerror());
00173                         cleanup(packet);
00174                         return RLM_MODULE_FAIL;
00175                 }
00176 
00177                 /*
00178                  *      We've sent it to at least one destination.
00179                  */
00180                 rcode = RLM_MODULE_OK;
00181         }
00182 
00183         cleanup(packet);
00184         return rcode;
00185 }
