00060 static void check_pair(VALUE_PAIR *check_item, VALUE_PAIR *reply_item,
00061                       int *pass, int *fail)
00062 {
00063         int compare;
00064 
00065         if (check_item->operator == T_OP_SET) return;
00066 
00067         compare = paircmp(check_item, reply_item);
00068         if (compare == 1) {
00069                 ++*(pass);
00070         } else {
00071                 ++*(fail);
00072         }
00073 
00074         return;
00075 }

00078 static int getattrsfile(const char *filename, PAIR_LIST **pair_list)
00079 {
00080         int rcode;
00081         PAIR_LIST *attrs = NULL;
00082         PAIR_LIST *entry;
00083         VALUE_PAIR *vp;
00084 
00085         rcode = pairlist_read(filename, &attrs, 1);
00086         if (rcode < 0) {
00087                 return -1;
00088         }
00089 
00090         /*
00091          * Walk through the 'attrs' file list.
00092          */
00093 
00094         entry = attrs;
00095         while (entry) {
00096 
00097                 entry->check = entry->reply;
00098                 entry->reply = NULL;
00099 
00100                 for (vp = entry->check; vp != NULL; vp = vp->next) {
00101 
00102                     /*
00103                      * If it's NOT a vendor attribute,
00104                      * and it's NOT a wire protocol
00105                      * and we ignore Fall-Through,
00106                      * then bitch about it, giving a good warning message.
00107                      */
00108                      if ((vp->vendor == 0) &&
00109                          (vp->attribute > 0xff) &&
00110                          (vp->attribute > 1000)) {
00111                         log_debug("[%s]:%d WARNING! Check item \"%s\"\n"
00112                                   "\tfound in filter list for realm \"%s\".\n",
00113                                   filename, entry->lineno, vp->name,
00114                                   entry->name);
00115                     }
00116                 }
00117 
00118                 entry = entry->next;
00119         }
00120 
00121         *pair_list = attrs;
00122         return 0;

00169  *      Common attr_filter checks
00170  */
00171 static int attr_filter_common(void *instance, REQUEST *request,
00172                               RADIUS_PACKET *packet)
00173 {
00174         struct attr_filter_instance *inst = instance;
00175         VALUE_PAIR      *vp;
00176         VALUE_PAIR      *output;
00177         VALUE_PAIR      **output_tail;
00178         VALUE_PAIR      *check_item;
00179         PAIR_LIST       *pl;
00180         int             found = 0;
00181         int             pass, fail = 0;
00182         char            *keyname = NULL;
00183         VALUE_PAIR      **input;
00184         char            buffer[256];
00185 
00186         if (!packet) return RLM_MODULE_NOOP;
00187 
00188         input = &(packet->vps);
00189 
00190         if (!inst->key) {
00191                 VALUE_PAIR      *namepair;
00192 
00193                 namepair = pairfind(request->packet->vps, PW_REALM, 0);
00194                 if (!namepair) {
00195                         return (RLM_MODULE_NOOP);
00196                 }
00197                 keyname = namepair->vp_strvalue;
00198         } else {
00199                 int len;
00200 
00201                 len = radius_xlat(buffer, sizeof(buffer), inst->key,
00202                                   request, NULL, NULL);
00203                 if (!len) {
00204                         return RLM_MODULE_NOOP;
00205                 }
00206                 keyname = buffer;
00207         }
00208 
00209         output = NULL;
00210         output_tail = &output;
00211 
00212         /*
00213          *      Find the attr_filter profile entry for the entry.
00214          */
00215         for (pl = inst->attrs; pl; pl = pl->next) {
00216                 int fall_through = 0;
00217                 int relax_filter = inst->relaxed;
00218 
00219                 /*
00220                  *  If the current entry is NOT a default,
00221                  *  AND the realm does NOT match the current entry,
00222                  *  then skip to the next entry.
00223                  */
00224                 if ((strcmp(pl->name, "DEFAULT") != 0) &&
00225                     (strcmp(keyname, pl->name) != 0))  {
00226                     continue;
00227                 }
00228 
00229                 RDEBUG2("Matched entry %s at line %d", pl->name,
00230                        pl->lineno);
00231                 found = 1;
00232 
00233                 for (check_item = pl->check;
00234                         check_item != NULL;
00235                         check_item = check_item->next) {
00236                         if ((check_item->attribute == PW_FALL_THROUGH) &&
00237                                 (check_item->vp_integer == 1)) {
00238                                 fall_through = 1;
00239                                 continue;
00240                         }
00241                         else if (check_item->attribute == PW_RELAX_FILTER) {
00242                                 relax_filter = check_item->vp_integer;
00243                                 continue;
00244                         }
00245 
00246                         /*
00247                          *    If it is a SET operator, add the attribute to
00248                          *    the output list without checking it.
00249                          */
00250                         if (check_item->operator == T_OP_SET ) {
00251                                 vp = paircopyvp(check_item);
00252                                 if (!vp) {
00253                                         pairfree(&output);
00254                                         return RLM_MODULE_FAIL;
00255                                 }
00256                                 pairxlatmove(request, output_tail, &vp);
00257                                 output_tail = &((*output_tail)->next);
00258                         }
00259                 }
00260 
00261                 /*
00262                  *      Iterate through the input items, comparing
00263                  *      each item to every rule, then moving it to the
00264                  *      output list only if it matches all rules
00265                  *      for that attribute.  IE, Idle-Timeout is moved
00266                  *      only if it matches all rules that describe an
00267                  *      Idle-Timeout.
00268                  */
00269                 for (vp = *input; vp != NULL; vp = vp->next ) {
00270                         /* reset the pass,fail vars for each reply item */
00271                         pass = fail = 0;
00272 
00273                         /*
00274                          *      reset the check_item pointer to
00275                          *      beginning of the list
00276                          */
00277                         for (check_item = pl->check;
00278                              check_item != NULL;
00279                              check_item = check_item->next) {
00280                                 /*
00281                                  *      Vendor-Specific is special, and
00282                                  *      matches any VSA if the comparison
00283                                  *      is always true.
00284                                  */
00285                                 if ((check_item->attribute == PW_VENDOR_SPECIFIC) &&
00286                                         (vp->vendor != 0) &&
00287                                         (check_item->operator == T_OP_CMP_TRUE)) {
00288                                         pass++;
00289                                         continue;
00290                                 }
00291 
00292                                 if (vp->attribute == check_item->attribute) {
00293                                         check_pair(check_item, vp,
00294                                                    &pass, &fail);
00295                                 }
00296                         }
00297 
00298                         /*  
00299                          *  Only move attribute if it passed all rules,
00300                          *  or if the config says we should copy unmatched
00301                          *  attributes ('relaxed' mode).
00302                          */
00303                         if (fail == 0 && (pass > 0 || relax_filter)) {
00304                                 if (!pass) {
00305                                         RDEBUG3("Attribute (%s) allowed by relaxed mode", vp->name);
00306                                 }
00307                                 *output_tail = paircopyvp(vp);
00308                                 if (!*output_tail) {
00309                                         pairfree(&output);
00310                                         return RLM_MODULE_FAIL;
00311                                 }
00312                                 output_tail = &((*output_tail)->next);
00313                         }
00314                 }
00315 
00316                 /* If we shouldn't fall through, break */
00317                 if (!fall_through)
00318                         break;
00319         }
00320 
00321         /*
00322          *      No entry matched.  We didn't do anything.
00323          */
00324         if (!found) {
00325                 rad_assert(output == NULL);
00326                 return RLM_MODULE_NOOP;
00327         }
00328 
00329         pairfree(input);
00330         *input = output;
00331 
00332         if (request->packet->code == PW_AUTHENTICATION_REQUEST) {
00333                 request->username = pairfind(request->packet->vps,
00334                                              PW_STRIPPED_USER_NAME, 0);
00335                 if (!request->username) 
00336                         request->username = pairfind(request->packet->vps,
00337                                                      PW_USER_NAME, 0);
00338                 request->password = pairfind(request->packet->vps,
00339                                              PW_USER_PASSWORD, 0);
00340         }
00341 
00342         return RLM_MODULE_UPDATED;
00343 }
