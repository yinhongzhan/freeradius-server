00045 /*
00046  *      ntpwdhash converts Unicode password to 16-byte NT hash
00047  *      with MD4
00048  */
00049 void mschap_ntpwdhash (uint8_t *szHash, const char *szPassword)
00050 {
00051         char szUnicodePass[513];
00052         int nPasswordLen;
00053         int i;
00054 
00055         /*
00056          *      NT passwords are unicode.  Convert plain text password
00057          *      to unicode by inserting a zero every other byte
00058          */
00059         nPasswordLen = strlen(szPassword);
00060         for (i = 0; i < nPasswordLen; i++) {
00061                 szUnicodePass[i << 1] = szPassword[i];
00062                 szUnicodePass[(i << 1) + 1] = 0;
00063         }
00064 
00065         /* Encrypt Unicode password to a 16-byte MD4 hash */
00066         fr_md4_calc(szHash, (uint8_t *) szUnicodePass, (nPasswordLen<<1) );
00067 }

00071  *      challenge_hash() is used by mschap2() and auth_response()
00072  *      implements RFC2759 ChallengeHash()
00073  *      generates 64 bit challenge
00074  */
00075 void mschap_challenge_hash( const uint8_t *peer_challenge,
00076                             const uint8_t *auth_challenge,
00077                             const char *user_name, uint8_t *challenge )
00078 {
00079         fr_SHA1_CTX Context;
00080         uint8_t hash[20];
00081 
00082         fr_SHA1Init(&Context);
00083         fr_SHA1Update(&Context, peer_challenge, 16);
00084         fr_SHA1Update(&Context, auth_challenge, 16);
00085         fr_SHA1Update(&Context, (const uint8_t *) user_name,
00086                       strlen(user_name));
00087         fr_SHA1Final(hash, &Context);
00088         memcpy(challenge, hash, 8);
