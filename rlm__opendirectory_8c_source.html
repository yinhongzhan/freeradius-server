00272 static int od_authenticate(UNUSED void *instance, REQUEST *request)
00273 {
00274         char *name, *passwd;
00275         int             ret;
00276         long odResult = eDSAuthFailed;
00277         
00278         /*
00279          *      We can only authenticate user requests which HAVE
00280          *      a User-Name attribute.
00281          */
00282         if (!request->username) {
00283                 RDEBUG("ERROR: You set 'Auth-Type = OpenDirectory' for a request that does not contain a User-Name attribute!");
00284                 return RLM_MODULE_INVALID;
00285         }
00286 
00287         /*
00288          *      Can't do OpenDirectory if there's no password.
00289          */
00290         if (!request->password ||
00291             (request->password->attribute != PW_PASSWORD)) {
00292                 RDEBUG("ERROR: You set 'Auth-Type = OpenDirectory' for a request that does not contain a User-Password attribute!");
00293                 return RLM_MODULE_INVALID;
00294         }
00295         
00296         name = (char *)request->username->vp_strvalue;
00297         passwd = (char *)request->password->vp_strvalue;
00298         
00299         odResult = od_check_passwd(name, passwd);
00300         switch(odResult)
00301         {
00302                 case eDSNoErr:
00303                         ret = RLM_MODULE_OK;
00304                         break;
00305                         
00306                 case eDSAuthUnknownUser:
00307                 case eDSAuthInvalidUserName:
00308                 case eDSAuthNewPasswordRequired:
00309                 case eDSAuthPasswordExpired:
00310                 case eDSAuthAccountDisabled:
00311                 case eDSAuthAccountExpired:
00312                 case eDSAuthAccountInactive:
00313                 case eDSAuthInvalidLogonHours:
00314                 case eDSAuthInvalidComputer:
00315                         ret = RLM_MODULE_USERLOCK;
00316                         break;
00317                 
00318                 default:
00319                         ret = RLM_MODULE_REJECT;
00320                         break;
00321         }
00322         
00323         if (ret != RLM_MODULE_OK) {
00324                 RDEBUG("[%s]: Invalid password", name);
00325                 return ret;
00326         }
00327                 
00328         return RLM_MODULE_OK;
