00082 static int eap_handler_cmp(const void *a, const void *b)
00083 {
00084         int rcode;
00085         const EAP_HANDLER *one = a;
00086         const EAP_HANDLER *two = b;
00087 
00088         if (one->eap_id < two->eap_id) return -1;
00089         if (one->eap_id > two->eap_id) return +1;
00090 
00091         rcode = memcmp(one->state, two->state, sizeof(one->state));
00092         if (rcode != 0) return rcode;
00093 
00094         /*
00095          *      As of 2.1.8, we don't key off of source IP.  This
00096          *      a NAS to send packets load-balanced (or fail-over)
00097          *      across multiple intermediate proxies, and still have
00098          *      EAP work.
00099          */
00100         if (fr_ipaddr_cmp(&one->src_ipaddr, &two->src_ipaddr) != 0) {
00101                 DEBUG("WARNING: EAP packets are arriving from two different upstream servers.  Has there been a proxy fail-over?");
00102         }
00103 
00104         return 0;

00486 /*
00487  * EAP authorization DEPENDS on other rlm authorizations,
00488  * to check for user existance & get their configured values.
00489  * It Handles EAP-START Messages, User-Name initilization.
00490  */
00491 static int eap_authorize(void *instance, REQUEST *request)
00492 {
00493         rlm_eap_t       *inst;
00494         int             status;
00495         VALUE_PAIR      *vp;
00496 
00497         inst = (rlm_eap_t *)instance;
00498 
00499 #ifdef WITH_PROXY
00500         /*
00501          *      We don't do authorization again, once we've seen the
00502          *      proxy reply (or the proxied packet)
00503          */
00504         if (request->proxy != NULL)
00505                 return RLM_MODULE_NOOP;
00506 #endif
00507 
00508         /*
00509          *      For EAP_START, send Access-Challenge with EAP Identity
00510          *      request.  even when we have to proxy this request
00511          *
00512          *      RFC 2869, Section 2.3.1 notes that the "domain" of the
00513          *      user, (i.e. where to proxy him) comes from the EAP-Identity,
00514          *      so we CANNOT proxy the user, until we know his identity.
00515          *
00516          *      We therefore send an EAP Identity request.
00517          */
00518         status = eap_start(inst, request);
00519         switch(status) {
00520         case EAP_NOOP:
00521                 return RLM_MODULE_NOOP;
00522         case EAP_FAIL:
00523                 return RLM_MODULE_FAIL;
00524         case EAP_FOUND:
00525                 return RLM_MODULE_HANDLED;
00526         case EAP_OK:
00527         case EAP_NOTFOUND:
00528         default:
00529                 break;
00530         }
00531 
00532         /*
00533          *      RFC 2869, Section 2.3.1.  If a NAS sends an EAP-Identity,
00534          *      it MUST copy the identity into the User-Name attribute.
00535          *
00536          *      But we don't worry about that too much.  We depend on
00537          *      each EAP sub-module to look for handler->request->username,
00538          *      and to get excited if it doesn't appear.
00539          */
00540 
00541         vp = pairfind(request->config_items, PW_AUTH_TYPE, 0);
00542         if ((!vp) ||
00543             (vp->vp_integer != PW_AUTHTYPE_REJECT)) {
00544                 vp = pairmake("Auth-Type", inst->xlat_name, T_OP_EQ);
00545                 if (!vp) {
00546                         RDEBUG2("Failed to create Auth-Type %s: %s\n",
00547                                 inst->xlat_name, fr_strerror());
00548                         return RLM_MODULE_FAIL;
00549                 }
00550                 pairadd(&request->config_items, vp);
00551         } else {
00552                 RDEBUG2("WARNING: Auth-Type already set.  Not setting to EAP");
00553         }
00554 
00555         if (status == EAP_OK) return RLM_MODULE_OK;
00556 
00557         return RLM_MODULE_UPDATED;
00558 }
