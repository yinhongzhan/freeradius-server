00051 /* Initialize the pwattr array for supported password encodings. */
00052 void
00053 otp_pwe_init(void)
00054 {
00055   DICT_ATTR *da;
00056 
00057   /*
00058    * Setup known password types.  These are pairs.
00059    * NB: Increase pwattr array size when adding a type.
00060    *     It should be sized as (number of password types * 2)
00061    * NB: Array indices must match otp_pwe_t! (see otp.h)
00062    */
00063   (void) memset(pwattr, 0, sizeof(pwattr));
00064 
00065   /* PAP */
00066   if ((da = dict_attrbyname("User-Password")) != NULL) {
00067     pwattr[0] = da;
00068     pwattr[1] = da;
00069   }
00070 
00071   /* CHAP */
00072   if ((da = dict_attrbyname("CHAP-Challenge")) != NULL) {
00073     pwattr[2] = da;
00074     if ((da = dict_attrbyname("CHAP-Password")) != NULL)
00075       pwattr[3] = da;
00076     else
00077       pwattr[2] = NULL;
00078   }
00079 
00080 #if 0
00081   /* MS-CHAP (recommended not to use) */
00082   if ((da = dict_attrbyname("MS-CHAP-Challenge")) != NULL) {
00083     pwattr[4] = da;
00084     if ((da = dict_attrbyname("MS-CHAP-Response")) != NULL)
00085       pwattr[5] = da;
00086     else
00087       pwattr[4] = NULL;
00088   }
00089 #endif /* 0 */
00090 
00091   /* MS-CHAPv2 */
00092   if ((da = dict_attrbyname("MS-CHAP-Challenge")) != NULL) {
00093     pwattr[6] = da;
00094     if ((da = dict_attrbyname("MS-CHAP2-Response")) != NULL)
00095       pwattr[7] = da;
00096     else
00097       pwattr[6] = NULL;
00098   }
00099 }

00102 /*
00103  * Test for password presence in an Access-Request packet.
00104  * Returns 0 for "no supported password present", or the
00105  * password encoding type.
00106  */
00107 otp_pwe_t
00108 otp_pwe_present(const REQUEST *request)
00109 {
00110   unsigned i;
00111 
00112   for (i = 0; i < SIZEOF_PWATTR; i += 2) {
00113     if (pairfind(request->packet->vps, pwattr[i]->attr, pwattr[i]->vendor) &&
00114         pairfind(request->packet->vps, pwattr[i + 1]->attr,  pwattr[i + 1]->vendor)) {
00115       DEBUG("rlm_otp: %s: password attributes %s, %s", __func__,
00116              pwattr[i]->name, pwattr[i + 1]->name);
00117       return i + 1; /* Can't return 0 (indicates failure) */
00118     }
00119   }
00120 
00121   DEBUG("rlm_otp: %s: no password attributes present", __func__);
00122   return 0;
00123 }
