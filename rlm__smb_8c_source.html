00032 /*
00033  *      Define a structure for our module configuration.
00034  *
00035  *      These variables do not need to be in a structure, but it's

00045 /*
00046  *      A mapping of configuration file names to internal variables.
00047  *
00048  *      Note that the string is dynamically allocated, so it MUST
00049  *      be freed.  When the configuration file parse re-reads the string,
00050  *      it free's the old one, and strdup's the new one, placing the pointer

00052  *      buffer over-flows.

00055   { "server",  PW_TYPE_STRING_PTR, offsetof(rlm_smb_t,server), NULL,  NULL},

00057   { "domain",  PW_TYPE_STRING_PTR, offsetof(rlm_smb_t,domain), NULL,  NULL},

00099 /*
00100  *      Authenticate the user with the given password.
00101  */
00102 static int smb_authenticate(void *instance, REQUEST *request)
00103 {
00104         rlm_smb_t *data = (rlm_smb_t *) instance;
00105         int rcode;
00106 
00107         /*
00108          *      We can only authenticate user requests which HAVE
00109          *      a User-Name attribute.
00110          */
00111         if (!request->username) {
00112                 radlog(L_AUTH, "rlm_smb: Attribute \"User-Name\" is required for authentication.");
00113                 return RLM_MODULE_INVALID;
00114         }
00115 
00116         /*
00117          *      We can only authenticate user requests which HAVE
00118          *      a User-Password attribute.
00119          */
00120         if (!request->password) {
00121                 radlog(L_AUTH, "rlm_smb: Attribute \"User-Password\" is required for authentication.");
00122                 return RLM_MODULE_INVALID;
00123         }
00124 
00125         /*
00126          *  Ensure that we're being passed a plain-text password,
00127          *  and not anything else.
00128          */
00129         if (request->password->attribute != PW_USER_PASSWORD) {
00130                 radlog(L_AUTH, "rlm_smb: Attribute \"User-Password\" is required for authentication.  Cannot use \"%s\".", request->password->name);
00131                 return RLM_MODULE_INVALID;
00132         }
00133 
00134         /*
00135          *  Call the SMB magic to do the work.
00136          */
00137         rcode = Valid_User(request->username->vp_strvalue,
00138                            request->password->vp_strvalue,
00139                            data->server, data->backup, data->domain);
00140 
00141         switch (rcode) {
00142         case 0:                 /* success */
00143           return RLM_MODULE_OK;
00144           break;
00145 
00146         case 1:                 /* network failure */
00147         case 2:                 /* protocol failure */
00148           return RLM_MODULE_FAIL;
00149           break;
00150 
00151         case 3:                 /* invalid user name or password */
00152           return RLM_MODULE_REJECT;
00153         }
00154 
00155         /*
00156          *  Something weird happened.  Give up.
00157          */
00158         return RLM_MODULE_INVALID;
00159 }
