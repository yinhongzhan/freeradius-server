00291 /*
00292  *      Push an item onto the state.
00293  */
00294 static int policy_stack_push(policy_state_t *state, const policy_item_t *item)
00295 {
00296         rad_assert(state->depth >= 0);
00297 
00298         /*
00299          *      Asked to push nothing.  Don't push it.
00300          */
00301         if (!item) return 1;
00302 
00303         /*
00304          *      State is full.  Die.
00305          */
00306         if (state->depth >= POLICY_MAX_STACK) {
00307                 return 0;
00308         }
00309 
00310         /*
00311          *      Walk back up the stack, looking for previous ocurrances
00312          *      of this name.  If found, we have infinite recursion,
00313          *      which we stop dead in the water!
00314          *
00315          *      This isn't strictly necessary right now, as we look up
00316          *      policies by name when they're first referenced.  This
00317          *      means that ALL references are backwards (to the start
00318          *      of the file), which means that there are no circular
00319          *      references.
00320          */
00321         if (item->type == POLICY_TYPE_NAMED_POLICY) {
00322                 int i;
00323 
00324                 for (i = 0; i < state->depth; i++) {
00325                         /*
00326                          *      Check for circular references, by seeing
00327                          *      if the function is already on the stack.
00328                          *
00329                          *      Hmmm... do we want to do this for any type?
00330                          */
00331                         if (state->stack[i] == item) {
00332                                 debug_evaluate("Circular call to policy %s\n",
00333                                                ((const policy_named_t *) item)->name);
00334                                 return 0;
00335                         }
00336                 }
00337         }
00338 
00339         debug_evaluate("push %d %p\n", state->depth, item);

00341         state->stack[state->depth] = item;

00351 static int policy_stack_pop(policy_state_t *state, const policy_item_t **item)

00355

00380                 debug_evaluate("pop %d %p\n", state->depth, *item);
