00793  *      only paircopy() those attributes that we're really going to

00796 void radius_pairmove(REQUEST *request, VALUE_PAIR **to, VALUE_PAIR *from)
00797 {
00798         int i, j, count, from_count, to_count, tailto;
00799         VALUE_PAIR *vp, *next, **last;
00800         VALUE_PAIR **from_list, **to_list;
00801         int *edited = NULL;
00802         REQUEST *fixup = NULL;
00803 
00804         /*
00805          *      Set up arrays for editing, to remove some of the
00806          *      O(N^2) dependencies.  This also makes it easier to
00807          *      insert and remove attributes.
00808          *
00809          *      It also means that the operators apply ONLY to the
00810          *      attributes in the original list.  With the previous
00811          *      implementation of pairmove(), adding two attributes
00812          *      via "+=" and then "=" would mean that the second one
00813          *      wasn't added, because of the existence of the first
00814          *      one in the "to" list.  This implementation doesn't
00815          *      have that bug.
00816          *
00817          *      Also, the previous implementation did NOT implement
00818          *      "-=" correctly.  If two of the same attributes existed
00819          *      in the "to" list, and you tried to subtract something
00820          *      matching the *second* value, then the pairdelete()
00821          *      function was called, and the *all* attributes of that
00822          *      number were deleted.  With this implementation, only
00823          *      the matching attributes are deleted.
00824          */
00825         count = 0;
00826         for (vp = from; vp != NULL; vp = vp->next) count++;
00827         from_list = rad_malloc(sizeof(*from_list) * count);
00828 
00829         for (vp = *to; vp != NULL; vp = vp->next) count++;
00830         to_list = rad_malloc(sizeof(*to_list) * count);
00831 
00832         /*
00833          *      Move the lists to the arrays, and break the list
00834          *      chains.
00835          */
00836         from_count = 0;
00837         for (vp = from; vp != NULL; vp = next) {
00838                 next = vp->next;
00839                 from_list[from_count++] = vp;
00840                 vp->next = NULL;
00841         }
00842 
00843         to_count = 0;
00844         for (vp = *to; vp != NULL; vp = next) {
00845                 next = vp->next;
00846                 to_list[to_count++] = vp;
00847                 vp->next = NULL;
00848         }
00849         tailto = to_count;
00850         edited = rad_malloc(sizeof(*edited) * to_count);
00851         memset(edited, 0, sizeof(*edited) * to_count);
00852 
00853         RDEBUG4("::: FROM %d TO %d MAX %d", from_count, to_count, count);
00854 
00855         /*
00856          *      Now that we have the lists initialized, start working
00857          *      over them.
00858          */
00859         for (i = 0; i < from_count; i++) {
00860                 int found;
00861 
00862                 RDEBUG4("::: Examining %s", from_list[i]->name);
00863 
00864                 /*
00865                  *      Attribute should be appended, OR the "to" list
00866                  *      is empty, and we're supposed to replace or
00867                  *      "add if not existing".
00868                  */
00869                 if (from_list[i]->operator == T_OP_ADD) goto append;
00870 
00871                 found = FALSE;
00872                 for (j = 0; j < to_count; j++) {
00873                         if (edited[j] || !to_list[j] || !from_list[i]) continue;
00874 
00875                         /*
00876                          *      Attributes aren't the same, skip them.
00877                          */
00878                         if ((from_list[i]->attribute != to_list[j]->attribute) ||
00879                             (from_list[i]->vendor != to_list[j]->vendor)) {
00880                                 continue;
00881                         }
00882 
00883                         /*
00884                          *      We don't use a "switch" statement here
00885                          *      because we want to break out of the
00886                          *      "for" loop over 'j' in most cases.
00887                          */
00888 
00889                         /*
00890                          *      Over-write the FIRST instance of the
00891                          *      matching attribute name.  We free the
00892                          *      one in the "to" list, and move over
00893                          *      the one in the "from" list.
00894                          */
00895                         if (from_list[i]->operator == T_OP_SET) {
00896                                 RDEBUG4("::: OVERWRITING %s FROM %d TO %d",
00897                                        to_list[j]->name, i, j);
00898                                 pairfree(&to_list[j]);
00899                                 to_list[j] = from_list[i];
00900                                 from_list[i] = NULL;
00901                                 edited[j] = TRUE;
00902                                 break;
00903                         }
00904 
00905                         /*
00906                          *      Add the attribute only if it does not
00907                          *      exist... but it exists, so we stop
00908                          *      looking.
00909                          */
00910                         if (from_list[i]->operator == T_OP_EQ) {
00911                                 found = TRUE;
00912                                 break;
00913                         }
00914 
00915                         /*
00916                          *      Delete every attribute, independent
00917                          *      of its value.
00918                          */
00919                         if (from_list[i]->operator == T_OP_CMP_FALSE) {
00920                                 goto delete;
00921                         }
00922 
00923                         /*
00924                          *      Delete all matching attributes from
00925                          *      "to"
00926                          */
00927                         if ((from_list[i]->operator == T_OP_SUB) ||
00928                             (from_list[i]->operator == T_OP_CMP_EQ) ||
00929                             (from_list[i]->operator == T_OP_LE) ||
00930                             (from_list[i]->operator == T_OP_GE)) {
00931                                 int rcode;
00932                                 int old_op = from_list[i]->operator;
00933 
00934                                 /*
00935                                  *      Check for equality.
00936                                  */
00937                                 from_list[i]->operator = T_OP_CMP_EQ;
00938 
00939                                 /*
00940                                  *      If equal, delete the one in
00941                                  *      the "to" list.
00942                                  */
00943                                 rcode = radius_compare_vps(NULL, from_list[i],
00944                                                            to_list[j]);
00945                                 /*
00946                                  *      We may want to do more
00947                                  *      subtractions, so we re-set the
00948                                  *      operator back to it's original
00949                                  *      value.
00950                                  */
00951                                 from_list[i]->operator = old_op;

00953                                 switch (old_op) {
00954                                 case T_OP_CMP_EQ:

00956                                         break;

00958                                 case T_OP_SUB:

00965                                         }

00968                                         /*
00969                                          *      Enforce <=.  If it's
00970                                          *      >, replace it.
00971                                          */
00972                                 case T_OP_LE:
00973                                         if (rcode > 0) {
00974                                                 RDEBUG4("::: REPLACING %s FROM %d TO %d",

00977                                                 to_list[j] = from_list[i];
00978                                                 from_list[i] = NULL;
00979                                                 edited[j] = TRUE;
00980                                         }
00981                                         break;
00982 
00983                                 case T_OP_GE:
00984                                         if (rcode < 0) {
00985                                                 RDEBUG4("::: REPLACING %s FROM %d TO %d",
00986                                                        from_list[i]->name, i, j);
00987                                                 pairfree(&to_list[j]);
00988                                                 to_list[j] = from_list[i];
00989                                                 from_list[i] = NULL;
00990                                                 edited[j] = TRUE;
00991                                         }
00992                                         break;
00993                                 }
00994 
00995                                 continue;
00996                         }
00997 
00998                         rad_assert(0 == 1); /* panic! */
00999                 }
01000 
01001                 /*
01002                  *      We were asked to add it if it didn't exist,
01003                  *      and it doesn't exist.  Move it over to the
01004                  *      tail of the "to" list, UNLESS it was already
01005                  *      moved by another operator.
01006                  */
01007                 if (!found && from_list[i]) {
01008                         if ((from_list[i]->operator == T_OP_EQ) ||
01009                             (from_list[i]->operator == T_OP_LE) ||
01010                             (from_list[i]->operator == T_OP_GE) ||
01011                             (from_list[i]->operator == T_OP_SET)) {
01012                         append:
01013                                 RDEBUG4("::: APPENDING %s FROM %d TO %d",
01014                                        from_list[i]->name, i, tailto);
01015                                 to_list[tailto++] = from_list[i];
01016                                 from_list[i] = NULL;
01017                         }
01018                 }
01019         }
01020 
01021         /*
01022          *      Delete attributes in the "from" list.
01023          */
01024         for (i = 0; i < from_count; i++) {
01025                 if (!from_list[i]) continue;
01026 
01027                 pairfree(&from_list[i]);
01028         }
01029         free(from_list);
