00036 /*
00037  * Generate the State attribute, suitable for passing to pairmake().
00038  * 'challenge' must be a null terminated string, and be sized at least
00039  * as large as indicated in the function definition.
00040  *
00041  * Returns 0 on success, non-zero otherwise.  For successful returns,
00042  * 'rad_state' (suitable for passing to pairmake()) and 'raw_state',
00043  * if non-NULL, will be filled in.
00044  *
00045  * In the simplest implementation, we would just use the challenge as state.
00046  * Unfortunately, the RADIUS secret protects only the User-Password
00047  * attribute; an attacker that can remove packets from the wire and insert
00048  * new ones can simply insert a replayed state without having to know
00049  * the secret.  If not for an attacker that can remove packets from the
00050  * network, I believe trivial state to be secure.
00051  *
00052  * So, we have to make up for that deficiency by signing our state with
00053  * data unique to this specific request.  A NAS would use the Request
00054  * Authenticator, but we don't know what that will be when the State is
00055  * returned to us, so we'll use the time.  So our replay prevention
00056  * is limited to a time interval (inst->challenge_delay).  We could keep
00057  * track of all challenges issued over that time interval for
00058  * better protection.
00059  *
00060  * Our state, then, is
00061  *   (challenge + flags + time + hmac(challenge + resync + time, key)),
00062  * where '+' denotes concatentation, 'challenge' is ... the challenge,
00063  * 'flags' is a 32-bit value that can be used to record additional info,
00064  * 'time' is the 32-bit time (LSB if time_t is 64 bits), and 'key' is a
00065  * random key, generated in otp_instantiate().  'flags' and 'time' are
00066  * in network byte order.
00067  *
00068  * As the signing key is unique to each server, only the server which
00069  * generates a challenge can verify it; this should be OK if your NAS's
00070  * load balance across RADIUS servers using a "first available" algorithm.
00071  * If your NAS's round-robin and don't "stick" to the same server if they
00072  * see a State attribute (ugh), you could use the RADIUS secret instead,
00073  * but read RFC 2104 first, and make very sure you really want to do this.
00074  *
00075  * Since only the "same server" can verify State, 'flags' and 'time' doesn't
00076  * really need to be in network byte order, but we do it anyway.
00077  *
00078  * The State attribute is an octet string, however some versions of Cisco
00079  * IOS and Catalyst OS (at least IOS 12.1(26)E4 and CatOS 7.6.12) treat it
00080  * as an ASCII string (they only return data up to the first NUL byte).
00081  * So we must handle state as an ASCII string (0x00 -> 0x3030).
00082  */
00083 int
00084 otp_gen_state(char rad_state[OTP_MAX_RADSTATE_LEN],
00085               unsigned char raw_state[OTP_MAX_RADSTATE_LEN],
00086               const unsigned char challenge[OTP_MAX_CHALLENGE_LEN],
00087               size_t clen,
00088               int32_t flags, int32_t when, const unsigned char key[16])
00089 {
00090   HMAC_CTX hmac_ctx;
00091   unsigned char hmac[MD5_DIGEST_LENGTH];
00092   char *p;
00093   char state[OTP_MAX_RADSTATE_LEN];
00094 
00095   /*
00096    * Generate the hmac.  We already have a dependency on openssl for
00097    * DES, so we'll use it's hmac functionality also -- saves us from
00098    * having to collect the data to be signed into one contiguous piece.
00099    */
00100   HMAC_Init(&hmac_ctx, key, sizeof(key[0] * 16), EVP_md5());
00101   HMAC_Update(&hmac_ctx, challenge, clen);
00102   HMAC_Update(&hmac_ctx, (unsigned char *) &flags, 4);
00103   HMAC_Update(&hmac_ctx, (unsigned char *) &when, 4);
00104   HMAC_Final(&hmac_ctx, hmac, NULL);
00105   HMAC_cleanup(&hmac_ctx);
00106 
00107   /*
00108    * Generate the state.  Note that it is in ASCII.  The challenge
00109    * value doesn't have to be ASCII encoded, as it is already
00110    * ASCII, but we do it anyway, for consistency.
00111    */
00112 #if 0
00113   /*
00114    * We used to malloc() state (here and in callers).  We leave this
00115    * here to show how OTP_MAX_RADSTATE_LEN is composed.  Note that
00116    * it has to be double all the values below to account for an
00117    * extra ASCII expansion (see Cisco notes, below).
00118    */
00119   state = rad_malloc(clen * 2 +                 /* challenge */
00120                      8 +                        /* flags     */
00121                      8 +                        /* time      */
00122                      sizeof(hmac) * 2 +         /* hmac      */
00123                      1);                        /* '\0'      */
00124 #endif
00125   p = state;
00126   /* Add the challenge. */
00127   otp_x2a(challenge, clen, p);
00128   p += clen * 2;
00129   /* Add the flags and time. */
00130   otp_x2a((unsigned char *) &flags, 4, p);
00131   p += 8;
00132   otp_x2a((unsigned char *) &when, 4, p);
00133   p += 8;
00134   /* Add the hmac. */
00135   otp_x2a(hmac, 16, p);
00136 
00137   /*
00138    * Expand state (already ASCII) into ASCII again (0x31 -> 0x3331).
00139    * pairmake() forces us to do this (it will reduce it back to binary),
00140    * and to include a leading "0x".
00141    */
00142   if (rad_state) {
00143     (void) sprintf(rad_state, "0x");
00144     p = rad_state + 2;
00145     otp_x2a(state, strlen(state), p);
00146   }
00147 
00148   if (raw_state)
00149     (void) memcpy(raw_state, state, sizeof(state));
00150 
00151   return 0;
00152 }
