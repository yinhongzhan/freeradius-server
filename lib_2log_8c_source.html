00083 {
00084         va_list ap;

00089         pthread_once(&fr_strerror_once, fr_strerror_make_key);

00091         buffer = pthread_getspecific(fr_strerror_key);
00092         if (!buffer) {
00093                 buffer = malloc(FR_STRERROR_BUFSIZE);
00094                 if (!buffer) return; /* panic and die! */
00095 
00096                 pthread_setspecific(fr_strerror_key, buffer);
