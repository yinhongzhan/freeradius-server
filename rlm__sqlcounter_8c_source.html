00039 /*
00040  *      Note: When your counter spans more than 1 period (ie 3 months
00041  *      or 2 weeks), this module probably does NOT do what you want! It
00042  *      calculates the range of dates to count across by first calculating
00043  *      the End of the Current period and then subtracting the number of
00044  *      periods you specify from that to determine the beginning of the
00045  *      range.
00046  *
00047  *      For example, if you specify a 3 month counter and today is June 15th,
00048  *      the end of the current period is June 30. Subtracting 3 months from
00049  *      that gives April 1st. So, the counter will sum radacct entries from
00050  *      April 1st to June 30. Then, next month, it will sum entries from
00051  *      May 1st to July 31st.
00052  *
00053  *      To fix this behavior, we need to add some way of storing the Next
00054  *      Reset Time.
00055  */
00056

00100 {
00101         int ret = 0;
00102         size_t len;

00104         char last = '\0';

00109         len = strftime(sCurrentTime, sizeof(sCurrentTime), "%Y-%m-%d %H:%M:%S", tm);
00110         if (len == 0) *sCurrentTime = '\0';
00111         tm->tm_sec = tm->tm_min = 0;
00112 
00113         if (data->reset == NULL)

00126                 /*

00139                 /*

00142                 tm->tm_hour = 0;
00143                 tm->tm_mday += (7 - tm->tm_wday) +(7*(num-1));

00146                 tm->tm_hour = 0;
00147                 tm->tm_mday = 1;
00148                 tm->tm_mon += num;
00149                 data->reset_time = mktime(tm);
00150         } else if (strcmp(data->reset, "never") == 0) {
00151                 data->reset_time = 0;
00152         } else {
00153                 radlog(L_ERR, "rlm_sqlcounter: Unknown reset timer \"%s\"",
00154                         data->reset);
00155                 return -1;
00156         }
00157 
00158         len = strftime(sNextTime, sizeof(sNextTime),"%Y-%m-%d %H:%M:%S",tm);
00159         if (len == 0) *sNextTime = '\0';

00248 static int sqlcounter_expand(char *out, int outlen, const char *fmt, void *instance)
00249 {
00250         rlm_sqlcounter_t *data = (rlm_sqlcounter_t *) instance;
00251         int c,freespace;
00252         const char *p;
00253         char *q;
00254         char tmpdt[40]; /* For temporary storing of dates */
00255 
00256         q = out;
00257         for (p = fmt; *p ; p++) {
00258         /* Calculate freespace in output */
00259         freespace = outlen - (q - out);
00260                 if (freespace <= 1)
00261                         break;
00262                 c = *p;
00263                 if ((c != '%') && (c != '\\')) {
00264                         *q++ = *p;
00265                         continue;
00266                 }
00267                 if (*++p == '\0') break;
00268                 if (c == '\\') switch(*p) {
00269                         case '\\':
00270                                 *q++ = *p;
00271                                 break;
00272                         case 't':
00273                                 *q++ = '\t';
00274                                 break;
00275                         case 'n':
00276                                 *q++ = '\n';
00277                                 break;
00278                         default:
00279                                 *q++ = c;
00280                                 *q++ = *p;
00281                                 break;
00282 
00283                 } else if (c == '%') switch(*p) {
00284 
00285                         case '%':
00286                                 *q++ = *p;
00287                                 break;
00288                         case 'b': /* last_reset */
00289                                 snprintf(tmpdt, sizeof(tmpdt), "%lu", data->last_reset);
00290                                 strlcpy(q, tmpdt, freespace);
00291                                 q += strlen(q);
00292                                 break;
00293                         case 'e': /* reset_time */
00294                                 snprintf(tmpdt, sizeof(tmpdt), "%lu", data->reset_time);
00295                                 strlcpy(q, tmpdt, freespace);
00296                                 q += strlen(q);
00297                                 break;
00298                         case 'k': /* Key Name */
00299                                 DEBUG2("WARNING: Please replace '%%k' with '${key}'");
00300                                 strlcpy(q, data->key_name, freespace);
00301                                 q += strlen(q);
00302                                 break;
00303                         default:
00304                                 *q++ = '%';
00305                                 *q++ = *p;
00306                                 break;
00307                 }
00308         }
00309         *q = '\0';
00310 
00311         DEBUG2("sqlcounter_expand:  '%s'", out);
00312 
00313         return strlen(out);
00314 }
