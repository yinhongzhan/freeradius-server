00947 pair_lists_t radius_list_name(const char **name, pair_lists_t unknown)
00948 {
00949         const char *p = *name;
00950         const char *q;
00951         
00952         /* This should never be a NULL pointer or zero length string */
00953         rad_assert(name && *name);
00954 
00955         /*
00956          *      We couldn't determine the list if:
00957          *      
00958          *      A colon delimiter was found, but the next char was a 
00959          *      number, indicating a tag, not a list qualifier.
00960          *
00961          *      No colon was found and the first char was upper case 
00962          *      indicating an attribute.
00963          *
00964          *      This allows the function to be used to resolve list names too.
00965          */
00966         q = strchr(p, ':');
00967         if (((q && (q[1] >= '0') && (q[1] <= '9'))) ||
00968             (!q && isupper((int) *p))) {
00969                 return unknown;
00970         }
00971         
00972         if (q) {
00973                 *name = (q + 1);        /* Consume the list and delimiter */
00974         } else {
00975                 q = (p + strlen(p));    /* Consume the entire string */
00976                 *name = q;
00977         }
00978         
00979         return fr_substr2int(pair_lists, p, PAIR_LIST_UNKNOWN, (q - p));
00980 }
