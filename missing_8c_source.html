00096 /*
00097  *      Get next token from string *stringp, where tokens are
00098  *      possibly-empty strings separated by characters from delim.
00099  *
00100  *      Writes NULs into the string at *stringp to end tokens.
00101  *      delim need not remain constant from call to call.  On
00102  *      return, *stringp points past the last NUL written (if there
00103  *      might be further tokens), or is NULL (if there are
00104  *      definitely no more tokens).
00105  *
00106  *      If *stringp is NULL, strsep returns NULL.
00107  */
00108 char *
00109 strsep(char **stringp, const char *delim)
00110 {
00111         char *s;
00112         const char *spanp;
00113         int c, sc;
00114         char *tok;
00115 
00116         if ((s = *stringp) == NULL)
00117                 return (NULL);
00118 
00119         for (tok = s;;) {
00120                 c = *s++;
00121                 spanp = delim;
00122                 do {
00123                         if ((sc = *spanp++) == c) {
00124                                 if (c == 0)
00125                                         s = NULL;
00126                                 else
00127                                         s[-1] = 0;
00128                                 *stringp = s;
00129                                 return (tok);
00130                         }
00131                 } while (sc != 0);
00132         }
00133 
00134         return NULL;            /* NOTREACHED, but the compiler complains */

00209 static uint64_t filetime_to_unix_epoch (const FILETIME *ft)
00210 {
00211         uint64_t res = (uint64_t) ft->dwHighDateTime << 32;
00212 
00213         res |= ft->dwLowDateTime;
00214         res /= 10;                   /* from 100 nano-sec periods to usec */
00215         res -= DELTA_EPOCH_IN_USEC;  /* from Win epoch to Unix epoch */
00216         return (res);
00217 }
