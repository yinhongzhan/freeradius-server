00213                 return RLM_MODULE_NOTFOUND;

00216 #endif /* OSFC2 */

00283         if (spwd && spwd->sp_expire > 0 &&

00300

00381 /*
00382  *      UUencode 4 bits base64. We use this to turn a 4 byte field
00383  *      (an IP address) into 6 bytes of ASCII. This is used for the
00384  *      wtmp file if we didn't find a short name in the naslist file.
00385  */
00386 static char *uue(void *in)
00387 {
00388         int i;
00389         static unsigned char res[7];
00390         unsigned char *data = (unsigned char *)in;
00391 
00392         res[0] = ENC( data[0] >> 2 );
00393         res[1] = ENC( ((data[0] << 4) & 060) + ((data[1] >> 4) & 017) );
00394         res[2] = ENC( ((data[1] << 2) & 074) + ((data[2] >> 6) & 03) );
00395         res[3] = ENC( data[2] & 077 );
00396 
00397         res[4] = ENC( data[3] >> 2 );
00398         res[5] = ENC( (data[3] << 4) & 060 );
00399         res[6] = 0;
00400 
00401         for(i = 0; i < 6; i++) {
00402                 if (res[i] == ' ') res[i] = '`';
00403                 if (res[i] < 32 || res[i] > 127)
00404                         printf("uue: protocol error ?!\n");
00405         }
00406         return (char *)res;
00407 }

00416         FILE            *fp;

00419         char            buf[64];

00422         int             status = -1;

00425 #ifdef USER_PROCESS
00426         int             protocol = -1;
00427 #endif
00428         int             nas_port = 0;

00431 
00432         /*
00433          *      No radwtmp.  Don't do anything.
00434          */
00435         if (!inst->radwtmp) {
00436                 RDEBUG2("No radwtmp file configured.  Ignoring accounting request.");
00437                 return RLM_MODULE_NOOP;
00438         }
00439 
00440         if (request->packet->src_ipaddr.af != AF_INET) {
00441                 RDEBUG2("IPv6 is not supported!");
00442                 return RLM_MODULE_NOOP;
00443         }
00444 
00445         /*
00446          *      Which type is this.
00447          */

00449                 RDEBUG("no Accounting-Status-Type attribute in request.");
00450                 return RLM_MODULE_NOOP;

00453 
00454         /*
00455          *      FIXME: handle PW_STATUS_ALIVE like 1.5.4.3 did.
00456          */
00457         if (status != PW_STATUS_START &&
00458             status != PW_STATUS_STOP)
00459                 return RLM_MODULE_NOOP;
00460 
00461         /*
00462          *      We're only interested in accounting messages
00463          *      with a username in it.
00464          */
00465         if (pairfind(request->packet->vps, PW_USER_NAME, 0) == NULL)
00466                 return RLM_MODULE_NOOP;
00467 
00468         t = request->timestamp;
00469         memset(&ut, 0, sizeof(ut));
00470 
00471         /*
00472          *      First, find the interesting attributes.

00475                 switch (vp->attribute) {
00476                         case PW_USER_NAME:
00477                                 if (vp->length >= sizeof(ut.ut_name)) {
00478                                         memcpy(ut.ut_name, (char *)vp->vp_strvalue, sizeof(ut.ut_name));
00479                                 } else {
00480                                         strlcpy(ut.ut_name, (char *)vp->vp_strvalue, sizeof(ut.ut_name));

00484                         case PW_FRAMED_IP_ADDRESS:
00485                                 framed_address = vp->vp_ipaddr;

00487 #ifdef USER_PROCESS

00489                                 protocol = vp->vp_integer;
00490                                 break;

00493                                 nas_address = vp->vp_ipaddr;
00494                                 break;

00497                                 port_seen = 1;
00498                                 break;
