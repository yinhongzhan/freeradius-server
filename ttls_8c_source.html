00043 /*
00044  *      Verify that the diameter packet is valid.
00045  */
00046 static int diameter_verify(REQUEST *request,
00047                            const uint8_t *data, unsigned int data_len)
00048 {
00049         uint32_t attr;
00050         uint32_t length;
00051         unsigned int hdr_len;
00052         unsigned int remaining = data_len;
00053 
00054         while (remaining > 0) {
00055                 hdr_len = 12;
00056 
00057                 if (remaining < hdr_len) {
00058                   RDEBUG2(" Diameter attribute is too small (%u) to contain a Diameter header", remaining);
00059                         return 0;
00060                 }
00061 
00062                 memcpy(&attr, data, sizeof(attr));
00063                 attr = ntohl(attr);
00064                 memcpy(&length, data + 4, sizeof(length));
00065                 length = ntohl(length);
00066 
00067                 if ((data[4] & 0x80) != 0) {
00068                         if (remaining < 16) {
00069                                 RDEBUG2(" Diameter attribute is too small to contain a Diameter header with Vendor-Id");
00070                                 return 0;
00071                         }
00072 
00073                         hdr_len = 16;
00074                 }
00075 
00076                 /*
00077                  *      Get the length.  If it's too big, die.
00078                  */
00079                 length &= 0x00ffffff;
00080 
00081                 /*
00082                  *      Too short or too long is bad.
00083                  */
00084                 if (length <= (hdr_len - 4)) {
00085                         RDEBUG2("Tunneled attribute %u is too short (%u < %u) to contain anything useful.", attr, length, hdr_len);
00086                         return 0;
00087                 }
00088 
00089                 if (length > remaining) {
00090                         RDEBUG2("Tunneled attribute %u is longer than room remaining in the packet (%u > %u).", attr, length, remaining);
00091                         return 0;
00092                 }
00093 
00094                 /*
00095                  *      Check for broken implementations, which don't
00096                  *      pad the AVP to a 4-octet boundary.
00097                  */
00098                 if (remaining == length) break;
00099 
00100                 /*
00101                  *      The length does NOT include the padding, so
00102                  *      we've got to account for it here by rounding up
00103                  *      to the nearest 4-byte boundary.
00104                  */
00105                 length += 0x03;
00106                 length &= ~0x03;
00107 
00108                 /*
00109                  *      If the rest of the diameter packet is larger than
00110                  *      this attribute, continue.
00111                  *
00112                  *      Otherwise, if the attribute over-flows the end
00113                  *      of the packet, die.
00114                  */
00115                 if (remaining < length) {
00116                         RDEBUG2("ERROR! Diameter attribute overflows packet!");
00117                         return 0;
00118                 }
00119 
00120                 /*
00121                  *      remaining > length, continue.
00122                  */
00123                 remaining -= length;
00124                 data += length;
00125         }
00126 
00127         /*
00128          *      We got this far.  It looks OK.
00129          */
00130         return 1;
00131 }

00514

00759                  *      send tunneled reply data.  This is because
00760                  *      an Access-Challenge means that we MUST tunnel
00761                  *      a Reply-Message to the client.
00762                  */
00763         case PW_ACCESS_CHALLENGE:

00768                  *
00769                  *      Get rid of the old State, too.
00770                  */

00773 
00774                 /*
00775                  *      We should really be a bit smarter about this,
00776                  *      and move over only those attributes which
00777                  *      are relevant to the authentication request,
00778                  *      but that's a lot more work, and this "dumb"
00779                  *      method works in 99.9% of the situations.
00780                  */
00781                 vp = NULL;
00782                 pairmove2(&vp, &reply->vps, PW_EAP_MESSAGE, 0);
00783 
00784                 /*
00785                  *      There MUST be a Reply-Message in the challenge,
00786                  *      which we tunnel back to the client.
00787                  *
00788                  *      If there isn't one in the reply VP's, then
00789                  *      we MUST create one, with an empty string as
00790                  *      it's value.
00791                  */
00792                 pairmove2(&vp, &reply->vps, PW_REPLY_MESSAGE, 0);
00793 
00794                 /*
00795                  *      Handle the ACK, by tunneling any necessary reply
00796                  *      VP's back to the client.
00797                  */
00798                 if (vp) {
00799                         vp2diameter(request, tls_session, vp);
00800                         pairfree(&vp);
00801                 }
00802                 rcode = RLM_MODULE_HANDLED;
00803                 break;
00804 
00805         default:
00806                 RDEBUG("Unknown RADIUS packet type %d: rejecting tunneled user", reply->code);
00807                 rcode = RLM_MODULE_INVALID;
00808                 break;
00809         }
00810 
00811         return rcode;
