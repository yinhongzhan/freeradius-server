00414                        VALUE_PAIR *request_pairs, VALUE_PAIR **reply_pairs)

00434                                   request, NULL, NULL);

00440 
00441         if (!ht) return RLM_MODULE_NOOP;

00443         my_pl.name = name;

00446         default_pl = fr_hash_table_finddata(ht, &my_pl);
00447 
00448         /*
00449          *      Find the entry for the user.
00450          */

00452                 const PAIR_LIST *pl;
00453 
00454                 if (!default_pl && user_pl) {
00455                         pl = user_pl;
00456                         match = name;
00457                         user_pl = user_pl->next;

00459                 } else if (!user_pl && default_pl) {

00461                         match = "DEFAULT";
00462                         default_pl = default_pl->next;
00463 
00464                 } else if (user_pl->order < default_pl->order) {
00465                         pl = user_pl;
00466                         match = name;
00467                         user_pl = user_pl->next;
00468 
00469                 } else {
00470                         pl = default_pl;
00471                         match = "DEFAULT";
00472                         default_pl = default_pl->next;
00473                 }
00474

00476                         RDEBUG2("%s: Matched entry %s at line %d",
00477                                filename, match, pl->lineno);

00479                         check_tmp = paircopy(pl->check);
00480                         reply_tmp = paircopy(pl->reply);
