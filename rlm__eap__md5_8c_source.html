00039 static int md5_initiate(void *type_data, EAP_HANDLER *handler)
00040 {
00041         int             i;
00042         MD5_PACKET      *reply;
00043

00046         /*
00047          *      Allocate an EAP-MD5 packet.
00048          */
00049         reply = eapmd5_alloc();
00050         if (reply == NULL)  {
00051                 radlog(L_ERR, "rlm_eap_md5: out of memory");
00052                 return 0;

00054

00056          *      Fill it with data.
00057          */
00058         reply->code = PW_MD5_CHALLENGE;
00059         reply->length = 1 + MD5_CHALLENGE_LEN; /* one byte of value size */
00060         reply->value_size = MD5_CHALLENGE_LEN;
00061 
00062         /*
00063          *      Allocate user data.
00064          */
00065         reply->value = malloc(reply->value_size);
00066         if (reply->value == NULL) {
00067                 radlog(L_ERR, "rlm_eap_md5: out of memory");
00068                 eapmd5_free(&reply);
00069                 return 0;
00070         }
00071 
00072         /*
00073          *      Get a random challenge.
00074          */
00075         for (i = 0; i < reply->value_size; i++) {
00076                 reply->value[i] = fr_rand();
00077         }
00078         DEBUG2("rlm_eap_md5: Issuing Challenge");
00079 
00080         /*
00081          *      Keep track of the challenge.
00082          */
00083         handler->opaque = malloc(reply->value_size);
00084         rad_assert(handler->opaque != NULL);
00085         memcpy(handler->opaque, reply->value, reply->value_size);
00086         handler->free_opaque = free;
00087 
00088         /*
00089          *      Compose the EAP-MD5 packet out of the data structure,
00090          *      and free it.
00091          */
00092         eapmd5_compose(handler->eap_ds, reply);
00093 
00094         /*
00095          *      We don't need to authorize the user at this point.
00096          *
00097          *      We also don't need to keep the challenge, as it's
00098          *      stored in 'handler->eap_ds', which will be given back
00099          *      to us...
00100          */
00101         handler->stage = AUTHENTICATE;
00102 
00103         return 1;

00128         /*
00129          *      Extract the EAP-MD5 packet.

00132                 return 0;
00133 
00134         /*
00135          *      Create a reply, and initialize it.
00136          */
00137         reply = eapmd5_alloc();

00139                 eapmd5_free(&packet);
00140                 return 0;

00142         reply->id = handler->eap_ds->request->id;
00143         reply->length = 0;
00144 
00145         /*
00146          *      Verify the received packet against the previous packet
00147          *      (i.e. challenge) which we sent out.
00148          */
00149         if (eapmd5_verify(packet, password, handler->opaque)) {
00150                 reply->code = PW_MD5_SUCCESS;

00161         eapmd5_free(&packet);
00162         return 1;
