00300 /*
00301  *      Do authentication, by letting EAP-TLS do most of the work.
00302  */
00303 static int eappeap_authenticate(void *arg, EAP_HANDLER *handler)
00304 {
00305         int rcode;
00306         fr_tls_status_t status;
00307         rlm_eap_peap_t *inst = (rlm_eap_peap_t *) arg;
00308         tls_session_t *tls_session = (tls_session_t *) handler->opaque;
00309         peap_tunnel_t *peap = tls_session->opaque;
00310         REQUEST *request = handler->request;
00311 
00312         /*
00313          *      Session resumption requires the storage of data, so
00314          *      allocate it if it doesn't already exist.
00315          */
00316         if (!tls_session->opaque) {
00317                 peap = tls_session->opaque = peap_alloc(inst);
00318                 tls_session->free_opaque = peap_free;
00319         }
00320 
00321         status = eaptls_process(handler);
00322         RDEBUG2("eaptls_process returned %d\n", status);
00323         switch (status) {
00324                 /*
00325                  *      EAP-TLS handshake was successful, tell the
00326                  *      client to keep talking.
00327                  *
00328                  *      If this was EAP-TLS, we would just return
00329                  *      an EAP-TLS-Success packet here.
00330                  */
00331         case FR_TLS_SUCCESS:
00332                 RDEBUG2("FR_TLS_SUCCESS");
00333                 peap->status = PEAP_STATUS_TUNNEL_ESTABLISHED;
00334                 break;
00335 
00336                 /*
00337                  *      The TLS code is still working on the TLS
00338                  *      exchange, and it's a valid TLS request.
00339                  *      do nothing.
00340                  */
00341         case FR_TLS_HANDLED:
00342           /*
00343            *    FIXME: If the SSL session is established, grab the state
00344            *    and EAP id from the inner tunnel, and update it with
00345            *    the expected EAP id!
00346            */
00347                 RDEBUG2("FR_TLS_HANDLED");
00348                 return 1;
00349 
00350                 /*
00351                  *      Handshake is done, proceed with decoding tunneled
00352                  *      data.
00353                  */
00354         case FR_TLS_OK:
00355                 RDEBUG2("FR_TLS_OK");
00356                 break;
00357 
00358                 /*
00359                  *      Anything else: fail.
00360                  */
00361         default:
00362                 RDEBUG2("FR_TLS_OTHERS");
00363                 return 0;
00364         }
00365 
00366         /*
00367          *      Session is established, proceed with decoding
00368          *      tunneled data.
00369          */
00370         RDEBUG2("Session established.  Decoding tunneled attributes.");
00371 
00372         /*
00373          *      We may need PEAP data associated with the session, so
00374          *      allocate it here, if it wasn't already alloacted.
00375          */
00376         if (!tls_session->opaque) {
00377                 tls_session->opaque = peap_alloc(inst);
00378                 tls_session->free_opaque = peap_free;
00379         }
00380 
00381         /*
00382          *      Process the PEAP portion of the request.
00383          */
00384         rcode = eappeap_process(handler, tls_session);
00385         switch (rcode) {
00386         case RLM_MODULE_REJECT:
00387                 eaptls_fail(handler, 0);
00388                 return 0;
00389 
00390         case RLM_MODULE_HANDLED:
00391                 eaptls_request(handler->eap_ds, tls_session);
00392                 return 1;
00393 
00394         case RLM_MODULE_OK:
00395                 /*
00396                  *      Move the saved VP's from the Access-Accept to
00397                  *      our Access-Accept.
00398                  */
00399                 peap = tls_session->opaque;
00400                 if (peap->soh_reply_vps) {
00401                         RDEBUG2("Using saved attributes from the SoH reply");
00402                         debug_pair_list(peap->soh_reply_vps);
00403                         pairadd(&handler->request->reply->vps, peap->soh_reply_vps);
00404                         peap->soh_reply_vps = NULL;
00405                 }
00406                 if (peap->accept_vps) {
00407                         RDEBUG2("Using saved attributes from the original Access-Accept");
00408                         debug_pair_list(peap->accept_vps);
00409                         pairadd(&handler->request->reply->vps, peap->accept_vps);
00410                         peap->accept_vps = NULL;
00411                 }
00412 
00413                 /*
00414                  *      Success: Automatically return MPPE keys.
00415                  */
00416                 return eaptls_success(handler, 0);
00417 
00418                 /*
00419                  *      No response packet, MUST be proxying it.
00420                  *      The main EAP module will take care of discovering
00421                  *      that the request now has a "proxy" packet, and
00422                  *      will proxy it, rather than returning an EAP packet.
00423                  */
00424         case RLM_MODULE_UPDATED:
00425 #ifdef WITH_PROXY
00426                 rad_assert(handler->request->proxy != NULL);
00427 #endif
00428                 return 1;
00429                 break;
00430 
00431         default:
00432                 break;
00433         }
00434 
00435         eaptls_fail(handler, 0);
00436         return 0;
00437 }
