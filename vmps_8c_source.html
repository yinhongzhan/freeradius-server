00042 {
00043         RADIUS_PACKET   *packet;
00044         RAD_REQUEST_FUNP fun = NULL;

00046 
00047         packet = vqp_recv(listener->fd);
00048         if (!packet) {
00049                 radlog(L_ERR, "%s", fr_strerror());
00050                 return 0;
00051         }
00052 
00053         if ((client = client_listener_find(listener,
00054                                            &packet->src_ipaddr,
00055                                            packet->src_port)) == NULL) {
00056                 rad_free(&packet);
00057                 return 0;
00058         }
00059 
00060         /*
00061          *      Do new stuff.
00062          */
00063         fun = vmps_process;
00064 
00065         if (!request_receive(listener, packet, client, fun)) {
00066                 rad_free(&packet);
00067                 return 0;
00068         }
00069 
00070         return 1;
00071 }
