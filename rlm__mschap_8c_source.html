00061 static int pdb_decode_acct_ctrl(const char *p)
00062 {
00063         int acct_ctrl = 0;
00064         int done = 0;
00065 
00066         /*
00067          * Check if the account type bits have been encoded after the
00068          * NT password (in the form [NDHTUWSLXI]).
00069          */
00070 
00071         if (*p != '[') return 0;
00072 
00073         for (p++; *p && !done; p++) {
00074                 switch (*p) {
00075                         case 'N': /* 'N'o password. */
00076                           acct_ctrl |= ACB_PWNOTREQ;
00077                           break;
00078 
00079                         case 'D':  /* 'D'isabled. */
00080                           acct_ctrl |= ACB_DISABLED ;
00081                           break;
00082 
00083                         case 'H':  /* 'H'omedir required. */
00084                           acct_ctrl |= ACB_HOMDIRREQ;
00085                           break;
00086 
00087                         case 'T': /* 'T'emp account. */
00088                           acct_ctrl |= ACB_TEMPDUP;
00089                           break;
00090 
00091                         case 'U': /* 'U'ser account (normal). */
00092                           acct_ctrl |= ACB_NORMAL;
00093                           break;
00094 
00095                         case 'M': /* 'M'NS logon user account. What is this? */
00096                           acct_ctrl |= ACB_MNS;
00097                           break;
00098 
00099                         case 'W': /* 'W'orkstation account. */
00100                           acct_ctrl |= ACB_WSTRUST;
00101                           break;
00102 
00103                         case 'S': /* 'S'erver account. */
00104                           acct_ctrl |= ACB_SVRTRUST;
00105                           break;
00106 
00107                         case 'L': /* 'L'ocked account. */
00108                           acct_ctrl |= ACB_AUTOLOCK;
00109                           break;
00110 
00111                         case 'X': /* No 'X'piry on password */
00112                           acct_ctrl |= ACB_PWNOEXP;
