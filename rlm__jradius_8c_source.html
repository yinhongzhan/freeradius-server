00930 static int rlm_jradius_call(char func, void *instance, REQUEST *req, int isproxy)
00931 {
00932   JRADIUS        * inst    = instance;
00933   RADIUS_PACKET  * request = req->packet;
00934   RADIUS_PACKET  * reply   = req->reply;
00935   JRSOCK         * jrsock  = 0;
00936   JRSOCK           sjrsock;
00937 
00938   int exitstatus = inst->onfail;
00939   unsigned char rcode, pcount;
00940 
00941   unsigned char buff[MESSAGE_LEN];
00942   byte_array ba;
00943 
00944   char * n = inst->name;
00945   unsigned int nlen = strlen(n);
00946   const char * err = 0;
00947   int rc, attempt2=0;
00948 
00949 #define W_ERR(s) { err=s; goto packerror;  }
00950 #define R_ERR(s) { err=s; goto parseerror; }
00951 
00952 #ifdef WITH_PROXY
00953   if (isproxy) {
00954           request = req->proxy;
00955           reply   = req->proxy_reply;
00956   }
00957 #endif
00958 
00959   if (inst->keepalive) {
00960     jrsock = get_socket(inst);
00961     if (!jrsock) return exitstatus;
00962   } else {
00963     jrsock = &sjrsock;
00964     memset(jrsock, 0, sizeof(*jrsock));
00965     jrsock->state = not_connected;
00966   }
00967 
00968   init_byte_array(&ba, buff, sizeof(buff));
00969 
00970   pcount = 0;
00971   if (request) pcount++;
00972   if (reply) pcount++;
00973 
00974   /*
00975    *     Create byte array to send to jradius
00976    */
00977   if ((rc = pack_uint32 (&ba, nlen))                  == -1)  W_ERR("pack_uint32(nlen)");
00978   if ((rc = pack_bytes  (&ba, (void *)n, nlen))       == -1)  W_ERR("pack_bytes(name)");
00979   if ((rc = pack_byte   (&ba, func))                  == -1)  W_ERR("pack_byte(fun)");
00980   if ((rc = pack_byte   (&ba, pcount))                == -1)  W_ERR("pack_byte(pcnt)");
00981   if (pcount > 0 && (rc = pack_packet (&ba, request)) == -1)  W_ERR("pack_packet(req)");
00982   if (pcount > 1 && (rc = pack_packet (&ba, reply))   == -1)  W_ERR("pack_packet(rep)");
00983   if ((rc = pack_request(&ba, req))                   == -1)  W_ERR("pack_request()");
00984 
00985   /*
00986    *     Send data
00987    */
00988  start_over:
00989   if (jrsock->state == not_connected) {
00990     if (attempt2) radlog(L_ERR, LOG_PREFIX "reconnecting socket id %d", jrsock->id);
00991     if (!connect_socket(jrsock, inst)) {
00992       if (attempt2) radlog(L_ERR, LOG_PREFIX "could not reconnect socket %d, giving up", jrsock->id);
00993       goto cleanup;
00994     }
00995   }
00996   radlog(L_DBG, LOG_PREFIX "sending %d bytes to socket %d", ba.pos, jrsock->id);
00997   if (sock_write(inst, jrsock, ba.b, ba.pos) != (int)ba.pos ||
00998       (rc = read_byte(inst, jrsock, &rcode)) == -1) {
00999     /*
01000      *   With an error on the write or the first read, try closing the socket
01001      *   and reconnecting to see if that improves matters any (tries this only once)
01002      */
01003     radlog(L_ERR, LOG_PREFIX "error sending request with socket %d", jrsock->id);
01004     if (!inst->keepalive || attempt2) W_ERR("socket_send/first_read");
01005     close_socket(inst, jrsock);
01006     attempt2 = 1;
01007     goto start_over;
01008   }
01009 
01010   /*
01011    *     Read result
01012    */
01013   if ((rc = read_byte(inst, jrsock, &pcount)) == -1)  R_ERR("read_byte(pcnt)");
01014 
01015   radlog(L_DBG, LOG_PREFIX "return code %d; receiving %d packets", (int)rcode, (int)pcount);
01016 
01017   if (pcount > 0 && request) if ((rc = read_packet (inst, jrsock, request)) == -1)  R_ERR("read_packet(req)");
01018   if (pcount > 1 && reply)   if ((rc = read_packet (inst, jrsock, reply))   == -1)  R_ERR("read_packet(rep)");
01019 
01020   if ((rc = read_request(inst, jrsock, req)) == -1) R_ERR("read_request()");
01021 
01022   /*
01023    *    Since we deleted all the attribute lists in the request,
01024    *    we need to reconfigure a few pointers in the REQUEST object
01025    */
01026   if (req->username) {
01027     req->username = pairfind(request->vps, PW_USER_NAME, 0);
01028   }
01029   if (req->password) {
01030     req->password = pairfind(request->vps, PW_PASSWORD, 0);
01031     if (!req->password) req->password = pairfind(request->vps, PW_CHAP_PASSWORD, 0);
01032   }
01033 
01034   /*
01035    *    All done, set return code and cleanup
01036    */
01037   exitstatus = (int)rcode;
01038   goto cleanup;
01039 
01040  parseerror:
01041   radlog(L_ERR, LOG_PREFIX "problem parsing the data [%s]",err);
01042   if (inst->keepalive) close_socket(inst, jrsock);
01043   goto cleanup;
01044 
01045  packerror:
01046   radlog(L_ERR, LOG_PREFIX "problem packing the data[%s]",err);
01047   if (inst->keepalive) close_socket(inst, jrsock);
01048 
01049  cleanup:
01050   if (inst->keepalive) 
01051     release_socket(inst, jrsock);
01052   else  
01053     close_socket(inst, jrsock);
01054 
01055   return exitstatus;
01056 }
