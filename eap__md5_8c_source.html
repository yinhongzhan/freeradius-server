00081 /*
00082  *      We expect only RESPONSE for which SUCCESS or FAILURE is sent back
00083  */
00084 MD5_PACKET *eapmd5_extract(EAP_DS *eap_ds)
00085 {
00086         md5_packet_t    *data;
00087         MD5_PACKET      *packet;
00088         unsigned short  name_len;
00089 
00090         /*
00091          *      We need a response, of type EAP-MD5, with at least
00092          *      one byte of type data (EAP-MD5) following the 4-byte
00093          *      EAP-Packet header.
00094          */
00095         if (!eap_ds                                      ||
00096             !eap_ds->response                            ||
00097             (eap_ds->response->code != PW_MD5_RESPONSE)  ||
00098             eap_ds->response->type.type != PW_EAP_MD5    ||
00099             !eap_ds->response->type.data                 ||
00100             (eap_ds->response->length <= MD5_HEADER_LEN) ||
00101             (eap_ds->response->type.data[0] <= 0)) {
00102                 radlog(L_ERR, "rlm_eap_md5: corrupted data");
00103                 return NULL;
00104         }
00105 
00106         packet = eapmd5_alloc();
00107         if (!packet) return NULL;
00108 
00109         /*
00110          *      Code & id for MD5 & EAP are same
00111          *
00112          *      but md5_length = length of the EAP-MD5 data, which
00113          *      doesn't include the EAP header, or the octet saying
00114          *      EAP-MD5.
00115          */
00116         packet->code = eap_ds->response->code;
00117         packet->id = eap_ds->response->id;
00118         packet->length = eap_ds->response->length - (MD5_HEADER_LEN + 1);
00119 
00120         /*
00121          *      Sanity check the EAP-MD5 packet sent to us
00122          *      by the client.
00123          */
00124         data = (md5_packet_t *)eap_ds->response->type.data;
00125 
00126         /*
00127          *      Already checked the size above.
00128          */
00129         packet->value_size = data->value_size;
00130 
00131         /*
00132          *      Allocate room for the data, and copy over the data.
00133          */
00134         packet->value = malloc(packet->value_size);
00135         if (packet->value == NULL) {
00136                 radlog(L_ERR, "rlm_eap_md5: out of memory");
00137                 eapmd5_free(&packet);
00138                 return NULL;
00139         }
00140         memcpy(packet->value, data->value_name, packet->value_size);
00141 
00142         /*
00143          *      Name is optional and is present after Value, but we
00144          *      need to check for it, as eapmd5_compose()
00145          */
00146         name_len =  packet->length - (packet->value_size + 1);
00147         if (name_len) {
00148                 packet->name = malloc(name_len + 1);
00149                 if (!packet->name) {
00150                         radlog(L_ERR, "rlm_eap_md5: out of memory");
00151                         eapmd5_free(&packet);
00152                         return NULL;
00153                 }
00154                 memcpy(packet->name, data->value_name + packet->value_size,
00155                        name_len);
00156                 packet->name[name_len] = 0;
00157         }
00158 
00159         return packet;
00160 }

00167                   uint8_t *challenge)
00168 {

00170         char    string[1 + MAX_STRING_LEN*2];

00173 
00174         /*
00175          *      Sanity check it.
00176          */
00177         if (packet->value_size != 16) {
00178                 radlog(L_ERR, "rlm_eap_md5: Expected 16 bytes of response to challenge, got %d", packet->value_size);
00179                 return 0;

00181 
00182         len = 0;
00183         ptr = string;
00184 
00185         /*
00186          *      This is really rad_chap_pwencode()...
00187          */
00188         *ptr++ = packet->id;
00189         len++;
00190         memcpy(ptr, password->vp_strvalue, password->length);
00191         ptr += password->length;
00192         len += password->length;
00193 
00194         /*
00195          *      The challenge size is hard-coded.
00196          */
00197         memcpy(ptr, challenge, MD5_CHALLENGE_LEN);
00198         len += MD5_CHALLENGE_LEN;
00199 
00200         fr_md5_calc((u_char *)output, (u_char *)string, len);
00201 
00202         /*
00203          *      The length of the response is always 16 for MD5.
00204          */
00205         if (memcmp(output, packet->value, 16) != 0) {
00206                 return 0;
00207         }
00208         return 1;
