00144  *      Return an interface-id from

00147 uint8_t *ifid_aton(const char *ifid_str, uint8_t *ifid)
00148 {
00149         static const char xdigits[] = "0123456789abcdef";
00150         const char *p, *pch;
00151         int num_id = 0, val = 0, idx = 0;
00152 
00153         for (p = ifid_str; ; ++p) {
00154                 if (*p == ':' || *p == '\0') {
00155                         if (num_id <= 0)
00156                                 return NULL;
00157 
00158                         /*
00159                          *      Drop 'val' into the array.
00160                          */
00161                         ifid[idx] = (val >> 8) & 0xff;
00162                         ifid[idx + 1] = val & 0xff;
00163                         if (*p == '\0') {
00164                                 /*
00165                                  *      Must have all entries before
00166                                  *      end of the string.
00167                                  */
00168                                 if (idx != 6)
00169                                         return NULL;
00170                                 break;
00171                         }
00172                         val = 0;
00173                         num_id = 0;
00174                         if ((idx += 2) > 6)
00175                                 return NULL;
00176                 } else if ((pch = strchr(xdigits, tolower(*p))) != NULL) {
00177                         if (++num_id > 4)
00178                                 return NULL;
00179                         /*
00180                          *      Dumb version of 'scanf'
00181                          */
00182                         val <<= 4;
00183                         val |= (pch - xdigits);
00184                 } else
00185                         return NULL;

00243  * inet_pton6(src, dst)
00244  *      convert presentation level address to network order binary form.
00245  * return:

00247  * notice:
00248  *      (1) does not touch `dst' unless it's returning 1.
00249  *      (2) :: in a full address is silently ignored.

00251  *      inspired by Mark Andrews.
00252  * author:
00253  *      Paul Vixie, 1996.
00254  */
00255 static int
00256 inet_pton6(const char *src, unsigned char *dst)
00257 {

00264 
00265         memset((tp = tmp), 0, IN6ADDRSZ);
00266         endp = tp + IN6ADDRSZ;
00267         colonp = NULL;
00268         /* Leading :: requires some special handling. */
00269         if (*src == ':')
00270                 if (*++src != ':')
00271                         return (0);
00272         curtok = src;
00273         saw_xdigit = 0;
00274         val = 0;
00275         while ((ch = *src++) != '\0') {
00276                 const char *pch;
00277 
00278                 if ((pch = strchr((xdigits = xdigits_l), ch)) == NULL)
00279                         pch = strchr((xdigits = xdigits_u), ch);
00280                 if (pch != NULL) {
00281                         val <<= 4;
00282                         val |= (pch - xdigits);
00283                         if (val > 0xffff)
00284                                 return (0);
00285                         saw_xdigit = 1;
00286                         continue;
00287                 }
00288                 if (ch == ':') {
00289                         curtok = src;
00290                         if (!saw_xdigit) {
00291                                 if (colonp)

00293                                 colonp = tp;
00294                                 continue;
00295                         }
00296                         if (tp + INT16SZ > endp)
00297                                 return (0);
00298                         *tp++ = (u_char) (val >> 8) & 0xff;
00299                         *tp++ = (u_char) val & 0xff;
00300                         saw_xdigit = 0;
00301                         val = 0;
00302                         continue;
00303                 }
00304                 if (ch == '.' && ((tp + INADDRSZ) <= endp) &&

00307                         saw_xdigit = 0;

00309                 }
00310                 return (0);

00313                 if (tp + INT16SZ > endp)

00324                 int i;
00325 
00326                 for (i = 1; i <= n; i++) {
00327                         endp[- i] = colonp[n - i];
00328                         colonp[n - i] = 0;
00329                 }
00330                 tp = endp;
00331         }
00332         if (tp != endp)
00333                 return (0);
00334         /* bcopy(tmp, dst, IN6ADDRSZ); */
00335         memcpy(dst, tmp, IN6ADDRSZ);
00336         return (1);
00337 }
