00343  *      Walk over the cache, finding entries with the matching NAS IP address.
00344  */
00345 static int nas_port_walk(void *context, void *data)
00346 {
00347         offset_walk_t   *walk = context;
00348         NAS_PORT        *nas_port = data;
00349 
00350         /*
00351          *      Doesn't match, keep going.
00352          */
00353         if (walk->nas_address != nas_port->nas_address) return 0;
00354 
00355         /*
00356          *      Insert it into the offset tree, for later deletion.
00357          */
00358         if (rbtree_insert(walk->offset_tree, nas_port) != 1) {
00359                 DEBUG2("  rlm_radumtp: Insertion failed in nas port walk.");
00360                 return 1;
00361         }
00362

00367 /*
00368  *      Walk through the offset tree, operating on the cache
00369  */
00370 static int offset_walk(void *context, void *data)
00371 {
00372         offset_walk_t   *walk = context;
00373         NAS_PORT        *nas_port = data;
00374         struct radutmp  utmp;
00375         radutmp_simul_t *user, myUser;
00376 
00377         /*
00378          *      Seek to the entry, and possibly re-write it.
00379          */
00380         if (lseek(walk->fd, nas_port->offset, SEEK_SET) < 0) {
00381                 rad_assert(0 == 1);
00382         }
00383 
00384         if (read(walk->fd, &utmp, sizeof(utmp)) != sizeof(utmp)) {
00385                 rad_assert(0 == 1);
00386         }
00387 
00388         /*
00389          *      If the entry in the file is NEWER than the reboot
00390          *      packet, don't re-write it, and don't delete it.
00391          */
00392         if (utmp.time > walk->now) {
00393                 return 0;
00394         }
00395 
00396         utmp.type = P_IDLE;
00397         utmp.time = walk->now;
00398 
00399         if (lseek(walk->fd, -(off_t)sizeof(utmp), SEEK_CUR) < 0) {
00400                 radlog(L_ERR, "rlm_radutmp: offset_walk: failed in lseek: %s",
00401                        strerror(errno));
00402                 return 1;
00403         }
00404 
00405         write(walk->fd, &utmp, sizeof(utmp));
00406 
00407         strlcpy(myUser.login, utmp.login, sizeof(myUser.login));
00408         user = rbtree_finddata(walk->inst->user_tree, &myUser);
00409         rad_assert(user != NULL);
00410         rad_assert(user->simul_count > 0);
00411         user->simul_count--;
00412         if (user->simul_count == 0) {
00413                 rbtree_deletebydata(walk->inst->user_tree, user);
00414         }
00415 
00416         if (rbtree_deletebydata(walk->cache->nas_ports, nas_port) == 0) {
00417                 radlog(L_ERR, "rlm_radutmp: Failed to delete entry from cache");
00418                 return 1;
00419         }
00420 
00421         /*
00422          *      Insert the entry into the free list.
00423          */
00424         nas_port->next = walk->cache->free_offsets;
00425         walk->cache->free_offsets = nas_port;
00426 
00427         return 0;
00428 }

00431 /*
00432  *      Zap all users on a NAS from the radutmp file.
00433  */
00434 static int radutmp_zap(rlm_radutmp_t *inst,
00435                        radutmp_cache_t *cache,
00436                        uint32_t nas_address,
00437                        time_t now)
00438 {
00439         int             rcode;
00440         rbtree_t        *offset_tree;
00441         offset_walk_t   walk;
00442 
00443         rad_assert(now != 0);
00444 
00445         /*
00446          *      If there's nothing in the file, do nothing,
00447          *      but truncate the file, just to be safe.
00448          */
00449         if (rbtree_num_elements(cache->nas_ports) == 0) {
00450                 truncate(cache->filename, (off_t) 0);
00451                 DEBUG2("  rlm_radutmp: No entries in file.  Quenching zap.");
00452                 return 1;
00453         }
00454 
00455         /*
00456          *      Create the offset tree, as we want to delete utmp
00457          *      entries starting from the start of the file, and we
00458          *      can't delete nodes from an rbtree while we're walking
00459          *      it.
00460          */
00461         offset_tree = rbtree_create(offset_cmp, NULL, 0);
00462         if (!offset_tree) {
00463                 radlog(L_ERR, "rlm_radutmp: Out of memory");
00464                 return 0;
00465         }
00466 
00467         pthread_mutex_lock(&cache->mutex);
00468 
00469         /*
00470          *      Walk through the cache, finding entries for this NAS,
00471          *      and add those entries to the offset tree.
00472          */
00473         memset(&walk, 0, sizeof(walk));
00474         walk.inst = inst;
00475         walk.offset_tree = offset_tree;
00476         walk.nas_address = nas_address;
00477         rcode = rbtree_walk(cache->nas_ports, PreOrder, nas_port_walk, &walk);
00478         if (rcode != 0) {
00479                 pthread_mutex_unlock(&cache->mutex);
00480                 rbtree_free(offset_tree);
00481                 radlog(L_ERR, "rlm_radutmp: Failed walking the cache.");
00482                 return 0;
00483         }
00484 
00485         /*
00486          *      If both trees have the same number of elements, then
00487          *      don't do anything special, as UDP packets may be
00488          *      received out of order, by several seconds.  The
00489          *      "offset_walk" routine MAY NOT delete the entries, if
00490          *      it sees that the entries in the file are newer than
00491          *      the reboot packet.
00492          */
00493 
00494         /*
00495          *      If there's nothing to do, don't do anything.
00496          */
00497         if (rbtree_num_elements(offset_tree) == 0) {
00498                 DEBUG2("  rlm_radutmp: NAS IP %08x has no users recorded in file %s.",
00499                        htonl(nas_address), cache->filename);
00500                 pthread_mutex_unlock(&cache->mutex);
00501                 rbtree_free(offset_tree);
00502                 return 1;
00503         }
00504 
00505         /*
00506          *      Open the file, to re-write only a few of the entries.
00507          */
00508         walk.fd = open(cache->filename, O_RDWR);
00509         if (walk.fd < 0) {
00510                 pthread_mutex_unlock(&cache->mutex);
00511                 rbtree_free(offset_tree);
00512                 radlog(L_ERR, "rlm_radutmp: Error accessing file %s: %s",
00513                        cache->filename, strerror(errno));
00514                 return 0;
00515         }
00516 
00517         /*
00518          *      Lock the utmp file, prefer lockf() over flock().
00519          *
00520          *      FIXME: maybe we want to lock per-record?
00521          */
00522         rad_lockfd(walk.fd, LOCK_LEN);
00523 
00524         /*
00525          *      Walk through the offset tree, from start to finish,
00526          *      deleting entries from the NAS tree, adding them to
00527          *      the "free offset" cache, and lseek'ing to that offset
00528          *      in the file, and clearing out the data.
00529          */
00530         walk.cache = cache;
00531         walk.now = now;
00532         rcode = rbtree_walk(offset_tree, InOrder, offset_walk, &walk);
00533         rbtree_free(offset_tree);
00534         if (rcode != 0) {
00535                 radlog(L_ERR, "rlm_radutmp: Failed walking the offsets.");
00536                 return 0;
00537         }
00538 
00539         close(walk.fd); /* and implicitly release the locks */
00540 
00541         /*
00542          *      Just to clean up the file.  If it's empty,
00543          *      nuke everything.
00544          */
00545         if (rbtree_num_elements(cache->nas_ports) == 0) {
00546                 NAS_PORT        *this, *next; /* too many copies of code */
00547 
00548                 for (this = inst->cache.free_offsets;
00549                      this != NULL;
00550                      this = next) {
00551                         next = this->next;
00552                         free(this);
00553                 }
00554 
00555                 truncate(cache->filename, 0);
00556                 rad_assert(rbtree_num_elements(inst->user_tree) == 0);
00557         }
00558 
00559         pthread_mutex_unlock(&cache->mutex);
00560 
00561         return 1;
00562 }

00566  *      Read a file, to cache all of its entries.
00567  */
00568 static int cache_file(rlm_radutmp_t *inst, radutmp_cache_t *cache)
00569 {
00570         int             fd;
00571         int             read_size;
00572         struct          stat buf;
00573         struct          radutmp utmp;
00574         NAS_PORT        **tail;
00575 
00576         rad_assert(cache->max_offset == 0);
00577         rad_assert(cache->free_offsets == NULL);
00578 
00579         /*
00580          *      Doesn't exist, we're fine.
00581          */
00582         if (stat(cache->filename, &buf) < 0) {
00583                 if (errno == ENOENT) {
00584                         cache->cached_file = 1;
00585                         return 0;
00586                 }
00587                 radlog(L_ERR, "rlm_radutmp: Cannot stat %s: %s",
00588                        cache->filename, strerror(errno));
00589                 return 1;
00590         }
00591 
00592         /*
00593          *      Nothing's there, we're OK.
00594          */
00595         if (buf.st_size == 0) {
00596                 cache->cached_file = 1;
00597                 return 0;
00598         }
00599 
00600         /*
00601          *      Don't let others much around with our data.
00602          */
00603         pthread_mutex_lock(&cache->mutex);
00604 
00605         /*
00606          *      Read the file and cache it's entries.
00607          */
00608         fd = open(cache->filename, O_RDONLY, cache->permission);
00609         if (fd < 0) {
00610                 pthread_mutex_unlock(&cache->mutex);
00611                 radlog(L_ERR, "rlm_radutmp: Error opening %s: %s",
00612                        cache->filename, strerror(errno));
00613                 return 1;
00614         }
00615 
00616         /*
00617          *      Insert free entries into the tail, so that entries
00618          *      get used from the start.
00619          */
00620         tail = &(cache->free_offsets);
00621 
00622         /*
00623          *      Don't lock the file, as we're only reading it.
00624          */

00627

00634

00636                         nas_port->offset = cache->max_offset;
00637                         cache->max_offset += sizeof(utmp);
00638 
00639                         /*
00640                          *      Idle.  Add it to the list of free
00641                          *      offsets.
00642                          */
00643                         if (utmp.type == P_IDLE) {
00644                                 *tail = nas_port;
00645                                 tail = &(nas_port->next);
00646                                 continue;
00647                         }
00648 
00649                         /*
00650                          *      It's a login record,
00651                          */
00652                         nas_port->nas_address = utmp.nas_address;
00653                         nas_port->nas_port = utmp.nas_port;
00654 
00655                         if (!rbtree_insert(cache->nas_ports, nas_port)) {
00656                                 rad_assert(0 == 1);
00657                         }
00658 
00659                         /*
00660                          *      Adds a trailing \0, so myUser.login has
00661                          *      an extra char allocated..
00662                          */
00663                         strlcpy(myUser.login, utmp.login, sizeof(myUser.login));
00664                         user = rbtree_finddata(inst->user_tree, &myUser);
00665                         if (user) {
00666                                 user->simul_count++;
00667                         } else {
00668                                 /*
00669                                  *      Allocate new entry, and add it
00670                                  *      to the tree.
00671                                  */
00672                                 user = rad_malloc(sizeof(user));
00673                                 strlcpy(user->login, utmp.login,
00674                                         sizeof(user->login));
00675                                 user->simul_count = 1;
00676 
00677                                 if (!rbtree_insert(inst->user_tree, user)) {
00678                                         rad_assert(0 == 1);
00679                                 }

00681                         continue;
