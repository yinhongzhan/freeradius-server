00115 static int PAM_conv (int num_msg,
00116                      const struct pam_message **msg,
00117                      struct pam_response **resp,
00118                      void *appdata_ptr) {
00119   int count;
00120   struct pam_response *reply;
00121   my_PAM *pam_config = (my_PAM *) appdata_ptr;
00122 
00123 /* strdup(NULL) doesn't work on some platforms */
00124 #define COPY_STRING(s) ((s) ? strdup(s) : NULL)
00125 
00126   reply = rad_malloc(num_msg * sizeof(struct pam_response));
00127   memset(reply, 0, num_msg * sizeof(struct pam_response));
00128   for (count = 0; count < num_msg; count++) {
00129     switch (msg[count]->msg_style) {
00130     case PAM_PROMPT_ECHO_ON:
00131       reply[count].resp_retcode = PAM_SUCCESS;
00132       reply[count].resp = COPY_STRING(pam_config->username);
00133       break;
00134     case PAM_PROMPT_ECHO_OFF:
00135       reply[count].resp_retcode = PAM_SUCCESS;
00136       reply[count].resp = COPY_STRING(pam_config->password);
00137       break;
00138     case PAM_TEXT_INFO:
00139       /* ignore it... */
00140       break;
00141     case PAM_ERROR_MSG:
00142     default:
00143       /* Must be an error of some sort... */
00144       for (count = 0; count < num_msg; count++) {
00145         if (reply[count].resp) {
00146           /* could be a password, let's be sanitary */
00147           memset(reply[count].resp, 0, strlen(reply[count].resp));
00148           free(reply[count].resp);
00149         }
00150       }
00151       free(reply);
00152       pam_config->error = 1;
00153       return PAM_CONV_ERR;
00154     }
00155   }
00156   *resp = reply;
00157   /* PAM frees reply (including reply[].resp) */
00158 
00159   return PAM_SUCCESS;
00160 }

00179 {

00230 static int pam_auth(void *instance, REQUEST *request)
00231 {
00232         int     r;
00233         VALUE_PAIR *pair;
00234         rlm_pam_t *data = (rlm_pam_t *) instance;
00235 
00236         const char *pam_auth_string = data->pam_auth_name;
00237 
00238         /*
00239          *      We can only authenticate user requests which HAVE
00240          *      a User-Name attribute.
00241          */
00242         if (!request->username) {
00243                 radlog(L_AUTH, "rlm_pam: Attribute \"User-Name\" is required for authentication.");
00244                 return RLM_MODULE_INVALID;
00245         }
00246 
00247         /*
00248          *      We can only authenticate user requests which HAVE
00249          *      a User-Password attribute.
00250          */
00251         if (!request->password) {
00252                 radlog(L_AUTH, "rlm_pam: Attribute \"User-Password\" is required for authentication.");
00253                 return RLM_MODULE_INVALID;
00254         }
00255 
00256         /*
00257          *  Ensure that we're being passed a plain-text password,
00258          *  and not anything else.
00259          */
00260         if (request->password->attribute != PW_USER_PASSWORD) {
00261                 radlog(L_AUTH, "rlm_pam: Attribute \"User-Password\" is required for authentication.  Cannot use \"%s\".", request->password->name);
00262                 return RLM_MODULE_INVALID;
00263         }
00264 
00265         /*
00266          *      Let the 'users' file over-ride the PAM auth name string,
00267          *      for backwards compatibility.
00268          */
00269         pair = pairfind(request->config_items, PAM_AUTH_ATTR, 0);
00270         if (pair) pam_auth_string = (char *)pair->vp_strvalue;
00271 
00272         r = pam_pass((char *)request->username->vp_strvalue,
00273                      (char *)request->password->vp_strvalue,
00274                      pam_auth_string);
00275 
00276         if (r == 0) {
00277                 return RLM_MODULE_OK;
00278         }
00279         return RLM_MODULE_REJECT;
00280 }
