00307 void virtual_servers_free(time_t when)
00308 {
00309         int i;
00310         virtual_server_t **last;
00311         
00312         for (i = 0; i < VIRTUAL_SERVER_HASH_SIZE; i++) {
00313                 virtual_server_t *server, *next;
00314 
00315                 last = &virtual_servers[i];
00316                 for (server = virtual_servers[i];
00317                      server != NULL;
00318                      server = next) {
00319                         next = server->next;
00320 
00321                         /*
00322                          *      If we delete it, fix the links so that
00323                          *      we don't orphan anything.  Also,
00324                          *      delete it if it's old, AND a newer one
00325                          *      was defined.

00327                          *      Otherwise, the last pointer gets set to
00328                          *      the one we didn't delete.

00331                             ((server->created < when) && server->can_free)) {
00332                                 *last = server->next;
00333                                 virtual_server_free(server);
00334                         } else {
00335                                 last = &(server->next);

00478  *      Find a module on disk or in memory, and link to it.

00480 static module_entry_t *linkto_module(const char *module_name,
00481                                      CONF_SECTION *cs)
00482 {
00483         module_entry_t myentry;
00484         module_entry_t *node;
00485         lt_dlhandle handle = NULL;
00486         char module_struct[256];
00487         char *p;
00488         const module_t *module;
00489 
00490         strlcpy(myentry.name, module_name, sizeof(myentry.name));
00491         node = rbtree_finddata(module_tree, &myentry);
00492         if (node) return node;
00493 
00494         /*
00495          *      Link to the module's rlm_FOO{} module structure.
00496          *
00497          *      The module_name variable has the version number
00498          *      embedded in it, and we don't want that here.
00499          */
00500         strcpy(module_struct, module_name);
00501         p = strrchr(module_struct, '-');
00502         if (p) *p = '\0';
00503 
00504 #if !defined(WITH_LIBLTDL) && defined(HAVE_DLFCN_H) && defined(RTLD_SELF)
00505         module = lt_dlsym(RTLD_SELF, module_struct);
00506         if (module) goto open_self;
00507 #endif
00508 
00509         /*
00510          *      Keep the handle around so we can dlclose() it.
00511          */
00512         handle = fr_dlopenext(module_name);
00513         if (handle == NULL) {
00514                 cf_log_err(cf_sectiontoitem(cs),
00515                            "Failed to link to module '%s': %s\n",
00516                            module_name, lt_dlerror());
00517                 return NULL;
00518         }
00519 
00520         DEBUG3("    (Loaded %s, checking if it's valid)", module_name);
00521 
00522         /*
00523          *      libltld MAY core here, if the handle it gives us contains
00524          *      garbage data.
00525          */
00526         module = lt_dlsym(handle, module_struct);
00527         if (!module) {
00528                 cf_log_err(cf_sectiontoitem(cs),
00529                            "Failed linking to %s structure: %s\n",
00530                            module_name, lt_dlerror());
00531                 lt_dlclose(handle);
00532                 return NULL;
00533         }
00534 
00535 #if !defined(WIT_LIBLTDL) && defined (HAVE_DLFCN_H) && defined(RTLD_SELF)
00536  open_self:
00537 #endif
00538         /*
00539          *      Before doing anything else, check if it's sane.
00540          */
00541         if (module->magic != RLM_MODULE_MAGIC_NUMBER) {
00542                 lt_dlclose(handle);
00543                 cf_log_err(cf_sectiontoitem(cs),
00544                            "Invalid version in module '%s'",
00545                            module_name);
00546                 return NULL;
00547 
00548         }
00549 
00550         /* make room for the module type */
00551         node = rad_malloc(sizeof(*node));
00552         memset(node, 0, sizeof(*node));
00553         strlcpy(node->name, module_name, sizeof(node->name));

00557         cf_log_module(cs, "Linked to module %s", module_name);

00559         /*

00563         if (!rbtree_insert(module_tree, node)) {

00685                 /*

00691                  *      The module is thread-safe.  Don't give it a mutex.

00693                 node->mutex = NULL;
00694         }

00698

00713 /*
00714  *      Create a new sublist.
00715  */
00716 static indexed_modcallable *new_sublist(rbtree_t *components, int comp, int idx)
00717 {
00718         indexed_modcallable *c;
00719 
00720         c = lookup_by_index(components, comp, idx);
00721 
00722         /* It is an error to try to create a sublist that already
00723          * exists. It would almost certainly be caused by accidental
00724          * duplication in the config file.
00725          *
00726          * index 0 is the exception, because it is used when we want
00727          * to collect _all_ listed modules under a single index by
00728          * default, which is currently the case in all components
00729          * except authenticate. */
00730         if (c) {
00731                 if (idx == 0) {
00732                         return c;
00733                 }
00734                 return NULL;
00735         }
00736 
00737         c = rad_malloc(sizeof(*c));
00738         c->modulelist = NULL;
00739         c->comp = comp;
00740         c->idx = idx;
00741 
00742         if (!rbtree_insert(components, c)) {
00743                 free(c);
00744                 return NULL;
00745         }
00746 
00747         return c;

00800 /*
00801  *      Load a sub-module list, as found inside an Auth-Type foo {}
00802  *      block
00803  */
00804 static int load_subcomponent_section(modcallable *parent, CONF_SECTION *cs,
00805                                      rbtree_t *components,
00806                                      const DICT_ATTR *dattr, int comp)
00807 {
00808         indexed_modcallable *subcomp;
00809         modcallable *ml;
00810         DICT_VALUE *dval;
00811         const char *name2 = cf_section_name2(cs);
00812 
00813         rad_assert(comp >= RLM_COMPONENT_AUTH);
00814         rad_assert(comp < RLM_COMPONENT_COUNT);
00815 
00816         /*
00817          *      Sanity check.
00818          */
00819         if (!name2) {
00820                 cf_log_err(cf_sectiontoitem(cs),
00821                            "No name specified for %s block",
00822                            section_type_value[comp].typename);
00823                 return 1;
00824         }
00825 
00826         /*
00827          *      Compile the group.
00828          */
00829         ml = compile_modgroup(parent, comp, cs);
00830         if (!ml) {
00831                 return 0;
00832         }
00833 
00834         /*
00835          *      We must assign a numeric index to this subcomponent.
00836          *      It is generated and placed in the dictionary
00837          *      automatically.  If it isn't found, it's a serious
00838          *      error.
00839          */
00840         dval = dict_valbyname(dattr->attr, dattr->vendor, name2);
00841         if (!dval) {
00842                 cf_log_err(cf_sectiontoitem(cs),
00843                            "%s %s Not previously configured",
00844                            section_type_value[comp].typename, name2);
00845                 modcallable_free(&ml);
00846                 return 0;
00847         }
00848 
00849         subcomp = new_sublist(components, comp, dval->value);
00850         if (!subcomp) {
00851                 modcallable_free(&ml);
00852                 return 1;
00853         }
00854 
00855         subcomp->modulelist = ml;
00856         return 1;               /* OK */
00857 }

00859 static int define_type(const DICT_ATTR *dattr, const char *name)
00860 {
00861         uint32_t value;
00862         DICT_VALUE *dval;
00863 
00864         /*
00865          *      If the value already exists, don't
00866          *      create it again.
00867          */
00868         dval = dict_valbyname(dattr->attr, dattr->vendor, name);
00869         if (dval) return 1;
00870 
00871         /*
00872          *      Create a new unique value with a
00873          *      meaningless number.  You can't look at
00874          *      it from outside of this code, so it
00875          *      doesn't matter.  The only requirement
00876          *      is that it's unique.
00877          */
00878         do {
00879                 value = fr_rand() & 0x00ffffff;
00880         } while (dict_valbyattr(dattr->attr, dattr->vendor, value));
00881 
00882         DEBUG2("  Module: Creating %s = %s", dattr->name, name);
00883         if (dict_addvalue(name, dattr->name, value) < 0) {
00884                 radlog(L_ERR, "%s", fr_strerror());
00885                 return 0;
00886         }
00887 
00888         return 1;
00889 }

00891 static int load_component_section(CONF_SECTION *cs,
00892                                   rbtree_t *components, int comp)
00893 {
00894         modcallable *this;
00895         CONF_ITEM *modref;
00896         int idx;
00897         indexed_modcallable *subcomp;
00898         const char *modname;
00899         const char *visiblename;
00900         const DICT_ATTR *dattr;
00901 
00902         /*
00903          *      Find the attribute used to store VALUEs for this section.
00904          */
00905         dattr = dict_attrbyvalue(section_type_value[comp].attr, 0);
00906         if (!dattr) {
00907                 cf_log_err(cf_sectiontoitem(cs),
00908                            "No such attribute %s",
00909                            section_type_value[comp].typename);
00910                 return -1;
00911         }
00912 
00913         /*
00914          *      Loop over the entries in the named section, loading
00915          *      the sections this time.
00916          */
00917         for (modref = cf_item_find_next(cs, NULL);
00918              modref != NULL;
00919              modref = cf_item_find_next(cs, modref)) {
00920                 const char *name1;
00921                 CONF_PAIR *cp = NULL;
00922                 CONF_SECTION *scs = NULL;
00923 
00924                 if (cf_item_is_section(modref)) {
00925                         scs = cf_itemtosection(modref);
00926 
00927                         name1 = cf_section_name1(scs);
00928 
00929                         if (strcmp(name1,
00930                                    section_type_value[comp].typename) == 0) {
00931                                 if (!load_subcomponent_section(NULL, scs,
00932                                                                components,
00933                                                                dattr,
00934                                                                comp)) {
00935                                         return -1; /* FIXME: memleak? */
00936                                 }
00937                                 continue;
00938                         }
00939 
00940                         cp = NULL;
00941 
00942                 } else if (cf_item_is_pair(modref)) {
00943                         cp = cf_itemtopair(modref);
00944 
00945                 } else {
00946                         continue; /* ignore it */
00947                 }
00948 
00949                 /*
00950                  *      Try to compile one entry.
00951                  */
00952                 this = compile_modsingle(NULL, comp, modref, &modname);
00953 
00954                 /*
00955                  *      It's OK for the module to not exist.
00956                  */
00957                 if (!this && modname && (modname[0] == '-')) {
00958                         DEBUG("WARNING: Not loading module \"%s\" as it is not enabled", modname + 1);
00959                         continue;
00960                 }
00961 
00962                 if (!this) {
00963                         cf_log_err(cf_sectiontoitem(cs),
00964                                    "Errors parsing %s section.\n",
00965                                    cf_section_name1(cs));
00966                         return -1;
00967                 }
00968 
00969                 /*
00970                  *      Look for Auth-Type foo {}, which are special
00971                  *      cases of named sections, and allowable ONLY
00972                  *      at the top-level.
00973                  *
00974                  *      i.e. They're not allowed in a "group" or "redundant"
00975                  *      subsection.
00976                  */
00977                 if (comp == RLM_COMPONENT_AUTH) {
00978                         DICT_VALUE *dval;
00979                         const char *modrefname = NULL;
00980                         if (cp) {
00981                                 modrefname = cf_pair_attr(cp);
00982                         } else {
00983                                 modrefname = cf_section_name2(scs);
00984                                 if (!modrefname) {
00985                                         modcallable_free(&this);
00986                                         cf_log_err(cf_sectiontoitem(cs),
00987                                                    "Errors parsing %s sub-section.\n",
00988                                                    cf_section_name1(scs));
00989                                         return -1;
00990                                 }
00991                         }
00992 
00993                         dval = dict_valbyname(PW_AUTH_TYPE, 0, modrefname);
00994                         if (!dval) {
00995                                 /*
00996                                  *      It's a section, but nothing we
00997                                  *      recognize.  Die!
00998                                  */
00999                                 modcallable_free(&this);
01000                                 cf_log_err(cf_sectiontoitem(cs),
01001                                            "Unknown Auth-Type \"%s\" in %s sub-section.",
01002                                            modrefname, section_type_value[comp].section);
01003                                 return -1;
01004                         }
01005                         idx = dval->value;
01006                 } else {
01007                         /* See the comment in new_sublist() for explanation
01008                          * of the special index 0 */
01009                         idx = 0;
01010                 }
01011 
01012                 subcomp = new_sublist(components, comp, idx);
01013                 if (subcomp == NULL) {
01014                         modcallable_free(&this);
01015                         continue;
01016                 }
01017 
01018                 /* If subcomp->modulelist is NULL, add_to_modcallable will
01019                  * create it */
01020                 visiblename = cf_section_name2(cs);
01021                 if (visiblename == NULL)
01022                         visiblename = cf_section_name1(cs);
01023                 add_to_modcallable(&subcomp->modulelist, this,
01024                                    comp, visiblename);
01025         }
01026 
01027         return 0;
01028 }

01030 static int load_byserver(CONF_SECTION *cs)
01031 {
01032         int comp, flag;
01033         const char *name = cf_section_name2(cs);
01034         rbtree_t *components;
01035         virtual_server_t *server = NULL;
01036         indexed_modcallable *c;
01037 
01038         if (name) {
01039                 cf_log_info(cs, "server %s { # from file %s",
01040                             name, cf_section_filename(cs));
01041         } else {
01042                 cf_log_info(cs, "server { # from file %s",
01043                             cf_section_filename(cs));
01044         }
01045 
01046         cf_log_info(cs, " modules {");
01047 
01048         components = rbtree_create(indexed_modcallable_cmp,
01049                                    indexed_modcallable_free, 0);
01050         if (!components) {
01051                 radlog(L_ERR, "Failed to initialize components\n");
01052                 goto error;
01053         }
01054 
01055         server = rad_malloc(sizeof(*server));
01056         memset(server, 0, sizeof(*server));
01057 
01058         server->name = name;
01059         server->created = time(NULL);
01060         server->cs = cs;
01061         server->components = components;
01062 
01063         /*
01064          *      Define types first.
01065          */
01066         for (comp = 0; comp < RLM_COMPONENT_COUNT; ++comp) {
01067                 CONF_SECTION *subcs;
01068                 CONF_ITEM *modref;
01069                 DICT_ATTR *dattr;
01070 
01071                 subcs = cf_section_sub_find(cs,
01072                                             section_type_value[comp].section);
01073                 if (!subcs) continue;
01074 
01075                 if (cf_item_find_next(subcs, NULL) == NULL) continue;
01076 
01077                 /*
01078                  *      Find the attribute used to store VALUEs for this section.
01079                  */
01080                 dattr = dict_attrbyvalue(section_type_value[comp].attr, 0);
01081                 if (!dattr) {
01082                         cf_log_err(cf_sectiontoitem(subcs),
01083                                    "No such attribute %s",
01084                                    section_type_value[comp].typename);
01085                 error:
01086                         if (debug_flag == 0) {
01087                                 radlog(L_ERR, "Failed to load virtual server %s",
01088                                        (name != NULL) ? name : "<default>");
01089                         }
01090                         virtual_server_free(server);
01091                         return -1;
01092                 }
01093 
01094                 /*
01095                  *      Define dynamic types, so that others can reference
01096                  *      them.
01097                  */
01098                 for (modref = cf_item_find_next(subcs, NULL);
01099                      modref != NULL;
01100                      modref = cf_item_find_next(subcs, modref)) {
01101                         const char *name1;
01102                         CONF_SECTION *subsubcs;
01103 
01104                         /*
01105                          *      Create types for simple references
01106                          *      only when parsing the authenticate
01107                          *      section.
01108                          */
01109                         if ((section_type_value[comp].attr == PW_AUTH_TYPE) &&
01110                             cf_item_is_pair(modref)) {
01111                                 CONF_PAIR *cp = cf_itemtopair(modref);
01112                                 if (!define_type(dattr, cf_pair_attr(cp))) {
01113                                         goto error;
01114                                 }
01115 
01116                                 continue;
01117                         }
01118 
01119                         if (!cf_item_is_section(modref)) continue;
01120                         
01121                         subsubcs = cf_itemtosection(modref);
01122                         name1 = cf_section_name1(subsubcs);
01123 
01124                         if (strcmp(name1, section_type_value[comp].typename) == 0) {
01125                                 if (!define_type(dattr,
01126                                                  cf_section_name2(subsubcs))) {
01127                                         goto error;
01128                                 }
01129                         }
01130                 }
01131         } /* loop over components */
01132 
01133         /*
01134          *      Loop over all of the known components, finding their
01135          *      configuration section, and loading it.
01136          */
01137         flag = 0;
01138         for (comp = 0; comp < RLM_COMPONENT_COUNT; ++comp) {
01139                 CONF_SECTION *subcs;
01140 
01141                 subcs = cf_section_sub_find(cs,
01142                                             section_type_value[comp].section);
01143                 if (!subcs) continue;
01144                         
01145                 if (cf_item_find_next(subcs, NULL) == NULL) continue;
01146                         
01147                 cf_log_module(cs, "Checking %s {...} for more modules to load",
01148                        section_type_value[comp].section);
01149 
01150                 /*
01151                  *      Skip pre/post-proxy sections if we're not
01152                  *      proxying.
01153                  */
01154                 if (
01155 #ifdef WITH_PROXY
01156                     !mainconfig.proxy_requests &&
01157 #endif
01158                     ((comp == RLM_COMPONENT_PRE_PROXY) ||
01159                      (comp == RLM_COMPONENT_POST_PROXY))) {
01160                         continue;
01161                 }
01162 
01163 #ifndef WITH_ACCOUNTING
01164                 if (comp == RLM_COMPONENT_ACCT) continue;
01165 #endif
01166 
01167 #ifndef WITH_SESSION_MGMT
01168                 if (comp == RLM_COMPONENT_SESS) continue;
01169 #endif
01170 
01171                 if (load_component_section(subcs, components, comp) < 0) {
01172                         goto error;
01173                 }
01174 
01175                 /*
01176                  *      Cache a default, if it exists.  Some people
01177                  *      put empty sections for some reason...
01178                  */
01179                 c = lookup_by_index(components, comp, 0);
01180                 if (c) server->mc[comp] = c->modulelist;
01181 
01182                 server->subcs[comp] = subcs;
01183 
01184                 flag = 1;
01185         } /* loop over components */
01186 
01187         /*
01188          *      We haven't loaded any of the normal sections.  Maybe we're
01189          *      supposed to load the vmps section.
01190          *
01191          *      This is a bit of a hack...
01192          */
01193         if (!flag) {
01194                 CONF_SECTION *subcs;
01195 
01196                 subcs = cf_section_sub_find(cs, "vmps");
01197                 if (subcs) {
01198                         cf_log_module(cs, "Checking vmps {...} for more modules to load");              
01199                         if (load_component_section(subcs, components,
01200                                                    RLM_COMPONENT_POST_AUTH) < 0) {
01201                                 goto error;
01202                         }
01203                         c = lookup_by_index(components,
01204                                             RLM_COMPONENT_POST_AUTH, 0);
01205                         if (c) server->mc[RLM_COMPONENT_POST_AUTH] = c->modulelist;
01206                         flag = 1;
01207                 }
01208 
01209 #ifdef WITH_DHCP
01210                 if (!flag) {
01211                         const DICT_ATTR *dattr;
01212 
01213                         dattr = dict_attrbyname("DHCP-Message-Type");
01214 
01215                         /*
01216                          *      Handle each DHCP Message type separately.
01217                          */
01218                         if (dattr) for (subcs = cf_subsection_find_next(cs, NULL, "dhcp");
01219                                         subcs != NULL;
01220                                         subcs = cf_subsection_find_next(cs, subcs,
01221                                                                         "dhcp")) {
01222                                 const char *name2 = cf_section_name2(subcs);
01223 
01224                                 DEBUG2(" Module: Checking dhcp %s {...} for more modules to load", name2);
01225                                 if (!load_subcomponent_section(NULL, subcs,
01226                                                                components,
01227                                                                dattr,
01228                                                                RLM_COMPONENT_POST_AUTH)) {
01229                                         goto error; /* FIXME: memleak? */
01230                                 }
01231                                 c = lookup_by_index(components,
01232                                                     RLM_COMPONENT_POST_AUTH, 0);
01233                                 if (c) server->mc[RLM_COMPONENT_POST_AUTH] = c->modulelist;
01234                                 flag = 1;
01235                         }
01236                 }
01237 #endif
01238         }
01239 
01240         cf_log_info(cs, " } # modules");
01241         cf_log_info(cs, "} # server");
01242 
01243         if (!flag && name) {
01244                 DEBUG("WARNING: Server %s is empty, and will do nothing!",
01245                       name);
01246         }
01247 
01248         if (debug_flag == 0) {
01249                 radlog(L_INFO, "Loaded virtual server %s",
01250                        (name != NULL) ? name : "<default>");
01251         }
01252 
01253         /*
01254          *      Now that it is OK, insert it into the list.
01255          *
01256          *      This is thread-safe...
01257          */
01258         comp = virtual_server_idx(name);
01259         server->next = virtual_servers[comp];
01260         virtual_servers[comp] = server;
01261 
01262         /*
01263          *      Mark OLDER ones of the same name as being unused.
01264          */
01265         server = server->next;
01266         while (server) {
01267                 if ((!name && !server->name) ||
01268                     (name && server->name &&
01269                      (strcmp(server->name, name) == 0))) {
01270                         server->can_free = TRUE;
01271                         break;
01272                 }
01273                 server = server->next;
01274         }
01275 
01276         return 0;
01277 }

01280 /*
01281  *      Load all of the virtual servers.
01282  */
01283 int virtual_servers_load(CONF_SECTION *config)
01284 {
01285         CONF_SECTION *cs;
01286         static int first_time = TRUE;
01287 
01288         DEBUG2("%s: #### Loading Virtual Servers ####", mainconfig.name);
01289 
01290         /*
01291          *      If we have "server { ...}", then there SHOULD NOT be
01292          *      bare "authorize", etc. sections.  if there is no such
01293          *      server, then try to load the old-style sections first.
01294          *
01295          *      In either case, load the "default" virtual server first.
01296          *      this matches better iwth users expectations.
01297          */
01298         cs = cf_section_find_name2(cf_subsection_find_next(config, NULL,
01299                                                            "server"),
01300                                    "server", NULL);
01301         if (cs) {
01302                 if (load_byserver(cs) < 0) {
01303                         return -1;
01304                 }
01305         } else {
01306                 if (load_byserver(config) < 0) {
01307                         return -1;
01308                 }
01309         }
01310 
01311         /*
01312          *      Load all of the virtual servers.
01313          */
01314         for (cs = cf_subsection_find_next(config, NULL, "server");
01315              cs != NULL;
01316              cs = cf_subsection_find_next(config, cs, "server")) {
01317                 const char *name2;
01318                 virtual_server_t *server;
01319 
01320                 name2 = cf_section_name2(cs);
01321                 if (!name2) continue; /* handled above */
01322 
01323                 server = virtual_server_find(name2);
01324                 if (server &&
01325                     (cf_top_section(server->cs) == config)) {
01326                         radlog(L_ERR, "Duplicate virtual server \"%s\" in file %s:%d and file %s:%d",
01327                                server->name,
01328                                cf_section_filename(server->cs),
01329                                cf_section_lineno(server->cs),
01330                                cf_section_filename(cs),
01331                                cf_section_lineno(cs));
01332                         return -1;
01333                 }
01334 
01335                 if (load_byserver(cs) < 0) {
01336                         /*
01337                          *      Once we successfully started once,
01338                          *      continue loading the OTHER servers,
01339                          *      even if one fails.
01340                          */
01341                         if (!first_time) continue;
01342                         return -1;
01343                 }
01344         }
01345 
01346         /*
01347          *      If we succeed the first time around, remember that.
01348          */
01349         first_time = FALSE;
01350 
01351         return 0;
01352 }

01482                  *      i.e. Static modules.
