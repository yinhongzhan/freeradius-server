00176 /*
00177  *      EAP packet format to be sent over the wire
00178  *
00179  *      i.e. code+id+length+data where data = null/type+typedata
00180  *      based on code.
00181  *
00182  * INPUT to function is reply->code
00183  *                      reply->id
00184  *                      reply->type   - setup with data
00185  *
00186  * OUTPUT reply->packet is setup with wire format, and will
00187  *                      be malloc()'ed to the right size.
00188  *
00189  */
00190 int eap_wireformat(EAP_PACKET *reply)
00191 {
00192         eap_packet_t    *hdr;
00193         uint16_t total_length = 0;
00194 
00195         if (reply == NULL) return EAP_INVALID;
00196 
00197         /*
00198          *      If reply->packet is set, then the wire format
00199          *      has already been calculated, just succeed.
00200          */
00201         if(reply->packet != NULL) return EAP_VALID;
00202 
00203         total_length = EAP_HEADER_LEN;
00204         if (reply->code < 3) {
00205                 total_length += 1/*EAPtype*/;
00206                 if (reply->type.data && reply->type.length > 0) {
00207                         total_length += reply->type.length;
00208                 }
00209         }
00210 
00211         reply->packet = (unsigned char *)malloc(total_length);
00212         hdr = (eap_packet_t *)reply->packet;
00213         if (!hdr) {
00214                 radlog(L_ERR, "rlm_eap: out of memory");
00215                 return EAP_INVALID;
00216         }
00217 
00218         hdr->code = (reply->code & 0xFF);
00219         hdr->id = (reply->id & 0xFF);
00220         total_length = htons(total_length);
00221         memcpy(hdr->length, &total_length, sizeof(total_length));
00222 
00223         /*
00224          *      Request and Response packets are special.
00225          */
00226         if ((reply->code == PW_EAP_REQUEST) ||
00227             (reply->code == PW_EAP_RESPONSE)) {
00228                 hdr->data[0] = (reply->type.type & 0xFF);
00229 
00230                 /*
00231                  * Here since we cannot know the typedata format and length
00232                  *
00233                  * Type_data is expected to be wired by each EAP-Type
00234                  *
00235                  * Zero length/No typedata is supported as long as
00236                  * type is defined
00237                  */
00238                 if (reply->type.data && reply->type.length > 0) {
00239                         memcpy(&hdr->data[1], reply->type.data, reply->type.length);
00240                         free(reply->type.data);
00241                         reply->type.data = reply->packet + EAP_HEADER_LEN + 1/*EAPtype*/;
00242                 }
00243         }
00244 
00245         return EAP_VALID;
00246 }

00249 /*
00250  *      compose EAP reply packet in EAP-Message attr of RADIUS.  If
00251  *      EAP exceeds 253, frame it in multiple EAP-Message attrs.
00252  */
00253 int eap_basic_compose(RADIUS_PACKET *packet, EAP_PACKET *reply)
00254 {
00255         VALUE_PAIR *vp;
00256         eap_packet_t *eap_packet;
00257         int rcode;
00258 
00259         if (eap_wireformat(reply) == EAP_INVALID) {
00260                 return RLM_MODULE_INVALID;
00261         }
00262         eap_packet = (eap_packet_t *)reply->packet;
00263 
00264         pairdelete(&(packet->vps), PW_EAP_MESSAGE, 0);
00265 
00266         vp = eap_packet2vp(eap_packet);
00267         if (!vp) return RLM_MODULE_INVALID;
00268         pairadd(&(packet->vps), vp);
00269 
00270         /*
00271          *      EAP-Message is always associated with
00272          *      Message-Authenticator but not vice-versa.
00273          *
00274          *      Don't add a Message-Authenticator if it's already
00275          *      there.
00276          */
00277         vp = pairfind(packet->vps, PW_MESSAGE_AUTHENTICATOR, 0);
00278         if (!vp) {
00279                 vp = paircreate(PW_MESSAGE_AUTHENTICATOR, 0, PW_TYPE_OCTETS);
00280                 memset(vp->vp_strvalue, 0, AUTH_VECTOR_LEN);
00281                 vp->length = AUTH_VECTOR_LEN;
00282                 pairadd(&(packet->vps), vp);
00283         }
00284 
00285         /* Set request reply code, but only if it's not already set. */
00286         rcode = RLM_MODULE_OK;
00287         if (!packet->code) switch(reply->code) {
00288         case PW_EAP_RESPONSE:
00289         case PW_EAP_SUCCESS:
00290                 packet->code = PW_AUTHENTICATION_ACK;
00291                 rcode = RLM_MODULE_HANDLED;
00292                 break;
00293         case PW_EAP_FAILURE:
00294                 packet->code = PW_AUTHENTICATION_REJECT;
00295                 rcode = RLM_MODULE_REJECT;
00296                 break;
00297         case PW_EAP_REQUEST:
00298                 packet->code = PW_ACCESS_CHALLENGE;
00299                 rcode = RLM_MODULE_HANDLED;
00300                 break;
00301         default:
00302                 /* Should never enter here */
00303                 radlog(L_ERR, "rlm_eap: reply code %d is unknown, Rejecting the request.", reply->code);
00304                 packet->code = PW_AUTHENTICATION_REJECT;

00347 /*
00348  * Handles multiple EAP-Message attrs
00349  * ie concatenates all to get the complete EAP packet.
00350  *
00351  * NOTE: Sometimes Framed-MTU might contain the length of EAP-Message,
00352  *      refer fragmentation in rfc2869.
00353  */
00354 eap_packet_t *eap_vp2packet(VALUE_PAIR *vps)
00355 {
00356         VALUE_PAIR *first, *vp;
00357         eap_packet_t *eap_packet;
00358         unsigned char *ptr;
00359         uint16_t len;
00360         int total_len;
00361 
00362         /*
00363          *      Get only EAP-Message attribute list
00364          */
00365         first = pairfind(vps, PW_EAP_MESSAGE, 0);
00366         if (first == NULL) {
00367                 DEBUG("rlm_eap: EAP-Message not found");
00368                 return NULL;
00369         }
00370 
00371         /*
00372          *      Sanity check the length before doing anything.
00373          */
00374         if (first->length < 4) {
00375                 DEBUG("rlm_eap: EAP packet is too short.");
00376                 return NULL;
00377         }
00378 
00379         /*
00380          *      Get the Actual length from the EAP packet
00381          *      First EAP-Message contains the EAP packet header
00382          */
00383         memcpy(&len, first->vp_strvalue + 2, sizeof(len));
00384         len = ntohs(len);
00385 
00386         /*
00387          *      Take out even more weird things.
00388          */
00389         if (len < 4) {
00390                 DEBUG("rlm_eap: EAP packet has invalid length.");
00391                 return NULL;
00392         }
00393 
00394         /*
00395          *      Sanity check the length, BEFORE malloc'ing memory.
00396          */
00397         total_len = 0;
00398         for (vp = first; vp; vp = pairfind(vp->next, PW_EAP_MESSAGE, 0)) {
00399                 total_len += vp->length;
00400 
00401                 if (total_len > len) {
00402                         DEBUG("rlm_eap: Malformed EAP packet.  Length in packet header does not match actual length");
00403                         return NULL;
00404                 }
00405         }
00406 
00407         /*
00408          *      If the length is SMALLER, die, too.
00409          */
00410         if (total_len < len) {
00411                 DEBUG("rlm_eap: Malformed EAP packet.  Length in packet header does not match actual length");
00412                 return NULL;
00413         }
00414 
00415         /*
00416          *      Now that we know the lengths are OK, allocate memory.
00417          */
00418         eap_packet = (eap_packet_t *) malloc(len);
00419         if (eap_packet == NULL) {
00420                 radlog(L_ERR, "rlm_eap: out of memory");
00421                 return NULL;
00422         }
00423 
00424         /*
00425          *      Copy the data from EAP-Message's over to our EAP packet.
00426          */
00427         ptr = (unsigned char *)eap_packet;
00428 
00429         /* RADIUS ensures order of attrs, so just concatenate all */
00430         for (vp = first; vp; vp = pairfind(vp->next, PW_EAP_MESSAGE, 0)) {
00431                 memcpy(ptr, vp->vp_strvalue, vp->length);
00432                 ptr += vp->length;
00433         }
00434 
00435         return eap_packet;
00436 }
