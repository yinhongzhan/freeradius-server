00313 static int ippool_accounting(void *instance, REQUEST *request)
00314 {
00315         rlm_ippool_t *data = (rlm_ippool_t *)instance;
00316         datum key_datum;
00317         datum data_datum;
00318         datum save_datum;
00319         int acctstatustype = 0;
00320         int rcode;
00321         ippool_info entry;
00322         ippool_key key;
00323         int num = 0;
00324         VALUE_PAIR *vp;
00325         char str[32];
00326         uint8_t key_str[17];
00327         char hex_str[35];
00328         char xlat_str[MAX_STRING_LEN];
00329         FR_MD5_CTX md5_context;
00330 
00331 
00332         if ((vp = pairfind(request->packet->vps, PW_ACCT_STATUS_TYPE, 0)) != NULL)
00333                 acctstatustype = vp->vp_integer;
00334         else {
00335                 RDEBUG("Could not find account status type in packet. Return NOOP.");
00336                 return RLM_MODULE_NOOP;
00337         }
00338         switch(acctstatustype){
00339                 case PW_STATUS_STOP:
00340                         if (!radius_xlat(xlat_str,MAX_STRING_LEN,data->key, request, NULL, NULL)){
00341                                 RDEBUG("xlat on the 'key' directive failed");
00342                                 return RLM_MODULE_NOOP;
00343                         }
00344                         fr_MD5Init(&md5_context);
00345                         fr_MD5Update(&md5_context, (uint8_t *)xlat_str,
00346                          strlen(xlat_str));
00347                         fr_MD5Final(key_str, &md5_context);
00348                         key_str[16] = '\0';
00349                         fr_bin2hex(key_str,hex_str,16);
00350                         hex_str[32] = '\0';
00351                         RDEBUG("MD5 on 'key' directive maps to: %s",hex_str);
00352                         memcpy(key.key,key_str,16);
00353                         break;
00354                 default:
00355                         /* We don't care about any other accounting packet */
00356                         RDEBUG("This is not an Accounting-Stop. Return NOOP.");
00357 
00358                         return RLM_MODULE_NOOP;
00359         }
00360 
00361         RDEBUG("Searching for an entry for key: '%s'",xlat_str);
00362         key_datum.dptr = (char *) &key;
00363         key_datum.dsize = sizeof(ippool_key);
00364 
00365         pthread_mutex_lock(&data->op_mutex);
00366         data_datum = gdbm_fetch(data->gdbm, key_datum);
00367         if (data_datum.dptr != NULL){
00368 
00369                 /*
00370                  * If the entry was found set active to zero
00371                  */
00372                 memcpy(&entry, data_datum.dptr, sizeof(ippool_info));
00373                 free(data_datum.dptr);
00374                 RDEBUG("Deallocated entry for ip: %s",ip_ntoa(str,entry.ipaddr));
00375                 entry.active = 0;
00376                 entry.timestamp = 0;
00377                 entry.timeout = 0;
00378 
00379                 /*
00380                  * Save the reference to the entry
00381                  */
00382                 save_datum.dptr = key_datum.dptr;
00383                 save_datum.dsize = key_datum.dsize;
00384 
00385                 data_datum.dptr = (char *) &entry;
00386                 data_datum.dsize = sizeof(ippool_info);
00387 
00388                 rcode = gdbm_store(data->gdbm, key_datum, data_datum, GDBM_REPLACE);
00389                 if (rcode < 0) {
00390                         radlog(L_ERR, "rlm_ippool: Failed storing data to %s: %s",
00391                                         data->session_db, gdbm_strerror(gdbm_errno));
00392                         pthread_mutex_unlock(&data->op_mutex);
00393                         return RLM_MODULE_FAIL;
00394                 }
00395 
00396                 /*
00397                  * Decrease allocated count from the ip index
00398                  */
00399                 key_datum.dptr = (char *) &entry.ipaddr;
00400                 key_datum.dsize = sizeof(uint32_t);
00401                 data_datum = gdbm_fetch(data->ip, key_datum);
00402                 if (data_datum.dptr != NULL){
00403                         memcpy(&num, data_datum.dptr, sizeof(int));
00404                         free(data_datum.dptr);
00405                         if (num >0){
00406                                 num--;
00407                                 RDEBUG("num: %d",num);
00408                                 data_datum.dptr = (char *) &num;
00409                                 data_datum.dsize = sizeof(int);
00410                                 rcode = gdbm_store(data->ip, key_datum, data_datum, GDBM_REPLACE);
00411                                 if (rcode < 0) {
00412                                         radlog(L_ERR, "rlm_ippool: Failed storing data to %s: %s",
00413                                                         data->ip_index, gdbm_strerror(gdbm_errno));
00414                                         pthread_mutex_unlock(&data->op_mutex);
00415                                         return RLM_MODULE_FAIL;
00416                                 }
00417                                 if (num >0 && entry.extra == 1){
00418                                         /*
00419                                          * We are doing MPPP and we still have nas/port entries referencing
00420                                          * this ip. Delete this entry so that eventually we only keep one
00421                                          * reference to this ip.
00422                                          */
00423                                         gdbm_delete(data->gdbm,save_datum);
00424                                 }
00425                         }
00426                 }
00427                 pthread_mutex_unlock(&data->op_mutex);
00428         }
00429         else{
00430                 pthread_mutex_unlock(&data->op_mutex);
00431                 RDEBUG("Entry not found");
00432         }
00433 
00434         return RLM_MODULE_OK;
