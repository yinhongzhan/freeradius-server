00174                 RDEBUG("SoH - eap payload too short");

00191         if (eap_type != 0x21) {
00192                 RDEBUG("SoH - response eap type %08x is not EAP-SoH", eap_type);

00202         }

00214         uint8_t eap_type;

00216 
00217         /*
00218          *      No data, OR only 1 byte of EAP type.
00219          */

00221             ((data_len <= 1) && (data[0] != PW_EAP_IDENTITY))) {
00222                 return 0;
00223         }
00224 
00225         eap_type = *data;
00226         switch (eap_type) {
00227         case PW_EAP_IDENTITY:
00228                 if (data_len == 1) {
00229                         RDEBUG2("Identity - ");
00230                         return 1;
00231                 }
00232                 RDEBUG2("Identity - %*s",
00233                        data_len - 1, data + 1);
00234                 return 1;
00235                 break;
00236 
00237                 /*
00238                  *      If the first byte of the packet is
00239                  *      EAP-Response, and the EAP data is a TLV,
00240                  *      then it looks OK...
00241                  */
00242         case PW_EAP_RESPONSE:
00243                 if (eap_packet->data[0] == PW_EAP_TLV) {
00244                         RDEBUG2("Received EAP-TLV response.");
00245                         return 1;
00246                 }
00247                 RDEBUG2("Got something weird.");

00250

00255         case PW_EAP_MSCHAPV2:
00256         default:
00257                 RDEBUG2("EAP type %s",
00258                        eaptype_type2name(eap_type,

00277

00279         if (!vp) {
00280                 RDEBUG2("Failure in creating VP");
00281                 return NULL;
00282         }
00283 
00284         total = data_len;
00285         if (total > 249) total = 249;
00286 
00287         /*
00288          *      Hand-build an EAP packet from the crap in PEAP version 0.
00289          */
00290         vp->vp_octets[0] = PW_EAP_RESPONSE;
00291         vp->vp_octets[1] = eap_ds->response->id;
00292         vp->vp_octets[2] = (data_len + EAP_HEADER_LEN) >> 8;
00293         vp->vp_octets[3] = (data_len + EAP_HEADER_LEN) & 0xff;
00294 
00295         memcpy(vp->vp_octets + EAP_HEADER_LEN, data, total);
00296         vp->length = EAP_HEADER_LEN + total;
00297 
00298         head = vp;
00299         tail = &(vp->next);
00300         while (total < data_len) {
00301                 int vp_len;
00302 
00303 
00304                 vp = paircreate(PW_EAP_MESSAGE, 0, PW_TYPE_OCTETS);
00305                 if (!vp) {
00306                         RDEBUG2("Failure in creating VP");
00307                         pairfree(&head);
00308                         return NULL;
00309                 }
00310                 vp_len = (data_len - total);
00311                 if (vp_len > 253) vp_len = 253;
00312 
00313                 memcpy(vp->vp_octets, data + total, vp_len);
00314                 vp->length = vp_len;
00315                 
00316                 total += vp_len;
00317                 *tail = vp;
00318                 tail = &(vp->next);
00319         }
00320 
00321         return head;

00415  *      Use a reply packet to determine what to do.
00416  */
00417 static int process_reply(EAP_HANDLER *handler, tls_session_t *tls_session,
00418                          REQUEST *request, RADIUS_PACKET *reply)
00419 {
00420         int rcode = RLM_MODULE_REJECT;
00421         VALUE_PAIR *vp;
00422         peap_tunnel_t *t = tls_session->opaque;
00423 
00424         if ((debug_flag > 0) && fr_log_fp) {
00425                 RDEBUG("Got tunneled reply RADIUS code %d",
00426                        reply->code);
00427 
00428                 debug_pair_list(reply->vps);
00429         }
00430 
00431         switch (reply->code) {
00432         case PW_AUTHENTICATION_ACK:
00433                 RDEBUG2("Tunneled authentication was successful.");
00434                 t->status = PEAP_STATUS_SENT_TLV_SUCCESS;
00435                 eappeap_success(handler, tls_session);
00436                 rcode = RLM_MODULE_HANDLED;
00437 
00438                 /*
00439                  *      If we've been told to use the attributes from
00440                  *      the reply, then do so.
00441                  *
00442                  *      WARNING: This may leak information about the
00443                  *      tunneled user!
00444                  */
00445                 if (t->use_tunneled_reply) {
00446                         RDEBUG2("Saving tunneled attributes for later");
00447 
00448                         /*
00449                          *      Clean up the tunneled reply.
00450                          */
00451                         pairdelete(&reply->vps, PW_PROXY_STATE, 0);
00452                         pairdelete(&reply->vps, PW_EAP_MESSAGE, 0);
00453                         pairdelete(&reply->vps, PW_MESSAGE_AUTHENTICATOR, 0);
00454 
00455                         /*
00456                          *      Delete MPPE keys & encryption policy.  We don't
00457                          *      want these here.
00458                          */
00459                         pairdelete(&reply->vps, 7, VENDORPEC_MICROSOFT);
00460                         pairdelete(&reply->vps, 8, VENDORPEC_MICROSOFT);
00461                         pairdelete(&reply->vps, 16, VENDORPEC_MICROSOFT);
00462                         pairdelete(&reply->vps, 17, VENDORPEC_MICROSOFT);
00463 
00464                         t->accept_vps = reply->vps;
00465                         reply->vps = NULL;
00466                 }
00467                 break;
00468 
00469         case PW_AUTHENTICATION_REJECT:
00470                 RDEBUG2("Tunneled authentication was rejected.");
00471                 t->status = PEAP_STATUS_SENT_TLV_FAILURE;
00472                 eappeap_failure(handler, tls_session);
00473                 rcode = RLM_MODULE_HANDLED;
00474                 break;
00475 
00476         case PW_ACCESS_CHALLENGE:
00477                 RDEBUG2("Got tunneled Access-Challenge");
00478 
00479                 /*
00480                  *      Keep the State attribute, if necessary.
00481                  *
00482                  *      Get rid of the old State, too.
00483                  */
00484                 pairfree(&t->state);
00485                 pairmove2(&t->state, &(reply->vps), PW_STATE, 0);
00486 
00487                 /*
00488                  *      PEAP takes only EAP-Message attributes inside
00489                  *      of the tunnel.  Any Reply-Message in the
00490                  *      Access-Challenge is ignored.
00491                  */
00492                 vp = NULL;
00493                 pairmove2(&vp, &(reply->vps), PW_EAP_MESSAGE, 0);
00494 
00495                 /*
00496                  *      Handle EAP-MSCHAP-V2, where Access-Accept's
00497                  *      from the home server may contain MS-CHAP-Success,
00498                  *      which the module turns into challenges, so that
00499                  *      the client may respond to the challenge with
00500                  *      an "ack" packet.
00501                  */
00502                 if (t->home_access_accept && t->use_tunneled_reply) {
00503                         RDEBUG2("Saving tunneled attributes for later");
00504 
00505                         /*
00506                          *      Clean up the tunneled reply.
00507                          */
00508                         pairdelete(&reply->vps, PW_PROXY_STATE, 0);
00509                         pairdelete(&reply->vps, PW_MESSAGE_AUTHENTICATOR, 0);
00510 
00511                         t->accept_vps = reply->vps;
00512                         reply->vps = NULL;
00513                 }
00514 
00515                 /*
00516                  *      Handle the ACK, by tunneling any necessary reply
00517                  *      VP's back to the client.
00518                  */
00519                 if (vp) {
00520                         vp2eap(request, tls_session, vp);
00521                         pairfree(&vp);
00522                 }
00523 
00524                 rcode = RLM_MODULE_HANDLED;
00525                 break;
00526 
00527         default:
00528                 RDEBUG2("Unknown RADIUS packet type %d: rejecting tunneled user", reply->code);
00529                 rcode = RLM_MODULE_REJECT;
00530                 break;
00531         }
00532 
00533         return rcode;
00534 }

00660 
00661         case RLM_MODULE_OK:

00663

00665                  *      Success: Automatically return MPPE keys.
00666                  */

00744 
00745         rad_assert(request != NULL);
00746 
00747         /*
00748          *      Just look at the buffer directly, without doing
00749          *      record_minus.  This lets us avoid another data copy.

00754

00758             !eapmessage_verify(request, data, data_len)) {

00774                                 eappeap_soh(handler, tls_session);

00778                         t->status = PEAP_STATUS_SENT_TLV_SUCCESS;
00779                         eappeap_success(handler, tls_session);
00780                         
00781                 } else {
00782                         /* send an identity request */
00783                         t->session_resumption_state = PEAP_RESUMPTION_NO;
00784                         t->status = PEAP_STATUS_INNER_IDENTITY_REQ_SENT;
00785                         eappeap_identity(handler, tls_session);

00787                 return RLM_MODULE_HANDLED;
00788 
00789         case PEAP_STATUS_INNER_IDENTITY_REQ_SENT:
00790                 /* we're expecting an identity response */
00791                 if (data[0] != PW_EAP_IDENTITY) {
00792                         RDEBUG("Expected EAP-Identity, got something else.");
00793                         return RLM_MODULE_REJECT;
00794                 }

00798                         return RLM_MODULE_REJECT;
00799                 }

00802                  *      Save it for later.
00803                  */

00808                 t->username->length = data_len - 1;
00809                 t->username->vp_strvalue[t->username->length] = 0;

00811                 if (t->soh) {
00812                         t->status = PEAP_STATUS_WAIT_FOR_SOH_RESPONSE;
00813                         RDEBUG2("Requesting SoH from client");
00814                         eappeap_soh(handler, tls_session);
00815                         return RLM_MODULE_HANDLED;
00816                 }
00817                 t->status = PEAP_STATUS_PHASE2_INIT;
00818                 break;
00819 
00820         case PEAP_STATUS_WAIT_FOR_SOH_RESPONSE:
00821                 fake = request_alloc_fake(request);
00822                 rad_assert(fake->packet->vps == NULL);
00823                 fake->packet->vps = eapsoh_verify(request, data, data_len);
00824                 setup_fake_request(request, fake, t);
00825 
00826                 if (t->soh_virtual_server) {
00827                         fake->server = t->soh_virtual_server;
00828                 }
00829                 RDEBUG("Sending SoH request to server %s", fake->server ? fake->server : "NULL");
00830                 debug_pair_list(fake->packet->vps);
00831                 RDEBUG("server %s {", fake->server);
00832                 rad_virtual_server(fake);
00833                 RDEBUG("} # server %s", fake->server);
00834                 RDEBUG("Got SoH reply");
00835                 debug_pair_list(fake->reply->vps);
00836 
00837                 if (fake->reply->code != PW_AUTHENTICATION_ACK) {
00838                         RDEBUG2("SoH was rejected");
00839                         request_free(&fake);
00840                         t->status = PEAP_STATUS_SENT_TLV_FAILURE;
00841                         eappeap_failure(handler, tls_session);
00842                         return RLM_MODULE_HANDLED;
00843                 }
00844 
00845                 /* save the SoH VPs */
00846                 t->soh_reply_vps = fake->reply->vps;
00847                 fake->reply->vps = NULL;
00848                 request_free(&fake);
00849 
00850                 if (t->session_resumption_state == PEAP_RESUMPTION_YES) {
00851                         /* we're good, send success TLV */
00852                         t->status = PEAP_STATUS_SENT_TLV_SUCCESS;

00854                         return RLM_MODULE_HANDLED;
00855                 }

00857                 t->status = PEAP_STATUS_PHASE2_INIT;
00858                 break;

00862          *      If we authenticated the user, then it's OK.
00863          */
00864         case PEAP_STATUS_SENT_TLV_SUCCESS:

00866                         RDEBUG2("Success");
00867                         return RLM_MODULE_OK;
00868                 }

00872                  *      resumption.  If the session is being re-used,
00873                  *      we need to do a full authentication.

00876                  *      inside of the PEAP tunnel.
00877                  */
00878                 if (t->session_resumption_state == PEAP_RESUMPTION_YES) {
00879                         RDEBUG2("Client rejected session resumption.  Re-starting full authentication");
00880

00883                          */
00884                         t->status = PEAP_STATUS_INNER_IDENTITY_REQ_SENT;
00885                         t->session_resumption_state = PEAP_RESUMPTION_NO;
00886                         
00887                         eappeap_identity(handler, tls_session);
00888                         return RLM_MODULE_HANDLED;
00889                 }
00890

00892                 return RLM_MODULE_REJECT;

00895          *      Damned if I know why the clients continue sending EAP

00897          */

00902                 RDEBUG(" *** Look for \"reject\" or \"fail\".  Those earlier messages will tell you.");

00938                 vp->vp_octets[3] = len & 0xff;

00940 
00941                 if (len > sizeof(vp->vp_octets)) len = sizeof(vp->vp_octets);
00942                 memcpy(vp->vp_octets + EAP_HEADER_LEN + 1,
00943                        t->username->vp_strvalue, len - EAP_HEADER_LEN - 1);
00944                 vp->length = len;
00945 
00946                 pairadd(&fake->packet->vps, vp);

00949                         RDEBUG2("Setting default EAP type for tunneled EAP session.");

00952                         pairadd(&fake->config_items, vp);

00958                 if (!fake->packet->vps) {
00959                         request_free(&fake);

00961                         return PW_AUTHENTICATION_REJECT;
00962                 }
00963                 break;
00964 
00965         default:
00966                 RDEBUG("Invalid state change in PEAP.");
00967                 return PW_AUTHENTICATION_REJECT;
00968         }
00969 
00970         if ((debug_flag > 0) && fr_log_fp) {
00971                 RDEBUG("Got tunneled request");
00972

00986                  *      EAP-Identity packet.
00987                  */
00988                 if ((data[0] == PW_EAP_IDENTITY) && (data_len > 1)) {
00989                         t->username = pairmake("User-Name", "", T_OP_EQ);
00990                         rad_assert(t->username != NULL);
00991 
00992                         memcpy(t->username->vp_strvalue, data + 1, data_len - 1);
00993                         t->username->length = data_len - 1;
00994                         t->username->vp_strvalue[t->username->length] = 0;

00996 
00997                         /*
00998                          *      If there's a default EAP type,
00999                          *      set it here.
01000                          */
01001                         if (t->default_eap_type != 0) {
01002                                 DEBUG2("  PEAP: Setting default EAP type for tunneled EAP session.");
01003                                 vp = pairmake("EAP-Type", "0", T_OP_EQ);
01004                                 vp->vp_integer = t->default_eap_type;
01005                                 pairadd(&fake->config_items, vp);
01006                         }
01007                 }

01011

01017

01020         if ((debug_flag > 0) && fr_log_fp) {
01021                 fprintf(fr_log_fp, "Sending tunneled request\n");
01022 
01023                 debug_pair_list(fake->packet->vps);
01024 
01025                 fprintf(fr_log_fp, "server %s {\n",
01026                         (fake->server == NULL) ? "" : fake->server);
01027         }
01028 
01029         /*
01030          *      Call authentication recursively, which will
01031          *      do PAP, CHAP, MS-CHAP, etc.
01032          */
01033         rad_virtual_server(fake);
01034 
01035         /*
01036          *      Note that we don't do *anything* with the reply
01037          *      attributes.
01038          */
01039         if ((debug_flag > 0) && fr_log_fp) {
01040                 fprintf(fr_log_fp, "} # server %s\n",
01041                         (fake->server == NULL) ? "" : fake->server);

01043                 RDEBUG("Got tunneled reply code %d", fake->reply->code);

01081 
01082

01085                                  */

01109                                 }

01113                                  *      EAP-Message into another set
01114                                  *      of attributes.
01115                                  */

01117                                            PW_EAP_MESSAGE, 0);

01175                                  *      So we associate the fake request with
01176                                  *      this request.
01177                                  */
01178                                 rcode = request_data_add(request,
01179                                                          request->proxy,
01180                                                          REQUEST_DATA_EAP_MSCHAP_TUNNEL_CALLBACK,

01183 
01184                                 /*
01185                                  *      Do NOT free the fake request!
01186                                  */

01189 
01190                         /*
01191                          *      Didn't authenticate the packet, but
01192                          *      we're proxying it.
01193                          */
01194                         rcode = RLM_MODULE_UPDATED;
01195 
01196                 } else
01197 #endif  /* WITH_PROXY */
01198                   {
01199                         DEBUG2("  PEAP: Unknown RADIUS packet type %d: rejecting tunneled user", fake->reply->code);
01200                         rcode = RLM_MODULE_REJECT;
01201                   }
01202                 break;
01203 
01204         default:
01205         do_process:
01206                 rcode = process_reply(handler, tls_session, request,
01207                                       fake->reply);
01208                 break;
01209         }
01210 
01211  done:
01212         request_free(&fake);
01213 
01214         return rcode;
01215 }
