00069 /*
00070  *      Decode the configuration file string to a pointer to
00071  *      a value-pair list in the REQUEST data structure.
00072  */
00073 static VALUE_PAIR **decode_string(REQUEST *request, const char *string)
00074 {
00075         if (!string) return NULL;
00076 
00077         /*
00078          *      Yuck.  We need a 'switch' over character strings
00079          *      in C.
00080          */
00081         if (strcmp(string, "request") == 0) {
00082                 return &request->packet->vps;
00083         }
00084 
00085         if (strcmp(string, "reply") == 0) {
00086                 if (!request->reply) return NULL;
00087 
00088                 return &request->reply->vps;
00089         }
00090 
00091 #ifdef WITH_PROXY
00092         if (strcmp(string, "proxy-request") == 0) {
00093                 if (!request->proxy) return NULL;
00094 
00095                 return &request->proxy->vps;
00096         }
00097 
00098         if (strcmp(string, "proxy-reply") == 0) {
00099                 if (!request->proxy_reply) return NULL;
00100 
00101                 return &request->proxy_reply->vps;
00102         }
00103 #endif
00104 
00105         if (strcmp(string, "config") == 0) {
00106                 return &request->config_items;
00107         }
00108 
00109         if (strcmp(string, "none") == 0) {
00110                 return NULL;
00111         }
00112 
00113         return NULL;
00114 }

00260 /*
00261  *  Dispatch an exec method
00262  */
00263 static int exec_dispatch(void *instance, REQUEST *request)
00264 {
00265         int result;
00266         VALUE_PAIR **input_pairs, **output_pairs;
00267         VALUE_PAIR *answer = NULL;
00268         rlm_exec_t *inst = (rlm_exec_t *) instance;
00269 
00270         /*
00271          *      We need a program to execute.
00272          */
00273         if (!inst->program) {
00274                 radlog(L_ERR, "rlm_exec (%s): We require a program to execute",
00275                        inst->xlat_name);
00276                 return RLM_MODULE_FAIL;
00277         }
00278 
00279         /*
00280          *      See if we're supposed to execute it now.
00281          */
00282         if (!((inst->packet_code == 0) ||
00283               (request->packet->code == inst->packet_code) ||
00284               (request->reply->code == inst->packet_code)
00285 #ifdef WITH_PROXY
00286               || (request->proxy &&
00287                (request->proxy->code == inst->packet_code)) ||
00288               (request->proxy_reply &&
00289                (request->proxy_reply->code == inst->packet_code))
00290 #endif
00291                     )) {
00292                 RDEBUG2("Packet type is not %s.  Not executing.",
00293                        inst->packet_type);
00294                 return RLM_MODULE_NOOP;
00295         }
00296 
00297         /*
00298          *      Decide what input/output the program takes.
00299          */
00300         input_pairs = decode_string(request, inst->input);
00301         output_pairs = decode_string(request, inst->output);
00302 
00303         if (!input_pairs) {
00304                 RDEBUG2("WARNING: Possible parse error in %s",
00305                         inst->input);
00306                 return RLM_MODULE_NOOP;
00307         }
00308 
00309         /*
00310          *      It points to the attribute list, but the attribute
00311          *      list is empty.
00312          */
00313         if (!*input_pairs) {
00314                 RDEBUG2("WARNING! Input pairs are empty.  No attributes will be passed to the script");
00315         }
00316 
00317         /*
00318          *      This function does it's own xlat of the input program
00319          *      to execute.
00320          *
00321          *      FIXME: if inst->program starts with %{, then
00322          *      do an xlat ourselves.  This will allow us to do
00323          *      program = %{Exec-Program}, which this module
00324          *      xlat's into it's string value, and then the
00325          *      exec program function xlat's it's string value
00326          *      into something else.
00327          */
00328         result = radius_exec_program(inst->program, request,
00329                                      inst->wait, NULL, 0,
00330                                      *input_pairs, &answer, inst->shell_escape);
00331         if (result < 0) {
00332                 radlog(L_ERR, "rlm_exec (%s): External script failed",
00333                        inst->xlat_name);
00334                 return RLM_MODULE_FAIL;
00335         }
00336 
00337         /*
00338          *      Move the answer over to the output pairs.
00339          *
00340          *      If we're not waiting, then there are no output pairs.
00341          */
00342         if (output_pairs) pairmove(output_pairs, &answer);
00343 
00344         pairfree(&answer);
00345 
00346         if (result == 0) {
00347                 return RLM_MODULE_OK;
00348         }
00349         if (result > RLM_MODULE_NUMCODES) {
00350                 return RLM_MODULE_FAIL;
00351         }
00352         return result-1;
00353 }

00356 /*
00357  *      First, look for Exec-Program && Exec-Program-Wait.
00358  *
00359  *      Then, call exec_dispatch.
00360  */
00361 static int exec_postauth(void *instance, REQUEST *request)
00362 {
00363         int result;
00364         int exec_wait = 0;
00365         VALUE_PAIR *vp, *tmp;
00366         rlm_exec_t *inst = (rlm_exec_t *) instance;
00367 
00368         vp = pairfind(request->reply->vps, PW_EXEC_PROGRAM, 0);
00369         if (vp) {
00370                 exec_wait = 0;
00371 
00372         } else if ((vp = pairfind(request->reply->vps, PW_EXEC_PROGRAM_WAIT, 0)) != NULL) {
00373                 exec_wait = 1;
00374         }
00375         if (!vp) {
00376                 if (!inst->program) return RLM_MODULE_NOOP;
00377                 
00378                 return exec_dispatch(instance, request);
00379         }
00380 
00381         tmp = NULL;
00382         result = radius_exec_program(vp->vp_strvalue, request, exec_wait,
00383                                      NULL, 0, request->packet->vps, &tmp,
00384                                      inst->shell_escape);
00385 
00386         /*
00387          *      Always add the value-pairs to the reply.
00388          */
00389         pairmove(&request->reply->vps, &tmp);
00390         pairfree(&tmp);
00391 
00392         if (result < 0) {
00393                 /*
00394                  *      Error. radius_exec_program() returns -1 on
00395                  *      fork/exec errors.
00396                  */
00397                 tmp = pairmake("Reply-Message", "Access denied (external check failed)", T_OP_SET);
00398                 pairadd(&request->reply->vps, tmp);
00399 
00400                 RDEBUG2("Login incorrect (external check failed)");
00401 
00402                 request->reply->code = PW_AUTHENTICATION_REJECT;
00403                 return RLM_MODULE_REJECT;
00404         }
00405         if (result > 0) {
00406                 /*
00407                  *      Reject. radius_exec_program() returns >0
00408                  *      if the exec'ed program had a non-zero
00409                  *      exit status.
00410                  */
00411                 request->reply->code = PW_AUTHENTICATION_REJECT;
00412                 RDEBUG2("Login incorrect (external check said so)");
00413                 return RLM_MODULE_REJECT;
00414         }
00415 
00416         return RLM_MODULE_OK;
00417 }

00447                                      NULL, 0, request->packet->vps, NULL,

00450                 return RLM_MODULE_REJECT;
