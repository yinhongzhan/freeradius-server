00870  *      if (...) {...}
00871  *      if (...) {...} else {...}
00872  *      if (...) {...} else if ...
00873  */
00874 static int parse_if(policy_lex_file_t *lexer, policy_item_t **tail)
00875 {
00876         int rcode;
00877         policy_lex_t token;
00878         char mystring[256];

00880 
00881         debug_tokens("[IF] ");
00882 
00883         this = rad_malloc(sizeof(*this));
00884         memset(this, 0, sizeof(*this));
00885 
00886         this->item.type = POLICY_TYPE_IF;
00887         this->item.lineno = lexer->lineno;
00888 
00889         rcode = parse_condition(lexer, &(this->condition));
00890         if (!rcode) {
00891                 rlm_policy_free_item((policy_item_t *) this);
00892                 return rcode;
00893         }
00894 
00895         rcode = parse_block(lexer, &(this->if_true));
00896         if (!rcode) {
00897                 rlm_policy_free_item((policy_item_t *) this);
00898                 return rcode;
00899         }
00900 
00901         token = policy_lex_file(lexer, POLICY_LEX_FLAG_PEEK,
00902                                 mystring, sizeof(mystring));
00903         if ((token == POLICY_LEX_BARE_WORD) &&
00904             (fr_str2int(policy_reserved_words, mystring,
00905                           POLICY_RESERVED_UNKNOWN) == POLICY_RESERVED_ELSE)) {
00906                 debug_tokens("[ELSE] ");

00908                 rad_assert(token == POLICY_LEX_BARE_WORD);
00909

00911                                         mystring, sizeof(mystring));

00913                     (fr_str2int(policy_reserved_words, mystring,
00914                                   POLICY_RESERVED_UNKNOWN) == POLICY_RESERVED_IF)) {
00915                         token = policy_lex_file(lexer, 0,
00916                                                 mystring, sizeof(mystring));
00917                         rad_assert(token == POLICY_LEX_BARE_WORD);
00918                         rcode = parse_if(lexer, &(this->if_false));
00919                 } else {
00920                         rcode = parse_block(lexer, &(this->if_false));
00921                 }
00922                 if (!rcode) {
00923                         rlm_policy_free_item((policy_item_t *) this);
00924                         return rcode;

00934         if (!this->if_true && !this->if_false) {

00936                              this->item.lineno);
00937                 rlm_policy_free_item((policy_item_t *) this);
00938                 return 1;
00939         }
00940 
00941         *tail = (policy_item_t *) this;
00942 
00943         return 1;
00944 }

01199 /*
01200  *      Parse one statement.  'foo = bar', or 'if (...) {...}', or '{...}',
01201  *      and so on.
01202  */
01203 static int parse_statement(policy_lex_file_t *lexer, policy_item_t **tail)
01204 {
01205         int rcode;
01206         policy_reserved_word_t reserved;
01207         policy_lex_t token, assign;
01208         char lhs[256], rhs[256];
01209         policy_assignment_t *this;
01210 
01211         /*
01212          *      See what kind of token we have.
01213          */
01214         token = policy_lex_file(lexer, 0, lhs, sizeof(lhs));
01215         switch (token) {
01216         case POLICY_LEX_LC_BRACKET:
01217                 rcode = parse_block(lexer, tail);
01218                 if (!rcode) {
01219                         return 0;
01220                 }
01221                 break;
01222 
01223         case POLICY_LEX_BARE_WORD:
01224                 reserved = fr_str2int(policy_reserved_words,
01225                                         lhs,
01226                                         POLICY_RESERVED_UNKNOWN);
01227                 switch (reserved) {
01228                 case POLICY_RESERVED_IF:
01229                         if (parse_if(lexer, tail)) {
01230                                 return 1;
01231                         }
01232                         return 0;
01233                         break;
01234 
01235                 case POLICY_RESERVED_CONTROL:
01236                 case POLICY_RESERVED_REQUEST:
01237                 case POLICY_RESERVED_REPLY:
01238                 case POLICY_RESERVED_PROXY_REQUEST:
01239                 case POLICY_RESERVED_PROXY_REPLY:
01240                         if (parse_attribute_block(lexer, tail,
01241                                                   reserved))
01242                                 return 1;
01243                         return 0;
01244                         break;
01245 
01246                 case POLICY_RESERVED_PRINT:
01247                         if (parse_print(lexer, tail)) {
01248                                 return 1;
01249                         }
01250                         return 0;
01251                         break;
01252 
01253                 case POLICY_RESERVED_RETURN:
01254                         if (parse_return(lexer, tail)) {
01255                                 return 1;
01256                         }
01257                         return 0;
01258                         break;
01259 
01260                 case POLICY_RESERVED_MODULE:
01261                         if (parse_module(lexer, tail)) {
01262                                 return 1;
01263                         }
01264                         return 0;
01265                         break;
01266 
01267                 case POLICY_RESERVED_UNKNOWN: /* wasn't a reserved word */
01268                         /*
01269                          *      Is a named policy, parse the reference to it.
01270                          */
01271                         if (rlm_policy_find(lexer->policies, lhs) != NULL) {
01272                                 if (!parse_call(lexer, tail, lhs)) {
01273                                         return 0;
01274                                 }
01275                                 return 1;
01276                         }
01277 
01278                         {
01279                                 const DICT_ATTR *dattr;
01280 
01281                                 /*
01282                                  *      Bare words MUST be dictionary attributes
01283                                  */
01284 
01285                                 dattr = dict_attrbyname(lhs);
01286                                 if (!dattr) {
01287                                         fprintf(stderr, "%s[%d]: Expected attribute name, got \"%s\"\n",
01288                                                 lexer->filename, lexer->lineno, lhs);
01289                                         return 0;
