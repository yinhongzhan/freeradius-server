00341 /*
00342  * We are the server, we always get the dirty data
00343  * (Handshake data is also considered as dirty data)
00344  * During handshake, since SSL API handles itself,
00345  * After clean-up, dirty_out will be filled with
00346  * the data required for handshaking. So we check
00347  * if dirty_out is empty then we simply send it back.
00348  * As of now, if handshake is successful, then we keep going,
00349  * otherwise we fail.
00350  *
00351  * Fill the Bio with the dirty data to clean it
00352  * Get the cleaned data from SSL, if it is not Handshake data
00353  */

00355 {

00357 
00358         err = BIO_write(ssn->into_ssl, ssn->dirty_in.data, ssn->dirty_in.used);
00359         if (err != (int) ssn->dirty_in.used) {
00360                 RDEBUG("Failed writing %d to SSL BIO: %d", ssn->dirty_in.used,
00361                         err);
00362                 record_init(&ssn->dirty_in);
00363                 return 0;
00364         }
00365         record_init(&ssn->dirty_in);
00366 
00367         err = SSL_read(ssn->ssl, ssn->clean_out.data + ssn->clean_out.used,
00368                        sizeof(ssn->clean_out.data) - ssn->clean_out.used);
00369         if (err > 0) {
00370                 ssn->clean_out.used += err;
00371                 return 1;
00372         }
00373 
00374         if (!int_ssl_check(request, ssn->ssl, err, "SSL_read")) {
00375                 return 0;
00376         }
00377

00379         if (SSL_is_init_finished(ssn->ssl)) {
00380                 DEBUG2("SSL Connection Established\n");

00384         }

00400                         ssn->dirty_out.used = err;
00401 
00402                 } else if (BIO_should_retry(ssn->from_ssl)) {
00403                         record_init(&ssn->dirty_in);
00404                         DEBUG2("  tls: Asking for more data in tunnel");

00414                 /* Its clean application data, do whatever we want */
00415                 record_init(&ssn->clean_out);
00416         }
00417

00419         record_init(&ssn->dirty_in);

01020 {

01025

01028

01032

01036                 char filename[256];

01038 
01039                 /* find out what length data we need */
01040                 blob_len = i2d_SSL_SESSION(sess, NULL);
01041                 if (blob_len < 1) {
01042                         /* something went wrong */
01043                         DEBUG2("  SSL: could not find buffer length to persist session");
01044                         return 0;
01045                 }
01046 
01047                 /* alloc and convert to ASN.1 */
01048                 sess_blob = malloc(blob_len);
01049                 if (!sess_blob) {
01050                         DEBUG2("  SSL: could not allocate buffer len=%d to persist session", blob_len);
01051                         return 0;
01052                 }
01053                 /* openssl mutates &p */
01054                 p = sess_blob;
01055                 rv = i2d_SSL_SESSION(sess, &p);
01056                 if (rv != blob_len) {
01057                         DEBUG2("  SSL: could not persist session");
01058                         goto error;
01059                 }
01060 
01061                 /* open output file */
01062                 rv = snprintf(filename, sizeof(filename), "%s%c%s.asn1",
01063                         conf->session_cache_path, FR_DIR_SEP, buffer
01064                         );
01065                 fd = open(filename, O_RDWR|O_CREAT|O_EXCL, 0600);
01066                 if (fd < 0) {
01067                         DEBUG2("  SSL: could not open session file %s: %s", filename, strerror(errno));
01068                         goto error;
01069                 }
01070 
01071                 todo = blob_len;
01072                 p = sess_blob;
01073                 while (todo > 0) {
01074                         rv = write(fd, p, todo);
01075                         if (rv < 1) {
01076                                 DEBUG2("  SSL: failed writing session: %s", strerror(errno));
01077                                 close(fd);
01078                                 goto error;
01079                         }
01080                         p += rv;
01081                         todo -= rv;
01082                 }
01083                 close(fd);
01084                 DEBUG2("  SSL: wrote session %s to %s len=%d", buffer, filename, blob_len);
01085         }
01086

01089 
01090         return 0;

01107

01109

01111 
01112         conf = (fr_tls_server_conf_t *)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CONF);

01114                 int rv, fd, todo;
01115                 char filename[256];

01117                 struct stat st;
01118                 VALUE_PAIR *vp;
01119 
01120                 /* read in the cached VPs from the .vps file */
01121                 rv = snprintf(filename, sizeof(filename), "%s%c%s.vps",
01122                         conf->session_cache_path, FR_DIR_SEP, buffer

01128                         goto err;
01129                 }

01132                 rv = snprintf(filename, sizeof(filename), "%s%c%s.asn1",
01133                         conf->session_cache_path, FR_DIR_SEP, buffer
01134                         );

01137                         DEBUG2("  SSL: could not find persisted session file %s: %s", filename, strerror(errno));

01139                 }

01142                 if (rv == -1) {
01143                         DEBUG2("  SSL: could not stat persisted session file %s: %s", filename, strerror(errno));
01144                         close(fd);
01145                         goto err;
01146                 }
01147 
01148                 sess_data = malloc(st.st_size);
01149                 if (!sess_data) {
01150                         DEBUG2("  SSL: could not alloc buffer for persisted session len=%d", st.st_size);
01151                         close(fd);
01152                         goto err;
01153                 }
01154 
01155                 p = sess_data;
01156                 todo = st.st_size;
01157                 while (todo > 0) {
01158                         rv = read(fd, p, todo);
01159                         if (rv < 1) {
01160                                 DEBUG2("  SSL: could not read from persisted session: %s", strerror(errno));
01161                                 close(fd);
01162                                 goto err;
01163                         }
01164                         todo -= rv;
01165                         p += rv;
01166                 }
01167                 close(fd);

01240         int ocsp_ok = 0;
01241         int status ;
01242         ASN1_GENERALIZEDTIME *rev, *thisupd, *nextupd;
01243         int reason;
01244 #if OPENSSL_VERSION_NUMBER >= 0x1000003f
01245         OCSP_REQ_CTX *ctx;
01246         int rc;
01247         struct timeval now;
01248         struct timeval when;
01249 #endif
01250 
01251         /*
01252          * Create OCSP Request
01253          */

01263          */

01271         }

01274                 DEBUG2("[ocsp] - Host / port / path missing.  Not doing OCSP.");

01279         DEBUG2("[ocsp] --> Responder URL = http://%s:%s%s", host, port, path);

01296         }
01297 #else

01311                 ocsp_ok = 2;
01312                 goto ocsp_end;
01313         }

01316         when.tv_sec += conf->ocsp_timeout;

01320                 if (conf->ocsp_timeout) {
01321                         gettimeofday(&now, NULL);
01322                         if (!timercmp(&now, &when, <))
01323                                 break;
01324                 }
01325         } while ((rc == -1) && BIO_should_retry(cbio));
01326 
01327         if (conf->ocsp_timeout && (rc == -1) && BIO_should_retry(cbio)) {
01328                 radlog(L_ERR, "Error: OCSP response timed out");
01329                 ocsp_ok = 2;
01330                 goto ocsp_end;
01331         }
01332 
01333         OCSP_REQ_CTX_free(ctx);
01334 
01335         if (rc == 0) {
01336                 radlog(L_ERR, "Error: Couldn't get OCSP response");
01337                 ocsp_ok = 2;
01338                 goto ocsp_end;
01339         }
01340 #endif
01341 
01342         /* Verify OCSP response status */
01343         status = OCSP_response_status(resp);
01344         DEBUG2("[ocsp] --> Response status: %s",OCSP_response_status_str(status));
01345         if(status != OCSP_RESPONSE_STATUS_SUCCESSFUL) {
01346                 radlog(L_ERR, "Error: OCSP response status: %s", OCSP_response_status_str(status));
01347                 goto ocsp_end;

01350         if(conf->ocsp_use_nonce && OCSP_check_nonce(req, bresp)!=1) {
01351                 radlog(L_ERR, "Error: OCSP response has wrong nonce value");
01352                 goto ocsp_end;
01353         }

01355                 radlog(L_ERR, "Error: Couldn't verify OCSP basic response");
01356                 goto ocsp_end;
01357         }

01361                                                       &rev, &thisupd, &nextupd)) {
01362                 radlog(L_ERR, "ERROR: No Status found.\n");
01363                 goto ocsp_end;
01364         }
01365 
01366         if (!OCSP_check_validity(thisupd, nextupd, nsec, maxage)) {
01367                 BIO_puts(bio_out, "WARNING: Status times invalid.\n");
01368                 ERR_print_errors(bio_out);
01369                 goto ocsp_end;
01370         }
01371         BIO_puts(bio_out, "\tThis Update: ");
01372         ASN1_GENERALIZEDTIME_print(bio_out, thisupd);
01373         BIO_puts(bio_out, "\n");

01375                 BIO_puts(bio_out, "\tNext Update: ");
01376                 ASN1_GENERALIZEDTIME_print(bio_out, nextupd);
01377                 BIO_puts(bio_out, "\n");
01378         }
01379

01381         case V_OCSP_CERTSTATUS_GOOD:
01382                 DEBUG2("[oscp] --> Cert status: good");
01383                 ocsp_ok = 1;
01384                 break;
01385 
01386         default:
01387                 /* REVOKED / UNKNOWN */
01388                 DEBUG2("[ocsp] --> Cert status: %s",OCSP_cert_status_str(status));
01389                 if (reason != -1)
01390                         DEBUG2("[ocsp] --> Reason: %s", OCSP_crl_reason_str(reason));
01391                 BIO_puts(bio_out, "\tRevocation Time: ");
01392                 ASN1_GENERALIZEDTIME_print(bio_out, rev);
01393                 BIO_puts(bio_out, "\n");
01394                 break;
01395         }
01396 
01397 ocsp_end:
01398         /* Free OCSP Stuff */
01399         OCSP_REQUEST_free(req);
01400         OCSP_RESPONSE_free(resp);
01401         free(host);
01402         free(port);
01403         free(path);
01404         BIO_free_all(cbio);
01405         OCSP_BASICRESP_free(bresp);
01406 
01407  ocsp_skip:
01408         switch (ocsp_ok) {
01409         case 1:
01410                 DEBUG2("[ocsp] --> Certificate is valid!");
01411                 break;
01412         case 2:
01413                 if (conf->ocsp_softfail) {
01414                         DEBUG2("[ocsp] --> Unable to check certificate; assuming valid.");
01415                         DEBUG2("[ocsp] --> Warning! This may be insecure.");
01416                         ocsp_ok = 1;
01417                 } else {
01418                         DEBUG2("[ocsp] --> Unable to check certificate; failing!");
01419                         ocsp_ok = 0;
01420                 }
01421                 break;
01422         default:
01423                 DEBUG2("[ocsp] --> Certificate has been expired/revoked!");
01424                 break;
01425         }
01426 
01427         return ocsp_ok;
01428 }

01450 /*
01451  *      Before trusting a certificate, you must make sure that the
01452  *      certificate is 'valid'. There are several steps that your
01453  *      application can take in determining if a certificate is
01454  *      valid. Commonly used steps are:
01455  *
01456  *      1.Verifying the certificate's signature, and verifying that
01457  *      the certificate has been issued by a trusted Certificate
01458  *      Authority.
01459  *
01460  *      2.Verifying that the certificate is valid for the present date
01461  *      (i.e. it is being presented within its validity dates).
01462  *
01463  *      3.Verifying that the certificate has not been revoked by its
01464  *      issuing Certificate Authority, by checking with respect to a
01465  *      Certificate Revocation List (CRL).
01466  *
01467  *      4.Verifying that the credentials presented by the certificate
01468  *      fulfill additional requirements specific to the application,
01469  *      such as with respect to access control lists or with respect
01470  *      to OCSP (Online Certificate Status Processing).
01471  *
01472  *      NOTE: This callback will be called multiple times based on the
01473  *      depth of the root certificate chain
01474  */
01475 int cbtls_verify(int ok, X509_STORE_CTX *ctx)
01476 {
01477         char subject[1024]; /* Used for the subject name */
01478         char issuer[1024]; /* Used for the issuer name */
01479         char common_name[1024];
01480         char cn_str[1024];

01483         SSL *ssl;
01484         int err, depth, lookup, loc;
01485         fr_tls_server_conf_t *conf;
01486         int my_ok = ok;
01487         REQUEST *request;
01488         ASN1_INTEGER *sn = NULL;
01489         ASN1_TIME *asn_time = NULL;
01490         VALUE_PAIR **certs;
01491         char **identity;
01492 #ifdef HAVE_OPENSSL_OCSP_H
01493         X509_STORE *ocsp_store = NULL;
01494         X509 *issuer_cert;
01495 #endif
01496 
01497         client_cert = X509_STORE_CTX_get_current_cert(ctx);
01498         err = X509_STORE_CTX_get_error(ctx);
01499         depth = X509_STORE_CTX_get_error_depth(ctx);
01500 
01501         lookup = depth;
01502 
01503         /*
01504          *      Log client/issuing cert.  If there's an error, log
01505          *      issuing cert.
01506          */
01507         if ((lookup > 1) && !my_ok) lookup = 1;
01508 
01509         /*
01510          * Retrieve the pointer to the SSL of the connection currently treated
01511          * and the application specific data stored into the SSL object.
01512          */
01513         ssl = X509_STORE_CTX_get_ex_data(ctx, SSL_get_ex_data_X509_STORE_CTX_idx());
01514         conf = (fr_tls_server_conf_t *)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CONF);
01515         if (!conf) return 1;
01516 
01517         request = (REQUEST *)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST);
01518 
01519         if (!request) return 1; /* FIXME: outbound TLS */
01520 
01521         rad_assert(request != NULL);
01522         certs = (VALUE_PAIR **)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CERTS);
01523         rad_assert(certs != NULL);
01524         identity = (char **)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_IDENTITY);
01525 #ifdef HAVE_OPENSSL_OCSP_H
01526         ocsp_store = (X509_STORE *)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_STORE);
01527 #endif
01528 
01529 
01530         /*
01531          *      Get the Serial Number
01532          */
01533         buf[0] = '\0';
01534         sn = X509_get_serialNumber(client_cert);
01535 
01536         /*
01537          *      For this next bit, we create the attributes *only* if
01538          *      we're at the client or issuing certificate, AND we
01539          *      have a user identity.  i.e. we don't create the
01540          *      attributes for RadSec connections.
01541          */
01542         if (identity && 
01543             (lookup <= 1) && sn && ((size_t) sn->length < (sizeof(buf) / 2))) {
01544                 char *p = buf;
01545                 int i;
01546 
01547                 for (i = 0; i < sn->length; i++) {
01548                         sprintf(p, "%02x", (unsigned int)sn->data[i]);
01549                         p += 2;
01550                 }
01551                 pairadd(certs,
01552                         pairmake(cert_attr_names[FR_TLS_SERIAL][lookup], buf, T_OP_SET));
01553         }
01554 
01555 
01556         /*
01557          *      Get the Expiration Date
01558          */
01559         buf[0] = '\0';
01560         asn_time = X509_get_notAfter(client_cert);
01561         if (identity && (lookup <= 1) && asn_time &&
01562             (asn_time->length < sizeof(buf))) {
01563                 memcpy(buf, (char*) asn_time->data, asn_time->length);
01564                 buf[asn_time->length] = '\0';
01565                 pairadd(certs,
01566                         pairmake(cert_attr_names[FR_TLS_EXPIRATION][lookup], buf, T_OP_SET));
01567         }
01568 
01569         /*
01570          *      Get the Subject & Issuer
01571          */
01572         subject[0] = issuer[0] = '\0';
01573         X509_NAME_oneline(X509_get_subject_name(client_cert), subject,
01574                           sizeof(subject));
01575         subject[sizeof(subject) - 1] = '\0';
01576         if (identity && (lookup <= 1) && subject[0] &&
01577             (strlen(subject) < MAX_STRING_LEN)) {
01578                 pairadd(certs,
01579                         pairmake(cert_attr_names[FR_TLS_SUBJECT][lookup], subject, T_OP_SET));
01580         }
01581 
01582         X509_NAME_oneline(X509_get_issuer_name(ctx->current_cert), issuer,
01583                           sizeof(issuer));
01584         issuer[sizeof(issuer) - 1] = '\0';
01585         if (identity && (lookup <= 1) && issuer[0] &&
01586             (strlen(issuer) < MAX_STRING_LEN)) {
01587                 pairadd(certs,
01588                         pairmake(cert_attr_names[FR_TLS_ISSUER][lookup], issuer, T_OP_SET));
01589         }
01590 
01591         /*
01592          *      Get the Common Name, if there is a subject.
01593          */
01594         X509_NAME_get_text_by_NID(X509_get_subject_name(client_cert),
01595                                   NID_commonName, common_name, sizeof(common_name));
01596         common_name[sizeof(common_name) - 1] = '\0';
01597         if (identity && (lookup <= 1) && common_name[0] && subject[0] &&
01598             (strlen(common_name) < MAX_STRING_LEN)) {
01599                 pairadd(certs,
01600                         pairmake(cert_attr_names[FR_TLS_CN][lookup], common_name, T_OP_SET));
01601         }
01602 
01603 #ifdef GEN_EMAIL
01604         /*
01605          *      Get the RFC822 Subject Alternative Name
01606          */

01608         if (lookup <= 1 && loc >= 0) {
01609                 X509_EXTENSION *ext = NULL;
01610                 GENERAL_NAMES *names = NULL;

01612

01614                     (names = X509V3_EXT_d2i(ext))) {

01621                                                 break;

01624                                                 pairmake(cert_attr_names[FR_TLS_SAN_EMAIL][lookup],
01625                                                          (char *) ASN1_STRING_data(name->d.rfc822Name), T_OP_SET));
01626                                         break;
01627                                 default:
01628                                         /* XXX TODO handle other SAN types */
01629                                         break;
01630                                 }
01631                         }
01632                 }
01633                 if (names != NULL)
01634                         sk_GENERAL_NAME_free(names);
01635         }
01636 #endif  /* GEN_EMAIL */
01637 
01638         /*
01639          *      If the CRL has expired, that might still be OK.
01640          */
01641         if (!my_ok &&
01642             (conf->allow_expired_crl) &&
01643             (err == X509_V_ERR_CRL_HAS_EXPIRED)) {
01644                 my_ok = 1;
01645                 X509_STORE_CTX_set_error( ctx, 0 );
01646         }
01647 
01648         if (!my_ok) {
01649                 const char *p = X509_verify_cert_error_string(err);
01650                 radlog(L_ERR,"--> verify error:num=%d:%s\n",err, p);
01651                 radius_pairmake(request, &request->packet->vps,
01652                                 "Module-Failure-Message", p, T_OP_SET);

01654         }

01657

01660                 break;
01661         case X509_V_ERR_CERT_NOT_YET_VALID:
01662         case X509_V_ERR_ERROR_IN_CERT_NOT_BEFORE_FIELD:
01663                 radlog(L_ERR, "notBefore=");
01664 #if 0
01665                 ASN1_TIME_print(bio_err, X509_get_notBefore(ctx->current_cert));
01666 #endif
01667                 break;
01668         case X509_V_ERR_CERT_HAS_EXPIRED:
01669         case X509_V_ERR_ERROR_IN_CERT_NOT_AFTER_FIELD:
01670                 radlog(L_ERR, "notAfter=");
01671 #if 0
01672                 ASN1_TIME_print(bio_err, X509_get_notAfter(ctx->current_cert));
01673 #endif
01674                 break;
01675         }
01676 
01677         /*
01678          *      If we're at the actual client cert, apply additional
01679          *      checks.
01680          */
01681         if (depth == 0) {
01682                 /*
01683                  *      If the conf tells us to, check cert issuer
01684                  *      against the specified value and fail
01685                  *      verification if they don't match.
01686                  */
01687                 if (conf->check_cert_issuer &&
01688                     (strcmp(issuer, conf->check_cert_issuer) != 0)) {
01689                         radlog(L_AUTH, "rlm_eap_tls: Certificate issuer (%s) does not match specified value (%s)!", issuer, conf->check_cert_issuer);
01690                         my_ok = 0;
01691                 }
01692 
01693                 /*
01694                  *      If the conf tells us to, check the CN in the
01695                  *      cert against xlat'ed value, but only if the
01696                  *      previous checks passed.
01697                  */
01698                 if (my_ok && conf->check_cert_cn) {
01699                         if (!radius_xlat(cn_str, sizeof(cn_str), conf->check_cert_cn, request, NULL, NULL)) {
01700                                 radlog(L_ERR, "rlm_eap_tls (%s): xlat failed.",
01701                                        conf->check_cert_cn);
01702                                 /* if this fails, fail the verification */
01703                                 my_ok = 0;
01704                         } else {
01705                                 RDEBUG2("checking certificate CN (%s) with xlat'ed value (%s)", common_name, cn_str);
01706                                 if (strcmp(cn_str, common_name) != 0) {
01707                                         radlog(L_AUTH, "rlm_eap_tls: Certificate CN (%s) does not match specified value (%s)!", common_name, cn_str);
01708                                         my_ok = 0;
01709                                 }
01710                         }
01711                 } /* check_cert_cn */
01712 
01713 #ifdef HAVE_OPENSSL_OCSP_H
01714                 if (my_ok && conf->ocsp_enable){
01715                         RDEBUG2("--> Starting OCSP Request");
01716                         if(X509_STORE_CTX_get1_issuer(&issuer_cert, ctx, client_cert)!=1) {
01717                                 radlog(L_ERR, "Error: Couldn't get issuer_cert for %s", common_name);
01718                         }

01720                 }
01721 #endif

01724                         char filename[256];

01726                         FILE *fp;
01727

01729                                  conf->verify_tmp_dir, progname);

01731                         if (fd < 0) {
01732                                 RDEBUG("Failed creating file in %s: %s",
01733                                        conf->verify_tmp_dir, strerror(errno));
01734                                 break;
01735                         }
01736 
01737                         fp = fdopen(fd, "w");
01738                         if (!fp) {
01739                                 RDEBUG("Failed opening file %s: %s",
01740                                        filename, strerror(errno));
01741                                 break;
01742                         }
01743

01745                                 fclose(fp);
01746                                 RDEBUG("Failed writing certificate to file");
01747                                 goto do_unlink;
01748                         }

01750 
01751                         if (!radius_pairmake(request, &request->packet->vps,
01752                                              "TLS-Client-Cert-Filename",

01754                                 RDEBUG("Failed creating TLS-Client-Cert-Filename");
01755 
01756                                 goto do_unlink;
01757                         }
01758
