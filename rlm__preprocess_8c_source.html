00111  *      ThomasJ --
00112  *      This hack strips out Cisco's VSA duplicities in lines
00113  *      (Cisco not implemented VSA's in standard way.
00114  *
00115  *      Cisco sends it's VSA attributes with the attribute name *again*
00116  *      in the string, like:  H323-Attribute = "h323-attribute=value".
00117  *      This sort of behaviour is nonsense.
00118  */
00119 static void cisco_vsa_hack(VALUE_PAIR *vp)
00120 {
00121         int             vendorcode;
00122         char            *ptr;
00123         char            newattr[MAX_STRING_LEN];
00124 
00125         for ( ; vp != NULL; vp = vp->next) {
00126                 vendorcode = vp->vendor;
00127                 if (!((vendorcode == 9) || (vendorcode == 6618))) continue; /* not a Cisco or Quintum VSA, continue */
00128 
00129                 if (vp->type != PW_TYPE_STRING) continue;
00130 
00131                 /*
00132                  *  No weird packing.  Ignore it.
00133                  */
00134                 ptr = strchr(vp->vp_strvalue, '='); /* find an '=' */
00135                 if (!ptr) continue;
00136 
00137                 /*
00138                  *      Cisco-AVPair's get packed as:
00139                  *
00140                  *      Cisco-AVPair = "h323-foo-bar = baz"
00141                  *      Cisco-AVPair = "h323-foo-bar=baz"
00142                  *
00143                  *      which makes sense only if you're a lunatic.
00144                  *      This code looks for the attribute named inside
00145                  *      of the string, and if it exists, adds it as a new
00146                  *      attribute.
00147                  */
00148                 if (vp->attribute == 1) {
00149                         const char *p;
00150                         DICT_ATTR       *dattr;
00151 
00152                         p = vp->vp_strvalue;
00153                         gettoken(&p, newattr, sizeof(newattr));
00154 
00155                         if ((dattr = dict_attrbyname(newattr)) != NULL) {
00156                                 VALUE_PAIR *newvp;
00157 
00158                                 /*
00159                                  *  Make a new attribute.
00160                                  */
00161                                 newvp = pairmake(newattr, ptr + 1, T_OP_EQ);
00162                                 if (newvp) {
00163                                         pairadd(&vp, newvp);
00164                                 }
00165                         }
00166                 } else {        /* h322-foo-bar = "h323-foo-bar = baz" */
00167                         /*
00168                          *      We strip out the duplicity from the
00169                          *      value field, we use only the value on
00170                          *      the right side of the '=' character.
00171                          */
00172                         strlcpy(newattr, ptr + 1, sizeof(newattr));
00173                         strlcpy((char *)vp->vp_strvalue, newattr,
00174                                 sizeof(vp->vp_strvalue));
00175                         vp->length = strlen((char *)vp->vp_strvalue);
00176                 }

00354 {

00410                 /*
00411                  *      Use "paircompare", which is a little more general...

00414                      (strcmp(i->name, name) == 0)) &&

00419                          *      Now add all attributes to the request list,
