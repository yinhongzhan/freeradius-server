00275 /*
00276  * Retrieve an fd (from pool) to use for socket connection.
00277  * It'd be simpler to use TLS but FR can have lots of threads
00278  * and we don't want to waste fd's that way.
00279  * We can't have a global fd because we'd then be pipelining
00280  * requests to otpd and we have no way to demultiplex
00281  * the responses.
00282  */
00283 static smsotp_fd_t * smsotp_getfd(const rlm_smsotp_t *opt)
00284 {
00285   int rc;
00286   smsotp_fd_t *fdp;
00287 
00288   /* walk the connection pool looking for an available fd */
00289   for (fdp = smsotp_fd_head; fdp; fdp = fdp->next) {
00290     rc = smsotp_pthread_mutex_trylock(&fdp->mutex);
00291     if (!rc)
00292       if (!strcmp(fdp->path, opt->smsotp_socket))       /* could just use == */
00293         break;
00294   }
00295 
00296   if (!fdp) {
00297     /* no fd was available, add a new one */
00298     fdp = rad_malloc(sizeof(*fdp));
00299     smsotp_pthread_mutex_init(&fdp->mutex, NULL);
00300     smsotp_pthread_mutex_lock(&fdp->mutex);
00301     /* insert new fd at head */
00302     smsotp_pthread_mutex_lock(&smsotp_fd_head_mutex);
00303     fdp->next = smsotp_fd_head;
00304     smsotp_fd_head = fdp;
00305     smsotp_pthread_mutex_unlock(&smsotp_fd_head_mutex);
00306     /* initialize */
00307     fdp->path = opt->smsotp_socket;
00308     fdp->fd = -1;
00309   }
00310 
00311   /* establish connection */
00312   if (fdp->fd == -1)
00313     fdp->fd = smsotp_connect(fdp->path);
00314 
00315   return fdp;
00316 }

00378 /*
00379  * Full write with logging, and close on failure.
00380  * Returns 0 on success, errno on failure.
00381  */
00382 static int smsotp_write(smsotp_fd_t *fdp, const char *buf, size_t len)
00383 {
00384   size_t nleft = len;
00385   ssize_t nwrote;
00386 
00387   while (nleft) {
00388     if ((nwrote = write(fdp->fd, &buf[len - nleft], nleft)) == -1) {
00389       if (errno == EINTR || errno == EPIPE) {
00390         continue;
00391       } else {
00392         (void) radlog(L_ERR, "rlm_smsotp: %s: write to socket: %s", __func__, strerror(errno));
00393         smsotp_putfd(fdp, 1);
00394         return errno;
00395       }
00396     }
00397     nleft -= nwrote;
00398   }
00399 
00400   return 0;
