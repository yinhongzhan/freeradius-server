00143 static uint8_t *dhcp_get_option(dhcp_packet_t *packet, size_t packet_size,
00144                                 unsigned int option)
00145                                 
00146 {
00147         int overload = 0;
00148         int field = DHCP_OPTION_FIELD;
00149         size_t where, size;
00150         uint8_t *data = packet->options;
00151 
00152         where = 0;
00153         size = packet_size - offsetof(dhcp_packet_t, options);
00154         data = &packet->options[where];
00155 
00156         while (where < size) {
00157                 if (data[0] == 0) { /* padding */
00158                         where++;
00159                         continue;
00160                 }
00161 
00162                 if (data[0] == 255) { /* end of options */
00163                         if ((field == DHCP_OPTION_FIELD) &&
00164                             (overload & DHCP_FILE_FIELD)) {
00165                                 data = packet->file;
00166                                 where = 0;
00167                                 size = sizeof(packet->file);
00168                                 field = DHCP_FILE_FIELD;
00169                                 continue;
00170 
00171                         } else if ((field == DHCP_FILE_FIELD) &&
00172                                    (overload && DHCP_SNAME_FIELD)) {
00173                                 data = packet->sname;
00174                                 where = 0;
00175                                 size = sizeof(packet->sname);
00176                                 field = DHCP_SNAME_FIELD;
00177                                 continue;
00178                         }
00179 
00180                         return NULL;
00181                 }
00182 
00183                 /*
00184                  *      We MUST have a real option here.
00185                  */
00186                 if ((where + 2) > size) {
00187                         fr_strerror_printf("Options overflow field at %u",
00188                                            (unsigned int) (data - (uint8_t *) packet));
00189                         return NULL;
00190                 }
00191 
00192                 if ((where + 2 + data[1]) > size) {
00193                         fr_strerror_printf("Option length overflows field at %u",
00194                                            (unsigned int) (data - (uint8_t *) packet));
00195                         return NULL;
00196                 }
00197 
00198                 if (data[0] == option) return data;
00199 
00200                 if (data[0] == 52) { /* overload sname and/or file */
00201                         overload = data[3];
00202                 }
00203 
00204                 where += data[1] + 2;
00205                 data += data[1] + 2;
00206         }

00427 {

00432          *      Take a pass at parsing it.
00433          */

00435         while (p < (data + data_len)) {
00436                 if ((p + 2) > (data + data_len)) goto make_tlv;
00437 
00438                 if ((p + p[1] + 2) > (data + data_len)) goto make_tlv;
00439                 p += 2 + p[1];

00441 
00442         /*
00443          *      Got here... must be well formed.

00450                 vp = paircreate(tlv->attribute | (p[0] << 8), DHCP_MAGIC_VENDOR, PW_TYPE_OCTETS);

00520                 if (alen > 253) return -1;
00521                 memcpy(vp->vp_strvalue, p , alen);
00522                 vp->vp_strvalue[alen] = '\0';
00523                 break;
00524

00526                 vp->type = PW_TYPE_OCTETS;
00527 
00528         case PW_TYPE_OCTETS:

00666                         fr_strerror_printf("Attribute too long %u %u",
00667                               p[0], p[1]);
00668                         continue;
00669                 }
00670                                 
00671                 da = dict_attrbyvalue(p[0], DHCP_MAGIC_VENDOR);
00672                 if (!da) {
00673                         fr_strerror_printf("Attribute not in our dictionary: %u",
00674                               p[0]);
00675                         continue;
00676                 }
00677 
00678                 vp = NULL;
00679                 num_entries = 1;
00680                 alen = p[1];
00681                 p += 2;
00682 
00683                 /*
00684                  *      Could be an array of bytes, integers, etc.
00685                  */
00686                 if (da->flags.array) {
00687                         switch (da->type) {
00688                         case PW_TYPE_BYTE:
00689                                 num_entries = alen;
00690                                 alen = 1;
00691                                 break;
00692 
00693                         case PW_TYPE_SHORT: /* ignore any trailing data */
00694                                 num_entries = alen >> 1;
00695                                 alen = 2;
00696                                 break;
00697 
00698                         case PW_TYPE_IPADDR:
00699                         case PW_TYPE_INTEGER:
00700                         case PW_TYPE_DATE: /* ignore any trailing data */
00701                                 num_entries = alen >> 2;
00702                                 alen = 4;
00703                                 break;
00704 
00705                         default:
00706 
00707                                 break; /* really an internal sanity failure */
00708                         }
00709                 }
00710 
00711                 /*
00712                  *      Loop over all of the entries, building VPs
00713                  */
00714                 for (i = 0; i < num_entries; i++) {
00715                         vp = pairmake(da->name, NULL, T_OP_EQ);
00716                         if (!vp) {
00717                                 fr_strerror_printf("Cannot build attribute %s",
00718                                         fr_strerror());
00719                                 pairfree(&head);
00720                                 return -1;
00721                         }
00722 
00723                         /*
00724                          *      Hack for ease of use.
00725                          */
00726                         if ((da->attr == 0x3d) &&
00727                             !da->flags.array &&
00728                             (alen == 7) && (*p == 1) && (num_entries == 1)) {
00729                                 vp->type = PW_TYPE_ETHERNET;
00730                                 memcpy(vp->vp_octets, p + 1, 6);
00731                                 vp->length = alen;
00732 
00733                         } else if (fr_dhcp_attr2vp(vp, p, alen) < 0) {
00734                                         pairfree(&vp);
00735                                         pairfree(&head);
00736                                         return -1;
00737                         }
00738 
00739                         *tail = vp;
00740                         while (*tail) {
00741                                 debug_pair(*tail);
00742                                 tail = &(*tail)->next;
00743                         }
00744                         p += alen;
00745                 } /* loop over array entries */
00746         } /* loop over the entire packet */
00747 
00748         /*
00749          *      If DHCP request, set ciaddr to zero.
00750          */
00751 
00752         /*
00753          *      Set broadcast flag for broken vendors, but only if
00754          *      giaddr isn't set.
00755          */
00756         memcpy(&giaddr, packet->data + 24, sizeof(giaddr));
00757         if (giaddr == htonl(INADDR_ANY)) {
00758                 /*
00759                  *      DHCP Opcode is request
00760                  */
00761                 vp = pairfind(head, 256, DHCP_MAGIC_VENDOR);
00762                 if (vp && vp->vp_integer == 3) {
00763                         /*
00764                          *      Vendor is "MSFT 98"
00765                          */
00766                         vp = pairfind(head, 63, DHCP_MAGIC_VENDOR);
00767                         if (vp && (strcmp(vp->vp_strvalue, "MSFT 98") == 0)) {
00768                                 vp = pairfind(head, 262, DHCP_MAGIC_VENDOR);
00769 
00770                                 /*
00771                                  *      Reply should be broadcast.
00772                                  */
00773                                 if (vp) vp->vp_integer |= 0x8000;
00774                                 packet->data[10] |= 0x80;                       
00775                         }
00776                 }
00777         }
00778 
00779         /*
00780          *      FIXME: Nuke attributes that aren't used in the normal
00781          *      header for discover/requests.
00782          */
00783         packet->vps = head;
00784 
00785         /*
00786          *      Client can request a LARGER size, but not a smaller
00787          *      one.  They also cannot request a size larger than MTU.
00788          */
00789         maxms = pairfind(packet->vps, 57, DHCP_MAGIC_VENDOR);
00790         mtu = pairfind(packet->vps, 26, DHCP_MAGIC_VENDOR);
00791 
00792         if (mtu && (mtu->vp_integer < DEFAULT_PACKET_SIZE)) {
00793                 fr_strerror_printf("DHCP Fatal: Client says MTU is smaller than minimum permitted by the specification.");
00794                 return -1;
00795         }
00796 
00797         if (maxms && (maxms->vp_integer < DEFAULT_PACKET_SIZE)) {
00798                 fr_strerror_printf("DHCP WARNING: Client says maximum message size is smaller than minimum permitted by the specification: fixing it");
00799                 maxms->vp_integer = DEFAULT_PACKET_SIZE;
00800         }
00801 
00802         if (maxms && mtu && (maxms->vp_integer > mtu->vp_integer)) {
00803                 fr_strerror_printf("DHCP WARNING: Client says MTU is smaller than maximum message size: fixing it");
00804                 maxms->vp_integer = mtu->vp_integer;
00805         }
00806 
00807         if (fr_debug_flag > 0) {
00808                 for (vp = packet->vps; vp != NULL; vp = vp->next) {
00809                         
00810                 }
00811         }
00812 
00813         if (fr_debug_flag) fflush(stdout);
00814 
00815         return 0;
00816 }

00819 static int attr_cmp(const void *one, const void *two)
00820 {
00821         const VALUE_PAIR * const *a = one;
00822         const VALUE_PAIR * const *b = two;
00823 
00824         /*
00825          *      DHCP-Message-Type is first, for simplicity.
00826          */
00827         if (((*a)->attribute == 53) &&
00828             (*b)->attribute != 53) return -1;
00829 
00830         /*
00831          *      Relay-Agent is last
00832          */
00833         if (((*a)->attribute == 82) &&
00834             (*b)->attribute != 82) return +1;
00835 
00836         return ((*a)->attribute - (*b)->attribute);
00837 }

00840 static size_t fr_dhcp_vp2attr(VALUE_PAIR *vp, uint8_t *p, size_t room)
00841 {
00842         size_t length;
00843         uint32_t lvalue;
00844 
00845         /*
00846          *      FIXME: Check room!
00847          */
00848         room = room;            /* -Wunused */
00849 
00850         /*
00851          *      Search for all attributes of the same
00852          *      type, and pack them into the same
00853          *      attribute.
00854          */
00855         switch (vp->type) {
00856         case PW_TYPE_BYTE:
00857                 length = 1;
00858                 *p = vp->vp_integer & 0xff;
00859                 break;
00860                 
00861         case PW_TYPE_SHORT:
00862                 length = 2;
00863                 p[0] = (vp->vp_integer >> 8) & 0xff;
00864                 p[1] = vp->vp_integer & 0xff;
00865                 break;
00866                 
00867         case PW_TYPE_INTEGER:
00868                 length = 4;
00869                 lvalue = htonl(vp->vp_integer);
00870                 memcpy(p, &lvalue, 4);
00871                 break;
00872                 
00873         case PW_TYPE_IPADDR:
00874                 length = 4;
00875                 memcpy(p, &vp->vp_ipaddr, 4);
00876                 break;
00877                 
00878         case PW_TYPE_ETHERNET:
00879                 length = 6;
00880                 memcpy(p, &vp->vp_ether, 6);
00881                 break;
00882                 
00883         case PW_TYPE_STRING:
00884                 memcpy(p, vp->vp_strvalue, vp->length);
00885                 length = vp->length;
00886                 break;
00887                 
00888         case PW_TYPE_TLV:       /* FIXME: split it on 255? */
00889                 memcpy(p, vp->vp_tlv, vp->length);
00890                 length = vp->length;
00891                 break;
00892 
00893         case PW_TYPE_OCTETS:
00894                 memcpy(p, vp->vp_octets, vp->length);
00895                 length = vp->length;
00896                 break;
00897                 
00898         default:
00899                 fr_strerror_printf("BAD TYPE2 %d", vp->type);
00900                 length = 0;
00901                 break;
00902         }
00903 
00904         return length;
00905 }

00907 static VALUE_PAIR *fr_dhcp_vp2suboption(VALUE_PAIR *vps)
00908 {
00909         int length;
00910         unsigned int attribute;
00911         uint8_t *ptr;
00912         VALUE_PAIR *vp, *tlv;
00913 
00914         attribute = vps->attribute & 0xffff00ff;
00915 
00916         tlv = paircreate(attribute, DHCP_MAGIC_VENDOR, PW_TYPE_TLV);
00917         if (!tlv) return NULL;
00918 
00919         tlv->length = 0;
00920         for (vp = vps; vp != NULL; vp = vp->next) {
00921                 /*
00922                  *      Group the attributes ONLY until we see a
00923                  *      non-TLV attribute.
00924                  */
00925                 if (!vp->flags.is_tlv ||
00926                     vp->flags.extended ||
00927                     ((vp->attribute & 0xffff00ff) != attribute)) {
00928                         break;
00929                 }
00930 
00931                 tlv->length += vp->length + 2;
00932         }
00933 
00934         if (!tlv->length) {
00935                 pairfree(&tlv);
00936                 return NULL;
00937         }
00938 
00939         tlv->vp_tlv = malloc(tlv->length);
00940         if (!tlv->vp_tlv) {
00941                 pairfree(&tlv);
00942                 return NULL;
00943         }
00944 
00945         ptr = tlv->vp_tlv;
00946         for (vp = vps; vp != NULL; vp = vp->next) {
00947                 if (!vp->flags.is_tlv ||
00948                     vp->flags.extended ||
00949                     ((vp->attribute & 0xffff00ff) != attribute)) {
00950                         break;
00951                 }
00952 
00953                 length = fr_dhcp_vp2attr(vp, ptr + 2,
00954                                          tlv->vp_tlv + tlv->length - ptr);
00955                 if (length > 255) return NULL;
00956 
00957                 /*
00958                  *      Pack the attribute.
00959                  */
00960                 ptr[0] = (vp->attribute & 0xff00) >> 8;
00961                 ptr[1] = length;
00962 
00963                 ptr += length + 2;
00964                 vp->flags.extended = 1;
00965         }
00966 
00967         return tlv;
00968 }

00970 
00971 int fr_dhcp_encode(RADIUS_PACKET *packet, RADIUS_PACKET *original)
00972 {
00973         int i, num_vps;
00974         uint8_t *p;
00975         VALUE_PAIR *vp;
00976         uint32_t lvalue, mms;
00977         size_t dhcp_size, length;
00978         dhcp_packet_t *dhcp;
00979         char buffer[1024];
00980 
00981         if (packet->data) return 0;
00982 
00983         packet->data = malloc(MAX_PACKET_SIZE);
00984         if (!packet->data) return -1;
00985 
00986         packet->data_len = MAX_PACKET_SIZE;
00987 
00988         if (packet->code == 0) packet->code = PW_DHCP_NAK;
00989 
00990         /*
00991          *      If there's a request, use it as a template.
00992          *      Otherwise, assume that the caller has set up
00993          *      everything appropriately.
00994          */
00995         if (original) {
00996                 packet->dst_ipaddr.af = AF_INET;
00997                 packet->src_ipaddr.af = AF_INET;
00998 
00999                 packet->dst_port = original->src_port;
01000                 packet->src_port = original->dst_port;
01001 
01002                 /*
01003                  *      Note that for DHCP, we NEVER send the response
01004                  *      to the source IP address of the request.  It
01005                  *      may have traversed multiple relays, and we
01006                  *      need to send the request to the relay closest
01007                  *      to the client.
01008                  *
01009                  *      if giaddr, send to giaddr.
01010                  *      if NAK, send broadcast.
01011                  *      if broadcast flag, send broadcast.
01012                  *      if ciaddr is empty, send broadcast.
01013                  *      otherwise unicast to ciaddr.
01014                  */
01015                 
01016                 /*
01017                  *      FIXME: alignment issues.  We likely don't want to
01018                  *      de-reference the packet structure directly..
01019                  */
01020                 dhcp = (dhcp_packet_t *) original->data;
01021                 
01022                 /*
01023                  *      Default to sending it via sendto(), without using
01024                  *      raw sockets.
01025                  */
01026                 packet->offset = 1;
01027 
01028                 if (dhcp->giaddr != htonl(INADDR_ANY)) {
01029                         packet->dst_ipaddr.ipaddr.ip4addr.s_addr = dhcp->giaddr;
01030                         
01031                         if (dhcp->giaddr != htonl(INADDR_LOOPBACK)) {
01032                                 packet->dst_port = original->dst_port;
01033                         } else {
01034                                 packet->dst_port = original->src_port; /* debugging */
01035                         }
01036                         
01037                 } else if ((packet->code == PW_DHCP_NAK) ||
01038                            ((dhcp->flags & 0x8000) != 0) ||
01039                            (dhcp->ciaddr == htonl(INADDR_ANY))) {
01040                         /*
01041                          *      The kernel will take care of sending it to
01042                          *      the broadcast MAC.
01043                          */
01044                         packet->dst_ipaddr.ipaddr.ip4addr.s_addr = htonl(INADDR_BROADCAST);
01045                         
01046                 } else {
01047                         /*
01048                          *      It was broadcast to us: we need to
01049                          *      broadcast the response.
01050                          */
01051                         if (packet->src_ipaddr.ipaddr.ip4addr.s_addr != dhcp->ciaddr) {
01052                                 packet->offset = 0;
01053                         }
01054                         packet->dst_ipaddr.ipaddr.ip4addr.s_addr = dhcp->ciaddr;
01055                 }
01056 
01057                 /*
01058                  *      Rewrite the source IP to be our own, if we know it.
01059                  */
01060                 if (packet->src_ipaddr.ipaddr.ip4addr.s_addr == htonl(INADDR_BROADCAST)) {
01061                         packet->src_ipaddr.ipaddr.ip4addr.s_addr = htonl(INADDR_ANY);
01062                 }
01063         } else {
01064                 memset(packet->data, 0, packet->data_len);
01065         }
01066 
01067         if (fr_debug_flag > 1) {
01068                 char type_buf[64];
01069                 const char *name = type_buf;
01070                 char src_ip_buf[256], dst_ip_buf[256];
01071                 
01072                 if ((packet->code >= PW_DHCP_DISCOVER) &&
01073                     (packet->code <= PW_DHCP_INFORM)) {
01074                         name = dhcp_message_types[packet->code - PW_DHCP_OFFSET];
01075                 } else {
01076                         snprintf(type_buf, sizeof(type_buf), "%d",
01077                                  packet->code - PW_DHCP_OFFSET);
01078                 }
01079 
01080                 DEBUG("Sending %s of id %08x from %s:%d to %s:%d\n",
01081                        name, (unsigned int) packet->id,
01082                        inet_ntop(packet->src_ipaddr.af,
01083                                  &packet->src_ipaddr.ipaddr,
01084                                  src_ip_buf, sizeof(src_ip_buf)),
01085                        packet->src_port,
01086                        inet_ntop(packet->dst_ipaddr.af,
01087                                  &packet->dst_ipaddr.ipaddr,
01088                                  dst_ip_buf, sizeof(dst_ip_buf)),
01089                        packet->dst_port);
01090 
01091                 if (fr_debug_flag) {
01092                         for (i = 256; i < 269; i++) {
01093                                 vp = pairfind(packet->vps, i,
01094                                               DHCP_MAGIC_VENDOR);
01095                                 if (!vp) continue;
01096 
01097                                 debug_pair(vp);
01098                         }
01099                 }
01100         }
01101 
01102         p = packet->data;
01103 
01104         mms = DEFAULT_PACKET_SIZE; /* maximum message size */
01105 
01106         if (original) {
01107                 /*
01108                  *      Clients can request a LARGER size, but not a
01109                  *      smaller one.  They also cannot request a size
01110                  *      larger than MTU.
01111                  */
01112                 vp = pairfind(original->vps, 57, DHCP_MAGIC_VENDOR);
01113                 if (vp && (vp->vp_integer > mms)) {
01114                         mms = vp->vp_integer;
01115                         
01116                         if (mms > MAX_PACKET_SIZE) mms = MAX_PACKET_SIZE;
01117                 }
01118         }
01119 
01120         /*
01121          *      RFC 3118: Authentication option.
01122          */
01123         vp = pairfind(packet->vps, 90, DHCP_MAGIC_VENDOR);
01124         if (vp) {
01125                 if (vp->length < 2) {
01126                         memset(vp->vp_octets + vp->length, 0,
01127                                2 - vp->length);
01128                         vp->length = 2;
01129                 }
01130 
01131                 if (vp->length < 3) {
01132                         struct timeval tv;
01133 
01134                         gettimeofday(&tv, NULL);
01135                         vp->vp_octets[2] = 0;
01136                         timeval2ntp(&tv, vp->vp_octets + 3);
01137                         vp->length = 3 + 8;
01138                 }
01139 
01140                 /*
01141                  *      Configuration token (clear-text token)
01142                  */
01143                 if (vp->vp_octets[0] == 0) {
01144                         VALUE_PAIR *pass;
01145                         vp->vp_octets[1] = 0;
01146 
01147                         pass = pairfind(packet->vps, PW_CLEARTEXT_PASSWORD, DHCP_MAGIC_VENDOR);
01148                         if (pass) {
01149                                 length = pass->length;
01150                                 if ((length + 11) > sizeof(vp->vp_octets)) {
01151                                         length -= ((length + 11) - sizeof(vp->vp_octets));
01152                                 }
01153                                 memcpy(vp->vp_octets + 11, pass->vp_strvalue,
01154                                        length);
01155                                 vp->length = length + 11;
01156                         } else {
01157                                 vp->length = 11 + 8;
01158                                 memset(vp->vp_octets + 11, 0, 8);
01159                                 vp->length = 11 + 8;
01160                         }
01161                 } else {        /* we don't support this type! */
01162                         fr_strerror_printf("DHCP-Authentication %d unsupported",
01163                                 vp->vp_octets[0]);
01164                 }
01165         }
01166 
01167         vp = pairfind(packet->vps, 256, DHCP_MAGIC_VENDOR);
01168         if (vp) {
01169                 *p++ = vp->vp_integer & 0xff;
01170         } else {
01171                 if (!original) {
01172                         *p++ = 1;       /* client message */
01173                 } else {
01174                         *p++ = 2;       /* server message */
01175                 }
01176         }
01177         *p++ = 1;               /* hardware type = ethernet */
01178         *p++ = 6;               /* 6 bytes of ethernet */
01179 
01180         vp = pairfind(packet->vps, 259, DHCP_MAGIC_VENDOR);
01181         if (vp) {
01182                 *p++ = vp->vp_integer & 0xff;
01183         } else {
01184                 *p++ = 0;               /* hops */
01185         }
01186 
01187         if (original) { /* Xid */
01188                 memcpy(p, original->data + 4, 4);
01189         } else {
01190                 lvalue = fr_rand();
01191                 memcpy(p, &lvalue, 4);
01192         }
01193         p += 4;
01194 
01195         memset(p, 0, 2);        /* secs are zero */
01196         p += 2;
01197 
01198         if (original) {
01199                 memcpy(p, original->data + 10, 6); /* copy flags && ciaddr */
01200         }
01201 
01202         /*
01203          *      Allow the admin to set the broadcast flag.
01204          */
01205         vp = pairfind(packet->vps, 262, DHCP_MAGIC_VENDOR);
01206         if (vp) {
01207                 p[0] |= (vp->vp_integer & 0xff00) >> 8;
01208                 p[1] |= (vp->vp_integer & 0xff);
01209         }
01210 
01211         p += 6;
01212 
01213         /*
01214          *      Set client IP address.
01215          */
01216         vp = pairfind(packet->vps, 264, DHCP_MAGIC_VENDOR); /* Your IP address */
01217         if (vp) {
01218                 lvalue = vp->vp_ipaddr;
01219         } else {
01220                 lvalue = htonl(INADDR_ANY);
01221         }
01222         memcpy(p, &lvalue, 4);  /* your IP address */
01223         p += 4;
01224 
01225         vp = pairfind(packet->vps, 265, DHCP_MAGIC_VENDOR); /* server IP address */
01226         if (!vp) vp = pairfind(packet->vps, 54, DHCP_MAGIC_VENDOR); /* identifier */
01227         if (vp) {
01228                 lvalue = vp->vp_ipaddr;
01229         } else {
01230                 lvalue = htonl(INADDR_ANY);
01231         }
01232         memcpy(p, &lvalue, 4);  /* Server IP address */
01233         p += 4;
01234 
01235         if (original) {
01236                 memcpy(p, original->data + 24, 4); /* copy gateway IP address */
01237         } else {
01238                 vp = pairfind(packet->vps, 266, DHCP_MAGIC_VENDOR);
01239                 if (vp) {
01240                         lvalue = vp->vp_ipaddr;
01241                 } else {
01242                         lvalue = htonl(INADDR_NONE);
01243                 }
01244                 memcpy(p, &lvalue, 4);
01245         }
01246         p += 4;
01247 
01248         if (original) {
01249                 memcpy(p, original->data + 28, DHCP_CHADDR_LEN);
01250         } else {
01251                 vp = pairfind(packet->vps, 267, DHCP_MAGIC_VENDOR);
01252                 if (vp) {
01253                         if (vp->length > DHCP_CHADDR_LEN) {
01254                                 memcpy(p, vp->vp_octets, DHCP_CHADDR_LEN);
01255                         } else {
01256                                 memcpy(p, vp->vp_octets, vp->length);
01257                         }
01258                 }
01259         }
01260         p += DHCP_CHADDR_LEN;
01261 
01262         /*
01263          *      Zero our sname && filename fields.
01264          */
01265         memset(p, 0, DHCP_SNAME_LEN + DHCP_FILE_LEN);
01266         p += DHCP_SNAME_LEN;
01267 
01268         /*
01269          *      Copy over DHCP-Boot-Filename.
01270          *
01271          *      FIXME: This copy should be delayed until AFTER the options
01272          *      have been processed.  If there are too many options for
01273          *      the packet, then they go into the sname && filename fields.
01274          *      When that happens, the boot filename is passed as an option,
01275          *      instead of being placed verbatim in the filename field.
01276          */
01277         vp = pairfind(packet->vps, 269, DHCP_MAGIC_VENDOR);
01278         if (vp) {
01279                 if (vp->length > DHCP_FILE_LEN) {
01280                         memcpy(p, vp->vp_strvalue, DHCP_FILE_LEN);
01281                 } else {
01282                         memcpy(p, vp->vp_strvalue, vp->length);
01283                 }
01284         }
01285         p += DHCP_FILE_LEN;
01286 
01287         lvalue = htonl(DHCP_OPTION_MAGIC_NUMBER); /* DHCP magic number */
01288         memcpy(p, &lvalue, 4);
01289         p += 4;
01290 
01291         /*
01292          *      Print the header.
01293          */
01294         if (fr_debug_flag > 1) {
01295                 uint8_t *pp = p;
01296 
01297                 p = packet->data;
01298 
01299                 for (i = 0; i < 14; i++) {
01300                         vp = pairmake(dhcp_header_names[i], NULL, T_OP_EQ);
01301                         if (!vp) {
01302                                 fr_strerror_printf("Parse error %s", fr_strerror());
01303                                 return -1;
01304                         }
01305                         
01306                         switch (vp->type) {
01307                         case PW_TYPE_BYTE:
01308                                 vp->vp_integer = p[0];
01309                                 vp->length = 1;
01310                                 break;
01311                                 
01312                         case PW_TYPE_SHORT:
01313                                 vp->vp_integer = (p[0] << 8) | p[1];
01314                                 vp->length = 2;
01315                                 break;
01316                                 
01317                         case PW_TYPE_INTEGER:
01318                                 memcpy(&vp->vp_integer, p, 4);
01319                                 vp->vp_integer = ntohl(vp->vp_integer);
01320                                 vp->length = 4;
01321                                 break;
01322                                 
01323                         case PW_TYPE_IPADDR:
01324                                 memcpy(&vp->vp_ipaddr, p, 4);
01325                                 vp->length = 4;
01326                                 break;
01327
