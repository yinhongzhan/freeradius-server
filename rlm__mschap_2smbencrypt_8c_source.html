00050 static void ntpwdhash (uint8_t *szHash, const char *szPassword)
00051 {
00052         char szUnicodePass[513];
00053         char nPasswordLen;
00054         int i;
00055 
00056         /*
00057          *      NT passwords are unicode.  Convert plain text password
00058          *      to unicode by inserting a zero every other byte
00059          */
00060         nPasswordLen = strlen(szPassword);
00061         for (i = 0; i < nPasswordLen; i++) {
00062                 szUnicodePass[i << 1] = szPassword[i];
00063                 szUnicodePass[(i << 1) + 1] = 0;
00064         }
00065 
00066         /* Encrypt Unicode password to a 16-byte MD4 hash */
00067         fr_md4_calc(szHash, (uint8_t *) szUnicodePass, (nPasswordLen<<1) );
