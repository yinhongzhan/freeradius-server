00026 /*
00027  * Standard sequence:
00028  *      INADDR_ANY : 68 -> INADDR_BROADCAST : 67        DISCOVER
00029  *      CLIENT_IP : 68 <- DHCP_SERVER_IP : 67           OFFER
00030  *      INADDR_ANY : 68 -> INADDR_BROADCAST : 67        REQUEST
00031  *      CLIENT_IP : 68 <- DHCP_SERVER_IP : 67           ACK
00032  *
00033  * Relay sequence:
00034  *      INADDR_ANY : 68 -> INADDR_BROADCAST : 67        DISCOVER
00035  *      RELAY_IP : 67 -> NEXT_SERVER_IP : 67            DISCOVER
00036  *                              (NEXT_SERVER_IP can be a relay itself)
00037  *      FIRST_RELAY_IP : 67 <- DHCP_SERVER_IP : 67      OFFER
00038  *      CLIENT_IP : 68 <- FIRST_RELAY_IP : 67           OFFER
00039  *      INADDR_ANY : 68 -> INADDR_BROADCAST : 67        REQUEST
00040  *      RELAY_IP : 67 -> NEXT_SERVER_IP : 67            REQUEST
00041  *                              (NEXT_SERVER_IP can be a relay itself)
00042  *      FIRST_RELAY_IP : 67 <- DHCP_SERVER_IP : 67      ACK
00043  *      CLIENT_IP : 68 <- FIRST_RELAY_IP : 67           ACK
00044  *
00045  * Note: NACK are broadcasted, rest is unicast, unless client asked
00046  * for a broadcast
00047  */

00070 {
00071         uint8_t maxhops = 16;
00072         VALUE_PAIR *vp;
00073         dhcp_socket_t *sock;
00074 
00075         rad_assert(request->packet->data[0] == 1);
00076 
00077         /*
00078          * Do the forward by ourselves, do not rely on dhcp_socket_send()
00079          */
00080         request->reply->code = 0;
00081 
00082         /*
00083          * It's invalid to have giaddr=0 AND a relay option
00084          */
00085         vp = pairfind(request->packet->vps, 266, DHCP_MAGIC_VENDOR); /* DHCP-Gateway-IP-Address */
00086         if ((vp && (vp->vp_ipaddr == htonl(INADDR_ANY))) &&
00087             pairfind(request->packet->vps, 82, DHCP_MAGIC_VENDOR)) { /* DHCP-Relay-Agent-Information */
00088                 DEBUG("DHCP: Received packet with giaddr = 0 and containing relay option: Discarding packet\n");
00089                 return 1;
00090         }
00091 
00092         /*
00093          * RFC 1542 (BOOTP), page 15
00094          *
00095          * Drop requests if hop-count > 16 or admin specified another value
00096          */
00097         if ((vp = pairfind(request->config_items, 271, DHCP_MAGIC_VENDOR))) { /* DHCP-Relay-Max-Hop-Count */
00098             maxhops = vp->vp_integer;
00099         }
00100         vp = pairfind(request->packet->vps, 259, DHCP_MAGIC_VENDOR); /* DHCP-Hop-Count */
00101         rad_assert(vp != NULL);
00102         if (vp->vp_integer > maxhops) {
00103                 DEBUG("DHCP: Number of hops is greater than %d: not relaying\n", maxhops);
00104                 return 1;
00105         } else {
00106             /* Increment hop count */
00107             vp->vp_integer++;
00108         }
00109 
00110         sock = request->listener->data;
00111 
00112         /*
00113          *      Forward the request to the next server using the
00114          *      incoming request as a template.
00115          */
00116         /* set SRC ipaddr/port to the listener ipaddr/port */
00117         request->packet->src_ipaddr.af = AF_INET;
00118         /* XXX sock->ipaddr == 0 (listening on '*') */
00119         request->packet->src_ipaddr.ipaddr.ip4addr.s_addr = sock->ipaddr.ipaddr.ip4addr.s_addr;
00120         request->packet->src_port = sock->port;
00121 
00122         vp = pairfind(request->config_items, 270, DHCP_MAGIC_VENDOR); /* DHCP-Relay-To-IP-Address */
00123         rad_assert(vp != NULL);
00124 
00125         /* set DEST ipaddr/port to the next server ipaddr/port */
00126         request->packet->dst_ipaddr.af = AF_INET;
00127         request->packet->dst_ipaddr.ipaddr.ip4addr.s_addr = vp->vp_ipaddr;
00128         request->packet->dst_port = request->packet->dst_port;
00129 
00130         if (fr_dhcp_encode(request->packet, NULL) < 0) {
00131                 DEBUG("dhcprelay_process_client_request: ERROR in fr_dhcp_encode\n");
00132                 return -1;
00133         }
00134 
00135         return fr_dhcp_send(request->packet);
00136 }

00238 {
00239         int rcode;
00240         VALUE_PAIR *vp;
00241 
00242         vp = pairfind(request->packet->vps, 53, DHCP_MAGIC_VENDOR); /* DHCP-Message-Type */
00243         if (vp) {
00244                 DICT_VALUE *dv = dict_valbyattr(53, DHCP_MAGIC_VENDOR, vp->vp_integer);
00245                 DEBUG("Trying sub-section dhcp %s {...}",
00246                       dv->name ? dv->name : "<unknown>");
00247                 rcode = module_post_auth(vp->vp_integer, request);
00248         } else {
00249                 DEBUG("DHCP: Failed to find DHCP-Message-Type in packet!");
00250                 rcode = RLM_MODULE_FAIL;
00251         }
00252 
00253         /*
00254          *      For messages from a client, look for Relay attribute,
00255          *      and forward it if necessary.
00256          */
00257         vp = NULL;
00258         if (request->packet->data[0] == 1) {
00259                 vp = pairfind(request->config_items, 270, DHCP_MAGIC_VENDOR);
00260         }
00261         if (vp) {
00262                 VALUE_PAIR *giaddr;
00263                 
00264                 /*
00265                  *      Find the original giaddr.
00266                  *      FIXME: Maybe look in the original packet?
00267                  *
00268                  *      It's invalid to have giaddr=0 AND a relay option
00269                  */
00270                 giaddr = pairfind(request->packet->vps, 266, DHCP_MAGIC_VENDOR);
00271                 if (giaddr && (giaddr->vp_ipaddr == htonl(INADDR_ANY))) {
00272                         if (pairfind(request->packet->vps, 82, DHCP_MAGIC_VENDOR)) {
00273                                 RDEBUG("DHCP: Received packet with giaddr = 0 and containing relay option: Discarding packet");
00274                                 return 1;
00275                         }
00276                 }
00277 
00278                 if (request->packet->data[3] > 10) {
00279                         RDEBUG("DHCP: Number of hops is greater than 10: not relaying");
00280                         return 1;
00281                 }
00282 
00283                 /*
00284                  *      Forward a reply...
00285                  */
00286                 pairfree(&request->reply->vps);
00287                 request->reply->vps = paircopy(request->packet->vps);
00288                 request->reply->code = request->packet->code;
00289                 request->reply->id = request->packet->id;
00290                 request->reply->src_ipaddr = request->packet->dst_ipaddr;
00291                 request->reply->src_port = request->packet->dst_port;
00292                 request->reply->dst_ipaddr.af = AF_INET;
00293                 request->reply->dst_ipaddr.ipaddr.ip4addr.s_addr = vp->vp_ipaddr;
00294                 /*
00295                  *      Don't change the destination port.  It's the
00296                  *      server port.
00297                  */
00298 
00299                 /*
00300                  *      Hop count goes up.
00301                  */
00302                 vp = pairfind(request->reply->vps, 259, DHCP_MAGIC_VENDOR);
00303                 if (vp) vp->vp_integer++;
00304                 
00305                 return 1;
00306         }
00307 
00308         /*
00309          *      Responses from a server.  Handle them differently.
00310          */
00311         if (request->packet->data[0] == 2) {
00312                 pairfree(&request->reply->vps);
00313                 request->reply->vps = paircopy(request->packet->vps);
00314                 request->reply->code = request->packet->code;
00315                 request->reply->id = request->packet->id;
00316 
00317                 /*
00318                  *      Delete any existing giaddr.  If we received a
00319                  *      message from the server, then we're NOT the
00320                  *      server.  So we must be the destination of the
00321                  *      giaddr field.
00322                  */
00323                 pairdelete(&request->packet->vps, 266, DHCP_MAGIC_VENDOR);
00324 
00325                 /*
00326                  *      Search for client IP address.
00327                  */
00328                 vp = pairfind(request->packet->vps, 264, DHCP_MAGIC_VENDOR);
00329                 if (!vp) {
00330                         request->reply->code = 0;
00331                         RDEBUG("DHCP: No YIAddr in the reply. Discarding packet");
00332                         return 1;
00333                 }
00334 
00335                 /*
00336                  *      FROM us, TO the client's IP, OUR port + 1.
00337                  */
00338                 request->reply->src_ipaddr = request->packet->dst_ipaddr;
00339                 request->reply->src_port = request->packet->dst_port;
00340                 request->reply->dst_ipaddr.af = AF_INET;
00341                 request->reply->dst_ipaddr.ipaddr.ip4addr.s_addr = vp->vp_ipaddr;
00342                 request->reply->dst_port = request->packet->dst_port + 1;
00343 
00344                 /*
00345                  *      Hop count goes down.
00346                  */
00347                 vp = pairfind(request->reply->vps, 259, DHCP_MAGIC_VENDOR);
00348                 if (vp && (vp->vp_integer > 0)) vp->vp_integer--;
00349 
00350                 /*
00351                  *      FIXME: Keep original somewhere?  If the
00352                  *      broadcast flags are set, use them here?
00353                  */
00354                 
00355                 return 1;
00356         }
00357 
00358         vp = pairfind(request->reply->vps, 53, DHCP_MAGIC_VENDOR); /* DHCP-Message-Type */
00359         if (vp) {
00360                 request->reply->code = vp->vp_integer;
00361                 if ((request->reply->code != 0) &&
00362                     (request->reply->code < PW_DHCP_OFFSET)) {
00363                         request->reply->code += PW_DHCP_OFFSET;
00364                 }
00365         }
00366         else switch (rcode) {
00367         case RLM_MODULE_OK:
00368         case RLM_MODULE_UPDATED:
00369                 if (request->packet->code == PW_DHCP_DISCOVER) {
00370                         request->reply->code = PW_DHCP_OFFER;
00371                         break;
00372 
00373                 } else if (request->packet->code == PW_DHCP_REQUEST) {
00374                         request->reply->code = PW_DHCP_ACK;
00375                         break;
00376                 }
00377                 request->reply->code = PW_DHCP_NAK;
00378                 break;
00379 
00380         default:
00381         case RLM_MODULE_REJECT:
00382         case RLM_MODULE_FAIL:
00383         case RLM_MODULE_INVALID:
00384         case RLM_MODULE_NOOP:
00385         case RLM_MODULE_NOTFOUND:       
00386                 if (request->packet->code == PW_DHCP_DISCOVER) {
00387                         request->reply->code = 0; /* ignore the packet */
00388                 } else {
00389                         request->reply->code = PW_DHCP_NAK;
00390                 }
00391                 break;
00392 
00393         case RLM_MODULE_HANDLED:
00394                 break;
00395         }
00396 
00397         /*
00398          *      Releases don't get replies.
00399          */
00400         if (request->packet->code == PW_DHCP_RELEASE) {
00401                 request->reply->code = 0;
00402         }
00403 
00404         return 1;
00405 }
