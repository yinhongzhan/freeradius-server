00068 static int sql_init_socket(SQLSOCK *sqlsocket, SQL_CONFIG *config)
00069 {
00070         LOGINREC *login;
00071         rlm_sql_freetds_sock *freetds_sock;
00072         
00073         if (!sqlsocket->conn) {
00074                 sqlsocket->conn = (rlm_sql_freetds_sock *)rad_malloc(sizeof(struct rlm_sql_freetds_sock));
00075                 if (!sqlsocket->conn) {
00076                         return -1;
00077                 }
00078         }
00079         
00080         if (dbinit() == FAIL) {
00081                 radlog(L_ERR, "rlm_sql_freetds: Unable to init FreeTDS");
00082                 return -1;              
00083         }
00084         
00085         dbsetversion(DBVERSION_80);
00086         dberrhandle(err_handler);
00087         
00088         // Timeout so that FreeTDS doesn't wait for ever.
00089         dbsetlogintime((unsigned long)config->query_timeout);
00090         dbsettime((unsigned long)config->query_timeout);
00091         
00092         freetds_sock = sqlsocket->conn;
00093         memset(freetds_sock, 0, sizeof(*freetds_sock));
00094         
00095         radlog(L_INFO, "rlm_sql_freetds (%s): Starting connect to FreeTDS/MSSQL server", config->xlat_name);
00096         
00097         if (!(login = dblogin())) {
00098                 radlog(L_ERR, "rlm_sql_freetds (%s): Unable to allocate login record", config->xlat_name);
00099                 return -1;
00100         }
00101         
00102         DBSETLUSER(login, config->sql_login);
00103         DBSETLPWD(login, config->sql_password);
00104         
00105         if ((freetds_sock->dbproc = dbopen(login, config->sql_server)) == FAIL) {
00106                 radlog(L_ERR, "rlm_sql_freetds (%s): Unable to connect to FreeTDS/MSSQL server %s@%s", 
00107                            config->xlat_name, config->sql_login, config->sql_server);
00108                 dbloginfree(login);
00109                 return -1;
00110         }
00111         
00112         dbloginfree(login);
00113         
00114         if ((dbuse(freetds_sock->dbproc, config->sql_db)) == FAIL) {
00115                 radlog(L_ERR, "rlm_sql_freetds (%s): Unable to select database on FreeTDS/MSSQL server %s@%s:%s", 
00116                            config->xlat_name, config->sql_login, config->sql_server, config->sql_db);
00117                 return -1;
00118         }
00119         
00120         /* I know this may look strange, but it sets a pointer to
00121          the freetds_sock struct so that it can be used within the
00122          query_timeout_handler function to be able to timeout properly */
00123         dbsetinterrupt(freetds_sock->dbproc, query_timeout_handler, query_timeout_handler);
00124         dbsetuserdata(freetds_sock->dbproc, (BYTE *)freetds_sock);
00125         
00126         return 0;
00127 }

00153 static int sql_query(SQLSOCK *sqlsocket, SQL_CONFIG *config, char *querystr)
00154 {
00155         rlm_sql_freetds_sock *freetds_sock = sqlsocket->conn;
00156         
00157         if (freetds_sock->dbproc == NULL || DBDEAD(freetds_sock->dbproc)) {
00158                 radlog(L_ERR, "rlm_sql_freetds (%s): Socket not connected", config->xlat_name);
00159                 return SQL_DOWN;
00160         }
00161         
00162         if ((dbcmd(freetds_sock->dbproc, querystr)) == FAIL) {
00163                 radlog(L_ERR, "rlm_sql_freetds (%s): Unable to allocate SQL query", config->xlat_name);
00164                 return -1;
00165         }
00166         
00167         if ((dbsqlexec(freetds_sock->dbproc)) == FAIL) {
00168                 radlog(L_ERR, "rlm_sql_freetds (%s): SQL query failed", config->xlat_name);
00169                 return -1;
00170         }
00171         
00172         return 0;
