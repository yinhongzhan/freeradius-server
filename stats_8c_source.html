00065 static void tv_sub(struct timeval *end, struct timeval *start,
00066                    struct timeval *elapsed)
00067 {
00068         elapsed->tv_sec = end->tv_sec - start->tv_sec;
00069         if (elapsed->tv_sec > 0) {
00070                 elapsed->tv_sec--;
00071                 elapsed->tv_usec = USEC;
00072         } else {
00073                 elapsed->tv_usec = 0;
00074         }
00075         elapsed->tv_usec += end->tv_usec;
00076         elapsed->tv_usec -= start->tv_usec;
00077         
00078         if (elapsed->tv_usec >= USEC) {
00079                 elapsed->tv_usec -= USEC;
00080                 elapsed->tv_sec++;
00081         }
00082 }

00084 static void stats_time(fr_stats_t *stats, struct timeval *start,
00085                        struct timeval *end)
00086 {
00087         struct timeval diff;
00088         uint32_t delay;
00089 
00090         if ((start->tv_sec == 0) || (end->tv_sec == 0) ||
00091             (end->tv_sec < start->tv_sec)) return;
00092 
00093         tv_sub(end, start, &diff);
00094 
00095         if (diff.tv_sec >= 10) {
00096                 stats->elapsed[7]++;
00097         } else {
00098                 int i;
00099                 uint32_t cmp;
00100 
00101                 delay = (diff.tv_sec * USEC) + diff.tv_usec;
00102 
00103                 cmp = 10;
00104                 for (i = 0; i < 7; i++) {
00105                         if (delay < cmp) {
00106                                 stats->elapsed[i]++;
00107                                 break;
00108                         }
00109                         cmp *= 10;
00110                 }
00111         }
00112 }

00802 #endif

00805         rad_assert(start->tv_sec <= end->tv_sec);

00807         /*
00808          *      Initialize it.

00812

00830                 ema->ema1 = micro;
00831                 ema->ema10 = micro;

00834

00836                 ema->ema1 += (diff / 1000000);
00837                 
00838                 diff = ema->f10 * (micro - ema->ema10);
00839                 ema->ema10 += (diff / 1000000);

00841

00844         DEBUG("time %d %d.%06d\t%d.%06d\t%d.%06d\n",
