00034 /*
00035  *      Return a short string showing the terminal server, port
00036  *      and calling station ID.
00037  */
00038 char *auth_name(char *buf, size_t buflen, REQUEST *request, int do_cli)
00039 {
00040         VALUE_PAIR      *cli;
00041         VALUE_PAIR      *pair;
00042         int             port = 0;
00043         const char      *tls = "";
00044 
00045         if ((cli = pairfind(request->packet->vps, PW_CALLING_STATION_ID, 0)) == NULL)
00046                 do_cli = 0;
00047         if ((pair = pairfind(request->packet->vps, PW_NAS_PORT, 0)) != NULL)
00048                 port = pair->vp_integer;
00049 
00050         if (request->packet->dst_port == 0) {
00051                 if (pairfind(request->packet->vps, PW_FREERADIUS_PROXIED_TO, 0)) {
00052                         tls = " via TLS tunnel";
00053                 } else {
00054                         tls = " via proxy to virtual server";
00055                 }
00056         }
00057 
00058         snprintf(buf, buflen, "from client %.128s port %u%s%.128s%s",
00059                         request->client->shortname, port,
00060                  (do_cli ? " cli " : ""), (do_cli ? (char *)cli->vp_strvalue : ""),
00061                  tls);
00062 
00063         return buf;
00064 }

00161  *
00162  *      Returns:        0  OK
00163  *                      -1 Password fail
00164  *                      -2 Rejected (Auth-Type = Reject, send Port-Message back)
00165  *                      1  End check & return, don't reply
00166  *
00167  *      NOTE: NOT the same as the RLM_ values !

00172         VALUE_PAIR *cur_config_item;

00174         int result;
00175         int auth_type_count = 0;

00177 
00178         /*
00179          *      Look for matching check items. We skip the whole lot
00180          *      if the authentication type is PW_AUTHTYPE_ACCEPT or
00181          *      PW_AUTHTYPE_REJECT.
00182          */
00183         cur_config_item = request->config_items;
00184         while(((auth_type_pair = pairfind(cur_config_item, PW_AUTH_TYPE, 0))) != NULL) {
00185                 auth_type = auth_type_pair->vp_integer;
00186                 auth_type_count++;
00187 
00188                 RDEBUG2("Found Auth-Type = %s",
00189                         dict_valnamebyattr(PW_AUTH_TYPE, 0, auth_type));
00190                 cur_config_item = auth_type_pair->next;
00191 
00192                 if (auth_type == PW_AUTHTYPE_REJECT) {
00193                         RDEBUG2("Auth-Type = Reject, rejecting user");
00194                         return -2;
00195                 }
00196         }
00197 
00198         /*
00199          *      Warn if more than one Auth-Type was found, because only the last
00200          *      one found will actually be used.
00201          */
00202         if (( auth_type_count > 1) && (debug_flag)) {
00203                 radlog_request(L_ERR, 0, request, "Warning:  Found %d auth-types on request for user '%s'",
00204                         auth_type_count, request->username->vp_strvalue);
00205         }
00206 
00207         /*
00208          *      This means we have a proxy reply or an accept and it wasn't
00209          *      rejected in the above loop. So that means it is accepted and we
00210          *      do no further authentication.
00211          */
00212         if ((auth_type == PW_AUTHTYPE_ACCEPT)
00213 #ifdef WITH_PROXY
00214             || (request->proxy)
00215 #endif
00216             ) {
00217                 RDEBUG2("Auth-Type = Accept, accepting the user");
00218                 return 0;
00219         }
00220 
00221         /*
00222          *      Check that Auth-Type has been set, and reject if not.
00223          *
00224          *      Do quick checks to see if Cleartext-Password or Crypt-Password have
00225          *      been set, and complain if so.
00226          */
00227         if (auth_type < 0) {
00228                 if (pairfind(request->config_items, PW_CRYPT_PASSWORD, 0) != NULL) {
00229                         RDEBUG2("WARNING: Please update your configuration, and remove 'Auth-Type = Crypt'");
00230                         RDEBUG2("WARNING: Use the PAP module instead.");
00231                 }
00232                 else if (pairfind(request->config_items, PW_CLEARTEXT_PASSWORD, 0) != NULL) {
00233                         RDEBUG2("WARNING: Please update your configuration, and remove 'Auth-Type = Local'");
00234                         RDEBUG2("WARNING: Use the PAP or CHAP modules instead.");
00235                 }
00236 
00237                 /*
00238                  *      The admin hasn't told us how to
00239                  *      authenticate the user, so we reject them!
00240                  *
00241                  *      This is fail-safe.
00242                  */
00243 
00244                 RDEBUG2("ERROR: No authenticate method (Auth-Type) found for the request: Rejecting the user");
00245                 return -2;
00246         }
00247 
00248         /*
00249          *      See if there is a module that handles
00250          *      this Auth-Type, and turn the RLM_ return
00251          *      status into the values as defined at
00252          *      the top of this function.
00253          */
00254         result = module_authenticate(auth_type, request);
00255         switch (result) {
00256                 /*
00257                  *      An authentication module FAIL
00258                  *      return code, or any return code that
00259                  *      is not expected from authentication,
00260                  *      is the same as an explicit REJECT!
00261                  */
00262                 case RLM_MODULE_FAIL:
00263                 case RLM_MODULE_INVALID:
00264                 case RLM_MODULE_NOOP:
00265                 case RLM_MODULE_NOTFOUND:
00266                 case RLM_MODULE_REJECT:
00267                 case RLM_MODULE_UPDATED:
00268                 case RLM_MODULE_USERLOCK:
00269                 default:
00270                         result = -1;
00271                         break;
00272                 case RLM_MODULE_OK:
00273                         result = 0;
00274                         break;
00275                 case RLM_MODULE_HANDLED:
00276                         result = 1;
00277                         break;
00278         }
00279 
00280         return result;
