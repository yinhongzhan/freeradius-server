00032 /*
00033  *      Define a structure for our module configuration.
00034  *
00035  *      These variables do not need to be in a structure, but it's
00036  *      a lot cleaner to do so, and a pointer to the structure can

00059  *      Check if account has expired, and if user may login now.

00061 static int expiration_authorize(void *instance, REQUEST *request)
00062 {
00063         rlm_expiration_t *data = (rlm_expiration_t *)instance;
00064         VALUE_PAIR *vp, *check_item = NULL;
00065         char msg[MAX_STRING_LEN];

00068                 /*
00069                 *      Has this user's password expired?
00070                 *
00071                 *      If so, remove ALL reply attributes,
00072                 *      and add our own Reply-Message, saying
00073                 *      why they're being rejected.
00074                 */
00075                 RDEBUG("Checking Expiration time: '%s'",check_item->vp_strvalue);
00076                 if (((time_t) check_item->vp_date) <= request->timestamp) {
00077                         char logstr[MAX_STRING_LEN];
00078                         VALUE_PAIR *module_fmsg_vp;
00079 
00080                         RDEBUG("Account has expired");
00081 
00082                         if (data->msg && data->msg[0]){
00083                                 if (!radius_xlat(msg, sizeof(msg), data->msg, request, NULL, NULL)) {
00084                                         radlog(L_ERR, "rlm_expiration: xlat failed.");
00085                                         return RLM_MODULE_FAIL;
00086                                 }
00087 
00088                                 vp = pairmake("Reply-Message", msg, T_OP_ADD);
00089                                 pairfree(&request->reply->vps);
00090                                 request->reply->vps = vp;
00091                         }
00092                         snprintf(logstr, sizeof(logstr), "Account has expired [Expiration %s]",check_item->vp_strvalue);
00093                         module_fmsg_vp = pairmake("Module-Failure-Message", logstr, T_OP_EQ);
00094                         pairadd(&request->packet->vps, module_fmsg_vp);

00096                         return RLM_MODULE_USERLOCK;
00097                 }
00098                 /*
00099                  *      Else the account hasn't expired, but it may do so
00100                  *      in the future.  Set Session-Timeout.
00101                  */
00102                 vp = pairfind(request->reply->vps, PW_SESSION_TIMEOUT, 0);
00103                 if (!vp) {

00105                                                PW_SESSION_TIMEOUT, 0,
00106                                                PW_TYPE_INTEGER);

00108 
00109                 } else if (vp->vp_date > ((uint32_t) (((time_t) check_item->vp_date) - request->timestamp))) {
00110                         vp->vp_date = (uint32_t) (((time_t) check_item->vp_date) - request->timestamp);
00111                 }
00112         }
