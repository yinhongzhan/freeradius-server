00035 /*
00036  * send an initial eap-leap request
00037  * ie access challenge to the user/peer.
00038 
00039  * Frame eap reply packet.
00040  * len = header + type + leap_typedata
00041  * leap_typedata = value_size + value
00042  */
00043 static int leap_initiate(UNUSED void *instance, EAP_HANDLER *handler)
00044 {
00045         leap_session_t  *session;
00046         LEAP_PACKET     *reply;
00047 
00048         DEBUG2("  rlm_eap_leap: Stage 2");
00049 
00050         /*
00051          *      LEAP requires a User-Name attribute
00052          */
00053         if (!handler->request->username) {
00054                 DEBUG2("  rlm_eap_leap: User-Name is required for EAP-LEAP authentication.");
00055                 return 0;
00056         }
00057 
00058         reply = eapleap_initiate(handler->eap_ds, handler->request->username);
00059         if (reply == NULL)
00060                 return 0;
00061 
00062         eapleap_compose(handler->eap_ds, reply);
00063 
00064         handler->opaque = malloc(sizeof(leap_session_t));
00065         if (!handler->opaque) {
00066           radlog(L_ERR, "rlm_eap_leap: Out of memory");
00067           eapleap_free(&reply);
00068           return 0;
00069         }
00070 
00071         /*
00072          *      Remember which stage we're in, and which challenge
00073          *      we sent to the AP.  The later stages will take care
00074          *      of filling in the peer response.
00075          */
00076         session = (leap_session_t *) handler->opaque;
00077         handler->free_opaque = free; /* just malloc'd memory */
00078 
00079         session->stage = 4;     /* the next stage we're in */
00080         memcpy(session->peer_challenge, reply->challenge, reply->count);
00081 
00082         DEBUG2("  rlm_eap_leap: Successfully initiated");
00083 
00084         /*
00085          *      The next stage to process the packet.
00086          */
00087         handler->stage = AUTHENTICATE;
00088 
00089         eapleap_free(&reply);
00090         return 1;
00091 }

00093 static int leap_authenticate(UNUSED void *instance, EAP_HANDLER *handler)
00094 {
00095         int             rcode;
00096         leap_session_t  *session;
00097         LEAP_PACKET     *packet;
00098         LEAP_PACKET     *reply;
00099         char*           username;
00100         VALUE_PAIR      *password;
00101 
00102         if (!handler->opaque) {
00103                 radlog(L_ERR, "rlm_eap_leap: Cannot authenticate without LEAP history");
00104                 return 0;
00105         }
00106         session = (leap_session_t *) handler->opaque;
00107         reply = NULL;
00108 
00109         /*
00110          *      Extract the LEAP packet.
00111          */
00112         if (!(packet = eapleap_extract(handler->eap_ds)))
00113                 return 0;
00114 
00115         username = (char *)handler->request->username->vp_strvalue;
00116 
00117         /*
00118          *      The password is never sent over the wire.
00119          *      Always get the configured password, for each user.
00120          */
00121         password = pairfind(handler->request->config_items, PW_CLEARTEXT_PASSWORD, 0);
00122         if (!password) password = pairfind(handler->request->config_items, PW_NT_PASSWORD, 0);
00123         if (!password) {
00124                 DEBUG2("rlm_eap_leap: No Cleartext-Password or NT-Password configured for this user");
00125                 eapleap_free(&packet);
00126                 return 0;
00127         }
00128 
00129         /*
00130          *      We've already sent the AP challenge.  This packet
00131          *      should contain the NtChallengeResponse
00132          */
00133         switch (session->stage) {
00134         case 4:                 /* Verify NtChallengeResponse */
00135                 DEBUG2("  rlm_eap_leap: Stage 4");
00136                 rcode = eapleap_stage4(packet, password, session);
00137                 session->stage = 6;
00138 
00139                 /*
00140                  *      We send EAP-Success or EAP-Fail, and not
00141                  *      any LEAP packet.  So we return here.
00142                  */
00143                 if (!rcode) {
00144                         handler->eap_ds->request->code = PW_EAP_FAILURE;
00145                         eapleap_free(&packet);
00146                         return 0;
00147                 }
00148 
00149                 handler->eap_ds->request->code = PW_EAP_SUCCESS;
00150 
00151                 /*
00152                  *      Do this only for Success.
00153                  */
00154                 handler->eap_ds->request->id = handler->eap_ds->response->id + 1;
00155                 handler->eap_ds->set_request_id = 1;
00156 
00157                 /*
00158                  *      LEAP requires a challenge in stage 4, not
00159                  *      an Access-Accept, which is normally returned
00160                  *      by eap_compose() in eap.c, when the EAP reply code
00161                  *      is EAP_SUCCESS.
00162                  */
00163                 handler->request->reply->code = PW_ACCESS_CHALLENGE;
00164                 eapleap_free(&packet);
00165                 return 1;
00166                 break;
00167 
00168         case 6:                 /* Issue session key */
00169                 DEBUG2("  rlm_eap_leap: Stage 6");
00170                 reply = eapleap_stage6(packet, handler->request,
00171                                        handler->request->username, password,
00172                                        session, &handler->request->reply->vps);
00173                 break;
00174 
00175                 /*
00176                  *      Stages 1, 3, and 5 are requests from the AP.
00177                  *      Stage 2 is handled by initiate()
00178                  */
00179         default:
00180                 radlog(L_ERR, "  rlm_eap_leap: Internal sanity check failed on stage");
00181                 break;
00182         }
00183 
00184         eapleap_free(&packet);
00185 
00186         /*
00187          *      Process the packet.  We don't care about any previous
00188          *      EAP packets, as
00189          */
00190         if (!reply) {
00191                 return 0;
00192         }
00193 
00194         eapleap_compose(handler->eap_ds, reply);
00195 
00196         eapleap_free(&reply);
00197         return 1;
