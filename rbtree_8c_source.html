00374 {

00379          *  delete node Z from tree  *

00385                 /* Y has a NIL node as a child */

00389                 Y = Z->Right;
00390                 while (Y->Left != NIL) Y = Y->Left;
00391         }
00392 
00393         /* X is Y's only child */
00394         if (Y->Left != NIL)
00395                 X = Y->Left;
00396         else
00397                 X = Y->Right;   /* may be NIL! */
00398 
00399         /* remove Y from the parent chain */
00400         Parent = Y->Parent;
00401         if (X != NIL) X->Parent = Parent;
00402 
00403         if (Parent)
00404                 if (Y == Parent->Left)
00405                         Parent->Left = X;
00406                 else
00407                         Parent->Right = X;
00408         else
00409                 tree->Root = X;
00410

00412                 if (tree->freeNode) tree->freeNode(Z->Data);
00413                 Z->Data = Y->Data;
00414                 Y->Data = NULL;

00416                 if (Y->Color == Black && X != NIL)
00417                         DeleteFixup(tree, X, Parent);
