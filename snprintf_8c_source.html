00083  * Note: this not a real log10()

00092 #else
00093 log_10(r)

00095 #endif

00099 
00100   if (r < 0.)
00101     r = -r;
00102 
00103   if (r < 1.) {
00104     while (result >= r) {result *= .1; i++;}
00105     return (-i);
00106   } else {
00107     while (result <= r) {result *= 10.; i++;}
00108     return (i - 1);
00109   }

00241 
00242   if (fract != (char **)0)

00399

00409              ((d > 0. && p->justify == RIGHT) ? 1:0) -

00413   PUT_SPACE(d, p);

00415     PUT_CHAR(*tmp, p);
