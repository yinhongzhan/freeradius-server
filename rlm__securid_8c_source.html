00058 static int securid_session_cmp(const void *a, const void *b)
00059 {
00060         int rcode;
00061         const SECURID_SESSION *one = a;
00062         const SECURID_SESSION *two = b;
00063 
00064         rad_assert(one != NULL);
00065         rad_assert(two != NULL);
00066 
00067         rcode = fr_ipaddr_cmp(&one->src_ipaddr, &two->src_ipaddr);

00436 static int securid_instantiate(CONF_SECTION *conf, void **instance)
00437 {
00438         rlm_securid_t *inst;
00439 
00440         /* Set up a storage area for instance data */
00441         inst = rad_malloc(sizeof(*inst));
00442         if (!inst)  return -1;
00443         memset(inst, 0, sizeof(*inst));
00444 
00445         /* If the configuration parameters can't be parsed, then fail. */
00446         if (cf_section_parse(conf, inst, module_config) < 0) {
00447                 radlog(L_ERR|L_CONS, "rlm_securid: Unable to parse configuration section.");
00448                 securid_detach(inst);
00449                 return -1;
00450         }
00451 
00452         /*
00453          *      Lookup sessions in the tree.  We don't free them in
00454          *      the tree, as that's taken care of elsewhere...
00455          */
00456         inst->session_tree = rbtree_create(securid_session_cmp, NULL, 0);
00457         if (!inst->session_tree) {
00458                 radlog(L_ERR|L_CONS, "rlm_securid: Cannot initialize session tree.");
00459                 securid_detach(inst);
00460                 return -1;
00461         }
00462 
00463         pthread_mutex_init(&(inst->session_mutex), NULL);
00464 
00465         *instance = inst;
00466         return 0;
00467 }

00470 /*
00471  *      Authenticate the user via one of any well-known password.
00472  */
00473 static int securid_authenticate(void *instance, REQUEST *request)
00474 {
00475         int rcode;
00476         rlm_securid_t *inst = instance;
00477         VALUE_PAIR *module_fmsg_vp;
00478         VALUE_PAIR *vp;
00479         char  buffer[MAX_STRING_LEN]="";
00480         const char *username=NULL, *password=NULL;
00481         char module_fmsg[MAX_STRING_LEN]="";
00482         
00483         /*
00484          *      We can only authenticate user requests which HAVE
00485          *      a User-Name attribute.
00486          */
00487         if (!request->username) {
00488                 radlog(L_AUTH, "rlm_securid: Attribute \"User-Name\" is required for authentication.");
00489                 return RLM_MODULE_INVALID;
00490         }
00491 
00492         if (!request->password) {
00493                 radlog_request(L_AUTH, 0, request, "Attribute \"Password\" is required for authentication.");
00494                 return RLM_MODULE_INVALID;
00495         }
00496 
00497         /*
00498          *      Clear-text passwords are the only ones we support.
00499          */
00500         if (request->password->attribute != PW_USER_PASSWORD) {
00501                 radlog_request(L_AUTH, 0, request, "Attribute \"User-Password\" is required for authentication. Cannot use \"%s\".", request->password->name);
00502                 return RLM_MODULE_INVALID;
00503         }
00504 
00505         /*
00506          *      The user MUST supply a non-zero-length password.
00507          */
00508         if (request->password->length == 0) {
00509                 snprintf(module_fmsg,sizeof(module_fmsg),"rlm_securid: empty password supplied");
00510                 module_fmsg_vp = pairmake("Module-Failure-Message", module_fmsg, T_OP_EQ);
00511                 pairadd(&request->packet->vps, module_fmsg_vp);
00512                 return RLM_MODULE_INVALID;
00513         }
00514 
00515         /*
00516          *      shortcuts
00517          */
00518         username = request->username->vp_strvalue;
00519         password = request->password->vp_strvalue;
00520         
00521         RDEBUG("User [%s] login attempt with password [%s]",
00522                username, password);
00523         
00524         rcode = securidAuth(inst, request, username, password,
00525                             buffer, sizeof(buffer));
00526         
00527         switch (rcode) {
00528         case RC_SECURID_AUTH_SUCCESS:
00529                 rcode = RLM_MODULE_OK;
00530                 break;
00531 
00532         case RC_SECURID_AUTH_CHALLENGE:
00533                 /* reply with Access-challenge message code (11) */
00534 
00535                 /* Generate Prompt attribute */
00536                 vp = paircreate(PW_PROMPT, 0, PW_TYPE_INTEGER);
00537                                 
00538                 rad_assert(vp != NULL);
00539                 vp->vp_integer = 0; /* no echo */
00540                 pairadd(&request->reply->vps, vp);
00541 
00542                 /* Mark the packet as a Acceess-Challenge Packet */
00543                 request->reply->code = PW_ACCESS_CHALLENGE;
00544                 RDEBUG("Sending Access-Challenge.");
00545                 rcode = RLM_MODULE_HANDLED;
00546                 break;
00547 
00548         case RC_SECURID_AUTH_FAILURE:
00549         case RC_SECURID_AUTH_ACCESS_DENIED_FAILURE:
00550         case RC_SECURID_AUTH_INVALID_SERVER_FAILURE:
00551         default:
00552                 rcode = RLM_MODULE_REJECT;
00553                 break;
00554         }
00555
