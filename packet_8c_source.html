00034 /*
00035  *      See if two packets are identical.
00036  *
00037  *      Note that we do NOT compare the authentication vectors.
00038  *      That's because if the authentication vector is different,
00039  *      it means that the NAS has given up on the earlier request.
00040  */
00041 int fr_packet_cmp(const RADIUS_PACKET *a, const RADIUS_PACKET *b)
00042 {
00043         int rcode;
00044 
00045         rcode = a->id - b->id;
00046         if (rcode != 0) return rcode;
00047 
00048         rcode = (int) a->src_port - (int) b->src_port;
00049         if (rcode != 0) return rcode;
00050 
00051         rcode = (int) a->dst_port - (int) b->dst_port;
00052         if (rcode != 0) return rcode;
00053 
00054         rcode = a->sockfd - b->sockfd;
00055         if (rcode != 0) return rcode;
00056 
00057         rcode = fr_ipaddr_cmp(&a->src_ipaddr, &b->src_ipaddr);
00058         if (rcode != 0) return rcode;
00059 
00060         return fr_ipaddr_cmp(&a->dst_ipaddr, &b->dst_ipaddr);
00061 }

00548 /*
00549  *      1 == ID was allocated & assigned
00550  *      0 == error allocating memory
00551  *      -1 == all ID's are used, caller should open a new socket.
00552  *
00553  *      Note that this ALSO assigns a socket to use, and updates
00554  *      packet->request->src_ipaddr && packet->request->src_port
00555  *
00556  *      In multi-threaded systems, the calls to id_alloc && id_free
00557  *      should be protected by a mutex.  This does NOT have to be
00558  *      the same mutex as the one protecting the insert/find/yank
00559  *      calls!
00560  *
00561  *      We assume that the packet has dst_ipaddr && dst_port
00562  *      already initialized.  We will use those to find an
00563  *      outgoing socket.  The request MAY also have src_ipaddr set.
00564  *
00565  *      We also assume that the sender doesn't care which protocol
00566  *      should be used.
00567  */
00568 int fr_packet_list_id_alloc(fr_packet_list_t *pl, int proto,
00569                             RADIUS_PACKET *request, void **pctx)
00570 {
00571         int i, j, k, fd, id, start_i, start_j, start_k;
00572         int src_any = 0;
00573         fr_packet_socket_t *ps;
00574 
00575         if ((request->dst_ipaddr.af == AF_UNSPEC) ||
00576             (request->dst_port == 0)) {
00577                 fr_strerror_printf("No destination address/port specified");
00578                 return 0;
00579         }
00580 
00581 #ifndef WITH_TCP
00582         if ((proto != 0) && (proto != IPPROTO_UDP)) {
00583                 fr_strerror_printf("Invalid destination protocol");
00584                 return 0;
00585         }
00586 #endif
00587 
00588         /*
00589          *      Special case: unspec == "don't care"
00590          */
00591         if (request->src_ipaddr.af == AF_UNSPEC) {
00592                 memset(&request->src_ipaddr, 0, sizeof(request->src_ipaddr));
00593                 request->src_ipaddr.af = request->dst_ipaddr.af;
00594         }
00595 
00596         src_any = fr_inaddr_any(&request->src_ipaddr);
00597         if (src_any < 0) return 0;
00598 
00599         /*
00600          *      MUST specify a destination address.
00601          */
00602         if (fr_inaddr_any(&request->dst_ipaddr) != 0) return 0;
00603 
00604         /*
00605          *      FIXME: Go to an LRU system.  This prevents ID re-use
00606          *      for as long as possible.  The main problem with that
00607          *      approach is that it requires us to populate the
00608          *      LRU/FIFO when we add a new socket, or a new destination,
00609          *      which can be expensive.
00610          *
00611          *      The LRU can be avoided if the caller takes care to free
00612          *      Id's only when all responses have been received, OR after
00613          *      a timeout.
00614          *
00615          *      Right now, the random approach is almost OK... it's
00616          *      brute-force over all of the available ID's, BUT using
00617          *      random numbers for everything spreads the load a bit.
00618          *
00619          *      The old method had a hash lookup on allocation AND
00620          *      on free.  The new method has brute-force on allocation,
00621          *      and near-zero cost on free.
00622          */
00623 
00624         id = fd = -1;
00625         start_i = fr_rand() & SOCKOFFSET_MASK;
00626 
00627 #define ID_i ((i + start_i) & SOCKOFFSET_MASK)
00628         for (i = 0; i < MAX_SOCKETS; i++) {
00629                 if (pl->sockets[ID_i].sockfd == -1) continue; /* paranoia */
00630 
00631                 ps = &(pl->sockets[ID_i]);
00632 
00633                 /*
00634                  *      This socket is marked as "don't use for new
00635                  *      packets".  But we can still receive packets
00636                  *      that are outstanding.
00637                  */
00638                 if (ps->dont_use) continue;
00639 
00640                 /*
00641                  *      All IDs are allocated: ignore it.
00642                  */
00643                 if (ps->num_outgoing == 256) continue;
00644 
00645 #ifdef WITH_TCP
00646                 if (ps->proto != proto) continue;
00647 #endif
00648 
00649                 /*
00650                  *      Address families don't match, skip it.
00651                  */
00652                 if (ps->src_ipaddr.af != request->dst_ipaddr.af) continue;
00653 
00654                 /*
00655                  *      MUST match dst port, if we have one.
00656                  */
00657                 if ((ps->dst_port != 0) && 
00658                     (ps->dst_port != request->dst_port)) continue;
00659 
00660                 /*
00661                  *      MUST match requested src port, if one has been given.
00662                  */
00663                 if ((request->src_port != 0) && 
00664                     (ps->src_port != request->src_port)) continue;
00665 
00666                 /*
00667                  *      We're sourcing from *, and they asked for a
00668                  *      specific source address: ignore it.
00669                  */
00670                 if (ps->src_any && !src_any) continue;
00671 
00672                 /*
00673                  *      We're sourcing from a specific IP, and they
00674                  *      asked for a source IP that isn't us: ignore
00675                  *      it.
00676                  */
00677                 if (!ps->src_any && !src_any &&
00678                     (fr_ipaddr_cmp(&request->src_ipaddr,
00679                                    &ps->src_ipaddr) != 0)) continue;
00680 
00681                 /*
00682                  *      UDP sockets are allowed to match
00683                  *      destination IPs exactly, OR a socket
00684                  *      with destination * is allowed to match
00685                  *      any requested destination.
00686                  *
00687                  *      TCP sockets must match the destination
00688                  *      exactly.  They *always* have dst_any=0,
00689                  *      so the first check always matches.
