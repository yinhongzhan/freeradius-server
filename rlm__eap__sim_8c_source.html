00194  * this code sends the challenge itself.
00195  *
00196  * Challenges will come from one of three places eventually:
00197  *
00198  * 1  from attributes like ATTRIBUTE_EAP_SIM_RANDx
00199  *            (these might be retrived from a database)
00200  *
00201  * 2  from internally implemented SIM authenticators
00202  *            (a simple one based upon XOR will be provided)
00203  *
00204  * 3  from some kind of SS7 interface.
00205  *
00206  * For now, they only come from attributes.
00207  * It might be that the best way to do 2/3 will be with a different
00208  * module to generate/calculate things.
00209  *
00210  */

00212 {

00215

00218         rad_assert(handler->request->reply);
00219 
00220         /* invps is the data from the client.
00221          * but, this is for non-protocol data here. We should
00222          * already have consumed any client originated data.
00223          */
00224         invps = &handler->request->packet->vps;
00225 
00226         /* outvps is the data to the client. */
00227         outvps= &handler->request->reply->vps;
00228 
00229         if ((debug_flag > 0) && fr_log_fp) {
00230                 fprintf(fr_log_fp, "+++> EAP-sim decoded packet:\n");
00231                 debug_pair_list(*invps);
00232         }
00233 
00234         /* okay, we got the challenges! Put them into an attribute */
00235         newvp = paircreate(ATTRIBUTE_EAP_SIM_BASE+PW_EAP_SIM_RAND,
00236                            0, PW_TYPE_OCTETS);
00237         memset(newvp->vp_strvalue,    0, 2); /* clear reserved bytes */
00238         memcpy(newvp->vp_strvalue+2+EAPSIM_RAND_SIZE*0, ess->keys.rand[0], EAPSIM_RAND_SIZE);
00239         memcpy(newvp->vp_strvalue+2+EAPSIM_RAND_SIZE*1, ess->keys.rand[1], EAPSIM_RAND_SIZE);
00240         memcpy(newvp->vp_strvalue+2+EAPSIM_RAND_SIZE*2, ess->keys.rand[2], EAPSIM_RAND_SIZE);
00241         newvp->length = 2+EAPSIM_RAND_SIZE*3;
00242         pairadd(outvps, newvp);
00243 
00244         /* set the EAP_ID - new value */
00245         newvp = paircreate(ATTRIBUTE_EAP_ID, 0, PW_TYPE_INTEGER);
00246         newvp->vp_integer = ess->sim_id++;
00247         pairreplace(outvps, newvp);
00248 
00249         /* make a copy of the identity */
00250         newvp = pairfind(*invps, ATTRIBUTE_EAP_SIM_BASE + PW_EAP_SIM_IDENTITY, 0);
00251         if (newvp) {
00252                 ess->keys.identitylen = newvp->length;
00253                 memcpy(ess->keys.identity, newvp->vp_octets, newvp->length);
00254         } else {
00255                 ess->keys.identitylen = strlen(handler->identity);
00256                 memcpy(ess->keys.identity, handler->identity, ess->keys.identitylen);
00257         }
00258 
00259         /* all set, calculate keys! */
00260         eapsim_calculate_keys(&ess->keys);
00261 
00262 #ifdef EAP_SIM_DEBUG_PRF

00264 #endif
00265 
00266         /*
00267          * need to include an AT_MAC attribute so that it will get
00268          * calculated. The NONCE_MT and the MAC are both 16 bytes, so
00269          * we store the NONCE_MT in the MAC for the encoder, which
00270          * will pull it out before it does the operation.
00271          */
00272 
00273         newvp = paircreate(ATTRIBUTE_EAP_SIM_BASE+PW_EAP_SIM_MAC,
00274                            0, PW_TYPE_OCTETS);
00275         memcpy(newvp->vp_strvalue, ess->keys.nonce_mt, 16);
00276         newvp->length = 16;
00277         pairreplace(outvps, newvp);
00278 
00279         newvp = paircreate(ATTRIBUTE_EAP_SIM_KEY, 0, PW_TYPE_OCTETS);
00280         memcpy(newvp->vp_strvalue, ess->keys.K_aut, 16);
00281         newvp->length = 16;
00282         pairreplace(outvps, newvp);
00283 
00284         /* the SUBTYPE, set to challenge. */
00285         newvp = paircreate(ATTRIBUTE_EAP_SIM_SUBTYPE, 0, PW_TYPE_INTEGER);
00286         newvp->vp_integer = eapsim_challenge;
00287         pairreplace(outvps, newvp);
00288 
00289         return 1;

00327 /*
00328  * run the server state machine.
00329  */
00330 static void eap_sim_stateenter(EAP_HANDLER *handler,
00331                                struct eap_sim_server_state *ess,
00332                                enum eapsim_serverstates newstate)
00333 {
00334         switch(newstate) {
00335         case eapsim_server_start:
00336                 /*
00337                  * send the EAP-SIM Start message, listing the
00338                  * versions that we support.
00339                  */
00340                 eap_sim_sendstart(handler);
00341                 break;
00342 
00343         case eapsim_server_challenge:
00344                 /*
00345                  * send the EAP-SIM Challenge message.
00346                  */
00347                 eap_sim_sendchallenge(handler);
00348                 break;
00349 
00350         case eapsim_server_success:
00351                 /*
00352                  * send the EAP Success message
00353                  */
00354                 eap_sim_sendsuccess(handler);
00355                 handler->eap_ds->request->code = PW_EAP_SUCCESS;
00356                 break;
00357 
00358         default:
00359                 /*
00360                  * nothing to do for this transition.
00361                  */
00362                 break;
00363         }
00364 
00365         ess->state = newstate;
00366 
00367         /* build the target packet */
00368         eap_sim_compose(handler);
00369 }

00382         outvps = handler->request->reply->vps;

00386         vp = pairfind(outvps, ATTRIBUTE_EAP_SIM_RAND1, 0);

00393         if(ess == NULL) {
00394                 DEBUG2("   no space for eap sim state");
00395                 return 0;
00396         }
00397 
00398         handler->opaque = ((void *)ess);
00399         handler->free_opaque = eap_sim_state_free;
00400 
00401         handler->stage = AUTHENTICATE;
00402 
00403         /*
00404          * save the keying material, because it could change on a subsequent
00405          * retrival.
00406          *
00407          */
00408         if((eap_sim_getchalans(outvps, 0, ess) +
00409             eap_sim_getchalans(outvps, 1, ess) +
00410             eap_sim_getchalans(outvps, 2, ess)) != 3)
00411         {
00412                 DEBUG2("   can not initiate sim, missing attributes");
00413                 return 0;
00414         }
00415 
00416         /*
00417          * this value doesn't have be strong, but it is good if it
00418          * is different now and then
00419          */
00420         time(&n);
00421         ess->sim_id = (n & 0xff);
00422 
00423         eap_sim_stateenter(handler, ess, eapsim_server_start);
00424 
00425         return 1;
00426 }

00429 /*
00430  * process an EAP-Sim/Response/Start.
00431  *
00432  * verify that client chose a version, and provided a NONCE_MT,
00433  * and if so, then change states to challenge, and send the new
00434  * challenge, else, resend the Request/Start.
00435  *
00436  */
00437 static int process_eap_sim_start(EAP_HANDLER *handler, VALUE_PAIR *vps)
00438 {
00439         VALUE_PAIR *nonce_vp, *selectedversion_vp;
00440         struct eap_sim_server_state *ess;
00441         uint16_t simversion;
00442 
00443         ess = (struct eap_sim_server_state *)handler->opaque;
00444 
00445         nonce_vp = pairfind(vps, ATTRIBUTE_EAP_SIM_BASE+PW_EAP_SIM_NONCE_MT, 0);
00446         selectedversion_vp = pairfind(vps, ATTRIBUTE_EAP_SIM_BASE+PW_EAP_SIM_SELECTED_VERSION, 0);
00447 
00448         if(nonce_vp == NULL ||
00449            selectedversion_vp == NULL) {
00450                 DEBUG2("   client did not select a version and send a NONCE");
00451                 eap_sim_stateenter(handler, ess, eapsim_server_start);
00452                 return 1;
00453         }
00454 
00455         /*
00456          * okay, good got stuff that we need. Check the version we found.
00457          */
00458         if(selectedversion_vp->length < 2) {
00459                 DEBUG2("   EAP-Sim version field is too short.");
00460                 return 0;
00461         }
00462         memcpy(&simversion, selectedversion_vp->vp_strvalue, sizeof(simversion));
00463         simversion = ntohs(simversion);
00464         if(simversion != EAP_SIM_VERSION) {
00465                 DEBUG2("   EAP-Sim version %d is unknown.", simversion);
00466                 return 0;
00467         }
00468 
00469         /* record it for later keying */
00470         memcpy(ess->keys.versionselect, selectedversion_vp->vp_strvalue,
00471                sizeof(ess->keys.versionselect));
00472 
00473         /*
00474          * double check the nonce size.
00475          */
00476         if(nonce_vp->length != 18) {
00477           DEBUG2("   EAP-Sim nonce_mt must be 16 bytes (+2 bytes padding), not %d", (int) nonce_vp->length);
00478                 return 0;
00479         }
00480         memcpy(ess->keys.nonce_mt, nonce_vp->vp_strvalue+2, 16);
00481 
00482         /* everything looks good, change states */
00483         eap_sim_stateenter(handler, ess, eapsim_server_challenge);
00484         return 1;
00485 }

00489  * process an EAP-Sim/Response/Challenge

00496 {
00497         struct eap_sim_server_state *ess;
00498         uint8_t srescat[EAPSIM_SRES_SIZE*3];
00499         uint8_t calcmac[EAPSIM_CALCMAC_SIZE];
00500 
00501         ess = (struct eap_sim_server_state *)handler->opaque;
00502 
00503         memcpy(srescat +(0*EAPSIM_SRES_SIZE), ess->keys.sres[0], EAPSIM_SRES_SIZE);
00504         memcpy(srescat +(1*EAPSIM_SRES_SIZE), ess->keys.sres[1], EAPSIM_SRES_SIZE);
00505         memcpy(srescat +(2*EAPSIM_SRES_SIZE), ess->keys.sres[2], EAPSIM_SRES_SIZE);
00506 
00507         /* verify the MAC, now that we have all the keys. */
00508         if(eapsim_checkmac(vps, ess->keys.K_aut,
00509                            srescat, sizeof(srescat),
00510                            calcmac)) {
00511                 DEBUG2("MAC check succeed\n");
00512         } else {
00513                 int i, j;
00514                 char macline[20*3];
00515                 char *m = macline;
00516 
00517                 j=0;
00518                 for (i = 0; i < EAPSIM_CALCMAC_SIZE; i++) {
00519                         if(j==4) {
00520                           *m++ = '_';
00521                           j=0;
00522                         }
00523                         j++;
00524 
00525                         sprintf(m, "%02x", calcmac[i]);
00526                         m = m + strlen(m);
00527                 }
00528                 DEBUG2("calculated MAC (%s) did not match", macline);
00529                 return 0;
00530         }
00531 
00532         /* everything looks good, change states */
00533         eap_sim_stateenter(handler, ess, eapsim_server_success);
00534         return 1;
00535 }
